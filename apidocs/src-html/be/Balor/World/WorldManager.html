017package be.Balor.World;
018
019import java.util.HashSet;
020import java.util.Set;
021import java.util.concurrent.ConcurrentMap;
022
023import org.bukkit.World;
024
025import be.Balor.Manager.Exceptions.WorldNotLoaded;
026import be.Balor.Tools.Converter.WorldConverter;
027import be.Balor.Tools.Debug.DebugLog;
028import be.Balor.Tools.Help.String.Str;
029
030import com.google.common.collect.MapMaker;
031
032/**
033 * @author Balor (aka Antoine Aflalo)
034 * 
035 */
036public class WorldManager {
037        private final ConcurrentMap<String, ACWorld> worlds = new MapMaker()
038                        .makeMap();
039        private IWorldFactory worldFactory;
040        private static final WorldManager INSTANCE = new WorldManager();
041
042        /**
043         * 
044         */
045        private WorldManager() {
046        }
047
048        /**
049         * @return the instance
050         */
051        public static WorldManager getInstance() {
052                return INSTANCE;
053        }
054
055        /**
056         * Add a new world
057         * 
058         * @param world
059         */
060        private synchronized boolean addWorld(final ACWorld world) {
061                final String name = world.getName();
062                if (name == null) {
063                        throw new NullPointerException();
064                }
065
066                final ACWorld ref = worlds.get(name);
067                if (ref != null) {
068                        return false;
069                }
070                worlds.put(name, world);
071                return true;
072        }
073
074        /**
075         * @param worldFactory
076         *            the worldFactory to set
077         */
078        public void setWorldFactory(final IWorldFactory worldFactory) {
079                if (this.worldFactory == null) {
080                        this.worldFactory = worldFactory;
081                }
082        }
083
084        /**
085         * To convert the ACWorld using an another factory
086         * 
087         * @param factory
088         */
089        public void convertFactory(final IWorldFactory factory) {
090                new WorldConverter(worldFactory, factory).convert();
091                DebugLog.INSTANCE.info("Setting new Factory");
092                this.worldFactory = factory;
093        }
094
095        ACWorld demandACWorld(final String name) throws WorldNotLoaded {
096                ACWorld result = worlds.get(name);
097                if (result == null) {
098                        final String found = Str.matchString(worlds.keySet(), name);
099                        if (found != null) {
100                                return worlds.get(found);
101                        }
102                        result = worldFactory.createWorld(name);
103                        addWorld(result);
104                        result = worlds.get(name);
105                }
106                return result;
107        }
108
109        ACWorld demandACWorld(final World world) {
110                final String name = world.getName();
111                ACWorld result = worlds.get(name);
112                if (result == null) {
113                        result = worldFactory.createWorld(world);
114                        addWorld(result);
115                        result = worlds.get(name);
116                }
117                return result;
118        }
119
120        /**
121         * Getting all the WarpsName using the world name as prefix
122         * 
123         * @return
124         */
125        public Set<String> getAllWarpList() {
126                final Set<String> warps = new HashSet<String>();
127                for (final ACWorld world : worlds.values()) {
128                        for (final String warp : world.getWarpList()) {
129                                warps.add(world.getName() + ":" + warp);
130                        }
131                }
132                return warps;
133
134        }
135
136}
