064        public static void initPrepStmt() {
065                SPAWN = Database.DATABASE
066                                .prepare("REPLACE INTO `ac_spawns` (`name`,`world_id`,`x`,`y`,`z`,`pitch`,`yaw`) VALUES (?,?,?,?,?,?,?)");
067                INSERT_INFO = Database.DATABASE
068                                .prepare("REPLACE INTO `ac_w_infos` (`key`,`world_id`,`info`) VALUES (?,?,?)");
069                DELETE_INFO = Database.DATABASE
070                                .prepare("DELETE FROM `ac_w_infos` WHERE `key`=? AND `world_id`=?");
071                INSERT_WARP = Database.DATABASE
072                                .prepare("REPLACE INTO `ac_warps` (`name`,`world_id`,`x`,`y`,`z`,`pitch`,`yaw`) VALUES (?,?,?,?,?,?,?)");
073                DELETE_WARP = Database.DATABASE
074                                .prepare("DELETE FROM `ac_warps` WHERE `name`=? AND `world_id`=?");
075                GET_INFOS = Database.DATABASE
076                                .prepare("SELECT `key`,`info` FROM `ac_w_infos` WHERE `world_id`=?");
077                GET_SPAWNS = Database.DATABASE
078                                .prepare("SELECT `name`,`x`,`y`,`z`,`yaw`,`pitch` FROM `ac_spawns` WHERE `world_id`=?");
079                GET_WARPS = Database.DATABASE
080                                .prepare("SELECT `name`,`x`,`y`,`z`,`yaw`,`pitch` FROM `ac_warps` WHERE `world_id`=?");
081        }

083        /**
084         * @param world
085         */
086        SQLWorld(final World world, final long id) {
087                super(world);
088                this.id = id;
089                synchronized (GET_INFOS) {
090                        try {
091                                GET_INFOS.clearParameters();
092                                GET_INFOS.setLong(1, id);
093                                ResultSet rs;
094                                synchronized (GET_INFOS.getConnection()) {
095                                        rs = GET_INFOS.executeQuery();
096                                }
097                                while (rs.next()) {
098                                        final String key = rs.getString("key");
099                                        if (key.startsWith("mobLimit:")) {
100                                                mobLimits.put(key.substring(9), rs.getInt("info"));
101                                        } else {
102                                                synchronized (SQLObjectContainer.yaml) {
103                                                        informations.put(key, SQLObjectContainer.yaml
104                                                                        .load(rs.getString("info")));
105                                                }
106                                        }
107                                }
108                                rs.close();
109                        } catch (final SQLException e) {
110                                ACLogger.severe(
111                                                "Problem while getting the informations of the world"
112                                                                + this.getName(), e);
113                        }
114                }
115                synchronized (GET_SPAWNS) {
116                        try {
117                                GET_SPAWNS.clearParameters();
118                                GET_SPAWNS.setLong(1, id);
119                                ResultSet rs;
120                                synchronized (GET_SPAWNS.getConnection()) {
121                                        rs = GET_SPAWNS.executeQuery();
122                                }
123                                while (rs.next()) {
124                                        final String name = rs.getString("name");
125                                        if (name.equals("none")) {
126                                                defaultSpawn = getLoc(rs);
127                                        } else {
128                                                final Location loc = getLoc(rs);
129                                                if (loc != null) {
130                                                        gSpawns.put(name, loc);
131                                                }
132                                        }
133                                }
134                        } catch (final SQLException e) {
135                                ACLogger.severe("Problem while getting the spawns of the world"
136                                                + this.getName(), e);
137                        }
138
139                }
140                synchronized (GET_WARPS) {
141                        try {
142                                GET_WARPS.clearParameters();
143                                GET_WARPS.setLong(1, id);
144                                ResultSet rs;
145                                synchronized (GET_WARPS.getConnection()) {
146                                        rs = GET_WARPS.executeQuery();
147                                }
148                                while (rs.next()) {
149                                        final String name = rs.getString("name");
150                                        final Location loc = getLoc(rs);
151                                        if (loc != null) {
152                                                warps.put(name, new Warp(name, loc));
153                                        }
154                                }
155                        } catch (final SQLException e) {
156                                ACLogger.severe("Problem while getting the warps of the world"
157                                                + this.getName(), e);
158                        }
159
160                }
161        }
162
163        private Location getLoc(final ResultSet rs) {
164                try {
165                        return new Location(this.getHandle(), rs.getDouble("x"),
166                                        rs.getDouble("y"), rs.getDouble("z"), rs.getFloat("yaw"),
167                                        rs.getFloat("pitch"));
168                } catch (final SQLException e) {
169                        ACLogger.severe(
170                                        "Problem while getting the location from SQL of world "
171                                                        + this.getName(), e);
172                        return null;
173                }
174        }

196                                }
197                        } catch (final SQLException e) {
198                                ACLogger.severe("Problem while setting the global spawn", e);

244        /*
245         * (Non javadoc)
246         * 
247         * @see be.Balor.World.ACWorld#getWarp(java.lang.String)
248         */
249        @Override
250        public Warp getWarp(final String name) throws WorldNotLoaded,
251                        IllegalArgumentException {
252                if (name == null || (name != null && name.isEmpty())) {
253                        throw new IllegalArgumentException("Name can't be null or Empty");
254                }
255                Warp warp = warps.get(name);
256                if (warp == null) {
257                        final String warpName = Str.matchString(getWarpList(), name);
258                        if (warpName == null) {
259                                return null;
260                        }
261                        warp = warps.get(warpName);
262                }
263                return warp;
264        }

371         * (Non javadoc)
372         * 
373         * @see be.Balor.World.ACWorld#getInformations()
374         */
375        @Override
376        public Map<String, String> getInformationsList() {
377                final TreeMap<String, String> result = new TreeMap<String, String>();
378                synchronized (informations) {
379                        for (final Entry<String, Object> entry : informations.entrySet()) {
380                                result.put(entry.getKey(), entry.getValue().toString());
381                        }
382                }
383                synchronized (mobLimits) {
384                        for (final Entry<String, Integer> entry : mobLimits.entrySet()) {
385                                result.put("Limit on " + entry.getKey(), entry.getValue()
386                                                .toString());
387                        }
388                }
389                return result;
390        }

443                                                DELETE_INFO.executeUpdate();
444                                        }
445                                } catch (final SQLException e) {
446                                        ACLogger.severe("Problem while deleting information ", e);

448                        }
449                        return true;
450                }
451                return false;
452        }
