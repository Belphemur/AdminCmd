017package be.Balor.World.sql;
018
019import java.io.File;
020import java.sql.PreparedStatement;
021import java.sql.ResultSet;
022import java.sql.SQLException;
023import java.util.HashMap;
024import java.util.Map;
025
026import lib.SQL.PatPeter.SQLibrary.Database;
027
028import org.bukkit.World;
029import org.bukkit.WorldCreator;
030
031import be.Balor.Manager.Exceptions.WorldNotLoaded;
032import be.Balor.Tools.Debug.ACLogger;
033import be.Balor.World.ACWorld;
034import be.Balor.World.IWorldFactory;
035import be.Balor.bukkit.AdminCmd.ACPluginManager;
036
037/**
038 * @author Balor (aka Antoine Aflalo)
039 * 
040 */
041public class SQLWorldFactory implements IWorldFactory {
042        private final PreparedStatement insertWorld, getWorld;
043        private final Map<String, World> bukkitWorlds = new HashMap<String, World>();
044
045        /**
046 * 
047 */
048        public SQLWorldFactory() {
049                insertWorld = Database.DATABASE
050                                .prepare("INSERT INTO `ac_worlds` (`name`) VALUES (?)");
051                getWorld = Database.DATABASE
052                                .prepare("SELECT id FROM ac_worlds WHERE name=?");
053                for (final World w : ACPluginManager.getServer().getWorlds()) {
054                        bukkitWorlds.put(w.getName().toLowerCase(), w);
055                }
056        }
057
058        /*
059         * (non-Javadoc)
060         * 
061         * @see be.Balor.World.IWorldFactory#createWorld(java.lang.String)
062         */
063        @Override
064        public synchronized ACWorld createWorld(final String worldName)
065                        throws WorldNotLoaded {
066                World w = bukkitWorlds.get(worldName.toLowerCase());
067                if (w == null) {
068                        File worldFile = new File(ACPluginManager.getServer()
069                                        .getWorldContainer(), worldName);
070                        if (!isExistingWorld(worldFile)) {
071                                worldFile = new File(ACPluginManager.getServer()
072                                                .getWorldContainer(), worldName.toLowerCase());
073                        }
074                        if (!isExistingWorld(worldFile)) {
075                                worldFile = new File(ACPluginManager.getServer()
076                                                .getWorldContainer(), worldName.substring(0, 1)
077                                                .toUpperCase() + worldName.substring(1).toLowerCase());
078                        }
079                        if (isExistingWorld(worldFile)) {
080                                w = ACPluginManager.getServer().createWorld(
081                                                new WorldCreator(worldFile.getName()));
082                                bukkitWorlds.put(w.getName().toLowerCase(), w);
083                                return createWorld(w);
084                        }
085                        throw new WorldNotLoaded(worldName);
086                }
087                return createWorld(w);
088        }
089
090        /**
091         * @param worldFile
092         * @return
093         */
094        private boolean isExistingWorld(final File worldFile) {
095                return worldFile.exists() && worldFile.isDirectory();
096        }
097
098        /*
099         * (non-Javadoc)
100         * 
101         * @see be.Balor.World.IWorldFactory#createWorld(org.bukkit.World)
102         */
103        @Override
104        public synchronized ACWorld createWorld(final World world) {
105                ResultSet rs = null;
106                final String worldName = world.getName();
107                try {
108                        getWorld.clearParameters();
109                        getWorld.setString(1, worldName);
110
111                        synchronized (getWorld.getConnection()) {
112                                rs = getWorld.executeQuery();
113                        }
114                        if (rs.next()) {
115                                return new SQLWorld(world, rs.getLong(1));
116                        } else {
117                                rs.close();
118                                insertWorld.clearParameters();
119                                insertWorld.setString(1, worldName);
120                                synchronized (insertWorld.getConnection()) {
121                                        insertWorld.executeUpdate();
122                                }
123                                rs = insertWorld.getGeneratedKeys();
124                                if (rs.next()) {
125                                        return new SQLWorld(world, rs.getLong(1));
126                                } else {
127                                        return null;
128                                }
129                        }
130                } catch (final SQLException e) {
131                        ACLogger.severe("Can't create an ACWorld for the World "
132                                        + worldName, e);
133                } finally {
134                        if (rs != null) {
135                                try {
136                                        rs.close();
137                                } catch (final SQLException e) {
138                                }
139                        }
140                }
141
142                return null;
143
144        }
145
146}
