017    package be.Balor.World;
018    
019    import javax.persistence.MappedSuperclass;
020    
021    import org.bukkit.Location;
022    import org.bukkit.World;
023    
024    import be.Balor.Manager.Exceptions.WorldNotLoaded;
025    import be.Balor.bukkit.AdminCmd.ACPluginManager;
026    
027    /**
028     * @author Balor (aka Antoine Aflalo)
029     * 
030     */
031    @MappedSuperclass
032    public class SimpleLocation {
033            private String world;
034            private double x;
035            private double y;
036            private double z;
037            private float pitch;
038            private float yaw;
039    
040            public SimpleLocation(final Location loc) {
041                    setLocationValue(loc);
042            }
043    
044            public void setLocationValue(final Location loc) {
045                    x = loc.getX();
046                    y = loc.getY();
047                    z = loc.getZ();
048                    yaw = loc.getYaw();
049                    pitch = loc.getPitch();
050                    world = loc.getWorld().getName();
051            }
052    
053            public Location getLocation() throws WorldNotLoaded {
054                    final World w = ACPluginManager.getServer().getWorld(world);
055                    if (w == null)
056                            throw new WorldNotLoaded(world);
057                    else {
058                            return new Location(w, x, y, z, yaw, pitch);
059                    }
060            }
061    
062            /**
063             * 
064             */
065            public SimpleLocation() {
066            }
067    
068            /**
069             * @return the world
070             */
071            public String getWorld() {
072                    return world;
073            }
074    
075            /**
076             * @return the x
077             */
078            public double getX() {
079                    return x;
080            }
081    
082            /**
083             * @return the y
084             */
085            public double getY() {
086                    return y;
087            }
088    
089            /**
090             * @return the z
091             */
092            public double getZ() {
093                    return z;
094            }
095    
096            /**
097             * @return the pitch
098             */
099            public float getPitch() {
100                    return pitch;
101            }
102    
103            /**
104             * @return the yaw
105             */
106            public float getYaw() {
107                    return yaw;
108            }
109    
110            /**
111             * @param world
112             *            the world to set
113             */
114            public void setWorld(final String world) {
115                    this.world = world;
116            }
117    
118            /**
119             * @param x
120             *            the x to set
121             */
122            public void setX(final double x) {
123                    this.x = x;
124            }
125    
126            /**
127             * @param y
128             *            the y to set
129             */
130            public void setY(final double y) {
131                    this.y = y;
132            }
133    
134            /**
135             * @param z
136             *            the z to set
137             */
138            public void setZ(final double z) {
139                    this.z = z;
140            }
141    
142            /**
143             * @param pitch
144             *            the pitch to set
145             */
146            public void setPitch(final float pitch) {
147                    this.pitch = pitch;
148            }
149    
150            /**
151             * @param yaw
152             *            the yaw to set
153             */
154            public void setYaw(final float yaw) {
155                    this.yaw = yaw;
156            }
157    
158            /*
159             * (non-Javadoc)
160             * 
161             * @see java.lang.Object#hashCode()
162             */
163            @Override
164            public int hashCode() {
165                    final int prime = 31;
166                    int result = 1;
167                    result = prime * result + Float.floatToIntBits(pitch);
168                    result = prime * result + ((world == null) ? 0 : world.hashCode());
169                    long temp;
170                    temp = Double.doubleToLongBits(x);
171                    result = prime * result + (int) (temp ^ (temp >>> 32));
172                    temp = Double.doubleToLongBits(y);
173                    result = prime * result + (int) (temp ^ (temp >>> 32));
174                    result = prime * result + Float.floatToIntBits(yaw);
175                    temp = Double.doubleToLongBits(z);
176                    result = prime * result + (int) (temp ^ (temp >>> 32));
177                    return result;
178            }
179    
180            /*
181             * (non-Javadoc)
182             * 
183             * @see java.lang.Object#equals(java.lang.Object)
184             */
185            @Override
186            public boolean equals(final Object obj) {
187                    if (this == obj)
188                            return true;
189                    if (obj == null)
190                            return false;
191                    if (!(obj instanceof SimpleLocation))
192                            return false;
193                    final SimpleLocation other = (SimpleLocation) obj;
194                    if (Float.floatToIntBits(pitch) != Float.floatToIntBits(other.pitch))
195                            return false;
196                    if (world == null) {
197                            if (other.world != null)
198                                    return false;
199                    } else if (!world.equals(other.world))
200                            return false;
201                    if (Double.doubleToLongBits(x) != Double.doubleToLongBits(other.x))
202                            return false;
203                    if (Double.doubleToLongBits(y) != Double.doubleToLongBits(other.y))
204                            return false;
205                    if (Float.floatToIntBits(yaw) != Float.floatToIntBits(other.yaw))
206                            return false;
207                    if (Double.doubleToLongBits(z) != Double.doubleToLongBits(other.z))
208                            return false;
209                    return true;
210            }
211    
212    }
