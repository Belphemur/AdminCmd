017package be.Balor.World;
018
019import org.bukkit.Location;
020import org.bukkit.World;
021
022import be.Balor.Manager.Exceptions.WorldNotLoaded;
023
024/**
025 * @author Balor (aka Antoine Aflalo)
026 * 
027 */
028public class SimpleLocation {
029        private String world;
030        private double x;
031        private double y;
032        private double z;
033        private float pitch;
034        private float yaw;
035
036        public SimpleLocation(final Location loc) {
037                setLocationValue(loc);
038        }
039
040        public void setLocationValue(final Location loc) {
041                x = loc.getX();
042                y = loc.getY();
043                z = loc.getZ();
044                yaw = loc.getYaw();
045                pitch = loc.getPitch();
046                world = loc.getWorld().getName();
047        }
048
049        public Location getLocation() throws WorldNotLoaded {
050                final World w = ACWorld.getWorld(world).getHandle();
051                if (w == null) {
052                        throw new WorldNotLoaded(world);
053                } else {
054                        return new Location(w, x, y, z, yaw, pitch);
055                }
056        }
057
058        /**
059         * 
060         */
061        public SimpleLocation() {
062        }
063
064        /**
065         * @return the world
066         */
067        public String getWorld() {
068                return world;
069        }
070
071        /**
072         * @return the x
073         */
074        public double getX() {
075                return x;
076        }
077
078        /**
079         * @return the y
080         */
081        public double getY() {
082                return y;
083        }
084
085        /**
086         * @return the z
087         */
088        public double getZ() {
089                return z;
090        }
091
092        /**
093         * @return the pitch
094         */
095        public float getPitch() {
096                return pitch;
097        }
098
099        /**
100         * @return the yaw
101         */
102        public float getYaw() {
103                return yaw;
104        }
105
106        /**
107         * @param world
108         *            the world to set
109         */
110        public void setWorld(final String world) {
111                this.world = world;
112        }
113
114        /**
115         * @param x
116         *            the x to set
117         */
118        public void setX(final double x) {
119                this.x = x;
120        }
121
122        /**
123         * @param y
124         *            the y to set
125         */
126        public void setY(final double y) {
127                this.y = y;
128        }
129
130        /**
131         * @param z
132         *            the z to set
133         */
134        public void setZ(final double z) {
135                this.z = z;
136        }
137
138        /**
139         * @param pitch
140         *            the pitch to set
141         */
142        public void setPitch(final float pitch) {
143                this.pitch = pitch;
144        }
145
146        /**
147         * @param yaw
148         *            the yaw to set
149         */
150        public void setYaw(final float yaw) {
151                this.yaw = yaw;
152        }
153
154        /*
155         * (non-Javadoc)
156         * 
157         * @see java.lang.Object#hashCode()
158         */
159        @Override
160        public int hashCode() {
161                final int prime = 31;
162                int result = 1;
163                result = prime * result + Float.floatToIntBits(pitch);
164                result = prime * result + ((world == null) ? 0 : world.hashCode());
165                long temp;
166                temp = Double.doubleToLongBits(x);
167                result = prime * result + (int) (temp ^ (temp >>> 32));
168                temp = Double.doubleToLongBits(y);
169                result = prime * result + (int) (temp ^ (temp >>> 32));
170                result = prime * result + Float.floatToIntBits(yaw);
171                temp = Double.doubleToLongBits(z);
172                result = prime * result + (int) (temp ^ (temp >>> 32));
173                return result;
174        }
175
176        /*
177         * (non-Javadoc)
178         * 
179         * @see java.lang.Object#equals(java.lang.Object)
180         */
181        @Override
182        public boolean equals(final Object obj) {
183                if (this == obj) {
184                        return true;
185                }
186                if (obj == null) {
187                        return false;
188                }
189                if (!(obj instanceof SimpleLocation)) {
190                        return false;
191                }
192                final SimpleLocation other = (SimpleLocation) obj;
193                if (Float.floatToIntBits(pitch) != Float.floatToIntBits(other.pitch)) {
194                        return false;
195                }
196                if (world == null) {
197                        if (other.world != null) {
198                                return false;
199                        }
200                } else if (!world.equals(other.world)) {
201                        return false;
202                }
203                if (Double.doubleToLongBits(x) != Double.doubleToLongBits(other.x)) {
204                        return false;
205                }
206                if (Double.doubleToLongBits(y) != Double.doubleToLongBits(other.y)) {
207                        return false;
208                }
209                if (Float.floatToIntBits(yaw) != Float.floatToIntBits(other.yaw)) {
210                        return false;
211                }
212                if (Double.doubleToLongBits(z) != Double.doubleToLongBits(other.z)) {
213                        return false;
214                }
215                return true;
216        }
217
218}
