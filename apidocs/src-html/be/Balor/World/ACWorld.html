017package be.Balor.World;
018
019import java.util.Map;
020import java.util.Set;
021
022import org.bukkit.Difficulty;
023import org.bukkit.Location;
024import org.bukkit.World;
025
026import be.Balor.Manager.Exceptions.WorldNotLoaded;
027import be.Balor.Tools.Warp;
028import be.Balor.Tools.Files.ObjectContainer;
029import be.Balor.bukkit.AdminCmd.ACPluginManager;
030
031/**
032 * @author Balor (aka Antoine Aflalo)
033 * 
034 */
035public abstract class ACWorld {
036        private final String name;
037        protected final World handler;
038        private final int hashCode;
039
040        /**
041         *
042         */
043        protected ACWorld(final World world) {
044                final int prime = 37;
045                int result = 5;
046                handler = world;
047                name = world.getName();
048                result = prime * result + ((handler == null) ? 0 : handler.hashCode());
049                result = prime * result + ((name == null) ? 0 : name.hashCode());
050                hashCode = result;
051        }
052
053        /**
054         * @return the handler
055         */
056        public World getHandle() {
057                return handler;
058        }
059
060        /**
061         * @return the name
062         */
063        public String getName() {
064                return name;
065        }
066
067        /**
068         * Get the wanted world
069         * 
070         * @param name
071         *            name of the wanted world
072         * @return the ACWorld
073         * @throws WorldNotLoaded
074         *             if the world is not loaded in bukkit or don't exist
075         */
076        public static ACWorld getWorld(final String name) throws WorldNotLoaded {
077                return WorldManager.getInstance().demandACWorld(name);
078        }
079
080        /**
081         * Get the wanted world
082         * 
083         * @param world
084         *            loaded bukkit world
085         * @return the ACWorld
086         */
087        public static ACWorld getWorld(final World world) {
088                return WorldManager.getInstance().demandACWorld(world);
089        }
090
091        /**
092         * Set the spawn location
093         * 
094         * @param loc
095         *            location of the spawn
096         */
097        public abstract void setSpawn(Location loc);
098
099        /**
100         * Get the spawn location
101         * 
102         * @return the spawn location
103         */
104        public abstract Location getSpawn();
105
106        /**
107         * Get the worlds difficulty
108         * 
109         * @return The difficulty
110         */
111        public Difficulty getDifficulty() {
112                final Difficulty dif = getInformation("difficulty").getDifficulty();
113                if (dif == null) {
114                        return handler.getDifficulty();
115                } else {
116                        return dif;
117                }
118        }
119
120        /**
121         * Set the difficulty of the world.
122         * 
123         * @param dif
124         *            The difficulty to set.
125         */
126        public void setDifficulty(final Difficulty dif) {
127                ACPluginManager.scheduleSyncTask(new Runnable() {
128                        @Override
129                        public void run() {
130                                handler.setDifficulty(dif);
131                        }
132                });
133                setInformation("difficulty", dif);
134        }
135
136        /**
137         * Add a warp point
138         * 
139         * @param name
140         *            name of the warp
141         * @param loc
142         *            location of the warp
143         */
144        public abstract void addWarp(String name, Location loc);
145        
146        /**
147         * Add a warp point with a permission
148         * 
149         * @param name
150         *            name of the warp
151         * @param loc
152         *            location of the warp
153         * @param perm
154         * ...........permission needed to use this warp
155         */
156        public abstract void addPermWarp(String name, Location loc, String perm);
157
158        /**
159         * Get the location of the Warp
160         * 
161         * @param name
162         *            name of the Warp
163         * @return location of the Warp
164         * @throws WorldNotLoaded
165         *             if the location's world is not loaded
166         * @throws IllegalArgumentException
167         *             if the name is null or empty
168         */
169        public abstract Warp getWarp(String name) throws WorldNotLoaded,
170                        IllegalArgumentException;
171
172        /**
173         * List of the warps' name
174         * 
175         * @return a List containing the name of each warp of the World
176         */
177        public abstract Set<String> getWarpList();
178        
179        /**
180         * Remove the warp
181         * 
182         * @param name
183         *            name of the warp to remove
184         */
185        public abstract void removeWarp(String name);
186
187        /**
188         * Set player information
189         * 
190         * @param info
191         *            key of the information
192         * @param value
193         *            information to set
194         */
195        public abstract void setInformation(String info, Object value);
196
197        /**
198         * Remove the information
199         * 
200         * @param info
201         *            key of the information
202         */
203        public abstract void removeInformation(String info);
204
205        /**
206         * Get the information
207         * 
208         * @param info
209         *            key of the information
210         */
211        public abstract ObjectContainer getInformation(String info);
212
213        /**
214         * Get all informations about the world
215         * 
216         * @return
217         */
218        public abstract Map<String, String> getInformationsList();
219
220        /**
221         * Force the save
222         */
223        protected abstract void forceSave();
224
225        /**
226         * To set a limit for a specific mob
227         * 
228         * @param mob
229         *            name of the mob
230         * @param limit
231         *            limit for that mob
232         */
233        public abstract void setMobLimit(String mob, int limit);
234
235        /**
236         * To remove the limit set on a specific mob
237         * 
238         * @param mob
239         *            name of the mob
240         * @return true if the limit is removed, false if there is no limit
241         */
242        public abstract boolean removeMobLimit(String mob);
243
244        /**
245         * Get the limit for the given mob.
246         * 
247         * @param mob
248         *            name of the mob
249         * @return the limit if is set, else return -1.
250         */
251        public abstract int getMobLimit(String mob);
252
253        /**
254         * Get the list of every limited mobs
255         * 
256         * @return
257         */
258        public abstract Set<String> getMobLimitList();
259
260        /**
261         * Get the location of the spawn for the wanted group
262         * 
263         * @param group
264         *            name of the group. If null, return the location of the default
265         *            spawn.
266         * @return location of the spawn point. If no location has been set for the
267         *         group, return the default spawn.
268         */
269        public abstract Location getGroupSpawn(String group);
270
271        /**
272         * Set the location of the spawn for the given group
273         * 
274         * @param group
275         *            name of the group.
276         * @param spawn
277         *            location of the new spawn.
278         */
279        public abstract void setGroupSpawn(String group, Location spawn);
280
281        /**
282         * Get all group spawn
283         * 
284         * @return
285         */
286        protected abstract Map<String, Location> getGroupSpawns();
287
288        /**
289         * Get all Informations
290         * 
291         * @return
292         */
293        protected abstract Map<String, Object> getInformations();
294
295        /*
296         * (non-Javadoc)
297         * 
298         * @see java.lang.Object#hashCode()
299         */
300        @Override
301        public int hashCode() {
302                return hashCode;
303        }
304
305        /*
306         * (non-Javadoc)
307         * 
308         * @see java.lang.Object#equals(java.lang.Object)
309         */
310        @Override
311        public boolean equals(final Object obj) {
312                if (this == obj) {
313                        return true;
314                }
315                if (obj == null) {
316                        return false;
317                }
318                if (!(obj instanceof ACWorld)) {
319                        return false;
320                }
321                final ACWorld other = (ACWorld) obj;
322                if (handler == null) {
323                        if (other.handler != null) {
324                                return false;
325                        }
326                } else if (!handler.equals(other.handler)) {
327                        return false;
328                }
329                if (name == null) {
330                        if (other.name != null) {
331                                return false;
332                        }
333                } else if (!name.equals(other.name)) {
334                        return false;
335                }
336                return true;
337        }
