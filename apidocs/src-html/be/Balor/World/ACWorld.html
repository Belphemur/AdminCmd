017    package be.Balor.World;
018    
019    import java.util.Map;
020    import java.util.Set;
021    
022    import org.bukkit.Difficulty;
023    import org.bukkit.Location;
024    import org.bukkit.World;
025    
026    import be.Balor.Manager.Exceptions.WorldNotLoaded;
027    import be.Balor.Tools.Warp;
028    import be.Balor.Tools.Files.ObjectContainer;
029    
030    /**
031     * @author Balor (aka Antoine Aflalo)
032     * 
033     */
034    public abstract class ACWorld {
035            private final String name;
036            protected final World handler;
037            private final int hashCode;
038    
039            /**
040             *
041             */
042            public ACWorld(final World world) {
043                    final int prime = 37;
044                    int result = 5;
045                    handler = world;
046                    name = world.getName();
047                    result = prime * result + ((handler == null) ? 0 : handler.hashCode());
048                    result = prime * result + ((name == null) ? 0 : name.hashCode());
049                    hashCode = result;
050            }
051    
052            /**
053             * @return the handler
054             */
055            public World getHandler() {
056                    return handler;
057            }
058    
059            /**
060             * @return the name
061             */
062            public String getName() {
063                    return name;
064            }
065    
066            /**
067             * Get the wanted world
068             * 
069             * @param name
070             *            name of the wanted world
071             * @return the ACWorld
072             * @throws WorldNotLoaded
073             *             if the world is not loaded in bukkit or don't exist
074             */
075            public static ACWorld getWorld(final String name) throws WorldNotLoaded {
076                    return WorldManager.getInstance().demandACWorld(name);
077            }
078    
079            /**
080             * Set the spawn location
081             * 
082             * @param loc
083             *            location of the spawn
084             */
085            public abstract void setSpawn(Location loc);
086    
087            /**
088             * Get the spawn location
089             * 
090             * @return the spawn location
091             */
092            public abstract Location getSpawn();
093    
094            /**
095             * Get the worlds difficulty
096             * 
097             * @return The difficulty
098             */
099            public abstract Difficulty getDifficulty() throws WorldNotLoaded;
100    
101            /**
102             * Set the difficulty of the world.
103             * 
104             * @param dif
105             *            The difficulty to set.
106             */
107            public abstract void setDifficulty(Difficulty dif);
108    
109            /**
110             * Add a warp point
111             * 
112             * @param name
113             *            name of the warp
114             * @param loc
115             *            location of the warp
116             */
117            public abstract void addWarp(String name, Location loc);
118    
119            /**
120             * Get the location of the Warp
121             * 
122             * @param name
123             *            name of the Warp
124             * @return location of the Warp
125             * @throws WorldNotLoaded
126             *             if the location's world is not loaded
127             * @throws IllegalArgumentException
128             *             if the name is null or empty
129             */
130            public abstract Warp getWarp(String name) throws WorldNotLoaded, IllegalArgumentException;
131    
132            /**
133             * List of the warps' name
134             * 
135             * @return a List containing the name of each warp of the World
136             */
137            public abstract Set<String> getWarpList();
138    
139            /**
140             * Remove the warp
141             * 
142             * @param name
143             *            name of the warp to remove
144             */
145            public abstract void removeWarp(String name);
146    
147            /**
148             * Set player information
149             * 
150             * @param info
151             *            key of the information
152             * @param value
153             *            information to set
154             */
155            public abstract void setInformation(String info, Object value);
156    
157            /**
158             * Remove the information
159             * 
160             * @param info
161             *            key of the information
162             */
163            public abstract void removeInformation(String info);
164    
165            /**
166             * Get the information
167             * 
168             * @param info
169             *            key of the information
170             */
171            public abstract ObjectContainer getInformation(String info);
172    
173            /**
174             * Get all informations about the world
175             * 
176             * @return
177             */
178            public abstract Map<String, String> getInformations();
179    
180            /**
181             * Force the save
182             */
183            abstract void forceSave();
184    
185            /**
186             * To set a limit for a specific mob
187             * 
188             * @param mob
189             *            name of the mob
190             * @param limit
191             *            limit for that mob
192             */
193            public abstract void setMobLimit(String mob, int limit);
194    
195            /**
196             * To remove the limit set on a specific mob
197             * 
198             * @param mob
199             *            name of the mob
200             * @return true if the limit is removed, false if there is no limit
201             */
202            public abstract boolean removeMobLimit(String mob);
203    
204            /**
205             * Get the limit for the given mob.
206             * 
207             * @param mob
208             *            name of the mob
209             * @return the limit if is set, else return -1.
210             */
211            public abstract int getMobLimit(String mob);
212    
213            /**
214             * Get the list of every limited mobs
215             * 
216             * @return
217             */
218            public abstract Set<String> getMobLimitList();
219    
220            /*
221             * (non-Javadoc)
222             * 
223             * @see java.lang.Object#hashCode()
224             */
225            @Override
226            public int hashCode() {
227                    return hashCode;
228            }
229    
230            /*
231             * (non-Javadoc)
232             * 
233             * @see java.lang.Object#equals(java.lang.Object)
234             */
235            @Override
236            public boolean equals(final Object obj) {
237                    if (this == obj)
238                            return true;
239                    if (obj == null)
240                            return false;
241                    if (!(obj instanceof ACWorld))
242                            return false;
243                    final ACWorld other = (ACWorld) obj;
244                    if (handler == null) {
245                            if (other.handler != null)
246                                    return false;
247                    } else if (!handler.equals(other.handler))
248                            return false;
249                    if (name == null) {
250                            if (other.name != null)
251                                    return false;
252                    } else if (!name.equals(other.name))
253                            return false;
254                    return true;
255            }
256    
257    }
