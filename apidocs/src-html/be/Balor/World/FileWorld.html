017    package be.Balor.World;
018    
019    import java.io.File;
020    import java.io.IOException;
021    import java.util.Map;
022    import java.util.Map.Entry;
023    import java.util.Set;
024    import java.util.TreeMap;
025    
026    import org.bukkit.Difficulty;
027    import org.bukkit.Location;
028    import org.bukkit.World;
029    import org.bukkit.configuration.ConfigurationSection;
030    
031    import be.Balor.Manager.Exceptions.WorldNotLoaded;
032    import be.Balor.Tools.Warp;
033    import be.Balor.Tools.Configuration.ExConfigurationSection;
034    import be.Balor.Tools.Configuration.File.ExtendedConfiguration;
035    import be.Balor.Tools.Debug.ACLogger;
036    import be.Balor.Tools.Files.ObjectContainer;
037    import be.Balor.Tools.Help.String.Str;
038    import be.Balor.bukkit.AdminCmd.ACPluginManager;
039    
040    import com.google.common.io.Files;
041    
042    /**
043     * @author Balor (aka Antoine Aflalo)
044     * 
045     */
046    public class FileWorld extends ACWorld {
047            private final ExtendedConfiguration datas;
048            private final ConfigurationSection warps;
049            private final ExConfigurationSection informations;
050            private final ExConfigurationSection mobLimits;
051    
052            /**
053             * @param name
054             */
055            public FileWorld(final World world, final String directory) {
056                    super(world);
057                    final File wFile = new File(directory, world.getName() + ".yml");
058                    try {
059                            Files.createParentDirs(wFile);
060                    } catch (final IOException e) {
061                    }
062                    datas = ExtendedConfiguration.loadConfiguration(wFile);
063    
064                    warps = datas.addSection("warps");
065                    informations = datas.addSection("informations");
066                    mobLimits = informations.addSection("mobLimits");
067                    forceSave();
068            }
069    
070            /*
071             * (non-Javadoc)
072             * 
073             * @see be.Balor.World.ACWorld#setSpawn(org.bukkit.Location)
074             */
075            @Override
076            public void setSpawn(final Location loc) {
077                    datas.set("spawn", new SimpleLocation(loc));
078                    writeFile();
079            }
080    
081            /*
082             * (non-Javadoc)
083             * 
084             * @see be.Balor.World.ACWorld#getSpawn()
085             */
086            @Override
087            public Location getSpawn() throws WorldNotLoaded {
088                    final Object spawn = datas.get("spawn");
089                    if (spawn == null)
090                            return handler.getSpawnLocation();
091                    else if (spawn instanceof SimpleLocation)
092                            return ((SimpleLocation) spawn).getLocation();
093                    else
094                            return handler.getSpawnLocation();
095            }
096    
097            /*
098             * (non-Javadoc)
099             * 
100             * @see be.Balor.World.ACWorld#getDifficulty()
101             */
102            @Override
103            public Difficulty getDifficulty() throws WorldNotLoaded {
104                    Difficulty dif;
105                    dif = (Difficulty) informations.get("difficulty");
106                    if (dif == null)
107                            return handler.getDifficulty();
108                    return dif;
109            }
110    
111            /*
112             * (non-Javadoc)
113             * 
114             * @see be.Balor.World.ACWorld#setDifficulty(org.bukkit.Difficulty)
115             */
116            @Override
117            public void setDifficulty(final Difficulty dif) {
118                    ACPluginManager.scheduleSyncTask(new Runnable() {
119                            @Override
120                            public void run() {
121                                    handler.setDifficulty(dif);
122                            }
123                    });
124    
125                    informations.set("difficulty", dif);
126            }
127    
128            /*
129             * (non-Javadoc)
130             * 
131             * @see be.Balor.World.ACWorld#addWarp(java.lang.String,
132             * org.bukkit.Location)
133             */
134            @Override
135            public void addWarp(final String name, final Location loc) {
136                    warps.set(name, new SimpleLocation(loc));
137                    writeFile();
138            }
139    
140            /*
141             * (non-Javadoc)
142             * 
143             * @see be.Balor.World.ACWorld#getWarp(java.lang.String)
144             */
145            @Override
146            public Warp getWarp(final String name) throws WorldNotLoaded, IllegalArgumentException {
147                    if (name == null || (name != null && name.isEmpty()))
148                            throw new IllegalArgumentException("Name can't be null or Empty");
149                    Object warp = warps.get(name);
150                    String warpName = name;
151                    if (warp == null) {
152                            warpName = Str.matchString(warps.getKeys(false), name);
153                            if (warpName == null)
154                                    return null;
155                            warp = warps.get(warpName);
156                    }
157    
158                    return new Warp(warpName, ((SimpleLocation) warp).getLocation());
159            }
160    
161            /*
162             * (non-Javadoc)
163             * 
164             * @see be.Balor.World.ACWorld#getWarpList()
165             */
166            @Override
167            public Set<String> getWarpList() {
168                    return warps.getKeys(false);
169            }
170    
171            /*
172             * (non-Javadoc)
173             * 
174             * @see be.Balor.World.ACWorld#removeWarp(java.lang.String)
175             */
176            @Override
177            public void removeWarp(final String name) {
178                    warps.set(name, null);
179                    writeFile();
180            }
181    
182            /*
183             * (non-Javadoc)
184             * 
185             * @see be.Balor.World.ACWorld#setInformation(java.lang.String,
186             * java.lang.Object)
187             */
188            @Override
189            public void setInformation(final String info, final Object value) {
190                    informations.set(info, value);
191                    writeFile();
192            }
193    
194            /*
195             * (non-Javadoc)
196             * 
197             * @see be.Balor.World.ACWorld#removeInformation(java.lang.String)
198             */
199            @Override
200            public void removeInformation(final String info) {
201                    informations.set(info, null);
202                    writeFile();
203            }
204    
205            /*
206             * (non-Javadoc)
207             * 
208             * @see be.Balor.World.ACWorld#getInformation(java.lang.String)
209             */
210            @Override
211            public ObjectContainer getInformation(final String info) {
212                    return new ObjectContainer(informations.get(info));
213            }
214    
215            /**
216             *
217             */
218            private void writeFile() {
219                    forceSave();
220            }
221    
222            /*
223             * (non-Javadoc)
224             * 
225             * @see be.Balor.World.ACWorld#forceSave()
226             */
227            @Override
228            void forceSave() {
229                    try {
230                            datas.save();
231                    } catch (final IOException e) {
232                            ACLogger.severe("Problem when saving the World File of " + getName(), e);
233                    }
234            }
235    
236            /*
237             * (non-Javadoc)
238             * 
239             * @see be.Balor.World.ACWorld#getInformations()
240             */
241            @Override
242            public Map<String, String> getInformations() {
243                    final TreeMap<String, String> result = new TreeMap<String, String>();
244                    for (final Entry<String, Object> entry : informations.getValues(false).entrySet())
245                            result.put(entry.getKey(), entry.getValue().toString());
246                    for (final Entry<String, Object> entry : mobLimits.getValues(false).entrySet())
247                            result.put("Limit on " + entry.getKey(), entry.getValue().toString());
248                    return result;
249            }
250    
251            /*
252             * (non-Javadoc)
253             * 
254             * @see be.Balor.World.ACWorld#setMobLimit(java.lang.String, int)
255             */
256            @Override
257            public void setMobLimit(final String mob, final int limit) {
258                    mobLimits.set(mob, limit);
259                    writeFile();
260            }
261    
262            /*
263             * (non-Javadoc)
264             * 
265             * @see be.Balor.World.ACWorld#removeMobLimit(java.lang.String)
266             */
267            @Override
268            public boolean removeMobLimit(final String mob) {
269                    if (mobLimits.isSet(mob)) {
270                            mobLimits.remove(mob);
271                            writeFile();
272                            return true;
273                    }
274                    return false;
275            }
276    
277            /*
278             * (non-Javadoc)
279             * 
280             * @see be.Balor.World.ACWorld#getMobLimit(java.lang.String)
281             */
282            @Override
283            public int getMobLimit(final String mob) {
284                    return mobLimits.getInt(mob, -1);
285            }
286    
287            /*
288             * (non-Javadoc)
289             * 
290             * @see be.Balor.World.ACWorld#getMobLimitList()
291             */
292            @Override
293            public Set<String> getMobLimitList() {
294                    return mobLimits.getKeys(false);
295            }
296
