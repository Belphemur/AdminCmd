041/**

043 * 
044 */
045public class FileWorld extends ACWorld {
046        private final ExtendedConfiguration datas;
047        private final ConfigurationSection warps;
048        private final ExConfigurationSection informations;
049        private final ExConfigurationSection mobLimits;
050        private final ExConfigurationSection spawns;
051
052        /**
053         * @param name
054         */
055        FileWorld(final World world, final String directory) {
056                super(world);
057                final File wFile = new File(directory, world.getName() + ".yml");
058                try {
059                        Files.createParentDirs(wFile);
060                } catch (final IOException e) {
061                }
062                datas = ExtendedConfiguration.loadConfiguration(wFile);
063
064                warps = datas.addSection("warps");
065                informations = datas.addSection("informations");
066                mobLimits = informations.addSection("mobLimits");
067                spawns = datas.addSection("spawns");
068                forceSave();
069        }
070
071        /*
072         * (non-Javadoc)
073         * 
074         * @see be.Balor.World.ACWorld#setSpawn(org.bukkit.Location)
075         */
076        @Override
077        public void setSpawn(final Location loc) {
078                datas.set("spawn", new SimpleLocation(loc));
079                writeFile();
080        }
081
082        /*
083         * (non-Javadoc)
084         * 
085         * @see be.Balor.World.ACWorld#getSpawn()
086         */
087        @Override
088        public Location getSpawn() throws WorldNotLoaded {
089                final Object spawn = datas.get("spawn");
090                if (spawn == null) {
091                        return handler.getSpawnLocation();
092                } else if (spawn instanceof SimpleLocation) {
093                        return ((SimpleLocation) spawn).getLocation();
094                } else {
095                        return handler.getSpawnLocation();
096                }
097        }
098
099        /*
100         * (non-Javadoc)
101         * 
102         * @see be.Balor.World.ACWorld#addWarp(java.lang.String,
103         * org.bukkit.Location)
104         */
105        @Override
106        public void addWarp(final String name, final Location loc) {
107                warps.set(name, new SimpleLocation(loc));
108                writeFile();
109        }
110
111        /*
112         * (non-Javadoc)
113         * 
114         * @see be.Balor.World.ACWorld#getWarp(java.lang.String)
115         */
116        @Override
117        public Warp getWarp(final String name) throws WorldNotLoaded,
118                        IllegalArgumentException {
119                if (name == null || (name != null && name.isEmpty())) {
120                        throw new IllegalArgumentException("Name can't be null or Empty");
121                }
122                Object warp = warps.get(name);
123                String warpName = name;
124                if (warp == null) {
125                        warpName = Str.matchString(warps.getKeys(false), name);
126                        if (warpName == null) {
127                                return null;
128                        }
129                        warp = warps.get(warpName);
130                }
131                try {
132                        return new Warp(warpName, ((SimpleLocation) warp).getLocation());
133                } catch (ClassCastException e) {
134                        return new Warp(warpName, ((PermLocation) warp).getLocation(), ((PermLocation) warp).getPerm());
135                }
136        }

201
202        /**
203         *
204         */
205        private void writeFile() {
206                forceSave();
207        }
208
209        /*
210         * (non-Javadoc)
211         * 
212         * @see be.Balor.World.ACWorld#forceSave()
213         */
214        @Override
215        protected void forceSave() {
216                try {
217                        datas.save();
218                } catch (final IOException e) {
219                        ACLogger.severe("Problem when saving the World File of "
220                                        + getName(), e);
221                }
222        }
223
224        /*
225         * (non-Javadoc)
226         * 
227         * @see be.Balor.World.ACWorld#getInformations()
228         */
229        @Override
230        public Map<String, String> getInformationsList() {
231                final TreeMap<String, String> result = new TreeMap<String, String>();
232                for (final Entry<String, Object> entry : informations.getValues(false)
233                                .entrySet()) {
234                        result.put(entry.getKey(), entry.getValue().toString());
235                }
236                for (final Entry<String, Object> entry : mobLimits.getValues(false)
237                                .entrySet()) {
238                        result.put("Limit on " + entry.getKey(), entry.getValue()
239                                        .toString());
240                }
241                return result;
242        }

252                writeFile();
253        }

255        /*
256         * (non-Javadoc)
257         * 
258         * @see be.Balor.World.ACWorld#removeMobLimit(java.lang.String)
259         */
260        @Override
261        public boolean removeMobLimit(final String mob) {
262                if (mobLimits.isSet(mob)) {
263                        mobLimits.remove(mob);
264                        writeFile();
265                        return true;
266                }
267                return false;
268        }

290        /*
291         * (Non javadoc)
292         * 
293         * @see be.Balor.World.ACWorld#getGroupSpawn(java.lang.String)
294         */
295        @Override
296        public Location getGroupSpawn(final String group) {
297                if (group == null) {
298                        return getSpawn();
299                }
300                final Object spawn = spawns.get(group);
301                if (spawn == null) {
302                        return getSpawn();
303                } else if (spawn instanceof SimpleLocation) {
304                        return ((SimpleLocation) spawn).getLocation();
305                } else {
306                        return getSpawn();
307                }
308        }

314         * org.bukkit.Location)
315         */
316        @Override
317        public void setGroupSpawn(final String group, final Location spawn) {
318                spawns.set(group, new SimpleLocation(spawn));
319
320        }
321
322        /*
323         * (Non javadoc)
324         * 
325         * @see be.Balor.World.ACWorld#getGroupSpawns()
326         */
327        @Override
328        protected Map<String, Location> getGroupSpawns() {
329                final Map<String, Location> result = new HashMap<String, Location>();
330                for (final Entry<String, Object> entry : spawns.getValues(false)
331                                .entrySet()) {
332                        final Object obj = entry.getValue();
333                        if (!(obj instanceof SimpleLocation)) {
334                                continue;
335                        }
336                        result.put(entry.getKey(), ((SimpleLocation) obj).getLocation());
337                }
338                return result;
339        }
340
341        /*
342         * (non-Javadoc)
343         * 
344         * @see be.Balor.World.ACWorld#getInformations()
345         */
346        @Override
347        protected Map<String, Object> getInformations() {
348                final Map<String, Object> result = new HashMap<String, Object>();
349                for (final Entry<String, Object> entry : informations.getValues(false)
350                                .entrySet()) {
351                        if (entry.getKey().contains("mobLimits")) {
352                                continue;
353                        }
354                        result.put(entry.getKey(), entry.getValue());
355                }
356                for (final Entry<String, Object> entry : mobLimits.getValues(false)
357                                .entrySet()) {
358                        result.put("mobLimit:" + entry.getKey(), entry.getValue()
359                                        .toString());
360                }
361                return result;
362        }
