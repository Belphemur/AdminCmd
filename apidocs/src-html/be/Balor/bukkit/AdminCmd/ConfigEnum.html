391        private final String confVal;
392        private final Object defaultVal;
393        private final String description;
394        private static ExtendedConfiguration config;
395        private static String pluginVersion;
396        private static String pluginName;
397
398        /**
399         * @param confVal
400         * @param defaultVal
401         * @param description
402         */
403        private ConfigEnum(final String confVal, final Object defaultVal,
404                        final String description) {
405                this.confVal = confVal;
406                this.defaultVal = defaultVal;
407                this.description = description;
408        }
409
410        public String getString() {
411                return config.getString(confVal);
412        }
413
414        public int getInt() {
415                return config.getInt(confVal);
416        }
417
418        public double getDouble() {
419                return config.getDouble(confVal);
420        }
421
422        public boolean getBoolean() {
423                return config.getBoolean(confVal);
424        }
425
426        public long getLong() {
427                return config.getLong(confVal);
428        }
429
430        public float getFloat() {
431                return Float.parseFloat(config.getString(confVal));
432        }
433
434        public List<String> getStringList() {
435                return config.getStringList(confVal);
436        }
437
438        public void setValue(final Object value) {
439                config.set(confVal, value);
440        }
441
442        /**
443         * @return the defaultvalues
444         */
445        private static Map<String, Object> getDefaultvalues() {
446                final Map<String, Object> values = new LinkedHashMap<String, Object>();
447                for (final ConfigEnum ce : values()) {
448                        values.put(ce.confVal, ce.defaultVal);
449                }
450                return values;
451        }
452
453        private static String getHeader() {
454                final StringBuffer buffer = new StringBuffer();
455                buffer.append("Configuration file of ").append(pluginName).append('\n');
456                buffer.append("Plugin Version: ").append(pluginVersion).append('\n')
457                                .append('\n');
458                for (final ConfigEnum ce : values()) {
459                        buffer.append(ce.confVal).append("\t:\t").append(ce.description)
460                                        .append(" (Default : ").append(ce.defaultVal).append(')')
461                                        .append('\n');
462                }
463                return buffer.toString();
464        }
465
466        /**
467         * @param config
468         *            the config to set
469         */
470        public static void setConfig(final ExtendedConfiguration config) {
471                ConfigEnum.config = config;
472                ConfigEnum.config.options().copyDefaults(true).header(getHeader());
473                ConfigEnum.config.addDefaults(getDefaultvalues());
474                try {
475                        ConfigEnum.save();
476                } catch (final IOException e) {
477                }
478        }
479
480        public static void setPluginInfos(final PluginDescriptionFile pdf) {
481                ConfigEnum.pluginVersion = pdf.getVersion();
482                ConfigEnum.pluginName = pdf.getName();
483        }
484
485        public static void save() throws IOException {
486                config.save();
487        }
488
489        public static void reload() throws IOException,
490                        InvalidConfigurationException {
491                config.reload();
492        }
