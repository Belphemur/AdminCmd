140            private final String confVal;
141            private final Object defaultVal;
142            private final String description;
143            private static ConfigurationSection config;
144            private static String pluginVersion;
145            private static String pluginName;
146    
147            /**
148             * @param confVal
149             * @param defaultVal
150             * @param description
151             */
152            private ConfigEnum(final String confVal, final Object defaultVal, final String description) {
153                    this.confVal = confVal;
154                    this.defaultVal = defaultVal;
155                    this.description = description;
156            }
157    
158            public String getString() {
159                    return config.getString(confVal);
160            }
161    
162            public int getInt() {
163                    return config.getInt(confVal);
164            }
165    
166            public double getDouble() {
167                    return config.getDouble(confVal);
168            }
169    
170            public boolean getBoolean() {
171                    return config.getBoolean(confVal);
172            }
173    
174            public long getLong() {
175                    return config.getLong(confVal);
176            }
177    
178            public float getFloat() {
179                    return Float.parseFloat(config.getString(confVal));
180            }
181    
182            public List<String> getStringList() {
183                    return config.getStringList(confVal);
184            }
185    
186            public void setValue(final Object value) {
187                    config.set(confVal, value);
188            }
189    
190            /**
191             * @return the defaultvalues
192             */
193            public static Map<String, Object> getDefaultvalues() {
194                    final Map<String, Object> values = new LinkedHashMap<String, Object>();
195                    for (final ConfigEnum ce : values())
196                            values.put(ce.confVal, ce.defaultVal);
197                    return values;
198            }
199    
200            public static String getHeader() {
201                    final StringBuffer buffer = new StringBuffer();
202                    buffer.append("Configuration file of ").append(pluginName).append('\n');
203                    buffer.append("Plugin Version: ").append(pluginVersion).append('\n').append('\n');
204                    for (final ConfigEnum ce : values())
205                            buffer.append(ce.confVal).append("\t:\t").append(ce.description).append(" (Default : ")
206                                            .append(ce.defaultVal).append(')').append('\n');
207                    return buffer.toString();
208            }
209    
210            /**
211             * @param config
212             *            the config to set
213             */
214            public static void setConfig(final ConfigurationSection config) {
215                    ConfigEnum.config = config;
216            }
217    
218            public static void setPluginInfos(final PluginDescriptionFile pdf) {
219                    ConfigEnum.pluginVersion = pdf.getVersion();
220                    ConfigEnum.pluginName = pdf.getName();
221            }
