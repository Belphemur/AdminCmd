160            @Override
161            public void onDisable() {
162                    final PluginDescriptionFile pdfFile = this.getDescription();
163                    getServer().getScheduler().cancelTasks(this);
164                    FilePlayer.forceSaveList();
165                    for (final ACPlayer p : PlayerManager.getInstance().getOnlineACPlayers()) {
166                            PlayerManager.getInstance().setOffline(p);
167                    }
168                    ACPluginManager.getInstance().stopChildrenPlugins();
169                    CommandManager.getInstance().stopAllExecutorThreads();
170                    worker = null;
171                    ACHelper.killInstance();
172                    InvisibleWorker.killInstance();
173                    AFKWorker.killInstance();
174                    CommandManager.killInstance();
175                    HelpLister.killInstance();
176                    DebugLog.stopLogging();
177                    System.gc();
178                    ACLogger.info("Plugin Disabled. (version " + pdfFile.getVersion() + ")");
179            }

181            @Override
182            public void onEnable() {
183                    ExtendedConfiguration.setClassLoader(this.getClassLoader());
184                    DebugLog.setFile(getDataFolder().getPath());
185                    final PluginDescriptionFile pdfFile = this.getDescription();
186                    DebugLog.INSTANCE.info("Plugin Version : " + pdfFile.getVersion());
187                    final PluginManager pm = getServer().getPluginManager();
188                    ACLogger.info("Plugin Enabled. (version " + pdfFile.getVersion() + ")");
189                    pm.registerEvents(new ACPluginListener(), this);
190                    worker = ACHelper.getInstance();
191                    worker.setCoreInstance(this);
192                    super.onEnable();
193                    TerminalCommandManager.getInstance().setPerm(this);
194                    worker.loadInfos();
195                    permissionLinker.registerAllPermParent();
196                    pm.registerEvents(new ACBlockListener(), this);
197                    pm.registerEvents(new ACEntityListener(), this);
198                    pm.registerEvents(new ACPlayerListener(), this);
199                    pm.registerEvents(new ACWeatherListener(), this);
200                    if (ConfigEnum.COLSIGN.getBoolean())
201                            pm.registerEvents(new ACColorSignListener(), this);
202                    if (ConfigEnum.RESET_POWERS.getBoolean())
203                            pm.registerEvents(new ACResetPowerListener(), this);
204                    try {
205                            // create a new metrics object
206                            final Metrics metrics = new Metrics();
207                            ACPluginManager.setMetrics(metrics);
208    
209                            metrics.addCustomData(this, new Metrics.Plotter() {
210                                    @Override
211                                    public String getColumnName() {
212                                            return "Total Banned Players";
213                                    }
214    
215                                    @Override
216                                    public int getValue() {
217                                            return worker.countBannedPlayers();
218                                    }
219                            });
220                            metrics.addCustomData(this, new Metrics.Plotter() {
221                                    @Override
222                                    public String getColumnName() {
223                                            return "Total Kits";
224                                    }
225    
226                                    @Override
227                                    public int getValue() {
228                                            return worker.getNbKit();
229                                    }
230                            });
231                            metrics.addCustomData(this, new Metrics.Plotter() {
232                                    @Override
233                                    public String getColumnName() {
234                                            return "Total Invisible Players";
235                                    }
236    
237                                    @Override
238                                    public int getValue() {
239                                            return InvisibleWorker.getInstance().nbInvisibles();
240                                    }
241                            });
242                            metrics.addCustomData(this, new Metrics.Plotter() {
243                                    @Override
244                                    public String getColumnName() {
245                                            return "Total Afk Players";
246                                    }
247    
248                                    @Override
249                                    public int getValue() {
250                                            return AFKWorker.getInstance().nbAfk();
251                                    }
252                            });
253                            getServer().getScheduler().scheduleAsyncDelayedTask(this, new Runnable() {
254    
255                                    @Override
256                                    public void run() {
257                                            metrics.beginMeasuringPlugin(AdminCmd.this);
258                                            DebugLog.INSTANCE.info("Stats started");
259                                    }
260                            }, 30 * Utils.secInTick);
261                    } catch (final IOException e) {
262                            DebugLog.INSTANCE.log(Level.SEVERE, "Stats loggin problem", e);
263                    }
264            }
