017package be.Balor.bukkit.AdminCmd;
018
019import java.util.Collections;
020import java.util.HashMap;
021import java.util.Map;
022import java.util.Map.Entry;
023
024import org.bukkit.Bukkit;
025import org.bukkit.Server;
026import org.bukkit.plugin.Plugin;
027import org.bukkit.scheduler.BukkitScheduler;
028
029import be.Balor.Manager.CommandManager;
030import be.Balor.Manager.Commands.CoreCommand;
031import be.Balor.Tools.Debug.ACLogger;
032import be.Balor.Tools.Debug.DebugLog;
033import be.Balor.Tools.Metrics.Metrics;
034import be.Balor.Tools.Metrics.Metrics.Graph;
035import be.Balor.Tools.Metrics.Metrics.Plotter;
036
037/**
038 * @author Balor (aka Antoine Aflalo)
039 * 
040 */
041public class ACPluginManager {
042        private final static ACPluginManager instance = new ACPluginManager();
043        private final Map<String, AbstractAdminCmdPlugin> pluginInstances = Collections
044                        .synchronizedMap(new HashMap<String, AbstractAdminCmdPlugin>());
045        private final static Server server = Bukkit.getServer();
046        private static AbstractAdminCmdPlugin corePlugin;
047        private static Graph graph = null;
048
049        /**
050         * @return the instance
051         */
052        protected static ACPluginManager getInstance() {
053                return instance;
054        }
055
056        public static AbstractAdminCmdPlugin getPluginInstance(final String name) {
057                return getInstance().getPlugin(name);
058        }
059
060        /**
061         * Get Bukkit Scheduler
062         * 
063         * @return
064         */
065        public static BukkitScheduler getScheduler() {
066                return server.getScheduler();
067        }
068
069        /**
070         * @return the server
071         */
072        public static Server getServer() {
073                return server;
074        }
075
076        public static void registerACPlugin(final AbstractAdminCmdPlugin addon)
077                        throws IllegalArgumentException {
078                getInstance().registerPlugin(addon);
079        }
080
081        /**
082         * Register a Plugin Command
083         * 
084         * @param clazz
085         */
086        public static void registerCommand(final Class<? extends CoreCommand> clazz)
087                        throws IllegalArgumentException {
088                CommandManager.getInstance().registerCommand(clazz);
089        }
090
091        /**
092         * Schedule a SyncTask
093         * 
094         * @param task
095         * @return
096         */
097        public static int scheduleSyncTask(final Runnable task) {
098                return server.getScheduler().scheduleSyncDelayedTask(corePlugin, task);
099        }
100
101        /**
102         * Schedule a AsyncDelayedTask
103         * 
104         * @param task
105         * @return
106         */
107        public static int runTaskLaterAsynchronously(final Runnable task) {
108                return server.getScheduler().runTaskAsynchronously(corePlugin, task)
109                                .getTaskId();
110        }
111
112        /**
113         * @param corePlugin
114         *            the corePlugin to set
115         */
116        static void setCorePlugin(final AbstractAdminCmdPlugin corePlugin) {
117                ACPluginManager.corePlugin = corePlugin;
118        }
119
120        /**
121         * @return the corePlugin
122         */
123        public static AbstractAdminCmdPlugin getCorePlugin() {
124                return corePlugin;
125        }
126
127        /**
128         * @param metrics
129         *            the metrics to set
130         */
131        static void setMetrics(final Metrics metrics) {
132                ACPluginManager.graph = metrics.createGraph("Plugins");
133        }
134
135        public static void unRegisterACPlugin(final Plugin addon) {
136                if (addon instanceof AbstractAdminCmdPlugin) {
137                        getInstance().unRegisterPlugin((AbstractAdminCmdPlugin) addon);
138                }
139        }
140
141        private ACPluginManager() {
142        }
143
144        /**
145         * Get registered plugin
146         * 
147         * @param name
148         *            name of the addon
149         * @return the addon or null if not registered
150         */
151        protected AbstractAdminCmdPlugin getPlugin(final String name) {
152                return pluginInstances.get(name);
153        }
154
155        /**
156         * Register a AdminCmd addon
157         * 
158         * @param addon
159         */
160        protected void registerPlugin(final AbstractAdminCmdPlugin addon)
161                        throws IllegalArgumentException {
162                if (!pluginInstances.containsKey(addon.getAddonName())) {
163                        pluginInstances.put(addon.getAddonName(), addon);
164                        DebugLog.INSTANCE.info("Registering : " + addon);
165                        if (corePlugin == null || addon.equals(corePlugin)) {
166                                return;
167                        }
168                        graph.addPlotter(new Plotter() {
169
170                                @Override
171                                public int getValue() {
172                                        return 1;
173                                }
174
175                                @Override
176                                public String getColumnName() {
177                                        return "Addon " + addon.getAddonName();
178                                }
179                        });
180                } else {
181                        throw new IllegalArgumentException("Plugin " + addon.getAddonName()
182                                        + " Already registered.");
183                }
184        }
185
186        void stopChildrenPlugins() {
187                ACLogger.info("Disabling all AdminCmd's plugins");
188                for (final Entry<String, AbstractAdminCmdPlugin> plugin : pluginInstances
189                                .entrySet()) {
190                        if (plugin.getValue().isEnabled()) {
191                                server.getPluginManager().disablePlugin(plugin.getValue());
192                        }
193                }
194        }
195
196        /**
197         * Unregister an AdminCmd addon
198         * 
199         * @param addon
200         */
201        protected void unRegisterPlugin(final AbstractAdminCmdPlugin addon) {
202                pluginInstances.remove(addon.getAddonName());
203                if (!addon.equals(corePlugin)) {
204                        graph.removePlotter(new Plotter() {
205
206                                @Override
207                                public int getValue() {
208                                        return 1;
209                                }
210
211                                @Override
212                                public String getColumnName() {
213                                        return "Addon " + addon.getAddonName();
214                                }
215                        });
216                }
217        }
218
219}
