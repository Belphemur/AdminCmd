017    package be.Balor.bukkit.AdminCmd;
018    
019    import org.bukkit.plugin.java.JavaPlugin;
020    
021    import be.Balor.Manager.CommandManager;
022    import be.Balor.Manager.Permissions.PermissionLinker;
023    
024    /**
025     * @author Balor (aka Antoine Aflalo)
026     */
027    public abstract class AbstractAdminCmdPlugin extends JavaPlugin {
028            protected final PermissionLinker permissionLinker;
029            protected final String name;
030            private final int hashCode;
031    
032            /**
033             * Create the AdminCmd plugin.
034             * 
035             * @param name
036             *            the name used for the plugin.
037             * 
038             */
039            public AbstractAdminCmdPlugin(final String name) {
040                    this.name = name;
041                    permissionLinker = PermissionLinker.getPermissionLinker(name);
042                    final int prime = 31;
043                    int result = 1;
044                    result = prime * result + ((name == null) ? 0 : name.hashCode());
045                    result = prime * result + ((permissionLinker == null) ? 0 : permissionLinker.hashCode());
046                    hashCode = result;
047            }
048    
049            public AbstractAdminCmdPlugin() {
050                    this.name = this.getClass().getSimpleName();
051                    permissionLinker = PermissionLinker.getPermissionLinker(name);
052                    final int prime = 31;
053                    int result = 1;
054                    result = prime * result + ((name == null) ? 0 : name.hashCode());
055                    result = prime * result + ((permissionLinker == null) ? 0 : permissionLinker.hashCode());
056                    hashCode = result;
057            }
058    
059            /**
060             * @return the name
061             */
062            public String getAddonName() {
063                    return name;
064            }
065    
066            /**
067             * @return the permissionLinker
068             */
069            public PermissionLinker getPermissionLinker() {
070                    return permissionLinker;
071            }
072    
073            /**
074             * Return the name of the plugin in the plugin.yml
075             * 
076             * @return the bukkit name of the plugin
077             */
078            public String getPluginName() {
079                    return getDescription().getName();
080            }
081    
082            /*
083             * (non-Javadoc)
084             * 
085             * @see org.bukkit.plugin.Plugin#onEnable()
086             */
087            @Override
088            public void onEnable() {
089                    ACPluginManager.registerACPlugin(this);
090                    registerPermParents();
091                    CommandManager.getInstance().registerACPlugin(this);
092                    registerCmds();
093                    CommandManager.getInstance().checkAlias(this);
094                    setDefaultLocale();
095            }
096    
097            /**
098             * Definitions of the command used by the plugin
099             */
100            public abstract void registerCmds();
101    
102            /**
103             * Definition of the Permissions used by the plugin
104             */
105            protected abstract void registerPermParents();
106    
107            /**
108             * Definitions of the locale used by the plugin
109             */
110            protected abstract void setDefaultLocale();
111    
112            /*
113             * (non-Javadoc)
114             * 
115             * @see java.lang.Object#toString()
116             */
117            @Override
118            public String toString() {
119                    return "AbstractAdminCmdPlugin [permissionLinker=" + permissionLinker + ", name=" + name
120                                    + ", hashCode=" + hashCode + "]";
121            }
122    
123    }
