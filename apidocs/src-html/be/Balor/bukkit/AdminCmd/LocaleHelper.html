197        private final String key;
198        private final String locale;
199
200        private LocaleHelper(final String key, final String locale) {
201                this.key = key;
202                this.locale = locale;
203        }
204
205        /**
206         * @return the key
207         */
208        public String getKey() {
209                return key;
210        }
211
212        public static void addAllLocales() {
213                for (final LocaleHelper lh : values()) {
214                        LocaleManager.getInstance().addLocale(lh.key, lh.locale);
215                }
216        }
217
218        /**
219         * Send the locale the the CommandSender
220         * 
221         * @param sender
222         *            CommandSender that will get the message
223         * @param replace
224         *            Replacement variables
225         */
226        public void sendLocale(final CommandSender sender,
227                        final Map<String, String> replace) {
228                Utils.sI18n(sender, this, replace);
229        }
230
231        /**
232         * Send the locale the the CommandSender
233         * 
234         * @param sender
235         *            CommandSender that will get the messages
236         */
237        public void sendLocale(final CommandSender sender) {
238                Utils.sI18n(sender, this);
239        }
240
241        /**
242         * Get the locale with replacement variables
243         * 
244         * @param replace
245         *            Replacement variables
246         * @return the locale with the variables replaced in.
247         */
248        public String getLocale(final Map<String, String> replace) {
249                return Utils.I18n(this, replace);
250        }
251
252        /**
253         * Get the locale without replacement variables
254         * 
255         * @return the locale
256         */
257        public String getLocale() {
258                return Utils.I18n(this);
259        }
