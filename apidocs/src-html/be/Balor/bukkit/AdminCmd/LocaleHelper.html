078            private final String key;
079            private final String locale;
080    
081            private LocaleHelper(final String key, final String locale) {
082                    this.key = key;
083                    this.locale = locale;
084            }
085    
086            /**
087             * @return the key
088             */
089            public String getKey() {
090                    return key;
091            }
092    
093            public static void addAllLocales() {
094                    for (final LocaleHelper lh : values())
095                            LocaleManager.getInstance().addLocale(lh.key, lh.locale);
096            }
097    
098            /**
099             * Send the locale the the CommandSender
100             * 
101             * @param sender
102             *            CommandSender that will get the message
103             * @param replace
104             *            Replacement variables
105             */
106            public void sendLocale(final CommandSender sender, final Map<String, String> replace) {
107                    Utils.sI18n(sender, this, replace);
108            }
109    
110            /**
111             * Send the locale the the CommandSender
112             * 
113             * @param sender
114             *            CommandSender that will get the messages
115             */
116            public void sendLocale(final CommandSender sender) {
117                    Utils.sI18n(sender, this);
118            }
119    
120            /**
121             * Get the locale with replacement variables
122             * 
123             * @param replace
124             *            Replacement variables
125             * @return the locale with the variables replaced in.
126             */
127            public String getLocale(final Map<String, String> replace) {
128                    return Utils.I18n(this, replace);
129            }
130    
131            /**
132             * Get the locale without replacement variables
133             * 
134             * @return the locale
135             */
136            public String getLocale() {
