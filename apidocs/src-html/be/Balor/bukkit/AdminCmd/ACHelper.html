165                    final MaterialContainer m = checkMaterial(sender, name);
166                    if (!m.isNull()) {
167                            final ExtendedConfiguration config = fManager.getYml("blacklist");
168                            List<Integer> list = config.getIntList("BlackListedBlocks", null);
169                            if (list == null)
170                                    list = new ArrayList<Integer>();
171                            list.add(m.getMaterial().getId());
172                            config.set("BlackListedBlocks", list);
173                            try {
174                                    config.save();
175                            } catch (final IOException e) {
176                            }
177                            if (blockBlacklist == null)
178                                    blockBlacklist = new ArrayList<Integer>();
179                            blockBlacklist.add(m.getMaterial().getId());
180                            final HashMap<String, String> replace = new HashMap<String, String>();
181                            replace.put("material", m.getMaterial().toString());
182                            Utils.sI18n(sender, "addBlacklistBlock", replace);
183                            return true;
184                    }
185                    return false;
186            }

324             * Used to check if the Ban is a Temporary ban, to relaunch the task to
325             * unBan the player or unban him if his time out.
326             * 
327             * @param player
328             * @return true if the ban is valid, false if invalid (expired)
329             */
330            private boolean checkBan(final BannedPlayer player) {
331                    if (player instanceof TempBannedPlayer) {
332                            final Long timeLeft = ((TempBannedPlayer) player).timeLeft();
333                            if (timeLeft <= 0) {
334                                    unBanPlayer(player.getPlayer());
335                                    return false;
336                            } else {
337                                    ACPluginManager.getScheduler().scheduleAsyncDelayedTask(
338                                                    ACHelper.getInstance().getCoreInstance(), new Runnable() {
339    
340                                                            @Override
341                                                            public void run() {
342                                                                    unBanPlayer(player.getPlayer());
343    
344                                                            }
345                                                    }, timeLeft * Utils.secondInMillis * 20);
346                                    return true;
347                            }
348                    } else
349                            return true;
350            }

377            private void convertSpawnWarp() {
378                    final File spawnFile = fManager.getFile("spawn", "spawnLocations.yml", false);
379                    if (spawnFile.exists()) {
380                            final ExtendedConfiguration spawn = ExtendedConfiguration.loadConfiguration(spawnFile);
381                            final ConfigurationSection spawnPoints = spawn.getConfigurationSection("spawn");
382                            if (spawnPoints != null)
383                                    for (final String key : spawnPoints.getKeys(false))
384                                            try {
385                                                    ACWorld.getWorld(key).setSpawn(
386                                                                    fManager.getLocation("spawn." + key, "spawnLocations", "spawn"));
387                                            } catch (final WorldNotLoaded e) {
388                                            }
389    
390                            spawnFile.delete();
391                            spawnFile.getParentFile().delete();
392                    }
393                    final File warpFile = fManager.getFile("warp", "warpPoints.yml", false);
394                    if (warpFile.exists()) {
395                            for (final String key : fManager.getKeys("warp", "warpPoints", "warp")) {
396                                    try {
397                                            final Location loc = fManager.getLocation("warp." + key, "warpPoints", "warp");
398                                            ACWorld.getWorld(loc.getWorld().getName()).addWarp(key, loc);
399                                    } catch (final WorldNotLoaded e) {
400                                    }
401                            }
402                            warpFile.delete();
403                            warpFile.getParentFile().delete();
404                    }
405            }

449            // translates a given color value/name into a real color value
450            // also does some checking (error = -1)
451            private short getColor(final String name, final Material mat) {
452                    short value = -1;
453                    // first try numbered colors
454                    try {
455                            value = Short.parseShort(name);
456                    } catch (final Exception e) {
457                            // try to find the name then
458                            for (short i = 0; i < materialsColors.get(mat).length; ++i)
459                                    if (materialsColors.get(mat)[i].equalsIgnoreCase(name)) {
460                                            value = i;
461                                            break;
462                                    }
463                    }
464                    // is the value OK?
465                    if (value < 0 || value >= materialsColors.get(mat).length)
466                            return -1;
467                    return value;
468            }

585            public void groupSpawn(final CommandSender sender) {
586                    if (Utils.isPlayer(sender)) {
587                            final Player player = ((Player) sender);
588                            Location loc = null;
589                            final String worldName = player.getWorld().getName();
590                            if (groups.isEmpty()) {
591                                    loc = ACWorld.getWorld(worldName).getSpawn();
592                                    if (loc == null)
593                                            loc = player.getWorld().getSpawnLocation();
594                                    player.teleport(loc);
595                                    Utils.sI18n(sender, "spawn");
596                                    return;
597                            }
598                            for (final String groupName : groups) {
599                                    try {
600                                            if (PermissionManager.isInGroup(groupName, player))
601                                                    loc = ACWorld.getWorld(worldName)
602                                                                    .getWarp("spawn" + groupName.toLowerCase()).loc;
603                                            break;
604                                    } catch (final NoPermissionsPlugin e) {
605                                            loc = ACWorld.getWorld(worldName).getSpawn();
606                                            break;
607                                    }
608                            }
609    
610                            if (loc == null)
611                                    loc = player.getWorld().getSpawnLocation();
612                            player.teleport(loc);
613                            Utils.sI18n(sender, "spawn");
614                    }
615            }

735            // changes the color of a colorable item in hand
736            public boolean itemColor(final CommandSender sender, final String color) {
737                    if (Utils.isPlayer(sender)) {
738                            // help?
739                            if (color.equalsIgnoreCase("help")) {
740                                    sender.sendMessage(ChatColor.RED + "Wool: " + ChatColor.WHITE
741                                                    + printColors(Material.WOOL));
742                                    sender.sendMessage(ChatColor.RED + "Dyes: " + ChatColor.WHITE
743                                                    + printColors(Material.INK_SACK));
744                                    sender.sendMessage(ChatColor.RED + "Logs: " + ChatColor.WHITE
745                                                    + printColors(Material.LOG));
746                                    sender.sendMessage(ChatColor.RED + "Slab: " + ChatColor.WHITE
747                                                    + printColors(Material.STEP));
748                                    return true;
749                            }
750                            // determine the value based on what you're holding
751                            short value = -1;
752                            final Material m = ((Player) sender).getItemInHand().getType();
753    
754                            if (materialsColors.containsKey(m))
755                                    value = getColor(color, m);
756                            else {
757                                    sender.sendMessage(ChatColor.RED + "You must hold a colorable material!");
758                                    return true;
759                            }
760                            // error?
761                            if (value < 0) {
762                                    sender.sendMessage(ChatColor.RED + "Color " + ChatColor.WHITE + color
763                                                    + ChatColor.RED + " is not usable for what you're holding!");
764                                    return true;
765                            }
766    
767                            final Player player = (Player) sender;
768                            final short colorVal = value;
769                            ACPluginManager.scheduleSyncTask(new Runnable() {
770                                    @Override
771                                    public void run() {
772                                            player.getItemInHand().setDurability(colorVal);
773    
774                                    }
775                            });
776                    }
777                    return true;
778            }

866                    final MaterialContainer m = checkMaterial(sender, name);

883                            replace.put("material", m.getMaterial().toString());
884                            Utils.sI18n(sender, "rmBlacklistBlock", replace);
885                            return true;
886                    }
887                    return false;
888            }

1123            public int undoLastModification(final String player) throws EmptyStackException {
1124                    if (!undoQueue.containsKey(player))
1125                            throw new EmptyStackException();
1126                    final Stack<Stack<BlockRemanence>> blockQueue = undoQueue.get(player);
1127                    if (blockQueue.isEmpty())
1128                            throw new EmptyStackException();
1129                    final Stack<BlockRemanence> undo = blockQueue.pop();
1130                    final Stack<BlockRemanence> undoCache = new Stack<BlockRemanence>();
1131                    int i = 0;
1132                    try {
1133                            while (!undo.isEmpty()) {
1134                                    undoCache.push(undo.pop());
1135                                    if (undoCache.size() == Utils.MAX_BLOCKS)
1136                                            ACPluginManager.getScheduler().scheduleSyncDelayedTask(coreInstance,
1137                                                            new UndoBlockTask(undoCache), 1);
1138                                    i++;
1139                            }
1140    
1141                    } catch (final Exception e) {
1142                            ACLogger.severe(e.getMessage(), e);
1143                            return i;
1144                    } finally {
1145                            ACPluginManager.getScheduler().scheduleSyncDelayedTask(coreInstance,
1146                                            new UndoBlockTask(undoCache), 1);
1147                    }
1148                    return i;
1149            }
