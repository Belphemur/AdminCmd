017package be.Balor.bukkit.AdminCmd;
018
019import java.io.IOException;
020import java.util.Collection;
021import java.util.HashMap;
022
023import org.bukkit.entity.Player;
024
025import be.Balor.Manager.LocaleManager;
026import be.Balor.Player.ACPlayer;
027import be.Balor.Tools.Type.Whois;
028import be.Balor.Tools.Utils;
029import be.Balor.Tools.Configuration.File.ExtendedConfiguration;
030import be.Balor.Tools.Files.FileManager;
031import be.Balor.Tools.Help.String.ACMinecraftFontWidthCalculator;
032import belgium.Balor.Workers.InvisibleWorker;
033
034import com.google.common.base.Joiner;
035
036/**
037 * @author Balor (aka Antoine Aflalo)
038 * 
039 */
040public enum TextLocale {
041        NEWS("NEWS"), RULES("Rules"), MOTD("MOTD"), MOTD_NEW(
042                        "MOTDNewUser",
043                        "motdNewUser");
044        private final String locale;
045        private final String file;
046        private static ExtendedConfiguration version;
047
048        /**
049 * 
050 */
051        private TextLocale(final String locale) {
052                this.locale = locale;
053                this.file = this.locale.toLowerCase();
054        }
055
056        private TextLocale(final String locale, final String file) {
057                this.locale = locale;
058                this.file = file;
059        }
060
061        /**
062         * Save the new content of the locale
063         * 
064         * @param content
065         *            new content
066         */
067        public void saveContent(final String content) {
068                final String parsedContent = parseContent(content);
069                LocaleManager.getInstance().addLocale(locale, parsedContent, true);
070                FileManager.getInstance().setTxtFile(file, parsedContent);
071                version.set(file, System.currentTimeMillis());
072                try {
073                        version.save();
074                } catch (final IOException e) {
075                }
076        }
077
078        /**
079         * @param content
080         */
081        private String parseContent(final String content) {
082                return Utils.colorParser(content.replaceAll("\\n", "\n").replaceAll(
083                                "//n", "\n"));
084        }
085
086        /**
087         * Reload the text file
088         */
089        public void reloadContent() {
090                final String result = parseContent(FileManager.getInstance()
091                                .getTextFile(file + ".txt"));
092                LocaleManager.getInstance().addLocale(locale, result, true);
093                version.set(file, System.currentTimeMillis());
094                try {
095                        version.save();
096                } catch (final IOException e) {
097                }
098
099        }
100
101        /**
102         * @param version
103         *            the version to set
104         */
105        static void setVersion(final ExtendedConfiguration version) {
106                TextLocale.version = version;
107        }
108
109        /**
110         * Get the last modification time of this locale
111         * 
112         * @return milliseconds
113         */
114        public long getModifTime() {
115                return version.getLong(file, 0);
116        }
117
118        /**
119         * Send the formated text of the locale to the player.
120         * 
121         * @return .
122         */
123        public void sendText(final Player p) {
124                final HashMap<String, String> replace = new HashMap<String, String>();
125                final ACPlayer acPlayer = ACPlayer.getPlayer(p);
126                final long total = acPlayer.getCurrentPlayedTime();
127                replace.putAll(Utils.playedTime(Utils.getPlayerName(p), total));
128                replace.put(
129                                "nb",
130                                String.valueOf(p.getServer().getOnlinePlayers().length
131                                                - InvisibleWorker.getInstance().nbInvisibles()));
132                replace.put("world", p.getWorld().getName());
133                replace.put("x", String.valueOf(p.getLocation().getX()));
134                replace.put("y", String.valueOf(p.getLocation().getY()));
135                replace.put("z", String.valueOf(p.getLocation().getZ()));
136                final Collection<String> list = Utils.getPlayerList(p);
137                String connected = Joiner.on(", ").join(list);
138                if (connected.length() >= ACMinecraftFontWidthCalculator.chatwidth) {
139                        final String tmp = connected.substring(0,
140                                        ACMinecraftFontWidthCalculator.chatwidth);
141                        final String tmp2 = connected.substring(
142                                        ACMinecraftFontWidthCalculator.chatwidth,
143                                        connected.length());
144                        connected = tmp + "//n" + tmp2;
145                }
146                replace.put("connected", connected);
147                final String serverTime = Utils.replaceDateAndTimeFormat(Utils
148                                .getServerRealTime("GMT" + ConfigEnum.DT_GMT.getString()));
149                replace.put("time", serverTime);
150                final String date = Utils.replaceDateAndTimeFormat(acPlayer,
151                                Whois.LOGIN);
152                if (date == null) {
153                        replace.put("lastlogin", Utils.I18n("noLoginInformation"));
154                } else {
155                        replace.put("lastlogin", date);
156                }
157                final String messageToSend = parseContent(Utils.I18n(locale, replace));
158                if (messageToSend != null) {
159                        for (final String toSend : messageToSend.split("\n")) {
160                                if (toSend.isEmpty()) {
161                                        continue;
162                                }
163                                p.sendMessage(toSend);
164                        }
165                }
166        }
167}
