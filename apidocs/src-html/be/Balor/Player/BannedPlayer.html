017    package be.Balor.Player;
018    
019    import java.sql.Date;
020    
021    /**
022     * @author Balor (aka Antoine Aflalo)
023     * 
024     */
025    public class BannedPlayer {
026            private String player;
027            private String reason;
028            private Date date;
029    
030            /**
031             * 
032             */
033            public BannedPlayer() {
034            }
035    
036            /**
037             * @param player
038             * @param reason
039             */
040            public BannedPlayer(final String player, final String reason) {
041                    this.player = player;
042                    this.reason = reason;
043                    date = new Date(System.currentTimeMillis());
044            }
045    
046            /**
047             * @return the player
048             */
049            public String getPlayer() {
050                    return player;
051            }
052    
053            /**
054             * @param player
055             *            the player to set
056             */
057            public void setPlayer(final String player) {
058                    this.player = player;
059            }
060    
061            /**
062             * @return the reason
063             */
064            public String getReason() {
065                    return reason;
066            }
067    
068            /**
069             * @param reason
070             *            the reason to set
071             */
072            public void setReason(final String reason) {
073                    this.reason = reason;
074            }
075    
076            /**
077             * @return the date
078             */
079            public Date getDate() {
080                    return date;
081            }
082    
083            /**
084             * @param date
085             *            the date to set
086             */
087            public void setDate(final Date date) {
088                    this.date = date;
089            }
090    
091            /*
092             * (non-Javadoc)
093             * 
094             * @see java.lang.Object#hashCode()
095             */
096            @Override
097            public int hashCode() {
098                    final int prime = 31;
099                    int result = 5;
100                    result = prime * result + ((date == null) ? 0 : date.hashCode());
101                    result = prime * result + ((player == null) ? 0 : player.hashCode());
102                    result = prime * result + ((reason == null) ? 0 : reason.hashCode());
103                    return result;
104            }
105    
106            /*
107             * (non-Javadoc)
108             * 
109             * @see java.lang.Object#equals(java.lang.Object)
110             */
111            @Override
112            public boolean equals(final Object obj) {
113                    if (this == obj)
114                            return true;
115                    if (obj == null)
116                            return false;
117                    if (!(obj instanceof BannedPlayer))
118                            return false;
119                    final BannedPlayer other = (BannedPlayer) obj;
120                    if (date == null) {
121                            if (other.date != null)
122                                    return false;
123                    } else if (!date.equals(other.date))
124                            return false;
125                    if (player == null) {
126                            if (other.player != null)
127                                    return false;
128                    } else if (!player.equals(other.player))
129                            return false;
130                    if (reason == null) {
131                            if (other.reason != null)
132                                    return false;
133                    } else if (!reason.equals(other.reason))
134                            return false;
135                    return true;
136            }
137    
138            /*
139             * (non-Javadoc)
140             * 
141             * @see java.lang.Object#toString()
142             */
143            @Override
144            public String toString() {
145                    return "BannedPlayer [player=" + player + ", reason=" + reason + ", date=" + date + "]";
146            }
147    
148    }
