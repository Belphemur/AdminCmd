017package be.Balor.Player;
018
019import java.util.Map.Entry;
020
021import org.bukkit.Location;
022
023import be.Balor.Manager.Exceptions.WorldNotLoaded;
024import be.Balor.Tools.Type;
025import be.Balor.Tools.Debug.ACLogger;
026import be.Balor.Tools.Debug.DebugLog;
027
028/**
029 * @author Antoine
030 * 
031 */
032public class PlayerConvertTask implements Runnable {
033
034        private final IPlayerFactory newFactory;
035        private final ACPlayer oldPlayer;
036
037        /**
038         * @param oldFactory
039         * @param newFactory
040         * @param name
041         */
042        public PlayerConvertTask(final IPlayerFactory oldFactory,
043                        final IPlayerFactory newFactory, final String name) {
044                super();
045                this.newFactory = newFactory;
046                this.oldPlayer = oldFactory.createPlayer(name);
047        }
048
049        public PlayerConvertTask(final IPlayerFactory newFactory,
050                        final ACPlayer oldPlayer) {
051                super();
052                this.newFactory = newFactory;
053                this.oldPlayer = oldPlayer;
054                this.oldPlayer.forceSave();
055        }
056
057        /*
058         * (non-Javadoc)
059         * 
060         * @see java.lang.Runnable#run()
061         */
062        @Override
063        public void run() {
064                final String name = oldPlayer.getName();
065                ACLogger.info("Convert player : " + name);
066                DebugLog.INSTANCE.info("Convert player : " + name);
067                newFactory.addExistingPlayer(name);
068                final ACPlayer newPlayer = newFactory.createPlayer(name);
069                DebugLog.INSTANCE.info("Convert lastLoc");
070                newPlayer.setLastLocation(oldPlayer.getLastLocation());
071                DebugLog.INSTANCE.info("Convert presentation");
072                newPlayer.setPresentation(oldPlayer.getPresentation());
073                DebugLog.INSTANCE.info("Convert homes");
074                for (final String home : oldPlayer.getHomeList()) {
075                        try {
076                                final Location homeLoc = oldPlayer.getHome(home);
077                                if (homeLoc == null) {
078                                        ACLogger.warning("The home "
079                                                        + home
080                                                        + " of player "
081                                                        + name
082                                                        + " has not been converted because the world is not loaded.");
083                                } else {
084                                        newPlayer.setHome(home, homeLoc);
085                                }
086                        } catch (final WorldNotLoaded e) {
087                                ACLogger.warning("The home " + home + " of player " + name
088                                                + " has not been converted because the world "
089                                                + e.getMessage() + " don't exist anymore");
090                        } catch (final Exception e) {
091                                ACLogger.warning("The home " + home + " of player " + name
092                                                + " has not been converted because : " + e.getMessage());
093                        }
094                }
095                DebugLog.INSTANCE.info("Convert Powers");
096                for (final Entry<Type, Object> entry : oldPlayer.getPowers().entrySet()) {
097                        newPlayer.setPower(entry.getKey(), entry.getValue());
098                }
099                DebugLog.INSTANCE.info("Convert Custom Powers");
100                for (final Entry<String, Object> entry : oldPlayer.getCustomPowers()
101                                .entrySet()) {
102                        newPlayer.setCustomPower(entry.getKey(), entry.getValue());
103                }
104                DebugLog.INSTANCE.info("Convert Infos");
105                for (final String info : oldPlayer.getInformationsList()) {
106                        if (info.equals("lastLoc") || info.equals("presentation")) {
107                                continue;
108                        }
109                        newPlayer.setInformation(info, oldPlayer.getInformation(info)
110                                        .getObj());
111                }
112                DebugLog.INSTANCE.info("Convert Kit");
113                for (final String kit : oldPlayer.getKitUseList()) {
114                        newPlayer.setLastKitUse(kit, oldPlayer.getLastKitUse(kit));
115                }
116                if (!(newPlayer instanceof FilePlayer)) {
117                        newPlayer.forceSave();
118                }
119
120        }
121}
