020    import java.util.List;

032    /**
033     * @author Balor (aka Antoine Aflalo)
034     * 
035     */
036    public class PlayerManager {
037            private final ConcurrentMap<String, ACPlayer> players = new MapMaker().concurrencyLevel(8)
038                            .weakValues().makeMap();
039            private final ConcurrentMap<ACPlayer, Boolean> onlinePlayers = new MapMaker().concurrencyLevel(
040                            8).makeMap();
041            private final static PlayerManager instance = new PlayerManager();
042            private IPlayerFactory playerFactory;
043    
044            /**
045             *
046             */
047            private PlayerManager() {
048                    final EmptyPlayer console = new EmptyPlayer("serverConsole");
049                    onlinePlayers.put(console, true);
050                    addPlayer(console);
051            }
052    
053            /**
054             * @return the instance
055             */
056            public static PlayerManager getInstance() {
057                    return instance;
058            }
059    
060            /**
061             * @param playerFactory
062             *            the playerFactory to set
063             */
064            public void setPlayerFactory(final IPlayerFactory playerFactory) {
065                    this.playerFactory = playerFactory;
066            }
067    
068            /**
069             * Convert the ACPlayer
070             * 
071             * @param playerFactory
072             */
073            public void convertFactory(final IPlayerFactory factory) {
074                    ACLogger.info("Converting player to the new type.");
075                    for (final String name : this.playerFactory.getExistingPlayers()) {
076                            factory.addExistingPlayer(name);
077                            final ACPlayer oldPlayer = playerFactory.createPlayer(name);
078                            final ACPlayer newPlayer = factory.createPlayer(name);
079                            newPlayer.setLastLocation(oldPlayer.getLastLocation());
080                            newPlayer.setPresentation(oldPlayer.getPresentation());
081    
082                            for (final String home : oldPlayer.getHomeList())
083                                    newPlayer.setHome(home, oldPlayer.getHome(home));
084                            for (final Entry<String, String> entry : oldPlayer.getPowers().entrySet()) {
085                                    final Type power = Type.matchType(entry.getKey());
086                                    if (power != null)
087                                            newPlayer.setPower(power, oldPlayer.getPower(power).getObj());
088                                    else
089                                            newPlayer.setCustomPower(entry.getKey(),
090                                                            oldPlayer.getCustomPower(entry.getKey()).getObj());
091                            }
092                            for (final String info : oldPlayer.getInformationsList()) {
093                                    if (info.equals("lastLoc") || info.equals("presentation"))
094                                            continue;
095                                    newPlayer.setInformation(info, oldPlayer.getInformation(info).getObj());
096                            }
097                            for (final String kit : oldPlayer.getKitUseList())
098                                    newPlayer.setLastKitUse(kit, oldPlayer.getLastKitUse(kit));
099                            newPlayer.forceSave();
100                    }
101                    this.playerFactory = factory;
102                    ACLogger.info("Convertion finished.");
103            }
104    
105            /**
106             * Add a new player
107             * 
108             * @param player
109             */
110            private synchronized boolean addPlayer(final ACPlayer player) {
111                    final String name = player.getName();
112                    if (name == null) {
113                            throw new NullPointerException();
114                    }
115    
116                    final ACPlayer ref = players.get(name);
117                    if (ref != null)
118                            return false;
119                    players.put(name, player);
120                    return true;
121            }
122    
123            /**
124             * Return online AC players
125             * 
126             * @return
127             */
128            public List<ACPlayer> getOnlineACPlayers() {
129                    return new ArrayList<ACPlayer>(onlinePlayers.keySet());
130            }
131    
132            /**
133             * Get Online Bukkit Player
134             * 
135             * @return
136             */
137            public List<Player> getOnlinePlayers() {
138                    final ArrayList<Player> list = new ArrayList<Player>(onlinePlayers.size());
139                    for (final ACPlayer p : onlinePlayers.keySet()) {
140                            final Player handler = p.getHandler();
141                            if (handler != null)
142                                    list.add(handler);
143                    }
144                    return list;
145            }
146    
147            /**
148             * Get the list of AC Player having the wanted custom power
149             * 
150             * @param power
151             * @return
152             */
153            List<ACPlayer> getACPlayerHavingPower(final String power) {
154                    final ArrayList<ACPlayer> list = new ArrayList<ACPlayer>();
155                    for (final ACPlayer p : getExistingPlayers()) {
156                            if (p.hasCustomPower(power))
157                                    list.add(p);
158                    }
159                    return list;
160            }
161    
162            /**
163             * Get the list of AC Player having the wanted power
164             * 
165             * @param power
166             * @return
167             */
168            List<ACPlayer> getACPlayerHavingPower(final Type power) {
169                    final ArrayList<ACPlayer> list = new ArrayList<ACPlayer>();
170                    for (final ACPlayer p : getExistingPlayers()) {
171                            if (p.hasPower(power))
172                                    list.add(p);
173                    }
174                    return list;
175            }
176    
177            private List<ACPlayer> getExistingPlayers() {
178                    final ArrayList<ACPlayer> list = new ArrayList<ACPlayer>();
179                    for (final String name : playerFactory.getExistingPlayers()) {
180                            final ACPlayer player = demandACPlayer(name);
181                            if (!(player instanceof EmptyPlayer))
182                                    list.add(player);
183                    }
184                    return list;
185            }
186    
187            /**
188             * Get the wanted player
189             * 
190             * @param name
191             *            name of the player
192             * @return the ACPlayer if found, else null
193             */
194            private synchronized ACPlayer getPlayer(final String name) {
195                    final ACPlayer result = players.get(name);
196                    if (result != null)
197                            result.reloadHandler();
198                    return result;
199            }
200    
201            /**
202             * Set Offline an online player. The player will lost his strong reference,
203             * when the gc will be called, the reference will be deleted.
204             * 
205             * @param player
206             *            player to setOffline
207             * @return
208             */
209            public boolean setOffline(final ACPlayer player) {
210                    player.updatePlayedTime();
211                    player.forceSave();
212                    player.setOnline(false);
213                    return onlinePlayers.remove(player) != null;
214            }
215    
216            public ACPlayer setOnline(final Player player) {
217                    playerFactory.addExistingPlayer(player.getName());
218                    final ACPlayer acPlayer = demandACPlayer(player);
219                    onlinePlayers.put(acPlayer, true);
220                    DebugLog.INSTANCE.info(player.getName() + " is put online.");
221                    return acPlayer;
222            }
223    
224            ACPlayer demandACPlayer(final String name) {
225                    if (name == null)
226                            return getPlayer("serverConsole");
227                    ACPlayer result = getPlayer(name);
228                    if (result == null) {
229                            result = playerFactory.createPlayer(name);
230                            addPlayer(result);
231                            result = getPlayer(name);
232                    } else if (result instanceof EmptyPlayer) {
233                            final ACPlayer tmp = playerFactory.createPlayer(name);
234                            if (tmp instanceof EmptyPlayer)
235                                    return result;
236                            players.remove(name);
237                            onlinePlayers.remove(result);
238                            result = tmp;
239                            addPlayer(result);
240                            result = getPlayer(name);
241                    }
242                    return result;
243            }
244    
245            ACPlayer demandACPlayer(final Player player) {
246                    if (player == null)
247                            return getPlayer("serverConsole");
248                    final String playerName = player.getName();
249                    ACPlayer result = getPlayer(playerName);
250                    if (result == null) {
251                            result = playerFactory.createPlayer(player);
252                            addPlayer(result);
253                            result = getPlayer(playerName);
254                    } else if (result instanceof EmptyPlayer) {
255                            final ACPlayer tmp = playerFactory.createPlayer(playerName);
256                            if (tmp instanceof EmptyPlayer)
257                                    return result;
258                            players.remove(playerName);
259                            onlinePlayers.remove(result);
260                            result = tmp;
261                            addPlayer(result);
262                            result = getPlayer(playerName);
263                    }
264                    return result;
265            }
