038public abstract class ACPlayer {

046        /**
047         *
048         */
049        protected ACPlayer(final String name) {
050                this.name = name;
051                final int prime = 41;
052                int result = 7;
053                result = prime * result
054                                + ((this.name == null) ? 0 : this.name.hashCode());
055                hashCode = result;
056                handler = ACPluginManager.getServer().getPlayer(this.name);
057        }
058
059        protected ACPlayer(final Player p) {
060                this.name = p.getName();
061                final int prime = 41;
062                int result = 7;
063                result = prime * result
064                                + ((this.name == null) ? 0 : this.name.hashCode());
065                hashCode = result;
066                handler = p;
067        }

352        /**
353         * 
354         * @return the total played time in Long
355         */
356        public long getCurrentPlayedTime() {
357                if (this.isOnline()) {
358                        return (getInformation("totalTime").getLong(0)
359                                        + System.currentTimeMillis() - getInformation(
360                                                "lastConnection").getLong(System.currentTimeMillis()));
361                } else {
362                        return getInformation("totalTime").getLong(0);
363                }
364        }

395        /**
396         * @param isOnline
397         *            the isOnline to set
398         */
399        void setOnline(final boolean isOnline) {
400                this.online = isOnline;
401                if (!this.online) {
402                        this.handler = null;
403                } else if (handler == null) {
404                        this.handler = ACPluginManager.getServer().getPlayer(this.name);
405                }
406        }

413                this.lastCmd = lastCmd;
414        }
415
416        /**
417         * Execute the last command
418         * 
419         * @throws NullPointerException
420         *             if last command is not defined
421         * @throws PlayerNotFound
422         */
423        public void executeLastCmd() throws NullPointerException, PlayerNotFound {
424                if (this.lastCmd == null) {
425                        throw new NullPointerException();
426                }
427                try {
428                        this.lastCmd.execute();
429                } catch (final ActionNotPermitedException e) {
430                        e.sendMessage();
431                }
432        }

470        /*
471         * (non-Javadoc)
472         * 
473         * @see java.lang.Object#equals(java.lang.Object)
474         */
475        @Override
476        public boolean equals(final Object obj) {
477                if (this == obj) {
478                        return true;
479                }
480                if (obj == null) {
481                        return false;
482                }
483                if (!(obj instanceof ACPlayer)) {
484                        return false;
485                }
486                final ACPlayer other = (ACPlayer) obj;
487                if (name == null) {
488                        if (other.name != null) {
489                                return false;
490                        }
491                } else if (!name.equals(other.name)) {
492                        return false;
493                }
494                return true;
495        }
