037    public abstract class ACPlayer {
038            protected final String name;
039            private final int hashCode;
040            protected boolean isOnline = false;
041            protected ACCommandContainer lastCmd = null;
042            protected Player handler = null;
043            protected TpRequest tpRequest = null;
044    
045            /**
046             *
047             */
048            protected ACPlayer(final String name) {
049                    this.name = name;
050                    final int prime = 41;
051                    int result = 7;
052                    result = prime * result + ((this.name == null) ? 0 : this.name.hashCode());
053                    hashCode = result;
054                    handler = ACPluginManager.getServer().getPlayer(this.name);
055            }
056    
057            protected ACPlayer(final Player p) {
058                    this.name = p.getName();
059                    final int prime = 41;
060                    int result = 7;
061                    result = prime * result + ((this.name == null) ? 0 : this.name.hashCode());
062                    hashCode = result;
063                    handler = p;
064            }

350             * 
351             * @return the total played time in Long
352             */
353            public long getCurrentPlayedTime() {
354                    return (getInformation("totalTime").getLong(0) + System.currentTimeMillis() - getInformation(
355                                    "lastConnection").getLong(System.currentTimeMillis()));
356            }

373            /**
374             * @param isOnline
375             *            the isOnline to set
376             */
377            void setOnline(final boolean isOnline) {
378                    this.isOnline = isOnline;
379                    if (!this.isOnline)
380                            this.handler = null;
381            }
382    
383            void reloadHandler() {
384                    if (handler == null)
385                            this.handler = ACPluginManager.getServer().getPlayer(this.name);
386            }

396            /**
397             * Execute the last command
398             * 
399             * @throws NullPointerException
400             *             if last command is not defined
401             * @throws PlayerNotFound
402             */
403            public void executeLastCmd() throws NullPointerException, PlayerNotFound {
404                    if (this.lastCmd == null)
405                            throw new NullPointerException();
406                    this.lastCmd.execute();
407            }

438            /*
439             * (non-Javadoc)
440             * 
441             * @see java.lang.Object#equals(java.lang.Object)
442             */
443            @Override
444            public boolean equals(final Object obj) {
445                    if (this == obj)
446                            return true;
447                    if (obj == null)
448                            return false;
449                    if (!(obj instanceof ACPlayer))
450                            return false;
451                    final ACPlayer other = (ACPlayer) obj;
452                    if (name == null) {
453                            if (other.name != null)
454                                    return false;
455                    } else if (!name.equals(other.name))
456                            return false;
457                    return true;
458            }
