034import com.google.common.base.Joiner;
035import com.google.common.collect.MapMaker;

037/**
038 * @author Balor (aka Antoine Aflalo)
039 * 
040 */
041public class SQLPlayerFactory implements IPlayerFactory {
042        private final PreparedStatement insertPlayer;
043        private final Map<String, Long> playersID = new MapMaker()
044                        .concurrencyLevel(6).makeMap();
045        private final PreparedStatement doubleCheckPlayer;
046
047        /**
048 * 
049 */
050        public SQLPlayerFactory() {
051                insertPlayer = Database.DATABASE
052                                .prepare("INSERT INTO `ac_players` (`name`) VALUES (?);");
053                final ResultSet rs = Database.DATABASE
054                                .query("SELECT `name`,`id` FROM `ac_players`");
055                doubleCheckPlayer = Database.DATABASE
056                                .prepare("SELECT `id` FROM `ac_players` WHERE `name` = ?");
057                try {
058                        while (rs.next()) {
059                                playersID.put(rs.getString("name"), rs.getLong("id"));
060                        }
061                        rs.close();
062                } catch (final SQLException e) {
063                        ACLogger.severe("Problem when getting players from the DB", e);
064                }
065                DebugLog.INSTANCE.info("Players found : "
066                                + Joiner.on(", ").join(playersID.keySet()));
067
068        }
069
070        private Long getPlayerID(final String playername) {
071                Long id = playersID.get(playername);
072                if (id != null) {
073                        return id;
074                }
075                try {
076                        Database.DATABASE.autoReconnect();
077                } catch (final Throwable e) {
078                }
079
080                ResultSet rs = null;
081                synchronized (doubleCheckPlayer) {
082                        try {
083                                doubleCheckPlayer.clearParameters();
084                                doubleCheckPlayer.setString(1, playername);
085                                doubleCheckPlayer.execute();
086                                rs = doubleCheckPlayer.getResultSet();
087                                if (rs == null) {
088                                        return null;
089                                }
090                                if (!rs.next()) {
091                                        return null;
092                                }
093                                id = rs.getLong(1);
094                                if (id != null) {
095                                        playersID.put(playername, id);
096                                }
097                        } catch (final SQLException e) {
098                                return null;
099                        } finally {
100                                try {
101                                        if (rs != null) {
102                                                rs.close();
103                                        }
104                                } catch (final SQLException e) {
105
106                                }
107                        }
108
109                }
110                return id;
111        }
112
113        /*
114         * (Non javadoc)
115         * 
116         * @see be.Balor.Player.IPlayerFactory#createPlayer(java.lang.String)
117         */
118        @Override
119        public ACPlayer createPlayer(final String playername) {
120                final Long id = getPlayerID(playername);
121                if (id == null) {
122                        return new EmptyPlayer(playername);
123                } else {
124                        return new SQLPlayer(playername, id);
125                }
126
127        }
128
129        /*
130         * (Non javadoc)
131         * 
132         * @see
133         * be.Balor.Player.IPlayerFactory#createPlayer(org.bukkit.entity.Player)
134         */
135        @Override
136        public ACPlayer createPlayer(final Player player) {
137                final Long id = getPlayerID(player.getName());
138                if (id == null) {
139                        return new EmptyPlayer(player);
140                } else {
141                        return new SQLPlayer(player, id);
142                }
143
144        }
145
146        /*
147         * (Non javadoc)
148         * 
149         * @see be.Balor.Player.IPlayerFactory#getExistingPlayers()
150         */
151        @Override
152        public Set<String> getExistingPlayers() {
153                return Collections.unmodifiableSet(playersID.keySet());
154        }
155
156        /*
157         * (Non javadoc)
158         * 
159         * @see be.Balor.Player.IPlayerFactory#addExistingPlayer(java.lang.String)
160         */
161        @Override
162        public void addExistingPlayer(final String player) {
163                if (!playersID.containsKey(player)) {
164                        ResultSet rs = null;
165                        try {
166                                synchronized (insertPlayer) {
167                                        insertPlayer.clearParameters();
168                                        insertPlayer.setString(1, player);
169                                        insertPlayer.executeUpdate();
170
171                                        rs = insertPlayer.getGeneratedKeys();
172                                        if (rs.next()) {
173                                                playersID.put(player, rs.getLong(1));
174                                        }
175                                        if (rs != null) {
176                                                rs.close();
177                                        }
178                                }
179                        } catch (final SQLException e) {
180                                ACLogger.severe("Problem when adding player to the DB", e);
181                                if (rs != null) {
182                                        try {
183                                                rs.close();
184                                        } catch (final SQLException e1) {
185                                        }
186                                }
187                        }
188                }
189
190        }
