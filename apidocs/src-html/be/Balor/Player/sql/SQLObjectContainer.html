015package be.Balor.Player.sql;
016
017import org.yaml.snakeyaml.DumperOptions;
018import org.yaml.snakeyaml.Yaml;
019
020import be.Balor.Tools.Configuration.File.ExtendedRepresenter;
021import be.Balor.Tools.Configuration.File.YamlConstructor;
022import be.Balor.Tools.Egg.EggType;
023import be.Balor.Tools.Files.ObjectContainer;
024
025/**
026 * @author Balor (aka Antoine Aflalo)
027 * 
028 */
029public class SQLObjectContainer extends ObjectContainer {
030        public final static Yaml yaml = new Yaml(new YamlConstructor(
031                        ObjectContainer.class.getClassLoader()), new ExtendedRepresenter(),
032                        new DumperOptions());
033
034        /**
035         * @param obj
036         */
037        public SQLObjectContainer(final Object obj) {
038                super(obj);
039        }
040
041        /*
042         * (Non javadoc)
043         * 
044         * @see be.Balor.Tools.Files.ObjectContainer#getEggType()
045         */
046        @Override
047        public EggType<?> getEggType() {
048                if (getObj() instanceof String) {
049                        synchronized (yaml) {
050                                return (EggType<?>) yaml.load((String) getObj());
051                        }
052                }
053                return super.getEggType();
054        }
055
056        /*
057         * (non-Javadoc)
058         * 
059         * @see be.Balor.Tools.Files.ObjectContainer#castBoolean(java.lang.Object)
060         */
061        @Override
062        protected Boolean castBoolean(final Object o) {
063                if (o instanceof String) {
064                        synchronized (yaml) {
065                                return super.castBoolean(yaml.load((String) o));
066                        }
067                } else {
068                        return super.castBoolean(o);
069                }
070        }
071
072        /*
073         * (non-Javadoc)
074         * 
075         * @see be.Balor.Tools.Files.ObjectContainer#castDouble(java.lang.Object)
076         */
077        @Override
078        protected Double castDouble(final Object o) {
079                if (o instanceof String) {
080                        synchronized (yaml) {
081                                return super.castDouble(yaml.load((String) o));
082                        }
083                } else {
084                        return super.castDouble(o);
085                }
086        }
087
088        /*
089         * (non-Javadoc)
090         * 
091         * @see be.Balor.Tools.Files.ObjectContainer#castFloat(java.lang.Object)
092         */
093        @Override
094        protected Float castFloat(final Object o) {
095                if (o instanceof String) {
096                        synchronized (yaml) {
097                                return super.castFloat(yaml.load((String) o));
098                        }
099                } else {
100                        return super.castFloat(o);
101                }
102        }
103
104        /*
105         * (non-Javadoc)
106         * 
107         * @see be.Balor.Tools.Files.ObjectContainer#castInt(java.lang.Object)
108         */
109        @Override
110        protected Integer castInt(final Object o) {
111                if (o instanceof String) {
112                        synchronized (yaml) {
113                                return super.castInt(yaml.load((String) o));
114                        }
115                } else {
116                        return super.castInt(o);
117                }
118        }
119
120        /*
121         * (non-Javadoc)
122         * 
123         * @see be.Balor.Tools.Files.ObjectContainer#castLong(java.lang.Object)
124         */
125        @Override
126        protected Long castLong(final Object o) {
127                if (o instanceof String) {
128                        synchronized (yaml) {
129                                return super.castLong(yaml.load((String) o));
130                        }
131                } else {
132                        return super.castLong(o);
133                }
134        }
135
136}
