110        /**
111         * To Schedule the Async task
112         */
113        public static void scheduleAsyncSave() {
114                if (ACPluginManager.getScheduler().isCurrentlyRunning(ioStackTaskId)
115                                || ACPluginManager.getScheduler().isQueued(ioStackTaskId)) {
116                        return;
117                }
118                final int delay = ConfigEnum.WDELAY.getInt() >= 30 ? ConfigEnum.WDELAY
119                                .getInt() : 30;
120                ioStackTaskId = ACPluginManager
121                                .getScheduler()
122                                .runTaskTimerAsynchronously(
123                                                ACHelper.getInstance().getCoreInstance(), IOSAVET_TASK,
124                                                20 * 60, 20 * delay).getTaskId();
125                DebugLog.INSTANCE.info("IO Save RepeatingTask created : "
126                                + ioStackTaskId);
127        }
128
129        /**
130         * To stop the saving task.
131         */
132        public static void stopSavingTask() {
133                if (!ACPluginManager.getScheduler().isCurrentlyRunning(ioStackTaskId)
134                                && !ACPluginManager.getScheduler().isQueued(ioStackTaskId)) {
135                        return;
136                }
137                ACPluginManager.getScheduler().cancelTask(ioStackTaskId);
138        }

168        @Override
169        public Location getHome(final String home) {
170                final ConfigurationSection homeSection = homes
171                                .getConfigurationSection(home);
172                if (homeSection == null) {
173                        final String found = Str.matchString(homes.getKeys(false), home);
174                        if (found == null) {
175                                return null;
176                        }
177                        return getLocation(homes.getConfigurationSection(found));
178                } else {
179                        return getLocation(homeSection);
180                }
181        }

256        private Location getLocation(final ConfigurationSection node) {
257                final String world = node.getString("world");
258                if (world == null) {
259                        return null;
260                }
261                World w = Bukkit.getWorld(world);
262                if (w == null) {
263                        w = ACWorld.getWorld(world).getHandle();
264                }
265                if (w != null) {
266                        return new Location(w, node.getDouble("x", 0), node.getDouble("y",
267                                        0), node.getDouble("z", 0), Float.parseFloat(node
268                                        .getString("yaw")), Float.parseFloat(node
269                                        .getString("pitch")));
270                } else {
271                        ACLogger.warning("The world " + world + " is not loaded !");
272                        return null;
273                }
274        }

361        /*
362         * (non-Javadoc)
363         * 
364         * @see be.Balor.Player.ACPlayer#removeAllSuperPower()
365         */
366        @Override
367        public void removeAllSuperPower() {
368                for (final String power : powers.getKeys(false)) {
369                        final Type matched = Type.matchType(power);
370                        if (matched != null
371                                        && matched.getCategory().equals(Category.SUPER_POWER)) {
372                                powers.set(power, null);
373                                if (matched != Type.FLY) {
374                                        continue;
375                                }
376                                if (handler == null) {
377                                        continue;
378                                }
379                                handler.setFlying(false);
380                                handler.setAllowFlight(false);
381                        }
382                }
383
384                if (handler != null
385                                && InvisibleWorker.getInstance().hasInvisiblePowers(handler)) {
386                        InvisibleWorker.getInstance().reappear(handler);
387                }
388                writeFile();
389
390        }

520        /*
521         * (Non javadoc)
522         * 
523         * @see be.Balor.Player.ACPlayer#getPowers()
524         */
525        @Override
526        public Map<Type, Object> getPowers() {
527                final Map<Type, Object> result = new EnumMap<Type, Object>(Type.class);
528                for (final Entry<String, Object> entry : powers.getValues(false)
529                                .entrySet()) {
530                        final Type power = Type.matchType(entry.getKey());
531                        if (power == null || (power != null && power == Type.CUSTOM)) {
532                                continue;
533                        }
534                        result.put(power, entry.getValue());
535                }
536                return result;
537        }
