067                    super(name);

072                    }

074                    informations = datas.addSection("infos");
075                    homes = datas.addSection("home");
076                    powers = datas.addSection("powers");
077                    kitsUse = datas.addSection("kitsUse");
078                    lastLoc = informations.addSection("lastLoc");

106            /**
107             * To Schedule the Async task
108             */
109            public static void scheduleAsyncSave() {
110                    if (ACPluginManager.getScheduler().isCurrentlyRunning(ioStackTaskId)
111                                    || ACPluginManager.getScheduler().isQueued(ioStackTaskId))
112                            return;
113                    final int delay = ConfigEnum.WDELAY.getInt() >= 30 ? ConfigEnum.WDELAY.getInt() : 30;
114                    ioStackTaskId = ACPluginManager.getScheduler().scheduleAsyncRepeatingTask(
115                                    ACHelper.getInstance().getCoreInstance(), IOSAVET_TASK, 20 * 60, 20 * delay);
116                    DebugLog.INSTANCE.info("IO Save RepeatingTask created : " + ioStackTaskId);
117            }
118    
119            /**
120             * To stop the saving task.
121             */
122            public static void stopSavingTask() {
123                    if (!ACPluginManager.getScheduler().isCurrentlyRunning(ioStackTaskId)
124                                    && !ACPluginManager.getScheduler().isQueued(ioStackTaskId))
125                            return;
126                    ACPluginManager.getScheduler().cancelTask(ioStackTaskId);
127            }

129            @Override
130            public void setHome(final String home, final Location loc) {
131                    final ConfigurationSection homeToSet = homes.createSection(home);
132                    homeToSet.set("world", loc.getWorld().getName());
133                    homeToSet.set("x", loc.getX());
134                    homeToSet.set("y", loc.getY());
135                    homeToSet.set("z", loc.getZ());
136                    homeToSet.set("yaw", loc.getYaw());
137                    homeToSet.set("pitch", loc.getPitch());
138                    writeFile();
139            }

155             * @see be.Balor.Player.Data.PlayerDataManager#getHome(java.lang.String)
156             */
157            @Override
158            public Location getHome(final String home) {
159                    final ConfigurationSection homeSection = homes.getConfigurationSection(home);
160                    if (homeSection == null) {
161                            final String found = Str.matchString(homes.getKeys(false), home);
162                            if (found == null)
163                                    return null;
164                            return getLocation(homes.getConfigurationSection(found));
165                    } else
166                            return getLocation(homeSection);
167            }

241            private Location getLocation(final ConfigurationSection node) throws WorldNotLoaded {
242                    final World w = ACPluginManager.getServer().getWorld(node.getString("world"));
243                    if (w != null)
244                            return new Location(w, node.getDouble("x", 0), node.getDouble("y", 0), node.getDouble(
245                                            "z", 0), Float.parseFloat(node.getString("yaw")), Float.parseFloat(node
246                                            .getString("pitch")));
247                    else
248                            throw new WorldNotLoaded(node.getString("world"));
249            }

316            private void writeFile() {
317                    IOSAVET_TASK.addConfigurationToSave(datas);
318            }

320            /*
321             * (non-Javadoc)
322             * 
323             * @see be.Balor.Player.ACPlayer#forceSave()
324             */
325            @Override
326            void forceSave() {
327                    try {
328                            IOSAVET_TASK.removeConfiguration(datas);
329                            datas.save();
330                    } catch (final IOException e) {
331                            ACLogger.severe("Problem while saving Player file of " + getName(), e);
332                    }
333            }

338             * @see be.Balor.Player.ACPlayer#removeAllSuperPower()
339             */
340            @Override
341            public void removeAllSuperPower() {
342                    for (final String power : powers.getKeys(false)) {
343                            final Type matched = Type.matchType(power);
344                            if (matched != null && matched.getCategory().equals(Category.SUPER_POWER))
345                                    powers.set(power, null);
346                    }
347                    writeFile();
348    
349            }
