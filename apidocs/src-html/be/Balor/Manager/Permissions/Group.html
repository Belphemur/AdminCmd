017package be.Balor.Manager.Permissions;
018
019import org.bukkit.entity.Player;
020
021import be.Balor.Tools.Type;
022import be.Balor.bukkit.AdminCmd.ACHelper;
023
024/**
025 * @author Balor (aka Antoine Aflalo)
026 * 
027 */
028public class Group implements Comparable<Group> {
029        private final String name;
030        private final int rank;
031
032        /**
033         * @param name
034         * @param rank
035         */
036        public Group(final String name, final int rank) {
037                this.name = name;
038                this.rank = rank;
039        }
040
041        /**
042         * @param name
043         */
044        public Group(final String name, final Player p) {
045                this(name, ACHelper.getInstance().getLimit(p, Type.Limit.IMMUNITY,
046                                "defaultImmunityLvl"));
047        }
048
049        /**
050         * 
051         */
052        public Group() {
053                this("", 0);
054        }
055
056        /*
057         * (non-Javadoc)
058         * 
059         * @see java.lang.Comparable#compareTo(java.lang.Object)
060         */
061        @Override
062        public int compareTo(final Group o) {
063                if (rank == o.rank) {
064                        return name.compareTo(o.name);
065                }
066                return (rank - o.rank) * -1;
067        }
068
069        /**
070         * @return the name
071         */
072        public String getName() {
073                return name;
074        }
075
076        /**
077         * @return the rank
078         */
079        public int getRank() {
080                return rank;
081        }
082
083        /*
084         * (non-Javadoc)
085         * 
086         * @see java.lang.Object#hashCode()
087         */
088        @Override
089        public int hashCode() {
090                final int prime = 31;
091                int result = 1;
092                result = prime * result + ((name == null) ? 0 : name.hashCode());
093                result = prime * result + rank;
094                return result;
095        }
096
097        /*
098         * (non-Javadoc)
099         * 
100         * @see java.lang.Object#equals(java.lang.Object)
101         */
102        @Override
103        public boolean equals(final Object obj) {
104                if (this == obj) {
105                        return true;
106                }
107                if (obj == null) {
108                        return false;
109                }
110                if (!(obj instanceof Group)) {
111                        return false;
112                }
113                final Group other = (Group) obj;
114                if (name == null) {
115                        if (other.name != null) {
116                                return false;
117                        }
118                } else if (!name.equals(other.name)) {
119                        return false;
120                }
121                if (rank != other.rank) {
122                        return false;
123                }
124                return true;
125        }
126
127        /*
128         * (non-Javadoc)
129         * 
130         * @see java.lang.Object#toString()
131         */
132        @Override
133        public String toString() {
134                return "Group [name=" + name + ", rank=" + rank + "]";
135        }
136
137}
