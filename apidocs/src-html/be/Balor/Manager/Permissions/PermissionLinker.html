034    public class PermissionLinker {
035            protected Map<String, PermParent> permissions = new HashMap<String, PermParent>();
036            protected Map<String, PermParent> childrenPermParents = new HashMap<String, PermParent>();
037            protected PermParent majorPerm;
038            protected String name;
039            private static int counter = 0;
040    
041            /**
042             * Add a perm on the fly
043             * 
044             * @param permNode
045             * @param parentNode
046             * @return
047             */
048            public static Permission addOnTheFly(final String permNode, final String parentNode) {
049                    Permission child;
050                    if (ACPluginManager.getServer() == null)
051                            return null;
052                    if ((child = ACPluginManager.getServer().getPluginManager().getPermission(permNode)) == null) {
053                            child = new Permission(permNode, PermissionDefault.OP);
054                            ACPluginManager.getServer().getPluginManager().addPermission(child);
055                            if (parentNode.isEmpty())
056                                    return child;
057                            Permission parent = ACPluginManager.getServer().getPluginManager()
058                                            .getPermission(parentNode);
059    
060                            if (parent == null) {
061                                    parent = new Permission(parentNode, PermissionDefault.OP);
062                                    ACPluginManager.getServer().getPluginManager().addPermission(parent);
063                            }
064                            child.addParent(parent, true);
065                    }
066                    return child;
067    
068            }
069    
070            /**
071             * Get the PermissionLinker Note: The PermissionManager may only retain a
072             * weak reference to the newly created PermissionLinker. It is important to
073             * understand that a previously created PermissionLinker with the given name
074             * may be garbage collected at any time if there is no strong reference to
075             * the PermissionLinker. In particular, this means that two back-to-back
076             * calls like
077             * {@code PermissionLinker("MyPermissionLinker").addPermParent(...)} may use
078             * different PermissionLinker objects named "MyPermissionLinker" if there is
079             * no strong reference to the PermissionLinker named "MyPermissionLinker"
080             * elsewhere in the program.
081             * 
082             * @param name
083             * @return
084             */
085            public static synchronized PermissionLinker getPermissionLinker(final String name) {
086                    return PermissionManager.getInstance().demandPermissionLinker(name);
087            }

089            private final int plId = counter++;

099             * Add permission child (like myplugin.item.add)
100             * 
101             * @param permNode
102             * @return
103             */
104            public Permission addPermChild(final String permNode) {
105                    return addPermChild(permNode, PermissionDefault.OP);
106            }
107    
108            /**
109             * Add permission child (like myplugin.item.add)
110             * 
111             * @param permNode
112             * @param bukkitDefault
113             * @return
114             */
115            public Permission addPermChild(final String permNode, final PermissionDefault bukkitDefault)
116                            throws NullPointerException {
117                    final PermParent parent = matchPermParent(permNode);
118                    final PermChild child = new PermChild(permNode, bukkitDefault);
119                    if (parent == null) {
120                            DebugLog.INSTANCE.info("No Permission Parent found for : " + permNode);
121                            return child.getBukkitPerm();
122                    }
123                    parent.addChild(child);
124                    DebugLog.INSTANCE.info(child + " registered as child of " + parent);
125                    return child.getBukkitPerm();
126            }

129             * Add some important node (like myplygin.item)
130             * 
131             * @param toAdd
132             */
133            public void addPermParent(final PermParent toAdd) {
134                    permissions.put(toAdd.getPermName(), toAdd);
135            }
136    
137            public void addPermParent(final String toAdd) {
138                    final PermParent pp = new PermParent(toAdd);
139                    permissions.put(pp.getPermName(), pp);
140            }
141    
142            /**
143             * Add a PermParent as the child of another PermParent.
144             * 
145             * @param toAdd
146             *            permParent to add.
147             * @param parent
148             *            PermParent that will be the father of the item to add.
149             * @return the PermParent added.
150             */
151            public PermParent addChildPermParent(final PermParent toAdd, final PermParent parent) {
152                    parent.addChild(toAdd);
153                    childrenPermParents.put(toAdd.getPermName(), toAdd);
154                    return toAdd;
155            }

157            /*
158             * (non-Javadoc)
159             * 
160             * @see java.lang.Object#equals(java.lang.Object)
161             */
162            @Override
163            public boolean equals(final Object obj) {
164                    if (this == obj)
165                            return true;
166                    if (obj == null)
167                            return false;
168                    if (!(obj instanceof PermissionLinker))
169                            return false;
170                    final PermissionLinker other = (PermissionLinker) obj;
171                    if (majorPerm == null) {
172                            if (other.majorPerm != null)
173                                    return false;
174                    } else if (!majorPerm.equals(other.majorPerm))
175                            return false;
176                    if (name == null) {
177                            if (other.name != null)
178                                    return false;
179                    } else if (!name.equals(other.name))
180                            return false;
181                    if (plId != other.plId)
182                            return false;
183                    return true;
184            }

189            public String getName() {

191            }

193            /**
194             * Getting the PermParent of the given node.
195             * 
196             * @param permNode
197             * @return
198             */
199            public PermParent getPermParent(final String permNode) {
200                    PermParent result = permissions.get(permNode);
201                    if (result == null)
202                            result = childrenPermParents.get(permNode);
203                    return result;
204            }
205    
206            /*
207             * (non-Javadoc)
208             * 
209             * @see java.lang.Object#hashCode()
210             */
211            @Override
212            public int hashCode() {
213                    final int prime = 31;
214                    int result = 1;
215                    result = prime * result + ((majorPerm == null) ? 0 : majorPerm.hashCode());
216                    result = prime * result + ((name == null) ? 0 : name.hashCode());
217                    result = prime * result + plId;
218                    return result;
219            }
220    
221            private PermParent matchPermParent(final String search) {
222                    PermParent found = null;
223                    final String lowerSearch = search.toLowerCase();
224                    int delta = Integer.MAX_VALUE;
225                    final Set<PermParent> values = new HashSet<PermParent>();
226                    values.addAll(permissions.values());
227                    values.addAll(childrenPermParents.values());
228                    for (final PermParent perm : values) {
229                            final String str = perm.getCompareName();
230                            if (lowerSearch.toLowerCase().startsWith(str)) {
231                                    final int curDelta = lowerSearch.length() - str.length();
232                                    if (curDelta < delta) {
233                                            found = perm;
234                                            delta = curDelta;
235                                    }
236                                    if (curDelta == 0)
237                                            break;
238                            }
239                    }
240                    return found;
241    
242            }
