041        /**
042         * Add a perm on the fly
043         * 
044         * @param permNode
045         * @param parentNode
046         * @return
047         */
048        public static Permission addOnTheFly(final String permNode,
049                        final String parentNode) {
050                Permission child;
051                if (ACPluginManager.getServer() == null) {
052                        return null;
053                }
054                if ((child = ACPluginManager.getServer().getPluginManager()
055                                .getPermission(permNode)) == null) {
056                        child = new Permission(permNode, PermissionDefault.OP);
057                        ACPluginManager.getServer().getPluginManager().addPermission(child);
058                        if (parentNode.isEmpty()) {
059                                return child;
060                        }
061                        Permission parent = ACPluginManager.getServer().getPluginManager()
062                                        .getPermission(parentNode);
063
064                        if (parent == null) {
065                                parent = new Permission(parentNode, PermissionDefault.OP);
066                                ACPluginManager.getServer().getPluginManager()
067                                                .addPermission(parent);
068                        }
069                        child.addParent(parent, true);
070                }
071                return child;
072
073        }
074
075        /**
076         * Get the PermissionLinker Note: The PermissionManager may only retain a
077         * weak reference to the newly created PermissionLinker. It is important to
078         * understand that a previously created PermissionLinker with the given name
079         * may be garbage collected at any time if there is no strong reference to
080         * the PermissionLinker. In particular, this means that two back-to-back
081         * calls like
082         * {@code PermissionLinker("MyPermissionLinker").addPermParent(...)} may use
083         * different PermissionLinker objects named "MyPermissionLinker" if there is
084         * no strong reference to the PermissionLinker named "MyPermissionLinker"
085         * elsewhere in the program.
086         * 
087         * @param name
088         * @return
089         */
090        public static synchronized PermissionLinker getPermissionLinker(
091                        final String name) {
092                return PermissionManager.getInstance().demandPermissionLinker(name);
093        }

115        /**
116         * Add permission child (like myplugin.item.add)
117         * 
118         * @param permNode
119         * @param bukkitDefault
120         * @return
121         */
122        public PermChild addPermChild(final String permNode,
123                        final PermissionDefault bukkitDefault) throws NullPointerException {
124                final PermParent parent = matchPermParent(permNode);
125                final PermChild child = new PermChild(permNode, bukkitDefault);
126                if (parent == null) {
127                        DebugLog.INSTANCE.info("No Permission Parent found for : "
128                                        + permNode);
129                        return child;
130                }
131                parent.addChild(child);
132                DebugLog.INSTANCE.info(child + " registered as child of " + parent);
133                return child;
134        }

136        /**
137         * Add some important node (like myplygin.item)
138         * 
139         * @param toAdd
140         */
141        public void addPermParent(final PermParent toAdd) {
142                addPermParent0(toAdd);
143        }
144
145        private PermParent addPermParent0(final PermParent toAdd) {
146                final PermParent registeredPermParent = permissions.get(toAdd
147                                .getPermName());
148                if (registeredPermParent == null) {
149                        permissions.put(toAdd.getPermName(), toAdd);
150                        return toAdd;
151                }
152                return registeredPermParent;
153        }
154
155        public void addPermParent(final String toAdd) {
156                addPermParent(new PermParent(toAdd));
157        }

159        /**
160         * Add a PermParent as the child of another PermParent.
161         * 
162         * @param toAdd
163         *            permParent to add.
164         * @param parent
165         *            PermParent that will be the father of the item to add.
166         * @return the PermParent added.
167         */
168        public PermParent addChildPermParent(final PermParent toAdd,
169                        final PermParent parent) {
170                parent.addChild(toAdd);
171                childrenPermParents.put(toAdd.getPermName(), toAdd);
172                return toAdd;
173        }

175        /*
176         * (non-Javadoc)
177         * 
178         * @see java.lang.Object#equals(java.lang.Object)
179         */
180        @Override
181        public boolean equals(final Object obj) {
182                if (this == obj) {
183                        return true;
184                }
185                if (obj == null) {
186                        return false;
187                }
188                if (!(obj instanceof PermissionLinker)) {
189                        return false;
190                }
191                final PermissionLinker other = (PermissionLinker) obj;
192                if (majorPerm == null) {
193                        if (other.majorPerm != null) {
194                                return false;
195                        }
196                } else if (!majorPerm.equals(other.majorPerm)) {
197                        return false;
198                }
199                if (name == null) {
200                        if (other.name != null) {
201                                return false;
202                        }
203                } else if (!name.equals(other.name)) {
204                        return false;
205                }
206                if (plId != other.plId) {
207                        return false;
208                }
209                return true;
210        }

219        /**
220         * Getting the PermParent of the given node.
221         * 
222         * @param permNode
223         * @return
224         */
225        public PermParent getPermParent(final String permNode) {
226                PermParent result = permissions.get(permNode);
227                if (result == null) {
228                        result = childrenPermParents.get(permNode);
229                }
230                return result;
231        }

233        /*
234         * (non-Javadoc)
235         * 
236         * @see java.lang.Object#hashCode()
237         */
238        @Override
239        public int hashCode() {
240                final int prime = 31;
241                int result = 1;
242                result = prime * result
243                                + ((majorPerm == null) ? 0 : majorPerm.hashCode());
244                result = prime * result + ((name == null) ? 0 : name.hashCode());
245                result = prime * result + plId;
246                return result;
247        }
248
249        private PermParent matchPermParent(final String search) {
250                PermParent found = null;
251                final String lowerSearch = search.toLowerCase();
252                int delta = Integer.MAX_VALUE;
253                final Set<PermParent> values = new HashSet<PermParent>();
254                values.addAll(permissions.values());
255                values.addAll(childrenPermParents.values());
256                for (final PermParent perm : values) {
257                        final String str = perm.getCompareName();
258                        if (lowerSearch.toLowerCase().startsWith(str)) {
259                                final int curDelta = lowerSearch.length() - str.length();
260                                if (curDelta < delta) {
261                                        found = perm;
262                                        delta = curDelta;
263                                }
264                                if (curDelta == 0) {
265                                        break;
266                                }
267                        }
268                }
269                return found;
270
271        }

278                try {

280                } finally {
281                        DebugLog.endInfo();
282                }

284        }

291        public void setMajorPerm(final PermParent major) {
292                majorPerm = major;
293                for (final PermParent pp : permissions.values()) {
294                        majorPerm.addChild(pp);
295                }
296        }
