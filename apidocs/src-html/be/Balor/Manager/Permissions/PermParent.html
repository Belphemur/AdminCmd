017    package be.Balor.Manager.Permissions;
018    
019    import org.bukkit.permissions.PermissionDefault;
020    
021    /**
022     * @author Balor (aka Antoine Aflalo)
023     * 
024     */
025    public class PermParent extends PermChild {
026            protected final String compareName;
027    
028            public PermParent(final String perm) {
029                    this(perm, perm == null ? null : perm.substring(0, perm.length() - 1), PermissionDefault.OP);
030            }
031    
032            public PermParent(final String perm, final String compare, final PermissionDefault def) {
033                    super(perm, def);
034                    this.compareName = compare;
035            }
036    
037            /**
038             * @return the compareName
039             */
040            public String getCompareName() {
041                    return compareName;
042            }
043    
044            /**
045             * Add a permission Child to the Permission Parent
046             * 
047             * @param perm
048             * @return the PermParent (this)
049             */
050            public PermParent addChild(final PermChild perm) throws IllegalArgumentException {
051                    if (perm.equals(this))
052                            throw new IllegalArgumentException("The Child can't be the parent.");
053                    perm.bukkitPerm.addParent(bukkitPerm, true);
054                    return this;
055            }
056    
057            /**
058             * Add a permission Child to the Permission Parent
059             * 
060             * @param perm
061             * @return the PermParent (this)
062             */
063            public PermParent addChild(final String perm) {
064                    this.addChild(new PermChild(perm));
065                    return this;
066            }
067    
068            /*
069             * (non-Javadoc)
070             * 
071             * @see java.lang.Object#hashCode()
072             */
073            @Override
074            public int hashCode() {
075                    final int prime = 31;
076                    int result = super.hashCode();
077                    result = prime * result + ((compareName == null) ? 0 : compareName.hashCode());
078                    return result;
079            }
080    
081            /*
082             * (non-Javadoc)
083             * 
084             * @see java.lang.Object#equals(java.lang.Object)
085             */
086            @Override
087            public boolean equals(final Object obj) {
088                    if (this == obj)
089                            return true;
090                    if (!super.equals(obj))
091                            return false;
092                    if (!(obj instanceof PermParent))
093                            return false;
094                    final PermParent other = (PermParent) obj;
095                    if (compareName == null) {
096                            if (other.compareName != null)
097                                    return false;
098                    } else if (!compareName.equals(other.compareName))
099                            return false;
100                    return true;
101            }
102    
103    }
