017package be.Balor.Manager.Permissions;
018
019import org.bukkit.command.CommandSender;
020import org.bukkit.permissions.Permission;
021import org.bukkit.permissions.PermissionDefault;
022
023import be.Balor.Tools.Debug.DebugLog;
024import be.Balor.bukkit.AdminCmd.ACPluginManager;
025
026/**
027 * @author Balor (aka Antoine Aflalo)
028 * 
029 */
030public class PermChild {
031        protected final String permName;
032        protected final PermissionDefault permissionDefault;
033        protected PermParent parent = null;
034
035        public PermChild(final String permName) {
036                this(permName, PermissionDefault.OP);
037        }
038
039        /**
040         * @param permName
041         * @param parent
042         * @param value
043         * @param permDefault
044         */
045        public PermChild(final String permName, final PermissionDefault permDefault) {
046                DebugLog.beginInfo("Creation of a PermChild for : " + permName);
047                try {
048                        this.permName = permName;
049                        this.permissionDefault = permDefault;
050                } finally {
051                        DebugLog.endInfo();
052                }
053        }
054
055        public boolean hasPermission(final CommandSender player) {
056                return PermissionManager.hasPerm(player, this);
057        }
058
059        /*
060         * (non-Javadoc)
061         * 
062         * @see java.lang.Object#toString()
063         */
064        @Override
065        public String toString() {
066                return "PermChild [getPermName()=" + getPermName()
067                                + ", getPermDefault()=" + getPermDefault() + "]";
068        }
069
070        /**
071         * @return the parent
072         */
073        public PermParent getParent() {
074                return parent;
075        }
076
077        /**
078         * @return the permName
079         */
080        public String getPermName() {
081                return permName;
082        }
083
084        /**
085         * @return the permissionDefault
086         */
087        public PermissionDefault getPermDefault() {
088                return permissionDefault;
089        }
090
091        /**
092         * @return the bukkitPerm
093         */
094        public Permission getBukkitPerm() {
095                DebugLog.beginInfo("Creation of a BukkitPerm for : " + permName);
096                try {
097                        if (permName == null) {
098                                return null;
099                        }
100                        Permission bukkitPerm;
101                        if ((bukkitPerm = ACPluginManager.getServer().getPluginManager()
102                                        .getPermission(permName)) != null) {
103                                bukkitPerm.setDefault(permissionDefault);
104                                return bukkitPerm;
105                        }
106                        bukkitPerm = new Permission(permName, permissionDefault);
107                        ACPluginManager.getServer().getPluginManager()
108                                        .addPermission(bukkitPerm);
109                        return bukkitPerm;
110                } finally {
111                        DebugLog.endInfo();
112                }
113        }
114
115        /*
116         * (non-Javadoc)
117         * 
118         * @see java.lang.Object#hashCode()
119         */
120        @Override
121        public int hashCode() {
122                final int prime = 31;
123                int result = 1;
124                result = prime * result
125                                + ((permName == null) ? 0 : permName.hashCode());
126                result = prime
127                                * result
128                                + ((permissionDefault == null) ? 0 : permissionDefault
129                                                .hashCode());
130                return result;
131        }
132
133        /*
134         * (non-Javadoc)
135         * 
136         * @see java.lang.Object#equals(java.lang.Object)
137         */
138        @Override
139        public boolean equals(final Object obj) {
140                if (this == obj) {
141                        return true;
142                }
143                if (obj == null) {
144                        return false;
145                }
146                if (getClass() != obj.getClass()) {
147                        return false;
148                }
149                final PermChild other = (PermChild) obj;
150                if (permName == null) {
151                        if (other.permName != null) {
152                                return false;
153                        }
154                } else if (!permName.equals(other.permName)) {
155                        return false;
156                }
157                if (permissionDefault != other.permissionDefault) {
158                        return false;
159                }
160                return true;
161        }
162
163}
