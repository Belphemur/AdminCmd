021package be.Balor.Manager.Permissions.Plugins;
022
023import org.anjocaido.groupmanager.GroupManager;
024import org.anjocaido.groupmanager.permissions.AnjoPermissionsHandler;
025import org.bukkit.entity.Player;
026import org.bukkit.plugin.Plugin;
027
028import be.Balor.Manager.Permissions.Group;
029
030/**
031 * @author Lathanael (aka Philippe Leipold)
032 * 
033 */
034public class EssentialsGroupManager extends SuperPermissions {
035
036        private final GroupManager groupManager;
037
038        public EssentialsGroupManager(final Plugin manager) {
039                groupManager = (GroupManager) manager;
040        }
041
042        /*
043         * (non-Javadoc)
044         * 
045         * @see
046         * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
047         * .String, org.bukkit.entity.Player)
048         */
049        @Override
050        public boolean isInGroup(final String groupName, final Player player) {
051                final AnjoPermissionsHandler handler = groupManager.getWorldsHolder()
052                                .getWorldPermissions(player);
053                if (handler == null) {
054                        return false;
055                }
056                return handler.inGroup(player.getName(), groupName);
057        }
058
059        /*
060         * (non-Javadoc)
061         * 
062         * @see
063         * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(org.bukkit.
064         * entity.Player)
065         */
066        @Override
067        public String getPrefix(final Player base) {
068                final String prefix = super.getPrefix(base);
069                if (prefix == null || prefix.isEmpty()) {
070                        final AnjoPermissionsHandler handler = groupManager
071                                        .getWorldsHolder().getWorldPermissions(base);
072                        if (handler == null) {
073                                return "";
074                        }
075                        return handler.getUserPrefix(base.getName());
076                }
077                return prefix;
078        }
079
080        /*
081         * (non-Javadoc)
082         * 
083         * @see
084         * be.Balor.Manager.Permissions.IPermissionPlugin#getSuffix(org.bukkit.entity
085         * .Player)
086         */
087        @Override
088        public String getSuffix(final Player base) {
089                final String suffix = super.getSuffix(base);
090                if (suffix == null || suffix.isEmpty()) {
091                        final AnjoPermissionsHandler handler = groupManager
092                                        .getWorldsHolder().getWorldPermissions(base);
093                        if (handler == null) {
094                                return "";
095                        }
096                        return handler.getUserSuffix(base.getName());
097                }
098                return suffix;
099        }
100
101        /*
102         * (non-Javadoc)
103         * 
104         * @see
105         * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getGroup(org.bukkit
106         * .entity.Player)
107         */
108        @Override
109        public Group getGroup(final Player player) {
110                final AnjoPermissionsHandler handler = groupManager.getWorldsHolder()
111                                .getWorldPermissions(player);
112                final String group = handler.getGroup(player.getName());
113                if (group == null) {
114                        return new Group();
115                }
116                return new Group(group, player);
117        }
118}
