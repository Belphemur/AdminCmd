017package be.Balor.Manager.Permissions.Plugins;
018
019import java.util.ArrayList;
020import java.util.HashSet;
021import java.util.List;
022import java.util.Set;
023
024import org.bukkit.entity.Player;
025
026import be.Balor.Player.ACPlayer;
027
028import com.platymuus.bukkit.permissions.Group;
029import com.platymuus.bukkit.permissions.PermissionsPlugin;
030
031/**
032 * @author Balor (aka Antoine Aflalo)
033 * 
034 */
035public class BukkitPermissions extends SuperPermissions {
036        protected PermissionsPlugin permBukkit = null;
037
038        /**
039         *
040         */
041        public BukkitPermissions(final PermissionsPlugin plugin) {
042                permBukkit = plugin;
043        }
044
045        /*
046         * (non-Javadoc)
047         * 
048         * @see
049         * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
050         * .String, org.bukkit.entity.Player)
051         */
052        @Override
053        public boolean isInGroup(final String groupName, final Player player) {
054                List<Group> groups = new ArrayList<Group>();
055                try {
056                        groups = permBukkit.getGroups(player.getName());
057                } catch (final Exception e) {
058                        return false;
059                }
060                if (groups.isEmpty()) {
061                        return false;
062                }
063                for (final Group group : groups) {
064                        if (group.getName().equalsIgnoreCase(groupName)) {
065                                return true;
066                        }
067                }
068                return false;
069        }
070
071        /*
072         * (non-Javadoc)
073         * 
074         * @see
075         * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
076         * .String)
077         */
078        @Override
079        public Set<Player> getUsers(final String groupName) {
080                final Set<Player> players = new HashSet<Player>();
081                List<String> playersString;
082                try {
083                        playersString = permBukkit.getGroup(groupName).getPlayers();
084                } catch (final Exception e) {
085                        return players;
086                }
087                if (playersString != null) {
088                        for (final String player : playersString) {
089                                players.add(ACPlayer.getPlayer(player).getHandler());
090                        }
091                        return players;
092                }
093                return null;
094        }
095
096        /*
097         * (non-Javadoc)
098         * 
099         * @see
100         * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getGroup(org.bukkit
101         * .entity.Player)
102         */
103        @Override
104        public be.Balor.Manager.Permissions.Group getGroup(final Player player) {
105                final List<Group> groups = permBukkit.getGroups(player.getName());
106                if (groups.isEmpty()) {
107                        return new be.Balor.Manager.Permissions.Group();
108                }
109                final Group group = groups.get(groups.size() - 1);
110                return new be.Balor.Manager.Permissions.Group(group.getName(), player);
111        }
112}
