017    package be.Balor.Manager.Permissions.Plugins;
018    
019    import java.util.ArrayList;
020    import java.util.HashSet;
021    import java.util.List;
022    import java.util.Set;
023    
024    import org.bukkit.entity.Player;
025    
026    import be.Balor.Player.ACPlayer;
027    
028    import com.platymuus.bukkit.permissions.Group;
029    import com.platymuus.bukkit.permissions.PermissionsPlugin;
030    
031    /**
032     * @author Balor (aka Antoine Aflalo)
033     * 
034     */
035    public class BukkitPermissions extends SuperPermissions {
036            protected PermissionsPlugin permBukkit = null;
037    
038            /**
039             *
040             */
041            public BukkitPermissions(final PermissionsPlugin plugin) {
042                    permBukkit = plugin;
043            }
044    
045            /*
046             * (non-Javadoc)
047             * 
048             * @see
049             * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
050             * .String, org.bukkit.entity.Player)
051             */
052            @Override
053            public boolean isInGroup(final String groupName, final Player player) {
054                    List<Group> groups = new ArrayList<Group>();
055                    try {
056                            groups = permBukkit.getGroups(player.getName());
057                    } catch (final Exception e) {
058                            return false;
059                    }
060                    if (groups.isEmpty())
061                            return false;
062                    for (final Group group : groups)
063                            if (group.getName().equalsIgnoreCase(groupName))
064                                    return true;
065                    return false;
066            }
067    
068            /*
069             * (non-Javadoc)
070             * 
071             * @see
072             * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
073             * .String)
074             */
075            @Override
076            public Set<Player> getUsers(final String groupName) {
077                    final Set<Player> players = new HashSet<Player>();
078                    List<String> playersString;
079                    try {
080                            playersString = permBukkit.getGroup(groupName).getPlayers();
081                    } catch (final Exception e) {
082                            return players;
083                    }
084                    if (playersString != null) {
085                            for (final String player : playersString) {
086                                    players.add(ACPlayer.getPlayer(player).getHandler());
087                            }
088                            return players;
089                    }
090                    return null;
091            }
092    
093            /*
094             * (non-Javadoc)
095             * 
096             * @see
097             * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getGroup(org.bukkit
098             * .entity.Player)
099             */
100            @Override
101            public String getGroup(final Player player) {
102                    final List<Group> groups = permBukkit.getGroups(player.getName());
103                    return groups.get(groups.size() - 1).getName();
104            }
105    }
