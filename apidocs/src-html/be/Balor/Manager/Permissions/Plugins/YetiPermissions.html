017package be.Balor.Manager.Permissions.Plugins;
018
019import java.util.Set;
020
021import org.bukkit.command.CommandSender;
022import org.bukkit.entity.Player;
023import org.bukkit.permissions.Permission;
024
025import be.Balor.Manager.Exceptions.NoPermissionsPlugin;
026import be.Balor.Manager.Permissions.Group;
027import be.Balor.Tools.Utils;
028
029import com.nijiko.permissions.PermissionHandler;
030
031/**
032 * @author Balor (aka Antoine Aflalo)
033 * 
034 */
035public class YetiPermissions implements IPermissionPlugin {
036        protected PermissionHandler permission = null;
037
038        /*
039         * (non-Javadoc)
040         * 
041         * @see
042         * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
043         * .CommandSender, java.lang.String, boolean)
044         */
045        /**
046         *
047         */
048        public YetiPermissions(final PermissionHandler perm) {
049                this.permission = perm;
050        }
051
052        /**
053         * @return the permission
054         */
055        public PermissionHandler getPermission() {
056                return permission;
057        }
058
059        @Override
060        public boolean hasPerm(final CommandSender player, final String perm,
061                        final boolean errorMsg) {
062                if (!(player instanceof Player)) {
063                        return true;
064                }
065                if (permission.has((Player) player, perm)) {
066                        return true;
067                } else {
068                        if (errorMsg) {
069                                Utils.sI18n(player, "errorNotPerm", "p", perm);
070                        }
071                        return false;
072                }
073        }
074
075        /*
076         * (non-Javadoc)
077         * 
078         * @see
079         * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
080         * .CommandSender, org.bukkit.permissions.Permission, boolean)
081         */
082        @Override
083        public boolean hasPerm(final CommandSender player, final Permission perm,
084                        final boolean errorMsg) {
085                if (!(player instanceof Player)) {
086                        return true;
087                }
088                if (permission.has((Player) player, perm.getName())) {
089                        return true;
090                } else {
091                        if (errorMsg) {
092                                Utils.sI18n(player, "errorNotPerm", "p", perm.getName());
093                        }
094                        return false;
095                }
096        }
097
098        /*
099         * (non-Javadoc)
100         * 
101         * @see
102         * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
103         * .String, org.bukkit.entity.Player)
104         */
105        @Override
106        public boolean isInGroup(final String groupName, final Player player) {
107                return permission.inGroup(player.getWorld().getName(),
108                                player.getName(), groupName);
109        }
110
111        /*
112         * (non-Javadoc)
113         * 
114         * @see
115         * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
116         * .String)
117         */
118        @Override
119        public Set<Player> getUsers(final String groupName)
120                        throws NoPermissionsPlugin {
121                throw new NoPermissionsPlugin(
122                                "To use this functionality you need a newer Permissions plugin!");
123        }
124
125        /*
126         * (non-Javadoc)
127         * 
128         * @see
129         * be.Balor.Manager.Permissions.AbstractPermission#getPermissionLimit(org
130         * .bukkit.entity.Player, java.lang.String)
131         */
132        @SuppressWarnings("deprecation")
133        @Override
134        public String getPermissionLimit(final Player p, final String type) {
135                Integer limitInteger = null;
136                try {
137                        limitInteger = permission.getInfoInteger(p.getWorld().getName(),
138                                        p.getName(), "admincmd." + type, false);
139                } catch (final NoSuchMethodError e) {
140                        try {
141
142                                limitInteger = permission.getPermissionInteger(p.getWorld()
143                                                .getName(), p.getName(), "admincmd." + type);
144                        } catch (final Throwable e2) {
145                                limitInteger = null;
146                        }
147                }
148                if (limitInteger != null && limitInteger != -1) {
149                        return limitInteger.toString();
150                } else {
151                        return null;
152                }
153        }
154
155        /*
156         * (non-Javadoc)
157         * 
158         * @see
159         * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(java.lang.String
160         * , java.lang.String)
161         */
162        @SuppressWarnings("deprecation")
163        @Override
164        public String getPrefix(final Player player) {
165                final String world = player.getWorld().getName();
166                final String pName = player.getName();
167                String prefixstring = null;
168                try {
169                        prefixstring = permission.safeGetUser(world, pName).getPrefix();
170                } catch (final Exception e) {
171                        final String group = permission.getGroup(world, pName);
172                        prefixstring = permission.getGroupPrefix(world, group);
173                } catch (final NoSuchMethodError e) {
174                        final String group = permission.getGroup(world, pName);
175                        prefixstring = permission.getGroupPrefix(world, group);
176                }
177                return prefixstring;
178        }
179
180        /*
181         * (non-Javadoc)
182         * 
183         * @see
184         * be.Balor.Manager.Permissions.IPermissionPlugin#getSuffix(org.bukkit.entity
185         * .Player)
186         */
187        @Override
188        @SuppressWarnings("deprecation")
189        public String getSuffix(final Player player) {
190                final String world = player.getWorld().getName();
191                final String pName = player.getName();
192                String prefixstring = null;
193                try {
194                        prefixstring = permission.safeGetUser(world, pName).getSuffix();
195                } catch (final Exception e) {
196                        final String group = permission.getGroup(world, pName);
197                        prefixstring = permission.getGroupSuffix(world, group);
198                } catch (final NoSuchMethodError e) {
199                        final String group = permission.getGroup(world, pName);
200                        prefixstring = permission.getGroupSuffix(world, group);
201                }
202                return prefixstring;
203        }
204
205        /*
206         * (non-Javadoc)
207         * 
208         * @see
209         * be.Balor.Manager.Permissions.Plugins.IPermissionPlugin#getGroup(org.bukkit
210         * .entity.Player)
211         */
212        @Override
213        public Group getGroup(final Player player) {
214                return new Group();
215        }
216
217}
