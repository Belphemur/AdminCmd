057        /*
058         * (non-Javadoc)
059         * 
060         * @see
061         * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
062         * .CommandSender, java.lang.String, boolean)
063         */
064        @Override
065        public boolean hasPerm(final CommandSender player, final String perm,
066                        final boolean errorMsg) {
067                if (!(player instanceof Player)) {
068                        return true;
069                }
070                if (PEX.has((Player) player, perm)) {
071                        return true;
072                } else {
073                        if (errorMsg) {
074                                Utils.sI18n(player, "errorNotPerm", "p", perm);
075                        }
076                        return false;
077                }
078        }
079
080        @Override
081        public boolean hasPerm(final CommandSender player, final Permission perm,
082                        final boolean errorMsg) {
083                return this.hasPerm(player, perm.getName(), errorMsg);
084        }

086        /*
087         * (non-Javadoc)
088         * 
089         * @see
090         * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
091         * .String)
092         */
093        @Override
094        public Set<Player> getUsers(final String groupName)
095                        throws NoPermissionsPlugin {
096                PermissionUser[] users = null;
097                users = PEX.getUsers(groupName);
098                final Set<Player> players = new HashSet<Player>();
099                if (users != null) {
100                        Player player = null;
101                        for (final PermissionUser user : users) {
102                                player = ACPlayer.getPlayer(user.getName()).getHandler();
103                                if (player == null) {
104                                        continue;
105                                }
106                                players.add(player);
107                        }
108                        return players;
109                }
110                return null;
111        }

113        /*
114         * (non-Javadoc)
115         * 
116         * @see
117         * be.Balor.Manager.Permissions.AbstractPermission#getPermissionLimit(org
118         * .bukkit.entity.Player, java.lang.String)
119         */
120        @Override
121        public String getPermissionLimit(final Player p, final String limit) {
122                String permLimit = null;
123                final PermissionUser permUser = PEX.getUser(p);
124                try {
125                        permLimit = permUser.getOption("admincmd." + limit);
126                } catch (final ConcurrentModificationException e) {
127                        final Future<String> permTask = ACPluginManager.getScheduler()
128                                        .callSyncMethod(ACPluginManager.getCorePlugin(),
129                                                        new Callable<String>() {
130
131                                                                @Override
132                                                                public String call() throws Exception {
133                                                                        return permUser.getOption("admincmd."
134                                                                                        + limit);
135                                                                }
136                                                        });
137                        try {
138                                permLimit = permTask.get();
139                        } catch (final InterruptedException e1) {
140                                permLimit = null;
141                                DebugLog.INSTANCE.info("Problem while gettings ASYNC perm of "
142                                                + p.getName());
143                        } catch (final ExecutionException e1) {
144                                permLimit = null;
145                                DebugLog.INSTANCE.info("Problem while gettings ASYNC perm of "
146                                                + p.getName());
147                        }
148                }
149
150                if (permLimit == null || (permLimit != null && permLimit.isEmpty())) {
151                        Set<String> permissions = new HashSet<String>();
152                        final String worldName = p.getWorld().getName();
153                        try {
154                                for (final String str : permUser.getPermissions(worldName)) {
155                                        permissions.add(str);
156                                }
157                                for (final PermissionGroup group : permUser.getGroups()) {
158                                        for (final String str : group.getPermissions(worldName)) {
159                                                permissions.add(str);
160                                        }
161                                }
162                        } catch (final ConcurrentModificationException e) {
163                                final Future<Set<String>> permTask = ACPluginManager
164                                                .getScheduler().callSyncMethod(
165                                                                ACPluginManager.getCorePlugin(),
166                                                                new Callable<Set<String>>() {
167
168                                                                        @Override
169                                                                        public Set<String> call() throws Exception {
170                                                                                final Set<String> permissions = new HashSet<String>();
171                                                                                for (final String str : permUser
172                                                                                                .getPermissions(worldName)) {
173                                                                                        permissions.add(str);
174                                                                                }
175                                                                                for (final PermissionGroup group : permUser
176                                                                                                .getGroups()) {
177                                                                                        for (final String str : group
178                                                                                                        .getPermissions(worldName)) {
179                                                                                                permissions.add(str);
180                                                                                        }
181                                                                                }
182                                                                                return permissions;
183                                                                        }
184                                                                });
185                                try {
186                                        permissions = permTask.get();
187                                } catch (final InterruptedException e1) {
188                                        DebugLog.INSTANCE
189                                                        .info("Problem while gettings ASYNC perm (PEX) of "
190                                                                        + p.getName());
191                                        return null;
192                                } catch (final ExecutionException e1) {
193                                        DebugLog.INSTANCE
194                                                        .info("Problem while gettings ASYNC perm (PEX) of "
195                                                                        + p.getName());
196                                        return null;
197                                }
198                        }
199                        return limitSearcher(limit, permissions);
200                }
201                return permLimit;
202        }
203
204        private String limitSearcher(final String limit, final Set<String> perms) {
205                final Pattern regex = Pattern.compile("admincmd\\."
206                                + limit.toLowerCase() + "\\.[0-9]+");
207                int max = Integer.MIN_VALUE;
208                for (final String perm : perms) {
209                        final Matcher regexMatcher = regex.matcher(perm.toLowerCase());
210                        if (!regexMatcher.find()) {
211                                continue;
212                        }
213                        final int current = Integer.parseInt(perm.split("\\.")[2]);
214                        if (current < max) {
215                                continue;
216                        }
217                        max = current;
218                }
219
220                if (max != Integer.MIN_VALUE) {
221                        return String.valueOf(max);
222                }
223                return null;
224        }

226        /*
227         * (non-Javadoc)
228         * 
229         * @see
230         * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(org.bukkit.
231         * entity.Player)
232         */
233        @Override
234        public String getPrefix(final Player player) {
235                PermissionUser user = null;
236                try {
237                        user = PEX.getUser(player);
238                } catch (final Exception e) {
239                        DebugLog.INSTANCE.log(Level.SEVERE,
240                                        "Problem when trying to get the prefix of the user "
241                                                        + player.getName(), e);
242                        return "";
243                }
244
245                if (user != null) {
246                        return user.getPrefix() == null ? "" : user.getPrefix();
247                }
248
249                String prefix = "";
250                for (final PermissionGroup group : PEX.getUser(player).getGroups()) {
251                        if ((prefix = group.getPrefix()) != null && !prefix.isEmpty()) {
252                                break;
253                        }
254                }
255                return prefix;
256        }

258        /*
259         * (non-Javadoc)
260         * 
261         * @see
262         * be.Balor.Manager.Permissions.IPermissionPlugin#getSuffix(org.bukkit.entity
263         * .Player)
264         */
265        @Override
266        public String getSuffix(final Player player) {
267                final PermissionUser user = PEX.getUser(player);
268                if (user != null) {
269                        return user.getSuffix() == null ? "" : user.getSuffix();
270                }
271
272                String suffix = "";
273                for (final PermissionGroup group : PEX.getUser(player).getGroups()) {
274                        if ((suffix = group.getSuffix()) != null && !suffix.isEmpty()) {
275                                break;
276                        }
277                }
278                return suffix;
279        }

281        /*
282         * (non-Javadoc)
283         * 
284         * @see
285         * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getGroup(org.bukkit
286         * .entity.Player)
287         */
288        @Override
289        public Group getGroup(final Player player) {
290                int max = Integer.MIN_VALUE;
291                PermissionGroup cur = null;
292                for (final PermissionGroup group : PEX.getUser(player).getGroups()) {
293                        final int rank = group.getRank();
294                        if (rank > max) {
295                                max = rank;
296                                cur = group;
297                        }
298                }
299                if (cur == null) {
300                        return new Group();
301                }
302                return new Group(cur.getName(), cur.getRank());
303        }
