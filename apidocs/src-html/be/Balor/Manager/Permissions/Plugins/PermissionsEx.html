039    public class PermissionsEx extends SuperPermissions {
040            private final PermissionManager PEX;
041    
042            /**
043             *
044             */
045            public PermissionsEx(final PermissionManager PEX) {
046                    this.PEX = PEX;
047            }
048    
049            /*
050             * (non-Javadoc)
051             * 
052             * @see
053             * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
054             * .CommandSender, java.lang.String, boolean)
055             * 
056             * @Override public boolean hasPerm(CommandSender player, String perm,
057             * boolean errorMsg) { if (!(player instanceof Player)) return true; if
058             * (PEX.has((Player) player, perm)) return true; else { if (errorMsg)
059             * Utils.sI18n(player, "errorNotPerm", "p", perm); return false; } }
060             */
061    
062            /*
063             * (non-Javadoc)
064             * 
065             * @see
066             * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
067             * .CommandSender, org.bukkit.permissions.Permission, boolean)
068             * 
069             * @Override public boolean hasPerm(CommandSender player, Permission perm,
070             * boolean errorMsg) { if (!(player instanceof Player)) return true; if
071             * (PEX.has((Player) player, perm.getName())) return true; else { if
072             * (errorMsg) Utils.sI18n(player, "errorNotPerm", "p", perm.getName());
073             * return false; } }
074             */
075    
076            /*
077             * (non-Javadoc)
078             * 
079             * @see
080             * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
081             * .String, org.bukkit.entity.Player)
082             */
083            @Override
084            public boolean isInGroup(final String groupName, final Player player) {
085                    PermissionGroup[] groups;
086                    groups = PEX.getUser(player).getGroups(player.getWorld().getName());
087                    if (groups.length == 0)
088                            return false;
089                    for (final PermissionGroup group : groups)
090                            if (group.getName().equalsIgnoreCase(groupName))
091                                    return true;
092                    return false;
093            }
094    
095            /*
096             * (non-Javadoc)
097             * 
098             * @see
099             * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
100             * .String)
101             */
102            @Override
103            public Set<Player> getUsers(final String groupName) throws NoPermissionsPlugin {
104                    PermissionUser[] users = null;
105                    users = PEX.getUsers(groupName);
106                    final Set<Player> players = new HashSet<Player>();
107                    if (users != null) {
108                            Player player = null;
109                            for (final PermissionUser user : users) {
110                                    player = ACPlayer.getPlayer(user.getName()).getHandler();
111                                    if (player == null)
112                                            continue;
113                                    players.add(player);
114                            }
115                            return players;
116                    }
117                    return null;
118            }
119    
120            /*
121             * (non-Javadoc)
122             * 
123             * @see
124             * be.Balor.Manager.Permissions.AbstractPermission#getPermissionLimit(org
125             * .bukkit.entity.Player, java.lang.String)
126             */
127            @Override
128            public String getPermissionLimit(final Player p, final String limit) {
129                    String permLimit = null;
130                    try {
131                            permLimit = PEX.getUser(p).getOption("admincmd." + limit);
132                    } catch (final ConcurrentModificationException e) {
133                            final CountDownLatch countDown = new CountDownLatch(1);
134                            final String[] limitString = new String[1];
135                            ACPluginManager.scheduleSyncTask(new Runnable() {
136                                    @Override
137                                    public void run() {
138                                            try {
139                                                    limitString[0] = PEX.getUser(p).getOption("admincmd." + limit);
140                                            } catch (final Exception e2) {
141                                                    DebugLog.INSTANCE.log(Level.SEVERE, "Cant' get the limit " + limit
142                                                                    + " for the user " + p.getName(), e2);
143                                            }
144    
145                                            countDown.countDown();
146                                    }
147                            });
148                            try {
149                                    countDown.await();
150                            } catch (final InterruptedException e1) {
151                            }
152                            permLimit = limitString[0];
153                    }
154    
155                    if (permLimit == null || (permLimit != null && permLimit.isEmpty()))
156                            permLimit = super.getPermissionLimit(p, limit);
157                    return permLimit;
158            }
159    
160            /*
161             * (non-Javadoc)
162             * 
163             * @see
164             * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(org.bukkit.
165             * entity.Player)
166             */
167            @Override
168            public String getPrefix(final Player player) {
169                    PermissionUser user = null;
170                    try {
171                            user = PEX.getUser(player);
172                    } catch (final Exception e) {
173                            DebugLog.INSTANCE.log(Level.SEVERE,
174                                            "Problem when trying to get the prefix of the user " + player.getName(), e);
175                            return "";
176                    }
177    
178                    if (user != null)
179                            return user.getPrefix() == null ? "" : user.getPrefix();
180    
181                    String prefix = "";
182                    for (final PermissionGroup group : PEX.getUser(player).getGroups())
183                            if ((prefix = group.getPrefix()) != null && !prefix.isEmpty())
184                                    break;
185                    return prefix;
186            }
187    
188            /*
189             * (non-Javadoc)
190             * 
191             * @see
192             * be.Balor.Manager.Permissions.IPermissionPlugin#getSuffix(org.bukkit.entity
193             * .Player)
194             */
195            @Override
196            public String getSuffix(final Player player) {
197                    final PermissionUser user = PEX.getUser(player);
198                    if (user != null)
199                            return user.getSuffix() == null ? "" : user.getSuffix();
200    
201                    String suffix = "";
202                    for (final PermissionGroup group : PEX.getUser(player).getGroups())
203                            if ((suffix = group.getSuffix()) != null && !suffix.isEmpty())
204                                    break;
205                    return suffix;
206            }
207    
208            /*
209             * (non-Javadoc)
210             * 
211             * @see
212             * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getGroup(org.bukkit
213             * .entity.Player)
214             */
215            @Override
216            public String getGroup(final Player player) {
217                    final String[] groups = PEX.getUser(player).getGroupsNames();
218                    return groups[groups.length - 1];
219            }
