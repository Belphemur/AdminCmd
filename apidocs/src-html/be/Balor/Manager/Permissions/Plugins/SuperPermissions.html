040    /**
041     * @author Lathanael (aka Philippe Leipold)
042     * 
043     */
044    public abstract class SuperPermissions implements IPermissionPlugin {
045            protected static MInfoReader mChatInfo = null;
046    
047            /**
048             *
049             */
050            public SuperPermissions() {
051            }
052    
053            /**
054             * @param mChatSuite
055             *            the mChatAPI to set
056             */
057            public static void setmChatapi(final mChatSuite mChatSuite) {
058                    if (SuperPermissions.mChatInfo == null && mChatSuite != null)
059                            mChatInfo = mChatSuite.getInfoReader();
060            }
061    
062            /**
063             * @return the mChatAPI
064             */
065            public static boolean isApiSet() {
066                    return mChatInfo != null;
067            }
068    
069            /*
070             * (non-Javadoc)
071             * 
072             * @see
073             * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
074             * .CommandSender, java.lang.String, boolean)
075             */
076            @Override
077            public boolean hasPerm(final CommandSender player, final String perm, final boolean errorMsg) {
078                    if (!(player instanceof Player))
079                            return true;
080                    if (player.hasPermission(perm))
081                            return true;
082                    else {
083                            if (errorMsg)
084                                    Utils.sI18n(player, "errorNotPerm", "p", perm);
085    
086                            return false;
087                    }
088            }
089    
090            /*
091             * (non-Javadoc)
092             * 
093             * @see
094             * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
095             * .CommandSender, org.bukkit.permissions.Permission, boolean)
096             */
097            @Override
098            public boolean hasPerm(final CommandSender player, final Permission perm, final boolean errorMsg) {
099                    if (!(player instanceof Player))
100                            return true;
101                    if (player.hasPermission(perm))
102                            return true;
103                    else {
104                            if (errorMsg)
105                                    Utils.sI18n(player, "errorNotPerm", "p", perm.getName());
106                            return false;
107                    }
108            }
109    
110            /*
111             * (non-Javadoc)
112             * 
113             * @see
114             * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
115             * .String, org.bukkit.entity.Player)
116             */
117            @Override
118            public boolean isInGroup(final String group, final Player player) throws NoPermissionsPlugin {
119                    throw new NoPermissionsPlugin("To use this functionality you need a Permission Plugin");
120            }
121    
122            /*
123             * (non-Javadoc)
124             * 
125             * @see
126             * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
127             * .String)
128             */
129            @Override
130            public Set<Player> getUsers(final String groupName) throws NoPermissionsPlugin {
131                    throw new NoPermissionsPlugin("To use this functionality you need a Permission Plugin");
132            }
133    
134            /*
135             * (non-Javadoc)
136             * 
137             * @see
138             * be.Balor.Manager.Permissions.AbstractPermission#getPermissionLimit(org
139             * .bukkit.entity.Player, java.lang.String)
140             */
141            @Override
142            public String getPermissionLimit(final Player p, final String limit) {
143                    String result = null;
144                    if (mChatInfo != null)
145                            result = mChatInfo.getInfo(p.getName(), InfoType.USER, p.getWorld().getName(),
146                                            "admincmd." + limit);
147                    if (result == null || (result != null && result.isEmpty())) {
148                            final Pattern regex = Pattern
149                                            .compile("admincmd\\." + limit.toLowerCase() + "\\.[0-9]+");
150                            final Set<PermissionAttachmentInfo> perms = new LinkedHashSet<PermissionAttachmentInfo>();
151                            final CountDownLatch countDown = new CountDownLatch(1);
152                            ACPluginManager.scheduleSyncTask(new Runnable() {
153    
154                                    @Override
155                                    public void run() {
156                                            perms.addAll(p.getEffectivePermissions());
157                                            countDown.countDown();
158    
159                                    }
160                            });
161                            try {
162                                    countDown.await();
163                            } catch (final InterruptedException e) {
164                                    DebugLog.INSTANCE.log(Level.WARNING,
165                                                    "Problem to get Permission of the user " + p.getName(), e);
166                            }
167                            int max = Integer.MIN_VALUE;
168                            for (final PermissionAttachmentInfo info : perms) {
169                                    final Matcher regexMatcher = regex.matcher(info.getPermission().toLowerCase());
170                                    if (!regexMatcher.find())
171                                            continue;
172                                    final int current = Integer.parseInt(info.getPermission().split("\\.")[2]);
173                                    if (current < max)
174                                            continue;
175                                    max = current;
176    
177                            }
178                            if (max != Integer.MIN_VALUE)
179                                    return String.valueOf(max);
180                    } else
181                            return result;
182                    return null;
183            }
184    
185            /*
186             * (non-Javadoc)
187             * 
188             * @see
189             * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(java.lang.String
190             * , java.lang.String)
191             */
192            @Override
193            public String getPrefix(final Player player) {
194                    if (mChatInfo != null)
195                            return mChatInfo
196                                            .getPrefix(player.getName(), InfoType.USER, player.getWorld().getName());
197                    else
198                            return "";
199            }
200    
201            /*
202             * (non-Javadoc)
203             * 
204             * @see
205             * be.Balor.Manager.Permissions.IPermissionPlugin#getSuffix(org.bukkit.entity
206             * .Player)
207             */
208            @Override
209            public String getSuffix(final Player player) {
210                    if (mChatInfo != null)
211                            return mChatInfo
212                                            .getSuffix(player.getName(), InfoType.USER, player.getWorld().getName());
213                    else
214                            return "";
215            }
216
