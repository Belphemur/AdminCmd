043/**
044 * @author Lathanael (aka Philippe Leipold)
045 * 
046 */
047public abstract class SuperPermissions implements IPermissionPlugin {
048        private static boolean mChat = false;
049
050        /**
051         *
052         */
053        public SuperPermissions() {
054        }
055
056        /**
057         * @param mChatSuite
058         *            the mChatAPI to set
059         */
060        public static void setmChatapi(final mChatSuite mChatSuite) {
061                if (!SuperPermissions.mChat && mChatSuite != null) {
062                        SuperPermissions.mChat = true;
063                }
064        }
065
066        /**
067         * @return the mChatAPI
068         */
069        public static boolean isApiSet() {
070                return mChat;
071        }
072
073        /*
074         * (non-Javadoc)
075         * 
076         * @see
077         * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
078         * .CommandSender, java.lang.String, boolean)
079         */
080        @Override
081        public boolean hasPerm(final CommandSender player, final String perm,
082                        final boolean errorMsg) {
083                if (!(player instanceof Player)) {
084                        return true;
085                }
086                if (player.hasPermission(perm)) {
087                        return true;
088                } else {
089                        if (errorMsg) {
090                                Utils.sI18n(player, "errorNotPerm", "p", perm);
091                        }
092
093                        return false;
094                }
095        }
096
097        /*
098         * (non-Javadoc)
099         * 
100         * @see
101         * be.Balor.Manager.Permissions.AbstractPermission#hasPerm(org.bukkit.command
102         * .CommandSender, org.bukkit.permissions.Permission, boolean)
103         */
104        @Override
105        public boolean hasPerm(final CommandSender player, final Permission perm,
106                        final boolean errorMsg) {
107                if (!(player instanceof Player)) {
108                        return true;
109                }
110                if (player.hasPermission(perm)) {
111                        return true;
112                } else {
113                        if (errorMsg) {
114                                Utils.sI18n(player, "errorNotPerm", "p", perm.getName());
115                        }
116                        return false;
117                }
118        }
119
120        /*
121         * (non-Javadoc)
122         * 
123         * @see
124         * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
125         * .String, org.bukkit.entity.Player)
126         */
127        @Override
128        public boolean isInGroup(final String group, final Player player)
129                        throws NoPermissionsPlugin {
130                throw new NoPermissionsPlugin(
131                                "To use this functionality you need a Permission Plugin");
132        }
133
134        /*
135         * (non-Javadoc)
136         * 
137         * @see
138         * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
139         * .String)
140         */
141        @Override
142        public Set<Player> getUsers(final String groupName)
143                        throws NoPermissionsPlugin {
144                throw new NoPermissionsPlugin(
145                                "To use this functionality you need a Permission Plugin");
146        }
147
148        /*
149         * (non-Javadoc)
150         * 
151         * @see
152         * be.Balor.Manager.Permissions.AbstractPermission#getPermissionLimit(org
153         * .bukkit.entity.Player, java.lang.String)
154         */
155        @Override
156        public String getPermissionLimit(final Player p, final String limit) {
157                String result = null;
158                if (mChat) {
159                        result = Reader.getInfo(p.getName(), InfoType.USER, p.getWorld()
160                                        .getName(), "admincmd." + limit);
161                }
162                if (result == null || (result != null && result.isEmpty())) {
163                        final Pattern regex = Pattern.compile("admincmd\\."
164                                        + limit.toLowerCase() + "\\.[0-9]+");
165                        Set<PermissionAttachmentInfo> permissions = null;
166                        if (ACHelper.isMainThread()) {
167                                permissions = p.getEffectivePermissions();
168                        } else {
169                                final Callable<Set<PermissionAttachmentInfo>> perms = new Callable<Set<PermissionAttachmentInfo>>() {
170
171                                        @Override
172                                        public Set<PermissionAttachmentInfo> call()
173                                                        throws Exception {
174                                                return p.getEffectivePermissions();
175                                        }
176                                };
177                                final Future<Set<PermissionAttachmentInfo>> permTask = ACPluginManager
178                                                .getScheduler().callSyncMethod(
179                                                                ACPluginManager.getCorePlugin(), perms);
180                                try {
181                                        permissions = permTask.get();
182                                        DebugLog.INSTANCE.info("Perms got for " + p.getName());
183                                } catch (final InterruptedException e) {
184                                        DebugLog.INSTANCE
185                                                        .info("Problem while gettings ASYNC perm of "
186                                                                        + p.getName());
187                                } catch (final ExecutionException e) {
188                                        DebugLog.INSTANCE
189                                                        .info("Problem while gettings ASYNC perm of "
190                                                                        + p.getName());
191                                } catch (final CancellationException e) {
192                                        DebugLog.INSTANCE.info("Getting ASYNC Perm cancelled"
193                                                        + p.getName());
194                                }
195                        }
196                        return permissionCheck(permissions, regex);
197
198                } else {
199                        return result;
200                }
201        }
202
203        /*
204         * (non-Javadoc)
205         * 
206         * @see
207         * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(java.lang.String
208         * , java.lang.String)
209         */
210        @Override
211        public String getPrefix(final Player player) {
212                if (mChat) {
213                        return Reader.getPrefix(player.getName(), InfoType.USER, player
214                                        .getWorld().getName());
215                } else {
216                        return "";
217                }
218        }
219
220        /*
221         * (non-Javadoc)
222         * 
223         * @see
224         * be.Balor.Manager.Permissions.IPermissionPlugin#getSuffix(org.bukkit.entity
225         * .Player)
226         */
227        @Override
228        public String getSuffix(final Player player) {
229                if (mChat) {
230                        return Reader.getSuffix(player.getName(), InfoType.USER, player
231                                        .getWorld().getName());
232                } else {
233                        return "";
234                }
235        }
236
237        private String permissionCheck(final Set<PermissionAttachmentInfo> perms,
238                        final Pattern regex) {
239                if (perms == null) {
240                        return null;
241                }
242                int max = Integer.MIN_VALUE;
243                for (final PermissionAttachmentInfo info : perms) {
244                        final Matcher regexMatcher = regex.matcher(info.getPermission()
245                                        .toLowerCase());
246                        if (!regexMatcher.find()) {
247                                continue;
248                        }
249                        final int current = Integer.parseInt(info.getPermission().split(
250                                        "\\.")[2]);
251                        if (current < max) {
252                                continue;
253                        }
254                        max = current;
255                }
256
257                if (max != Integer.MIN_VALUE) {
258                        return String.valueOf(max);
259                }
260                return null;
261        }
