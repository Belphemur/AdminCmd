017package be.Balor.Manager.Permissions.Plugins;
018
019import java.util.HashSet;
020import java.util.Set;
021
022import org.bukkit.entity.Player;
023
024import be.Balor.Manager.Exceptions.NoPermissionsPlugin;
025import be.Balor.Manager.Permissions.Group;
026import be.Balor.Tools.Utils;
027import de.bananaco.bpermissions.api.ApiLayer;
028import de.bananaco.bpermissions.api.util.CalculableType;
029
030/**
031 * @author Lathanael (aka Philippe Leipold)
032 * 
033 */
034public class bPermissions extends SuperPermissions {
035
036        public bPermissions() {
037        }
038
039        /*
040         * (non-Javadoc)
041         * 
042         * @see
043         * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
044         * .String, org.bukkit.entity.Player)
045         */
046        @Override
047        public boolean isInGroup(final String groupName, final Player player) {
048                final String[] groups = ApiLayer.getGroups(player.getWorld().getName(),
049                                CalculableType.USER, player.getName());
050                if (groups == null) {
051                        return false;
052                }
053                if (groups.length == 0) {
054                        return false;
055                }
056                for (final String group : groups) {
057                        if (group.equalsIgnoreCase(groupName)) {
058                                return true;
059                        }
060                }
061                return false;
062        }
063
064        /*
065         * (non-Javadoc)
066         * 
067         * @see
068         * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
069         * .String)
070         */
071        @Override
072        public Set<Player> getUsers(final String groupName)
073                        throws NoPermissionsPlugin {
074                final Set<Player> players = new HashSet<Player>();
075                for (final Player player : Utils.getOnlinePlayers()) {
076                        if (ApiLayer.hasGroup(player.getWorld().getName(),
077                                        CalculableType.USER, player.getName(), groupName)) {
078                                players.add(player);
079                        }
080                }
081                return players;
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see
088         * be.Balor.Manager.Permissions.AbstractPermission#getPermissionLimit(org
089         * .bukkit.entity.Player, java.lang.String)
090         */
091        @Override
092        public String getPermissionLimit(final Player p, final String limit) {
093                String result = ApiLayer.getValue(p.getWorld().getName(),
094                                CalculableType.USER, p.getName(), limit);
095                if (result == null || (result != null && result.isEmpty())) {
096                        result = super.getPermissionLimit(p, limit);
097                }
098                return result;
099        }
100
101        /*
102         * (non-Javadoc)
103         * 
104         * @see
105         * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(java.lang.String
106         * , java.lang.String)
107         */
108        @Override
109        public String getPrefix(final Player player) {
110                String prefix = super.getPrefix(player);
111                if (prefix == null || prefix.isEmpty()) {
112                        prefix = ApiLayer.getValue(player.getWorld().getName(),
113                                        CalculableType.USER, player.getName(), "prefix");
114                }
115                return prefix;
116        }
117
118        /*
119         * (non-Javadoc)
120         * 
121         * @see
122         * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getSuffix(org.bukkit
123         * .entity.Player)
124         */
125        @Override
126        public String getSuffix(final Player player) {
127                String suffix = super.getSuffix(player);
128                if (suffix == null || suffix.isEmpty()) {
129                        suffix = ApiLayer.getValue(player.getWorld().getName(),
130                                        CalculableType.USER, player.getName(), "suffix");
131                }
132                return suffix;
133        }
134
135        /*
136         * (non-Javadoc)
137         * 
138         * @see
139         * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getGroup(org.bukkit
140         * .entity.Player)
141         */
142        @Override
143        public Group getGroup(final Player player) {
144                final String[] groups = ApiLayer.getGroups(player.getWorld().getName(),
145                                CalculableType.USER, player.getName());
146                if (groups.length == 0) {
147                        return new Group();
148                }
149                int maxPriority = Integer.MIN_VALUE;
150                String bestGroup = groups[0];
151                for (final String group : groups) {
152                        try {
153                                final int currentPriority = Integer.parseInt(ApiLayer.getValue(
154                                                player.getWorld().getName(), CalculableType.GROUP,
155                                                group, "priority"));
156                                if (currentPriority > maxPriority) {
157                                        maxPriority = currentPriority;
158                                        bestGroup = group;
159                                }
160                        } catch (final NumberFormatException e) {}
161                }
162                final String priority = ApiLayer.getValue(player.getWorld().getName(),
163                                CalculableType.USER, player.getName(), "priority");
164                return new Group(bestGroup, priority == null
165                                || (priority != null && priority.isEmpty())
166                                ? 0
167                                : Integer.parseInt(priority));
168        }
169
170}
