316        private final Hashtable<String, WeakReference<PermissionLinker>> permissionLinkers = new Hashtable<String, WeakReference<PermissionLinker>>();
317
318        /**
319         *
320         */
321        private PermissionManager() {
322                if (permissionHandler == null) {
323                        permissionHandler = new DinnerPermissions();
324                }
325        }
326
327        public synchronized boolean addPermissionLinker(final PermissionLinker perm) {
328                final String name = perm.getName();
329                if (name == null) {
330                        throw new NullPointerException();
331                }
332
333                final WeakReference<PermissionLinker> ref = permissionLinkers.get(name);
334                if (ref != null) {
335                        if (ref.get() == null) {
336                                // Hashtable holds stale weak reference
337                                // to a logger which has been GC-ed.
338                                // Allow to register new one.
339                                permissionLinkers.remove(name);
340                        } else {
341                                // We already have a registered logger with the given name.
342                                return false;
343                        }
344                }
345                // We're adding a new logger.
346                // Note that we are creating a weak reference here.
347                permissionLinkers.put(name, new WeakReference<PermissionLinker>(perm));
348                return true;
349        }
350
351        PermissionLinker demandPermissionLinker(final String name) {
352                PermissionLinker result = getPermissionLinker(name);
353                if (result == null) {
354                        result = new PermissionLinker(name);
355                        addPermissionLinker(result);
356                        result = getPermissionLinker(name);
357                }
358                return result;
359        }
360
361        public synchronized PermissionLinker getPermissionLinker(final String name) {
362                final WeakReference<PermissionLinker> ref = permissionLinkers.get(name);
363                if (ref == null) {
364                        return null;
365                }
366                final PermissionLinker perm = ref.get();
367                if (perm == null) {
368                        // Hashtable holds stale weak reference
369                        // to a logger which has been GC-ed.
370                        permissionLinkers.remove(name);
371                }
372                return perm;
373        }
