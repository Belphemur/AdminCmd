040    public class LocaleManager {
041            private static LocaleManager instance = new LocaleManager();
042            private final Map<String, ExtendedConfiguration> localesFiles = new HashMap<String, ExtendedConfiguration>();
043            public static final String PRIMARY_LOCALE = "primary";
044            private boolean noMsg = false;
045            private final Pattern recursiveLocale = Pattern.compile("#([\\w]+)#");
046            private final Pattern replaceLocale = Pattern.compile("%([\\w]+)");
047            private final Pattern buggedLocale = Pattern.compile("\\?(\\w)");

076             * Save all the change made to the locale.
077             */
078            public void save() {
079                    try {
080                            for (final ExtendedConfiguration conf : localesFiles.values())
081                                    conf.save();
082                    } catch (final Exception e) {
083                            ACLogger.severe("Problem while saving locale file", e);
084                    }

086            }

096            public void addLocale(final String key, final String value) {
097                    addLocale(key, value, false);
098            }

100            public void addLocale(final String key, final String value, final boolean override) {
101                    if (override)
102                            localesFiles.get(PRIMARY_LOCALE).set(key, value);
103                    else
104                            localesFiles.get(PRIMARY_LOCALE).add(key, value);
105            }

107            /**
108             * Add a locale to the wanted file.
109             * 
110             * @param file
111             *            name of the file.
112             * @param key
113             *            key of the locale
114             * @param value
115             *            text of the locale
116             * @param override
117             *            if true, override the existing locale
118             * @return true if everything went fine, else false.
119             */
120            public boolean addLocale(final String file, final String key, final String value,
121                            final boolean override) {
122                    final ExtendedConfiguration conf = localesFiles.get(file);
123                    if (conf == null)
124                            return false;
125                    if (override) {
126                            conf.set(key, value);
127                            return true;
128                    } else
129                            return conf.add(key, value);
130            }

133             * Add a locale to the wanted file.
134             * 
135             * @param file
136             *            name of the file.

139             * @param value
140             *            text of the locale
141             * @return true if everything went fine, else false.
142             */
143            public boolean addLocale(final String file, final String key, final String value) {
144                    return addLocale(file, key, value, false);
145            }

157            /**
158             * Get the locale, replace the keyword by the given strings
159             * 
160             * @param key
161             * @param values
162             * @return
163             */
164            public String get(final String key, final Map<String, String> values) {
165                    if (noMsg)
166                            return null;
167                    String locale = getLocale(key);
168                    if (locale != null && values != null)
169                            locale = recursiveReplaceLocale(locale, values);
170                    return locale;
171            }

181            public String get(final String file, final String key, final Map<String, String> values) {
182                    if (noMsg)
183                            return null;
184                    String locale = getLocale(file, key);
185                    if (locale != null && values != null)
186                            locale = recursiveReplaceLocale(locale, values);
187                    return locale;

190            /**
191             * Get the locale in the wanted file.
192             * 
193             * @param file
194             * @param key
195             * @return
196             */
197            private String getLocale(final String file, final String key) {
198                    final ExtendedConfiguration ex = localesFiles.get(file);
199                    if (ex == null)
200                            return null;
201                    return ex.getString(key);
202            }
203    
204            /**
205             * Looking in every locale file for the locale asked, begin by the
206             * principalLocale;
207             * 
208             * @param key
209             * @return
210             */
211            private String getLocale(final String key) {
212                    String result = localesFiles.get(PRIMARY_LOCALE).getString(key);
213                    if (result == null || (result != null && result.isEmpty()))
214                            for (final Entry<String, ExtendedConfiguration> e : localesFiles.entrySet()) {
215                                    if (e.getKey().equals(PRIMARY_LOCALE))
216                                            continue;
217                                    result = e.getValue().getString(key);
218                                    if (result != null && !result.isEmpty())
219                                            break;
220    
221                            }
222                    return result;
223    
224            }
225    
226            private String recursiveReplaceLocale(final String locale, final Map<String, String> values) {
227                    String ResultString = null;
228                    String result = locale;
229                    try {
230                            Matcher regexMatcher = recursiveLocale.matcher(locale);
231                            while (regexMatcher.find()) {
232                                    ResultString = regexMatcher.group(1);
233                                    final String recLocale = getLocale(ResultString);
234                                    if (recLocale != null)
235                                            result = regexMatcher.replaceFirst(recLocale);
236                                    else
237                                            result = regexMatcher.replaceFirst("");
238                                    regexMatcher = recursiveLocale.matcher(result);
239                            }
240                            regexMatcher = replaceLocale.matcher(result);
241                            while (regexMatcher.find()) {
242                                    ResultString = regexMatcher.group(1);
243                                    final String replaceValue = values.get(ResultString);
244                                    if (replaceValue != null) {
245                                            try {
246                                                    result = regexMatcher.replaceFirst(replaceValue);
247                                            } catch (final StringIndexOutOfBoundsException e) {
248                                                    result = regexMatcher.replaceFirst(replaceValue.replaceAll("\\W", ""));
249                                            }
250    
251                                    } else
252                                            result = regexMatcher.replaceFirst("");
253                                    regexMatcher = replaceLocale.matcher(result);
254                            }
255                    } catch (final PatternSyntaxException ex) {
256                            // Syntax error in the regular expression
257                    }
258                    // To correct interrogation point (?) problem in the locale file.
259                    final Matcher regexMatcher = buggedLocale.matcher(result);
260                    result = regexMatcher.replaceAll("§$1");
261                    return result;
262            }
263    
264            public String get(final String key, final String alias, final String replaceBy) {
265                    if (noMsg)
266                            return null;
267                    String locale = getLocale(key);
268                    if (locale != null && alias != null) {
269                            final HashMap<String, String> replace = new HashMap<String, String>();
270                            replace.put(alias, replaceBy);
271                            locale = recursiveReplaceLocale(locale, replace);
272                    }
273    
274                    return locale;
275            }
276    
277            public String get(final String file, final String key, final String alias,
278                            final String replaceBy) {
279                    if (noMsg)
280                            return null;
281                    String locale = getLocale(file, key);
282                    if (locale != null && alias != null) {
283                            final HashMap<String, String> replace = new HashMap<String, String>();
284                            replace.put(alias, replaceBy);
285                            locale = recursiveReplaceLocale(locale, replace);
286                    }
287    
288                    return locale;
289            }

291            /**
292             * Get all the possible locale key.
293             * 
294             * @return
295             */
296            public Set<String> getKeys() {
297                    final Set<String> locales = new HashSet<String>();
298                    for (final ExtendedConfiguration exConf : localesFiles.values())
299                            locales.addAll(exConf.getKeys(false));
300                    return locales;
301            }
302    
303            /**
304             * Reload the locale file
305             */
306            public void reload() {
307                    try {
308                            for (final ExtendedConfiguration exConf : localesFiles.values())
309                                    exConf.reload();
310                    } catch (final FileNotFoundException e) {
311                            ACLogger.severe("Locale Reload Problem :", e);
312                    } catch (final IOException e) {
313                            ACLogger.severe("Locale Reload Problem :", e);
314                    } catch (final InvalidConfigurationException e) {
315                            ACLogger.severe("Locale Reload Problem :", e);
316                    }
317            }
