017    package be.Balor.Manager.Terminal.Commands;
018    
019    import java.io.BufferedReader;
020    import java.io.File;
021    import java.io.InputStreamReader;
022    
023    import org.bukkit.command.CommandSender;
024    
025    import be.Balor.Manager.Terminal.TerminalCommand;
026    
027    /**
028     * @author Balor (aka Antoine Aflalo)
029     * 
030     */
031    public class UnixTerminalCommand extends TerminalCommand {
032    
033            /**
034             * @param commandName
035             * @param execution
036             * @param args
037             * @param workingDir
038             */
039            public UnixTerminalCommand(final String commandName, final String execution, final String args,
040                            final File workingDir) {
041                    super(commandName, execution, args, workingDir);
042            }
043    
044            /**
045             * @param commandName
046             * @param execution
047             * @param args
048             * @param workingDir
049             */
050            public UnixTerminalCommand(final String commandName, final String execution, final String args,
051                            final String workingDir) {
052                    super(commandName, execution, args, workingDir);
053            }
054    
055            /*
056             * (non-Javadoc)
057             * 
058             * @see
059             * be.Balor.Manager.Terminal.TerminalCommand#execute(org.bukkit.command.
060             * CommandSender)
061             */
062            @Override
063            public void execute(final CommandSender sender) {
064                    try {
065                            ProcessBuilder pb;
066                            if (args != null)
067                                    pb = new ProcessBuilder(execution, args);
068                            else
069                                    pb = new ProcessBuilder(execution);
070                            pb.redirectErrorStream(true);
071                            pb.directory(workingDir);
072                            final Process p = pb.start();
073                            final BufferedReader in = new BufferedReader(new InputStreamReader(p.getInputStream()));
074                            String line = null;
075                            while ((line = in.readLine()) != null) {
076                                    sender.sendMessage(line);
077                            }
078                    } catch (final Throwable e) {
079                            sender.sendMessage("CMD ERROR : " + e.getMessage());
080                            e.printStackTrace();
081                    }
082            }
083    
084    }
