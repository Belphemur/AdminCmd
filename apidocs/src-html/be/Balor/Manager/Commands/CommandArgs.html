017    package be.Balor.Manager.Commands;
018    
019    import java.util.ArrayList;
020    import java.util.Arrays;
021    import java.util.HashMap;
022    import java.util.HashSet;
023    import java.util.Iterator;
024    import java.util.List;
025    import java.util.Map;
026    import java.util.Map.Entry;
027    import java.util.Set;
028    
029    /**
030     * @author Balor (aka Antoine Aflalo)
031     * 
032     */
033    public class CommandArgs implements Iterable<String> {
034            protected final List<String> parsedArgs;
035            protected final Map<Character, String> valueFlags = new HashMap<Character, String>();
036            protected final Set<Character> booleanFlags = new HashSet<Character>();
037            public int length;
038    
039            public CommandArgs(final String args) {
040                    this(args.split(" "));
041            }
042    
043            /**
044                     *
045                     */
046            public CommandArgs(final String[] args) {
047                    final List<Integer> argIndexList = new ArrayList<Integer>(args.length);
048                    final List<String> argList = new ArrayList<String>(args.length);
049                    for (int i = 0; i < args.length; ++i) {
050                            String arg = args[i];
051                            if (arg.length() == 0) {
052                                    continue;
053                            }
054    
055                            argIndexList.add(i);
056    
057                            switch (arg.charAt(0)) {
058                            case '\'':
059                            case '"':
060                                    final StringBuilder build = new StringBuilder();
061                                    final char quotedChar = arg.charAt(0);
062    
063                                    int endIndex;
064                                    for (endIndex = i; endIndex < args.length; ++endIndex) {
065                                            final String arg2 = args[endIndex];
066                                            if (arg2.charAt(arg2.length() - 1) == quotedChar) {
067                                                    if (endIndex != i)
068                                                            build.append(' ');
069                                                    build.append(arg2.substring(endIndex == i ? 1 : 0, arg2.length() - 1));
070                                                    break;
071                                            } else if (endIndex == i) {
072                                                    build.append(arg2.substring(1));
073                                            } else {
074                                                    build.append(' ').append(arg2);
075                                            }
076                                    }
077    
078                                    if (endIndex < args.length) {
079                                            arg = build.toString();
080                                            i = endIndex;
081                                    }
082    
083                                    // In case there is an empty quoted string
084                                    if (arg.length() == 0) {
085                                            continue;
086                                    }
087                                    // else raise exception about hanging quotes?
088                            }
089                            argList.add(arg);
090                    }
091                    // Then flags
092    
093                    final List<Integer> originalArgIndices = new ArrayList<Integer>(argIndexList.size());
094                    this.parsedArgs = new ArrayList<String>(argList.size());
095    
096                    for (int nextArg = 0; nextArg < argList.size();) {
097                            // Fetch argument
098                            final String arg = argList.get(nextArg++);
099    
100                            if (arg.charAt(0) == '-' && arg.length() > 1 && arg.matches("^-[a-zA-Z]+$")) {
101                                    for (int i = 1; i < arg.length(); ++i) {
102                                            final char flagName = arg.charAt(i);
103                                            if (this.valueFlags.containsKey(flagName)) {
104                                                    continue;
105                                            }
106    
107                                            if (nextArg >= argList.size())
108                                                    this.booleanFlags.add(flagName);
109                                            else
110                                                    this.valueFlags.put(flagName, argList.get(nextArg));
111    
112                                    }
113                                    continue;
114                            }
115                            originalArgIndices.add(argIndexList.get(nextArg - 1));
116                            parsedArgs.add(arg);
117    
118                    }
119                    this.length = parsedArgs.size();
120            }
121    
122            public String getString(final int index) {
123                    try {
124                            return parsedArgs.get(index);
125                    } catch (final IndexOutOfBoundsException e) {
126                            return null;
127                    }
128    
129            }
130    
131            /**
132             * Try to parse the argument to an int
133             * 
134             * @param index
135             * @return
136             * @throws NumberFormatException
137             */
138            public int getInt(final int index) throws NumberFormatException {
139                    return Integer.parseInt(getString(index));
140            }
141    
142            /**
143             * Try to parse the argument to a float
144             * 
145             * @param index
146             * @return
147             * @throws NumberFormatException
148             */
149            public float getFloat(final int index) throws NumberFormatException {
150                    return Float.parseFloat(getString(index));
151            }
152    
153            /**
154             * Try to parse the argument to a double
155             * 
156             * @param index
157             * @return
158             * @throws NumberFormatException
159             */
160            public double getDouble(final int index) throws NumberFormatException {
161                    return Double.parseDouble(getString(index));
162            }
163    
164            /**
165             * Try to parse the argument to a long
166             * 
167             * @param index
168             * @return
169             * @throws NumberFormatException
170             */
171            public long getLong(final int index) throws NumberFormatException {
172                    return Long.parseLong(getString(index));
173            }
174    
175            /**
176             * Check if the arguments contain the wanted flagF
177             * 
178             * @param ch
179             *            flag to be searched
180             * @return true if found.
181             */
182            public boolean hasFlag(final char ch) {
183                    return booleanFlags.contains(ch) || valueFlags.containsKey(ch);
184            }
185    
186            /**
187             * Get the Value associated with the given flag and remove the flag from the
188             * normal arguments.
189             * 
190             * @param flag
191             *            flag to look for.
192             * @return null if not found else the value of the flag
193             */
194            public String getValueFlag(final char flag) {
195                    final String result = valueFlags.get(flag);
196                    if (result == null)
197                            return null;
198                    if (parsedArgs.remove(result))
199                            length--;
200                    return result;
201            }
202    
203            /*
204             * (non-Javadoc)
205             * 
206             * @see java.lang.Object#toString()
207             */
208            @Override
209            public String toString() {
210                    return Arrays.toString(parsedArgs.toArray(new String[] {})) + " BoolFlags : "
211                                    + Arrays.toString(booleanFlags.toArray(new Character[] {})) + " ValFlags : "
212                                    + (valueFlags.isEmpty() ? "[]" : "[" + valueFlagsToString() + "]");
213            }
214    
215            private String valueFlagsToString() {
216                    final StringBuffer buffer = new StringBuffer();
217                    for (final Entry<Character, String> entry : valueFlags.entrySet()) {
218                            buffer.append(entry.getKey()).append("->").append(entry.getValue()).append(", ");
219                    }
220                    buffer.deleteCharAt(buffer.length() - 1).deleteCharAt(buffer.length() - 1);
221                    return buffer.toString();
222            }
223    
224            /*
225             * (non-Javadoc)
226             * 
227             * @see java.lang.Iterable#iterator()
228             */
229            @Override
230            public Iterator<String> iterator() {
231                    return parsedArgs.iterator();
232            }
233    }
