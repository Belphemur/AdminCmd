019import java.util.HashMap;
020
021import org.bukkit.command.CommandSender;
022import org.bukkit.entity.Player;
023import org.bukkit.inventory.ItemStack;
024
025import be.Balor.Kit.ArmoredKitInstance;

028import be.Balor.Manager.Exceptions.PlayerNotFound;
029import be.Balor.Manager.Permissions.ActionNotPermitedException;
030import be.Balor.Manager.Permissions.PermChild;
031import be.Balor.Manager.Permissions.PermissionManager;
032import be.Balor.Player.ACPlayer;

037/**
038 * @author Balor (aka Antoine Aflalo)
039 * 
040 */
041public class Kit extends ItemCommand {
042
043        /**
044     *
045     */
046        public Kit() {
047                cmdName = "bal_kit";
048                other = true;
049        }
050
051        /*
052         * (non-Javadoc)
053         * 
054         * @see
055         * be.Balor.Manager.CoreCommand#permissionCheck(org.bukkit.command.CommandSender
056         * )
057         */
058        @Override
059        public boolean permissionCheck(final CommandSender sender) {
060                return true;
061        }
062
063        /*
064         * (non-Javadoc)
065         * 
066         * @see be.Balor.Manager.ACCommand#execute(org.bukkit.command.CommandSender,
067         * java.lang.String[])
068         */
069        @Override
070        public void execute(final CommandSender sender, final CommandArgs args)
071                        throws ActionNotPermitedException, PlayerNotFound {
072                Player target;
073                if (args.length == 0) {
074                        Utils.sI18n(sender, "kitList", "list", ACHelper.getInstance()
075                                        .getKitList(sender));
076                        return;
077                }
078                final KitInstance kit = ACHelper.getInstance()
079                                .getKit(args.getString(0));
080                if (kit == null) {
081                        Utils.sI18n(sender, "kitNotFound", "kit", args.getString(0));
082                        return;
083                }
084
085                target = Utils.getUser(sender, args, permNode, 1, true);
086                if (target == null) {
087                        return;
088                }
089                if (!kit.canUse(sender)) {
090                        return;
091                }
092                final ACPlayer actarget = ACPlayer.getPlayer(target);
093                if (!PermissionManager.hasPerm(sender, "admincmd.item.nodelay", false)) {
094                        final int kitDelay = kit.getDelay();
095                        final long kitLastUse = actarget.getLastKitUse(kit.getName());
096                        if (kitDelay == -1) {
097                                if (kitLastUse == 0) {
098                                        actarget.updateLastKitUse(kit.getName());
099                                } else {
100                                        Utils.sI18n(sender, "kitOnce", "kit", kit.getName());
101                                        return;
102                                }
103                        } else {
104                                final long nextuse = kitLastUse + kitDelay * 1000L;
105                                final long now = System.currentTimeMillis();
106                                if (now < nextuse) {
107                                        final long diff = nextuse - now;
108                                        final Long[] timeLeft = Utils.transformToElapsedTime(diff);
109                                        final HashMap<String, String> replace = new HashMap<String, String>();
110                                        replace.put("d", timeLeft[0].toString());
111                                        replace.put("h", timeLeft[1].toString());
112                                        replace.put("m", timeLeft[2].toString());
113                                        replace.put("s", timeLeft[3].toString());
114                                        final String timestamp = (timeLeft[0] > 0 ? (Utils.I18n(
115                                                        "days", "d", timeLeft[0].toString())) : "")
116                                                        + (timeLeft[1] > 0 ? (timeLeft[1] + "h ") : "")
117                                                        + (timeLeft[2] > 0 ? (timeLeft[2] + "m ") : "")
118                                                        + (timeLeft[3] > 0 ? (timeLeft[3] + "s") : "");
119                                        Utils.sI18n(sender, "kitDelayNotUp", "delay", timestamp);
120                                        return;
121                                }
122
123                                actarget.updateLastKitUse(kit.getName());
124                        }
125                }
126                final HashMap<String, String> replace = new HashMap<String, String>();
127                replace.put("kit", args.getString(0));
128                if (Utils.isPlayer(sender, false)) {
129                        if (!target.equals(sender)) {
130                                replace.put("sender", Utils.getPlayerName((Player) sender));
131                                Utils.sI18n(target, "kitOtherPlayer", replace);
132                                replace.remove("sender");
133                                replace.put("target", Utils.getPlayerName(target));
134                                Utils.sI18n(sender, "kitCommandSender", replace);
135                        } else {
136                                Utils.sI18n(sender, "kitYourself", replace);
137                        }
138                } else {
139                        replace.put("sender", "Server Admin");
140                        Utils.sI18n(target, "kitOtherPlayer", replace);
141                        replace.remove("sender");
142                        replace.put("target", Utils.getPlayerName(target));
143                        Utils.sI18n(sender, "kitCommandSender", replace);
144                }
145                final ItemStack[] items = kit.getItemStacks().toArray(
146                                new ItemStack[] {});
147                final Player taskTarget = target;
148                ACPluginManager.scheduleSyncTask(new Runnable() {
149                        @Override
150                        public void run() {
151                                taskTarget.getInventory().addItem(items);
152
153                        }
154                });
155                if (kit instanceof ArmoredKitInstance) {
156                        ((ArmoredKitInstance) kit).setPlayerArmorParts(target);
157                }
158
159        }
160
161        /*
162         * (non-Javadoc)
163         * 
164         * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
165         */
166        @Override
167        public void registerBukkitPerm() {
168                permParent
169                                .addChild(new PermChild("admincmd.item.nodelay", bukkitDefault))
170                                .addChild(new PermChild("admincmd.item.kithelp", bukkitDefault));
171                super.registerBukkitPerm();
172        }
173
174        /*
175         * (non-Javadoc)
176         * 
177         * @see be.Balor.Manager.ACCommand#argsCheck(java.lang.String[])
178         */
179        @Override
180        public boolean argsCheck(final String... args) {
181                return args != null;
182        }
183}
