035    /**
036     * @author Balor (aka Antoine Aflalo)
037     * 
038     */
039    public class Kit extends ItemCommand {
040    
041            /**
042         *
043         */
044            public Kit() {
045                    cmdName = "bal_kit";
046                    other = true;
047            }
048    
049            /*
050             * (non-Javadoc)
051             * 
052             * @see
053             * be.Balor.Manager.CoreCommand#permissionCheck(org.bukkit.command.CommandSender
054             * )
055             */
056            @Override
057            public boolean permissionCheck(final CommandSender sender) {
058                    return true;
059            }
060    
061            /*
062             * (non-Javadoc)
063             * 
064             * @see be.Balor.Manager.ACCommand#execute(org.bukkit.command.CommandSender,
065             * java.lang.String[])
066             */
067            @Override
068            public void execute(final CommandSender sender, final CommandArgs args) {
069                    // which material?
070                    Player target;
071                    if (args.length == 0) {
072                            Utils.sI18n(sender, "kitList", "list", ACHelper.getInstance().getKitList(sender));
073                            return;
074                    }
075                    final KitInstance kit = ACHelper.getInstance().getKit(args.getString(0));
076                    if (kit == null) {
077                            Utils.sI18n(sender, "kitNotFound", "kit", args.getString(0));
078                            return;
079                    }
080    
081                    target = Utils.getUser(sender, args, permNode, 1, true);
082                    if (target == null) {
083                            return;
084                    }
085                    if (!PermissionManager.hasPerm(sender, "admincmd.kit." + args.getString(0))) {
086                            return;
087                    }
088                    final ACPlayer actarget = ACPlayer.getPlayer(target);
089                    if (!PermissionManager.hasPerm(sender, "admincmd.item.nodelay", false)) {
090                            final int kitDelay = kit.getDelay();
091                            final long kitLastUse = actarget.getLastKitUse(kit.getName());
092                            if (kitDelay == -1) {
093                                    if (kitLastUse == 0)
094                                            actarget.updateLastKitUse(kit.getName());
095                                    else {
096                                            Utils.sI18n(sender, "kitOnce", "kit", kit.getName());
097                                            return;
098                                    }
099                            } else {
100                                    final long nextuse = kitLastUse + kitDelay * 1000L;
101                                    final long now = System.currentTimeMillis();
102                                    if (now < nextuse) {
103                                            final long diff = nextuse - now;
104                                            final Long[] timeLeft = Utils.transformToElapsedTime(diff);
105                                            final HashMap<String, String> replace = new HashMap<String, String>();
106                                            replace.put("d", timeLeft[0].toString());
107                                            replace.put("h", timeLeft[1].toString());
108                                            replace.put("m", timeLeft[2].toString());
109                                            replace.put("s", timeLeft[3].toString());
110                                            final String timestamp = (timeLeft[0] > 0 ? (Utils.I18n("days", "d",
111                                                            timeLeft[0].toString())) : "")
112                                                            + (timeLeft[1] > 0 ? (timeLeft[1] + "h ") : "")
113                                                            + (timeLeft[2] > 0 ? (timeLeft[2] + "m ") : "")
114                                                            + (timeLeft[3] > 0 ? (timeLeft[3] + "s") : "");
115                                            Utils.sI18n(sender, "kitDelayNotUp", "delay", timestamp);
116                                            return;
117                                    }
118    
119                                    actarget.updateLastKitUse(kit.getName());
120                            }
121                    }
122                    final HashMap<String, String> replace = new HashMap<String, String>();
123                    replace.put("kit", args.getString(0));
124                    if (Utils.isPlayer(sender, false)) {
125                            if (!target.equals(sender)) {
126                                    replace.put("sender", Utils.getPlayerName((Player) sender));
127                                    Utils.sI18n(target, "kitOtherPlayer", replace);
128                                    replace.remove("sender");
129                                    replace.put("target", Utils.getPlayerName(target));
130                                    Utils.sI18n(sender, "kitCommandSender", replace);
131                            } else
132                                    Utils.sI18n(sender, "kitYourself", replace);
133                    } else {
134                            replace.put("sender", "Server Admin");
135                            Utils.sI18n(target, "kitOtherPlayer", replace);
136                            replace.remove("sender");
137                            replace.put("target", Utils.getPlayerName(target));
138                            Utils.sI18n(sender, "kitCommandSender", replace);
139                    }
140                    final ItemStack[] items = kit.getItemStacks().toArray(new ItemStack[] {});
141                    final Player taskTarget = target;
142                    ACPluginManager.scheduleSyncTask(new Runnable() {
143                            @Override
144                            public void run() {
145                                    taskTarget.getInventory().addItem(items);
146    
147                            }
148                    });
149                    if (kit instanceof ArmoredKitInstance)
150                            ((ArmoredKitInstance) kit).setPlayerArmorParts(target);
151    
152            }
153    
154            /*
155             * (non-Javadoc)
156             * 
157             * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
158             */
159            @Override
160            public void registerBukkitPerm() {
161                    permParent.addChild(new PermChild("admincmd.item.nodelay", bukkitDefault)).addChild(
162                                    new PermChild("admincmd.item.kithelp", bukkitDefault));
163                    super.registerBukkitPerm();
164            }
165

167             * (non-Javadoc)
168             * 
169             * @see be.Balor.Manager.ACCommand#argsCheck(java.lang.String[])
170             */
171            @Override
172            public boolean argsCheck(final String... args) {
173                    return args != null;
174            }
175    }
