017    package be.Balor.Manager.Commands.Tp;
018    
019    import org.bukkit.Location;
020    import org.bukkit.command.CommandSender;
021    import org.bukkit.entity.Player;
022    
023    import be.Balor.Manager.Commands.CommandArgs;
024    import be.Balor.Manager.Exceptions.PlayerNotFound;
025    import be.Balor.Tools.Utils;
026    import be.Balor.Tools.Threads.TeleportTask;
027    import be.Balor.bukkit.AdminCmd.ACPluginManager;
028    
029    /**
030     * @author Balor (aka Antoine Aflalo)
031     * 
032     */
033    public class TpLoc extends TeleportCommand {
034    
035            /**
036             * 
037             */
038            public TpLoc() {
039                    permNode = "admincmd.tp.location";
040                    cmdName = "bal_tpthere";
041                    other = true;
042            }
043    
044            /*
045             * (non-Javadoc)
046             * 
047             * @see
048             * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
049             * java.lang.String[])
050             */
051            @Override
052            public void execute(final CommandSender sender, final CommandArgs args) throws PlayerNotFound {
053                    final Player target = Utils.getUserParam(sender, args, permNode);
054                    if (target == null)
055                            return;
056                    final double x;
057                    final double y;
058                    final double z;
059                    try {
060                            x = args.getDouble(0);
061                            y = args.getDouble(1);
062                            z = args.getDouble(2);
063                    } catch (final Exception e) {
064                            Utils.sI18n(sender, "errorLocation");
065                            return;
066                    }
067                    ACPluginManager.scheduleSyncTask(new TeleportTask(target, new Location(target.getWorld(),
068                                    x, y, z)));
069    
070            }
071    
072            /*
073             * (non-Javadoc)
074             * 
075             * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
076             */
077            @Override
078            public boolean argsCheck(final String... args) {
079                    return args != null && args.length >= 3;
080            }
081    
082    }
