040    /**
041     * @author Lathanael (aka Philippe Leipold)
042     * 
043     */
044    public class TpWorld extends TeleportCommand {
045    
046            private final PermChild list;
047    
048            public TpWorld() {
049                    permNode = "admincmd.tp.world";
050                    cmdName = "bal_tpworld";
051                    other = true;
052                    list = new PermChild(permNode + ".list");
053            }
054    
055            /*
056             * (non-Javadoc)
057             * 
058             * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
059             * CommandSender, be.Balor.Manager.Commands.CommandArgs)
060             */
061            @Override
062            public void execute(final CommandSender sender, final CommandArgs args) throws PlayerNotFound {
063                    final Player target = Utils.getUserParam(sender, args, permNode);
064                    assert (target != null);
065                    final Map<String, String> replace = new HashMap<String, String>();
066                    if (args.length < 1) {
067                            if (PermissionManager.hasPerm(target, permNode + ".list")) {
068                                    final List<World> worlds = ACPluginManager.getServer().getWorlds();
069                                    String worldList = "";
070                                    for (final World w : worlds) {
071                                            worldList += w.getName() + ", ";
072                                    }
073                                    worldList = worldList.substring(0, worldList.length() - 2);
074                                    replace.put("list", worldList);
075                                    LocaleHelper.TP_DIM_LIST.sendLocale(target, replace);
076                            }
077                            return;
078                    }
079                    if (args.hasFlag('l')) {
080                            // TODO: Location teleport with safety so one gets not teleported
081                            // into blocks
082                            return;
083                    }
084                    final String worldName = args.getString(0);
085                    if (worldName == null || worldName == "") {
086                            replace.put("arg", "world");
087                            replace.put("cmdName", "/tpd");
088                            LocaleHelper.MISSING_ARG.sendLocale(sender, replace);
089                            return;
090                    }
091                    final ACWorld world = ACWorld.getWorld(worldName);
092                    final Location loc = world.getSpawn();
093                    if (loc == null)
094                            return;
095                    ACPluginManager.scheduleSyncTask(new TeleportTask(target, loc));
096                    replace.put("world", worldName);
097                    LocaleHelper.TP_DIM.sendLocale(target, replace);
098            }
099    
100            /*
101             * (non-Javadoc)
102             * 
103             * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
104             */
105            @Override
106            public boolean argsCheck(final String... args) {
107                    return args != null;
108            }
109    
110            /*
111             * (non-Javadoc)
112             * 
113             * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
114             */
115            @Override
116            public void registerBukkitPerm() {
117                    final PermParent parent = new PermParent(permNode + ".*");
118                    plugin.getPermissionLinker().addChildPermParent(parent, permParent);
119                    final PermChild child = new PermChild(permNode, bukkitDefault);
120                    parent.addChild(child).addChild(list);
121                    bukkitPerm = child.getBukkitPerm();
122            }
123    }
