017package be.Balor.Manager.Commands.Tp;
018
019import org.bukkit.Location;
020import org.bukkit.command.CommandSender;
021import org.bukkit.entity.Player;
022
023import be.Balor.Manager.Commands.CommandArgs;
024import be.Balor.Manager.Exceptions.PlayerNotFound;
025import be.Balor.Manager.Permissions.ActionNotPermitedException;
026import be.Balor.Player.ACPlayer;
027import be.Balor.Tools.Utils;
028import be.Balor.Tools.Threads.TeleportTask;
029import be.Balor.bukkit.AdminCmd.ACPluginManager;
030
031/**
032 * @author Balor (aka Antoine Aflalo)
033 * 
034 */
035public class LastLocation extends TeleportCommand {
036
037        /**
038         *
039         */
040        public LastLocation() {
041                permNode = "admincmd.tp.back";
042                cmdName = "bal_back";
043        }
044
045        /*
046         * (non-Javadoc)
047         * 
048         * @see be.Balor.Manager.ACCommand#execute(org.bukkit.command.CommandSender,
049         * java.lang.String[])
050         */
051        @Override
052        public void execute(final CommandSender sender, final CommandArgs args)
053                        throws ActionNotPermitedException, PlayerNotFound {
054                final Player player = Utils.getUserParam(sender, args, permNode);
055                Location loc = null;
056                final ACPlayer target = ACPlayer.getPlayer(player);
057                loc = target.getLastLocation();
058                if (loc == null) {
059                        Utils.sI18n(sender, "noLastLocation");
060                        return;
061                }
062                ACPluginManager.scheduleSyncTask(new TeleportTask(target.getHandler(),
063                                loc));
064                Utils.sI18n(sender, "telportSuccess");
065                target.setLastLocation(null);
066
067        }
068
069        /*
070         * (non-Javadoc)
071         * 
072         * @see be.Balor.Manager.ACCommand#argsCheck(java.lang.String[])
073         */
074        @Override
075        public boolean argsCheck(final String... args) {
076                return true;
077        }
078
079}
