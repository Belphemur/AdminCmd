038    /**
039     * @authors Balor, Lathanael
040     * 
041     */
042    public class TpToWarp extends WarpCommand {
043            private PermChild tpAll;
044    
045            /**
046             *
047             */
048            public TpToWarp() {
049                    permNode = "admincmd.warp.tp";
050                    cmdName = "bal_tpwarp";
051                    other = true;
052    
053            }
054    
055            /*
056             * (non-Javadoc)
057             * 
058             * @see
059             * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
060             * java.lang.String[])
061             */
062            @Override
063            public void execute(final CommandSender sender, final CommandArgs args) {
064                    final Player target = Utils.getUser(sender, args, permNode, 1, true);
065                    if (Utils.isPlayer(sender)) {
066                            final Player p = (Player) sender;
067                            Location loc = null;
068                            if (target != null) {
069                                    final HashMap<String, String> replace = new HashMap<String, String>();
070    
071                                    if (args.getString(0).contains(":")) {
072                                            if (!PermissionManager.hasPerm(sender, tpAll.getBukkitPerm()))
073                                                    return;
074                                            final String[] split = args.getString(0).split(":");
075                                            final String world = split[0];
076                                            final String warp = split[1];
077                                            replace.put("name", world + ":" + warp);
078                                            try {
079                                                    final ACWorld acWorld = ACWorld.getWorld(world);
080                                                    final Warp warpPoint = acWorld.getWarp(warp);
081                                                    if (warpPoint == null) {
082                                                            replace.put("name", args.getString(0));
083                                                            Utils.sI18n(sender, "errorWarp", replace);
084                                                            return;
085                                                    }
086                                                    loc = warpPoint.loc;
087                                                    replace.put("name", acWorld.getName() + ":" + warpPoint.name);
088                                            } catch (final WorldNotLoaded e) {
089                                                    Utils.sI18n(sender, "worldNotFound", "world", world);
090                                                    return;
091                                            }
092                                    } else {
093                                            replace.put("name", args.getString(0));
094    
095                                            try {
096                                                    final Warp warpPoint = ACWorld.getWorld(p.getWorld().getName()).getWarp(
097                                                                    args.getString(0));
098                                                    if (warpPoint == null) {
099                                                            replace.put("name", args.getString(0));
100                                                            Utils.sI18n(sender, "errorWarp", replace);
101                                                            return;
102                                                    }
103                                                    loc = warpPoint.loc;
104                                                    replace.put("name", warpPoint.name);
105                                            } catch (final WorldNotLoaded e) {
106                                            }
107                                    }
108                                    if (loc == null) {
109                                            Utils.sI18n(sender, "errorWarp", replace);
110                                            return;
111                                    } else {
112                                            ACPluginManager.getScheduler()
113                                                            .scheduleSyncDelayedTask(
114                                                                            ACHelper.getInstance().getCoreInstance(),
115                                                                            new DelayedTeleport(target.getLocation(), loc, target, replace,
116                                                                                            sender), ConfigEnum.TP_DELAY.getLong());
117                                    }
118                            }
119                    }
120    
121            }
122    
123            /*
124             * (non-Javadoc)
125             * 
126             * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
127             */
128            @Override
129            public boolean argsCheck(final String... args) {
130                    return args != null && args.length >= 1;
131            }
132    
133            /*
134             * (non-Javadoc)
135             * 
136             * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
137             */
138            @Override
139            public void registerBukkitPerm() {
140                    super.registerBukkitPerm();
141                    tpAll = new PermChild("admincmd.warp.tp.all", bukkitDefault);
142                    permParent.addChild(tpAll);
143            }
144    
145            private class DelayedTeleport implements Runnable {
146    
147                    protected Location locBefore, teleportToLoc;
148                    protected Player target;
149                    protected HashMap<String, String> replace;
150                    protected CommandSender sender;
151    
152                    public DelayedTeleport(final Location locBefore, final Location teleportLoc,
153                                    final Player target, final HashMap<String, String> replace,
154                                    final CommandSender sender) {
155                            this.target = target;
156                            this.locBefore = locBefore;
157                            this.teleportToLoc = teleportLoc;
158                            this.replace = replace;
159                            this.sender = sender;
160                    }
161    
162                    @Override
163                    public void run() {
164                            if (locBefore.equals(target.getLocation()) && ConfigEnum.CHECKTP.getBoolean()) {
165                                    Utils.teleportWithChunkCheck(target, teleportToLoc);
166                                    sendMessage(sender, target, "tpWarp", replace);
167                            } else if (!ConfigEnum.CHECKTP.getBoolean()) {
168                                    Utils.teleportWithChunkCheck(target, teleportToLoc);
169                                    sendMessage(sender, target, "tpWarp", replace);
170                            } else {
171                                    replace.clear();
172                                    replace.put("cmdname", "Warp");
173                                    sendMessage(sender, target, "errorMoved", replace);
174                            }
175                    }
176            }
177    }
