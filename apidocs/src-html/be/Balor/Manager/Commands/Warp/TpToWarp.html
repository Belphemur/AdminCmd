045public class TpToWarp extends WarpCommand {
046        private PermChild tpAll;
047
048        /**
049         *
050         */
051        public TpToWarp() {
052                permNode = "admincmd.warp.tp";
053                cmdName = "bal_tpwarp";
054                other = true;
055
056        }
057
058        /*
059         * (non-Javadoc)
060         * 
061         * @see
062         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
063         * java.lang.String[])
064         */
065        @Override
066        public void execute(final CommandSender sender, final CommandArgs args)
067                        throws ActionNotPermitedException, PlayerNotFound {
068                if (args.length == 0) {
069                        Utils.sI18n(sender, "errorWarp", new HashMap<String, String>());
070                        return;
071                }
072                final Player target = Utils.getUser(sender, args, permNode, 1, true);
073                if (Utils.isPlayer(sender)) {
074                        final Player p = (Player) sender;
075                        Location loc = null;
076                        if (target != null) {
077                                final HashMap<String, String> replace = new HashMap<String, String>();
078
079                                if (args.getString(0).contains(":")) {
080                                        if (!PermissionManager.hasPerm(sender, tpAll)) {
081                                                return;
082                                        }
083                                        final String[] split = args.getString(0).split(":");
084                                        final String world = split[0];
085                                        final String warp = split[1];
086                                        replace.put("name", world + ":" + warp);
087                                        try {
088                                                final ACWorld acWorld = ACWorld.getWorld(world);
089                                                final Warp warpPoint = acWorld.getWarp(warp);
090                                                if (warpPoint == null) {
091                                                        replace.put("name", args.getString(0));
092                                                        Utils.sI18n(sender, "errorWarp", replace);
093                                                        return;
094                                                }
095                                                if (warpPoint.permission != null
096                                                                && !warpPoint.permission.isEmpty()
097                                                                && !warpPoint.permission.equalsIgnoreCase("")
098                                                                && !PermissionManager.hasPerm(sender, permNode
099                                                                                + "." + warpPoint.permission, false)) {
100                                                        replace.put("point", warp);
101                                                        LocaleHelper.WARP_NO_PERM.sendLocale(sender,
102                                                                        replace);
103                                                        return;
104                                                }
105                                                loc = warpPoint.loc;
106                                                replace.put("name", acWorld.getName() + ":"
107                                                                + warpPoint.name);
108                                        } catch (final WorldNotLoaded e) {
109                                                Utils.sI18n(sender, "worldNotFound", "world", world);
110                                                return;
111                                        }
112                                } else {
113                                        replace.put("name", args.getString(0));
114
115                                        try {
116                                                final Warp warpPoint = ACWorld.getWorld(
117                                                                p.getWorld().getName()).getWarp(
118                                                                args.getString(0));
119                                                if (warpPoint == null) {
120                                                        replace.put("name", args.getString(0));
121                                                        Utils.sI18n(sender, "errorWarp", replace);
122                                                        return;
123                                                }
124                                                if (warpPoint.permission != null
125                                                                && !warpPoint.permission.isEmpty()
126                                                                && !warpPoint.permission.equalsIgnoreCase("")
127                                                                && !PermissionManager.hasPerm(sender, permNode
128                                                                                + "." + warpPoint.permission, false)) {
129                                                        replace.put("point", args.getString(0));
130                                                        LocaleHelper.WARP_NO_PERM.sendLocale(sender,
131                                                                        replace);
132                                                        return;
133                                                }
134                                                loc = warpPoint.loc;
135                                                replace.put("name", warpPoint.name);
136                                        } catch (final WorldNotLoaded e) {
137                                        }
138                                }
139                                if (loc == null) {
140                                        Utils.sI18n(sender, "errorWarp", replace);
141                                        return;
142                                } else {
143                                        ACPluginManager.getScheduler().scheduleSyncDelayedTask(
144                                                        ACHelper.getInstance().getCoreInstance(),
145                                                        new DelayedTeleport(target.getLocation(), loc,
146                                                                        target, replace, sender),
147                                                        ConfigEnum.TP_DELAY.getLong());
148                                }
149                        }
150                }
151
152        }
153
154        /*
155         * (non-Javadoc)
156         * 
157         * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
158         */
159        @Override
160        public boolean argsCheck(final String... args) {
161                return args != null && args.length >= 1;
162        }
163
164        /*
165         * (non-Javadoc)
166         * 
167         * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
168         */
169        @Override
170        public void registerBukkitPerm() {
171                super.registerBukkitPerm();
172                tpAll = new PermChild("admincmd.warp.tp.all", bukkitDefault);
173                permParent.addChild(tpAll);
174        }
175
176        private class DelayedTeleport implements Runnable {
177
178                protected Location locBefore, teleportToLoc;
179                protected Player target;
180                protected HashMap<String, String> replace;
181                protected CommandSender sender;
182
183                public DelayedTeleport(final Location locBefore,
184                                final Location teleportLoc, final Player target,
185                                final HashMap<String, String> replace,
186                                final CommandSender sender) {
187                        this.target = target;
188                        this.locBefore = locBefore;
189                        this.teleportToLoc = teleportLoc;
190                        this.replace = replace;
191                        this.sender = sender;
192                }
193
194                @Override
195                public void run() {
196                        if (locBefore.equals(target.getLocation())
197                                        && ConfigEnum.CHECKTP.getBoolean()) {
198                                Utils.teleportWithChunkCheck(target, teleportToLoc);
199                                sendMessage(sender, target, "tpWarp", replace);
200                        } else if (!ConfigEnum.CHECKTP.getBoolean()) {
201                                Utils.teleportWithChunkCheck(target, teleportToLoc);
202                                sendMessage(sender, target, "tpWarp", replace);
203                        } else {
204                                replace.clear();
205                                replace.put("cmdname", "Warp");
206                                sendMessage(sender, target, "errorMoved", replace);
207                        }
208                }
209        }
