017package be.Balor.Manager.Commands.Warp;
018
019import java.util.Set;
020
021import org.bukkit.ChatColor;
022import org.bukkit.command.CommandSender;
023import org.bukkit.entity.Player;
024
025import be.Balor.Manager.Commands.CommandArgs;
026import be.Balor.Manager.Exceptions.PlayerNotFound;
027import be.Balor.Manager.Permissions.ActionNotPermitedException;
028import be.Balor.Manager.Permissions.PermChild;
029import be.Balor.Manager.Permissions.PermissionManager;
030import be.Balor.Tools.Utils;
031import be.Balor.Tools.Help.String.ACMinecraftFontWidthCalculator;
032import be.Balor.World.ACWorld;
033import be.Balor.World.WorldManager;
034
035/**
036 * @author Balor (aka Antoine Aflalo)
037 * 
038 */
039public class WarpList extends WarpCommand {
040        private PermChild tpAll;
041
042        /**
043         * 
044         */
045        public WarpList() {
046                permNode = "admincmd.warp.tp";
047                cmdName = "bal_warplist";
048        }
049
050        // TODO: Remove PermWarps a player is not allowed to warp to from the list
051        /*
052         * (non-Javadoc)
053         * 
054         * @see
055         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
056         * java.lang.String[])
057         */
058        @Override
059        public void execute(final CommandSender sender, final CommandArgs args)
060                        throws ActionNotPermitedException, PlayerNotFound {
061                if (Utils.isPlayer(sender)) {
062                        final Player p = (Player) sender;
063                        String msg = "";
064                        Set<String> wp;
065                        if (args.hasFlag('a')) {
066                                if (!PermissionManager.hasPerm(sender, tpAll)) {
067                                        return;
068                                }
069                                wp = WorldManager.getInstance().getAllWarpList();
070                        } else {
071                                wp = ACWorld.getWorld(p.getWorld()).getWarpList();
072                        }
073                        sender.sendMessage(ChatColor.GOLD + "Warp Point(s) : "
074                                        + ChatColor.WHITE + wp.size());
075                        for (final String name : wp) {
076                                msg += name + ", ";
077                                if (msg.length() >= ACMinecraftFontWidthCalculator.chatwidth) {
078                                        sender.sendMessage(msg);
079                                        msg = "";
080                                }
081                        }
082                        if (!msg.equals("")) {
083                                if (msg.endsWith(", ")) {
084                                        msg = msg.substring(0, msg.lastIndexOf(","));
085                                }
086                                sender.sendMessage(msg);
087                        }
088                }
089
090        }
091
092        /*
093         * (non-Javadoc)
094         * 
095         * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
096         */
097        @Override
098        public void registerBukkitPerm() {
099                super.registerBukkitPerm();
100                tpAll = new PermChild("admincmd.warp.tp.all", bukkitDefault);
101                permParent.addChild(tpAll);
102        }
103
104        /*
105         * (non-Javadoc)
106         * 
107         * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
108         */
109        @Override
110        public boolean argsCheck(final String... args) {
111                return true;
112        }
113
114}
