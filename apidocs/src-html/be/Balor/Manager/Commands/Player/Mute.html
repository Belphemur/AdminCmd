038/**
039 * @author Balor (aka Antoine Aflalo)
040 * 
041 */
042public class Mute extends PlayerCommand {
043        private PermChild cmdMute;
044
045        /**
046         *
047         */
048        public Mute() {
049                permNode = "admincmd.player.mute";
050                cmdName = "bal_mute";
051        }
052
053        /*
054         * (non-Javadoc)
055         * 
056         * @see
057         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
058         * java.lang.String[])
059         */
060        @Override
061        public void execute(final CommandSender sender, final CommandArgs args)
062                        throws ActionNotPermitedException, PlayerNotFound {
063                final Player player = Utils.getPlayer(args.getString(0));
064                if (args.hasFlag('c')
065                                && !PermissionManager.hasPerm(sender, cmdMute.getPermName())) {
066                        return;
067                }
068                if (player != null) {
069                        final HashMap<String, String> replace = new HashMap<String, String>();
070                        replace.put("player", Utils.getPlayerName(player));
071                        final ACPlayer acp = ACPlayer.getPlayer(player.getName());
072                        if (acp instanceof EmptyPlayer) {
073                                Utils.sI18n(sender, "playerNotFound", replace);
074                                return;
075                        }
076                        if (!Utils.checkImmunity(sender, player)) {
077                                Utils.sI18n(sender, "insufficientLvl");
078                                return;
079                        }
080                        String reason;
081                        if (args.hasFlag('r')) {
082                                reason = args.getValueFlag('r');
083                        } else {
084                                reason = "None";
085                        }
086                        replace.put("reason", reason);
087                        if (args.hasFlag('c')) {
088                                if (!acp.hasPower(Type.MUTED_COMMAND)) {
089                                        String msg = "Server Admin";
090                                        if (Utils.isPlayer(sender, false)) {
091                                                msg = Utils.getPlayerName((Player) sender);
092                                        }
093
094                                        if (!player.equals(sender)) {
095                                                Utils.sI18n(sender, "commandMuteEnabledTarget", replace);
096                                        }
097                                        if (args.length >= 2) {
098                                                Integer tmpMute = null;
099                                                try {
100                                                        tmpMute = args.getInt(args.length - 1);
101                                                        final String unmute = player.getName();
102                                                        final CommandSender senderFinal = sender;
103                                                        acp.setPower(Type.MUTED_COMMAND,
104                                                                        "Muted(including commands) by " + msg);
105                                                        ACPluginManager.getScheduler()
106                                                                        .runTaskLaterAsynchronously(getPlugin(),
107                                                                                        new Runnable() {
108
109                                                                                                @Override
110                                                                                                public void run() {
111                                                                                                        ACPlayer.getPlayer(unmute)
112                                                                                                                        .removePower(
113                                                                                                                                        Type.MUTED_COMMAND);
114                                                                                                        Utils.sI18n(
115                                                                                                                        senderFinal,
116                                                                                                                        "commandMuteDisabledTarget",
117                                                                                                                        "player", unmute);
118                                                                                                        final Lister list = Lister
119                                                                                                                        .getLister(
120                                                                                                                                        Lister.List.MUTE,
121                                                                                                                                        false);
122                                                                                                        if (list != null) {
123                                                                                                                list.update();
124                                                                                                        }
125                                                                                                }
126                                                                                        }, 20 * 60 * tmpMute);
127
128                                                } catch (final Exception e) {
129                                                }
130                                                if (tmpMute == null) {
131                                                        acp.setPower(Type.MUTED_COMMAND,
132                                                                        "Permanently muted(including commands) by "
133                                                                                        + msg);
134                                                        replace.put("time", "permanently");
135                                                        Utils.sI18n(player, "commandMuteEnabled", "reason",
136                                                                        reason);
137                                                } else {
138                                                        acp.setPower(Type.MUTED_COMMAND,
139                                                                        "Muted(including commands) by " + msg
140                                                                                        + " for " + tmpMute + " minutes");
141                                                        replace.put("minutes", tmpMute.toString());
142                                                        replace.put("time", "for" + tmpMute.toString()
143                                                                        + " minutes");
144                                                        Utils.sI18n(player, "commandTmpMuteEnabled",
145                                                                        replace);
146                                                }
147                                        } else {
148                                                acp.setPower(Type.MUTED_COMMAND,
149                                                                "Permanently muted(including commands) by "
150                                                                                + msg);
151                                                replace.put("time", "permanently");
152                                                Utils.sI18n(player, "commandMuteEnabled", replace);
153                                        }
154                                        if (args.hasFlag('b')) {
155                                                replace.put("muter", msg);
156                                                Utils.broadcastMessage(LocaleHelper.MUTE_BROADCAST
157                                                                .getLocale(replace));
158                                        }
159                                } else {
160                                        Utils.sI18n(sender, "alreadyCommandMuted");
161                                }
162                                return;
163                        }
164                        if (!acp.hasPower(Type.MUTED)) {
165                                String msg = "Server Admin";
166                                if (Utils.isPlayer(sender, false)) {
167                                        msg = Utils.getPlayerName((Player) sender);
168                                }
169                                if (!player.equals(sender)) {
170                                        Utils.sI18n(sender, "muteEnabledTarget", replace);
171                                }
172                                if (args.length >= 2) {
173                                        Integer tmpMute = null;
174                                        try {
175                                                tmpMute = args.getInt(args.length - 1);
176                                                final String unmute = player.getName();
177                                                final CommandSender senderFinal = sender;
178                                                ACPluginManager.getScheduler()
179                                                                .runTaskLaterAsynchronously(getPlugin(),
180                                                                                new Runnable() {
181
182                                                                                        @Override
183                                                                                        public void run() {
184                                                                                                ACPlayer.getPlayer(unmute)
185                                                                                                                .removePower(Type.MUTED);
186                                                                                                Utils.sI18n(senderFinal,
187                                                                                                                "muteDisabledTarget",
188                                                                                                                "player", unmute);
189                                                                                                final Lister list = Lister
190                                                                                                                .getLister(
191                                                                                                                                Lister.List.MUTE,
192                                                                                                                                false);
193                                                                                                if (list != null) {
194                                                                                                        list.update();
195                                                                                                }
196                                                                                        }
197                                                                                }, 20 * 60 * tmpMute);
198
199                                        } catch (final Exception e) {
200                                        }
201                                        if (tmpMute == null) {
202                                                acp.setPower(Type.MUTED, "Permanently muted by " + msg);
203                                                replace.put("time", "permanently");
204                                                Utils.sI18n(player, "muteEnabled", replace);
205                                        } else {
206                                                acp.setPower(Type.MUTED, "Muted by " + msg + " for "
207                                                                + tmpMute + " minutes");
208                                                replace.put("minutes", tmpMute.toString());
209                                                replace.put("time", "for" + tmpMute.toString()
210                                                                + " minutes");
211                                                Utils.sI18n(player, "tmpMuteEnabled", replace);
212                                        }
213                                } else {
214                                        acp.setPower(Type.MUTED, "Permanently muted by " + msg);
215                                        replace.put("time", "permanently");
216                                        Utils.sI18n(player, "muteEnabled", replace);
217                                }
218                                if (args.hasFlag('b')) {
219                                        replace.put("muter", msg);
220                                        Utils.broadcastMessage(LocaleHelper.MUTE_BROADCAST
221                                                        .getLocale(replace));
222                                }
223                        } else {
224                                Utils.sI18n(sender, "alreadyMuted");
225                        }
226                        final Lister list = Lister.getLister(Lister.List.MUTE, false);
227                        if (list != null) {
228                                list.update();
229                        }
230                }
231
232        }
233
234        /*
235         * (non-Javadoc)
236         * 
237         * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
238         */
239        @Override
240        public boolean argsCheck(final String... args) {
241                return args != null && args.length >= 1;
242        }
243
244        /*
245         * (non-Javadoc)
246         * 
247         * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
248         */
249        @Override
250        public void registerBukkitPerm() {
251                cmdMute = new PermChild(permNode + ".command");
252                final PermParent parent = new PermParent(permNode + ".*");
253                plugin.getPermissionLinker().addChildPermParent(parent, permParent);
254                final PermChild child = new PermChild(permNode, bukkitDefault);
255                parent.addChild(child).addChild(cmdMute);
256                permChild = child;
257        }
