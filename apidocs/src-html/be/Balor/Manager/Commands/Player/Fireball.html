017    package be.Balor.Manager.Commands.Player;
018    
019    import java.util.HashMap;
020    
021    import org.bukkit.command.CommandSender;
022    import org.bukkit.entity.Player;
023    
024    import be.Balor.Manager.Commands.CommandArgs;
025    import be.Balor.Player.ACPlayer;
026    import be.Balor.Tools.Type;
027    import be.Balor.Tools.Utils;
028    import be.Balor.Tools.Threads.RemovePowerTask;
029    import be.Balor.bukkit.AdminCmd.ACPluginManager;
030    import be.Balor.bukkit.AdminCmd.ConfigEnum;
031    
032    /**
033     * @author Balor (aka Antoine Aflalo)
034     * 
035     */
036    public class Fireball extends PlayerCommand {
037            public Fireball() {
038                    permNode = "admincmd.player.fireball";
039                    cmdName = "bal_fireball";
040                    other = true;
041            }
042    
043            /*
044             * (non-Javadoc)
045             * 
046             * @see
047             * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
048             * java.lang.String[])
049             */
050            @Override
051            public void execute(final CommandSender sender, final CommandArgs args) {
052                    Player player = null;
053                    float power = ConfigEnum.DFB.getFloat();
054                    final String timeOut = args.getValueFlag('t');
055                    if (args.length >= 1) {
056                            try {
057                                    player = Utils.getUser(sender, args, permNode, 1, false);
058                                    power = args.getFloat(0);
059                            } catch (final NumberFormatException e) {
060                                    power = ConfigEnum.DFB.getFloat();
061                                    player = Utils.getUser(sender, args, permNode);
062                            }
063                            if (args.length >= 2)
064                                    player = Utils.getUser(sender, args, permNode, 1, true);
065                    } else
066                            player = Utils.getUser(sender, args, permNode);
067                    if (player != null) {
068                            final HashMap<String, String> replace = new HashMap<String, String>();
069                            replace.put("player", Utils.getPlayerName(player));
070                            final ACPlayer acp = ACPlayer.getPlayer(player);
071                            if (acp.hasPower(Type.FIREBALL)) {
072                                    acp.removePower(Type.FIREBALL);
073                                    Utils.sI18n(player, "fireballDisabled");
074                                    if (!player.equals(sender))
075                                            Utils.sI18n(sender, "fireballDisabledTarget", replace);
076                            } else {
077                                    acp.setPower(Type.FIREBALL, power);
078                                    Utils.sI18n(player, "fireballEnabled");
079                                    if (!player.equals(sender))
080                                            Utils.sI18n(sender, "fireballEnabledTarget", replace);
081                                    if (timeOut == null)
082                                            return;
083                                    int timeOutValue;
084                                    try {
085                                            timeOutValue = Integer.parseInt(timeOut);
086                                    } catch (final Exception e) {
087                                            Utils.sI18n(sender, "NaN", "number", timeOut);
088                                            return;
089                                    }
090                                    ACPluginManager.getScheduler().scheduleAsyncDelayedTask(
091                                                    ACPluginManager.getCorePlugin(),
092                                                    new RemovePowerTask(acp, Type.FIREBALL, sender),
093                                                    Utils.secInTick * ConfigEnum.SCALE_TIMEOUT.getInt() * timeOutValue);
094                            }
095                    }
096    
097            }
098    
099            /*
100             * (non-Javadoc)
101             * 
102             * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
103             */
104            @Override
105            public boolean argsCheck(final String... args) {
106                    return args != null;
107            }
108    }
