017    package be.Balor.Manager.Commands.Player;
018    
019    import java.util.ArrayList;
020    import java.util.HashMap;
021    import java.util.List;
022    
023    import org.bukkit.ChatColor;
024    import org.bukkit.command.CommandSender;
025    import org.bukkit.entity.Player;
026    import org.bukkit.potion.PotionEffect;
027    import org.bukkit.potion.PotionEffectType;
028    
029    import be.Balor.Manager.Commands.CommandArgs;
030    import be.Balor.Manager.Exceptions.PlayerNotFound;
031    import be.Balor.Tools.Utils;
032    import be.Balor.Tools.Help.String.Str;
033    import be.Balor.bukkit.AdminCmd.LocaleHelper;
034    
035    import com.google.common.base.Joiner;
036    
037    /**
038     * @author Balor (aka Antoine Aflalo)
039     * 
040     */
041    public class Potion extends PlayerCommand {
042            private final static List<String> potions = new ArrayList<String>();
043            static {
044                    for (final PotionEffectType type : PotionEffectType.values())
045                            if (type != null && type.getName() != null)
046                                    potions.add(type.getName());
047            }
048    
049            /**
050             * 
051             */
052            public Potion() {
053                    super("bal_potion", "admincmd.player.potion");
054                    other = true;
055            }
056    
057            /*
058             * (non-Javadoc)
059             * 
060             * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
061             * CommandSender, be.Balor.Manager.Commands.CommandArgs)
062             */
063            @Override
064            public void execute(final CommandSender sender, final CommandArgs args) throws PlayerNotFound {
065                    final Player target = Utils.getUserParam(sender, args, permNode);
066                    final String potion = args.getString(0);
067                    final String potionFound = Str.matchString(potions, potion);
068                    final HashMap<String, String> replace = new HashMap<String, String>();
069                    if (potionFound == null) {
070                            replace.put("value", potion);
071                            replace.put("type", LocaleHelper.TYPE_POTION.getLocale());
072                            LocaleHelper.DONT_EXISTS.sendLocale(sender, replace);
073                            sender.sendMessage(ChatColor.GREEN + "Potion list :");
074                            sender.sendMessage(Joiner.on(", ").skipNulls().join(potions).toLowerCase());
075                            return;
076                    }
077                    final String potionDurationString = args.getString(1);
078                    final String potionAmplifierString = args.getString(2);
079                    int amplifier = 5;
080                    int duration = 20 * Utils.secInTick;
081                    if (potionAmplifierString != null)
082                            try {
083                                    amplifier = Integer.parseInt(potionAmplifierString);
084                            } catch (final NumberFormatException e) {
085                                    Utils.sI18n(sender, "NaN", "number", potionAmplifierString);
086                                    return;
087                            }
088                    if (potionDurationString != null)
089                            try {
090                                    duration = Integer.parseInt(potionDurationString);
091                            } catch (final NumberFormatException e) {
092                                    Utils.sI18n(sender, "NaN", "number", potionDurationString);
093                                    return;
094                            }
095                    target.addPotionEffect(new PotionEffect(PotionEffectType.getByName(potionFound), duration,
096                                    amplifier));
097                    replace.put("player", Utils.getPlayerName(target, sender));
098                    replace.put("potion", potionFound);
099                    LocaleHelper.POTION_EFFECT.sendLocale(sender, replace);
100                    if (!target.equals(sender))
101                            LocaleHelper.POTION_EFFECT.sendLocale(target, replace);
102    
103            }
104    
105            /*
106             * (non-Javadoc)
107             * 
108             * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
109             */
110            @Override
111            public boolean argsCheck(final String... args) {
112                    return args != null;
113            }
114    
115    }
