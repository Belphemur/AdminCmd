017package be.Balor.Manager.Commands.Player;
018
019import java.util.HashMap;
020
021import org.bukkit.Location;
022import org.bukkit.command.CommandSender;
023import org.bukkit.entity.ExperienceOrb;
024import org.bukkit.entity.Player;
025
026import be.Balor.Manager.Commands.CommandArgs;
027import be.Balor.Manager.Exceptions.PlayerNotFound;
028import be.Balor.Manager.Permissions.ActionNotPermitedException;
029import be.Balor.Tools.Utils;
030import be.Balor.bukkit.AdminCmd.ACPluginManager;
031
032/**
033 * @author Lathanael (aka Philippe Leipold)
034 * 
035 */
036public class Experience extends PlayerCommand {
037
038        public Experience() {
039                permNode = "admincmd.player.experience";
040                cmdName = "bal_exp";
041                other = true;
042        }
043
044        /*
045         * (non-Javadoc)
046         * 
047         * @see
048         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
049         * java.lang.String[])
050         */
051        @Override
052        public void execute(final CommandSender sender, final CommandArgs args)
053                        throws ActionNotPermitedException, PlayerNotFound {
054                float amount = 0;
055                Player target = null;
056                final HashMap<String, String> replace = new HashMap<String, String>();
057                boolean self = false;
058                if (args.hasFlag('p')) {
059                        target = Utils.getPlayer(args.getValueFlag('p'));
060                } else {
061                        if (!Utils.isPlayer(sender)) {
062                                return;
063                        }
064                        target = (Player) sender;
065                        self = true;
066                }
067                if (0 < args.length) {
068                        if (!args.hasFlag('t')) {
069                                try {
070                                        amount = args.getFloat(0);
071                                } catch (final NumberFormatException e) {
072                                        replace.put("number", args.getString(0));
073                                        Utils.I18n("NaN", replace);
074                                        return;
075                                }
076                        }
077                } else {
078                        if (Utils.isPlayer(sender, true)) {
079                                if (args.hasFlag('t')) {
080                                        target = (Player) sender;
081                                        replace.put("exp",
082                                                        String.valueOf(target.getTotalExperience()));
083                                        sender.sendMessage(Utils.I18n("expTotal", replace));
084                                        return;
085                                }
086                        } else {
087                                return;
088                        }
089                }
090                if (target == null) {
091                        return;
092                }
093                replace.put("amount", String.valueOf(amount));
094                final Player taskTarget = target;
095                final float amountXp = amount;
096                if (args.hasFlag('d')) {
097                        final Location loc = target.getLocation();
098                        loc.setX(loc.getX() + 2);
099                        ACPluginManager.scheduleSyncTask(new Runnable() {
100                                @Override
101                                public void run() {
102                                        taskTarget.getLocation().getWorld()
103                                                        .spawn(loc, ExperienceOrb.class)
104                                                        .setExperience((int) amountXp);
105                                }
106                        });
107
108                        if (self) {
109                                target.sendMessage(Utils.I18n("expDropped", replace));
110                        } else {
111                                replace.put("target", Utils.getPlayerName(target));
112                                target.sendMessage(Utils.I18n("expDropped", replace));
113                                sender.sendMessage(Utils.I18n("expDroppedTarget", replace));
114                        }
115                } else if (args.hasFlag('a')) {
116
117                        ACPluginManager.scheduleSyncTask(new Runnable() {
118                                @Override
119                                public void run() {
120                                        taskTarget.giveExp((int) amountXp);
121                                }
122                        });
123                        if (self) {
124                                target.sendMessage(Utils.I18n("expAdded", replace));
125                        } else {
126                                replace.put("target", Utils.getPlayerName(target));
127                                sender.sendMessage(Utils.I18n("expAddedTarget", replace));
128                                target.sendMessage(Utils.I18n("expAdded", replace));
129                        }
130                } else if (args.hasFlag('b')) {
131                        final float exp = (amount > 1 ? 1 : amount);
132                        ACPluginManager.scheduleSyncTask(new Runnable() {
133                                @Override
134                                public void run() {
135                                        taskTarget.setExp(exp);
136                                }
137                        });
138                        replace.put("amount", String.valueOf(exp * 100.0F));
139                        if (self) {
140                                target.sendMessage(Utils.I18n("expProgressionSet", replace));
141                        } else {
142                                replace.put("target", Utils.getPlayerName(target));
143                                target.sendMessage(Utils.I18n("expProgressionSet", replace));
144                                sender.sendMessage(Utils.I18n("expProgressionSetTarget",
145                                                replace));
146                        }
147                } else if (args.hasFlag('l')) {
148                        ACPluginManager.scheduleSyncTask(new Runnable() {
149                                @Override
150                                public void run() {
151                                        taskTarget.setLevel((int) amountXp);
152                                }
153                        });
154                        if (self) {
155                                target.sendMessage(Utils.I18n("expLevelSet", replace));
156                        } else {
157                                replace.put("target", Utils.getPlayerName(target));
158                                target.sendMessage(Utils.I18n("expLevelSet", replace));
159                                sender.sendMessage(Utils.I18n("expLevelSetTarget", replace));
160                        }
161                } else if (args.hasFlag('t')) {
162                        replace.put("exp", String.valueOf(target.getTotalExperience()));
163                        if (self) {
164                                sender.sendMessage(Utils.I18n("expTotal", replace));
165                        } else {
166                                replace.put("target", Utils.getPlayerName(target));
167                                sender.sendMessage(Utils.I18n("expTotalTarget", replace));
168                        }
169                }
170        }
171
172        /*
173         * (non-Javadoc)
174         * 
175         * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
176         */
177        @Override
178        public boolean argsCheck(final String... args) {
179                return args != null && args.length >= 1;
180        }
181
182}
