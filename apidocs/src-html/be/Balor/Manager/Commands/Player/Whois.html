040    /**
041     * @author Balor (aka Antoine Aflalo)
042     * 
043     */
044    public class Whois extends PlayerCommand {
045            /**
046             *
047             */
048            public Whois() {
049                    super("bal_whois", "admincmd.player.whois");
050                    other = true;
051            }
052    
053            /*
054             * (non-Javadoc)
055             * 
056             * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
057             * CommandSender, be.Balor.Manager.Commands.CommandArgs)
058             */
059            @Override
060            public void execute(final CommandSender sender, final CommandArgs args) {
061                    if (args.hasFlag('w')) {
062                            ACWorld w;
063                            try {
064                                    if (args.length >= 1)
065                                            w = ACWorld.getWorld(args.getString(0));
066                                    else if (Utils.isPlayer(sender))
067                                            w = ACWorld.getWorld(((Player) sender).getWorld().getName());
068                                    else
069                                            return;
070                            } catch (final WorldNotLoaded e) {
071                                    Utils.sI18n(sender, "worldNotFound", "world", args.getString(0));
072                                    return;
073                            }
074                            sender.sendMessage(ChatColor.GREEN
075                                            + ACMinecraftFontWidthCalculator.strPadCenterChat(
076                                                            ChatColor.AQUA + " " + w.getName() + " " + ChatColor.GREEN, '='));
077                            for (final Entry<String, String> power : w.getInformations().entrySet()) {
078                                    String line = ChatColor.GOLD + power.getKey() + ChatColor.WHITE + " : ";
079                                    final int sizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
080                                                    - ACMinecraftFontWidthCalculator.getStringWidth(line);
081                                    line += ACMinecraftFontWidthCalculator.strPadLeftChat(
082                                                    ChatColor.GREEN + power.getValue(), sizeRemaining, ' ');
083                                    sender.sendMessage(line);
084                            }
085                            return;
086                    }
087                    final ACPlayer actarget = Utils.getACPlayer(sender, args, permNode);
088                    if (actarget == null)
089                            return;
090                    sender.sendMessage(ChatColor.AQUA
091                                    + ACMinecraftFontWidthCalculator.strPadCenterChat(
092                                                    ChatColor.DARK_GREEN
093                                                                    + " "
094                                                                    + (actarget.getHandler() != null ? Utils.getPlayerName(
095                                                                                    actarget.getHandler(), sender) : actarget.getName()) + " "
096                                                                    + ChatColor.AQUA, '='));
097                    // Login
098                    String loginDate = ChatColor.GOLD + "Last Login" + ChatColor.WHITE + " : ";
099                    final int logSizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
100                                    - ACMinecraftFontWidthCalculator.getStringWidth(loginDate);
101                    loginDate += ACMinecraftFontWidthCalculator.strPadLeftChat(
102                                    ChatColor.GREEN + Utils.replaceDateAndTimeFormat(actarget, Type.Whois.LOGIN),
103                                    logSizeRemaining, ' ');
104                    sender.sendMessage(loginDate);
105    
106                    // Logout
107                    String logoutDate = ChatColor.GOLD + "Last Quit" + ChatColor.WHITE + " : ";
108                    final int logoutSizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
109                                    - ACMinecraftFontWidthCalculator.getStringWidth(logoutDate);
110                    logoutDate += ACMinecraftFontWidthCalculator.strPadLeftChat(
111                                    ChatColor.GREEN + Utils.replaceDateAndTimeFormat(actarget, Type.Whois.LOGOUT),
112                                    logoutSizeRemaining, ' ');
113                    sender.sendMessage(logoutDate);
114    
115                    // Presentation
116                    String presentation = ChatColor.GOLD + "Presentation" + ChatColor.WHITE + " : ";
117                    final int presSizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
118                                    - ACMinecraftFontWidthCalculator.getStringWidth(presentation);
119                    presentation += ACMinecraftFontWidthCalculator.strPadLeftChat(
120                                    ChatColor.GREEN + actarget.getPresentation(), presSizeRemaining, ' ');
121                    sender.sendMessage(presentation);
122    
123                    // Played
124                    final long total = actarget.getCurrentPlayedTime();
125                    final Long[] time = Utils.transformToElapsedTime(total);
126                    final HashMap<String, String> replace = new HashMap<String, String>();
127                    replace.put("d", time[0].toString());
128                    replace.put("h", time[1].toString());
129                    replace.put("m", time[2].toString());
130                    replace.put("s", time[3].toString());
131                    String played = ChatColor.GOLD + "Time Played" + ChatColor.WHITE + " : ";
132                    int strSizeRem = ACMinecraftFontWidthCalculator.chatwidth
133                                    - ACMinecraftFontWidthCalculator.getStringWidth(played);
134                    played += ACMinecraftFontWidthCalculator.strPadLeftChat(
135                                    ChatColor.GREEN + Utils.I18n("elapsedTotalTime", replace), strSizeRem, ' ');
136                    sender.sendMessage(played);
137                    // Banned
138                    final BannedPlayer ban = ACHelper.getInstance().getBan(actarget.getName());
139                    if (ban != null) {
140                            String banned = ChatColor.GOLD + "Banned" + ChatColor.WHITE + " : ";
141                            final int banSizeRem = ACMinecraftFontWidthCalculator.chatwidth
142                                            - ACMinecraftFontWidthCalculator.getStringWidth(banned);
143                            banned += ACMinecraftFontWidthCalculator.strPadLeftChat(ChatColor.GREEN
144                                            + (ban instanceof TempBannedPlayer ? ((TempBannedPlayer) ban).getEndBan()
145                                                            .toString() : "Permanent"), banSizeRem, ' ');
146                            sender.sendMessage(banned);
147                    }
148                    // Powers
149                    for (final Entry<String, String> power : actarget.getPowers().entrySet()) {
150                            String line = ChatColor.GOLD + Type.matchType(power.getKey()).display()
151                                            + ChatColor.WHITE + " : ";
152                            final int sizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
153                                            - ACMinecraftFontWidthCalculator.getStringWidth(line);
154                            line += ACMinecraftFontWidthCalculator.strPadLeftChat(
155                                            ChatColor.GREEN + power.getValue(), sizeRemaining, ' ');
156                            sender.sendMessage(line);
157                    }
158                    // Invisible
159                    String line = ChatColor.GOLD + "invisible" + ChatColor.WHITE + " : ";
160                    final int sizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
161                                    - ACMinecraftFontWidthCalculator.getStringWidth(line);
162                    line += ACMinecraftFontWidthCalculator.strPadLeftChat(
163                                    ChatColor.GREEN
164                                                    + String.valueOf((InvisibleWorker.getInstance().hasInvisiblePowers(
165                                                                    actarget.getName()) && PermissionManager.hasPerm(sender,
166                                                                    "admincmd.invisible.cansee", false))), sizeRemaining, ' ');
167                    sender.sendMessage(line);
168    
169                    // Immunity Level
170                    final int level = actarget.getInformation("immunityLvl").getInt(
171                                    ConfigEnum.DIMMUNITY.getInt());
172                    String immuLvl = ChatColor.GOLD + "Immunity Level" + ChatColor.WHITE + " : ";
173                    strSizeRem = ACMinecraftFontWidthCalculator.chatwidth
174                                    - ACMinecraftFontWidthCalculator.getStringWidth(immuLvl);
175                    immuLvl += ACMinecraftFontWidthCalculator.strPadLeftChat(
176                                    ChatColor.GREEN + String.valueOf(level), strSizeRem, ' ');
177                    sender.sendMessage(immuLvl);
178            }
179    
180            /*
181             * (non-Javadoc)
182             * 
183             * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
184             */
185            @Override
186            public boolean argsCheck(final String... args) {
187                    return args != null;
188            }
