044/**
045 * @author Balor (aka Antoine Aflalo)
046 * 
047 */
048public class Whois extends PlayerCommand {
049        /**
050         *
051         */
052        public Whois() {
053                super("bal_whois", "admincmd.player.whois");
054                other = true;
055        }
056
057        /*
058         * (non-Javadoc)
059         * 
060         * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
061         * CommandSender, be.Balor.Manager.Commands.CommandArgs)
062         */
063        @Override
064        public void execute(final CommandSender sender, final CommandArgs args)
065                        throws ActionNotPermitedException, PlayerNotFound {
066                if (args.hasFlag('w')) {
067                        ACWorld w;
068                        try {
069                                if (args.length >= 1) {
070                                        w = ACWorld.getWorld(args.getString(0));
071                                } else if (Utils.isPlayer(sender)) {
072                                        w = ACWorld
073                                                        .getWorld(((Player) sender).getWorld().getName());
074                                } else {
075                                        return;
076                                }
077                        } catch (final WorldNotLoaded e) {
078                                Utils.sI18n(sender, "worldNotFound", "world", args.getString(0));
079                                return;
080                        }
081                        sender.sendMessage(ChatColor.GREEN
082                                        + ACMinecraftFontWidthCalculator.strPadCenterChat(
083                                                        ChatColor.AQUA + " " + w.getName() + " "
084                                                                        + ChatColor.GREEN, '='));
085                        for (final Entry<String, String> power : w.getInformationsList()
086                                        .entrySet()) {
087                                String line = ChatColor.GOLD + power.getKey() + ChatColor.WHITE
088                                                + " : ";
089                                final int sizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
090                                                - ACMinecraftFontWidthCalculator.getStringWidth(line);
091                                line += ACMinecraftFontWidthCalculator.strPadLeftChat(
092                                                ChatColor.GREEN + power.getValue(), sizeRemaining, ' ');
093                                sender.sendMessage(line);
094                        }
095                        return;
096                }
097                final ACPlayer actarget = Utils.getACPlayer(sender, args, permNode);
098                if (actarget == null) {
099                        return;
100                }
101                sender.sendMessage(ChatColor.AQUA
102                                + ACMinecraftFontWidthCalculator.strPadCenterChat(
103                                                ChatColor.DARK_GREEN
104                                                                + " "
105                                                                + (actarget.isOnline()
106                                                                                ? Utils.getPlayerName(
107                                                                                                actarget.getHandler(), sender)
108                                                                                : actarget.getName()) + " "
109                                                                + ChatColor.AQUA, '='));
110                // Login
111                String loginDate = ChatColor.GOLD + "Last Login" + ChatColor.WHITE
112                                + " : ";
113                final int logSizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
114                                - ACMinecraftFontWidthCalculator.getStringWidth(loginDate);
115                loginDate += ACMinecraftFontWidthCalculator.strPadLeftChat(
116                                ChatColor.GREEN
117                                                + Utils.replaceDateAndTimeFormat(actarget,
118                                                                Type.Whois.LOGIN), logSizeRemaining, ' ');
119                sender.sendMessage(loginDate);
120
121                // Logout
122                String logoutDate = ChatColor.GOLD + "Last Quit" + ChatColor.WHITE
123                                + " : ";
124                final int logoutSizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
125                                - ACMinecraftFontWidthCalculator.getStringWidth(logoutDate);
126                logoutDate += ACMinecraftFontWidthCalculator.strPadLeftChat(
127                                ChatColor.GREEN
128                                                + Utils.replaceDateAndTimeFormat(actarget,
129                                                                Type.Whois.LOGOUT), logoutSizeRemaining, ' ');
130                sender.sendMessage(logoutDate);
131
132                // Presentation
133                String presentation = ChatColor.GOLD + "Presentation" + ChatColor.WHITE
134                                + " : ";
135                final int presSizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
136                                - ACMinecraftFontWidthCalculator.getStringWidth(presentation);
137                presentation += ACMinecraftFontWidthCalculator.strPadLeftChat(
138                                ChatColor.GREEN + actarget.getPresentation(),
139                                presSizeRemaining, ' ');
140                sender.sendMessage(presentation);
141
142                // Played
143                final long total = actarget.getCurrentPlayedTime();
144                final Map<String, String> replace = Utils.playedTime(
145                                actarget.getName(), total);
146                String played = ChatColor.GOLD + "Time Played" + ChatColor.WHITE
147                                + " : ";
148                int strSizeRem = ACMinecraftFontWidthCalculator.chatwidth
149                                - ACMinecraftFontWidthCalculator.getStringWidth(played);
150                played += ACMinecraftFontWidthCalculator.strPadLeftChat(ChatColor.GREEN
151                                + Utils.I18n("elapsedTotalTime", replace), strSizeRem, ' ');
152                sender.sendMessage(played);
153                // Banned
154                final IBan ban = ACHelper.getInstance().getBan(actarget.getName());
155                if (ban != null) {
156                        String banned = ChatColor.GOLD + "Banned" + ChatColor.WHITE + " : ";
157                        final int banSizeRem = ACMinecraftFontWidthCalculator.chatwidth
158                                        - ACMinecraftFontWidthCalculator.getStringWidth(banned);
159                        banned += ACMinecraftFontWidthCalculator.strPadLeftChat(
160                                        ChatColor.GREEN
161                                                        + (ban instanceof ITempBan ? Utils
162                                                                        .replaceDateAndTimeFormat(((ITempBan) ban)
163                                                                                        .getEndBan()) : "Permanent"),
164                                        banSizeRem, ' ');
165                        sender.sendMessage(banned);
166                }
167                // Powers
168                for (final Entry<String, String> power : actarget.getPowersString()
169                                .entrySet()) {
170                        final Type powerType = Type.matchType(power.getKey());
171                        if (powerType == Type.INVISIBLE) {
172                                continue;
173                        }
174                        String line = ChatColor.GOLD + powerType.display()
175                                        + ChatColor.WHITE + " : ";
176                        final int sizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
177                                        - ACMinecraftFontWidthCalculator.getStringWidth(line);
178                        line += ACMinecraftFontWidthCalculator.strPadLeftChat(
179                                        ChatColor.GREEN + power.getValue(), sizeRemaining, ' ');
180                        sender.sendMessage(line);
181                }
182                // Invisible
183                String line = ChatColor.GOLD + "invisible" + ChatColor.WHITE + " : ";
184                final int sizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
185                                - ACMinecraftFontWidthCalculator.getStringWidth(line);
186                line += ACMinecraftFontWidthCalculator
187                                .strPadLeftChat(
188                                                ChatColor.GREEN
189                                                                + String.valueOf((InvisibleWorker.getInstance()
190                                                                                .hasInvisiblePowers(
191                                                                                                actarget.getHandler()) && PermissionManager
192                                                                                .hasPerm(sender,
193                                                                                                "admincmd.invisible.cansee",
194                                                                                                false))), sizeRemaining, ' ');
195                sender.sendMessage(line);
196
197                // Immunity Level
198                final int level = actarget.getInformation("immunityLvl").getInt(
199                                ConfigEnum.DIMMUNITY.getInt());
200                String immuLvl = ChatColor.GOLD + "Immunity Level" + ChatColor.WHITE
201                                + " : ";
202                strSizeRem = ACMinecraftFontWidthCalculator.chatwidth
203                                - ACMinecraftFontWidthCalculator.getStringWidth(immuLvl);
204                immuLvl += ACMinecraftFontWidthCalculator.strPadLeftChat(
205                                ChatColor.GREEN + String.valueOf(level), strSizeRem, ' ');
206                sender.sendMessage(immuLvl);
207
208                // GameMode
209                String gameMode = ChatColor.GREEN + "Current GameMode: "
210                                + ChatColor.GOLD;
211                String currentMode = "";
212                if (actarget.isOnline()) {
213                        if (actarget.getHandler().getGameMode() == GameMode.CREATIVE) {
214                                currentMode = "Creative";
215                        } else {
216                                currentMode = "Survival";
217                        }
218                } else {
219                        currentMode = actarget.getInformation("gameMode").getString();
220                        if (currentMode == null) {
221                                currentMode = LocaleHelper.UNKNOWN.getLocale();
222                        }
223                }
224
225                strSizeRem = ACMinecraftFontWidthCalculator.chatwidth
226                                - ACMinecraftFontWidthCalculator.getStringWidth(gameMode);
227                gameMode += ACMinecraftFontWidthCalculator.strPadLeftChat(
228                                ChatColor.GREEN + String.valueOf(currentMode), strSizeRem, ' ');
229                sender.sendMessage(gameMode);
230        }
231
232        /*
233         * (non-Javadoc)
234         * 
235         * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
236         */
237        @Override
238        public boolean argsCheck(final String... args) {
239                return args != null;
240        }
