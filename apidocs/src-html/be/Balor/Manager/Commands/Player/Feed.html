017    package be.Balor.Manager.Commands.Player;
018    
019    import org.bukkit.command.CommandSender;
020    
021    import be.Balor.Manager.Commands.CommandArgs;
022    import be.Balor.Tools.Type.Health;
023    import be.Balor.Tools.Utils;
024    import be.Balor.bukkit.AdminCmd.ACPluginManager;
025    
026    /**
027     * @author Balor (aka Antoine Aflalo)
028     * @author Lathanael (aka Philippe Leipold)
029     * 
030     */
031    public class Feed extends PlayerCommand {
032    
033            /**
034             *
035             */
036            public Feed() {
037                    permNode = "admincmd.player.feed";
038                    cmdName = "bal_feed";
039                    other = true;
040            }
041    
042            /*
043             * (non-Javadoc)
044             * 
045             * @see
046             * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
047             * java.lang.String[])
048             */
049            @Override
050            public void execute(final CommandSender sender, final CommandArgs args) {
051                    ACPluginManager.scheduleSyncTask(new Runnable() {
052                            @Override
053                            public void run() {
054                                    Utils.setPlayerHealth(sender, args, Health.FEED);
055                            }
056                    });
057    
058            }
059    
060            /*
061             * (non-Javadoc)
062             * 
063             * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
064             */
065            @Override
066            public boolean argsCheck(final String... args) {
067                    return args != null;
068            }
069    
070    }
