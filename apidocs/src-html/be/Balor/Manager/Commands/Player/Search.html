019package be.Balor.Manager.Commands.Player;
020
021import java.net.InetAddress;
022import java.util.List;
023import java.util.TreeSet;
024
025import org.bukkit.command.CommandSender;
026import org.bukkit.entity.Player;
027
028import be.Balor.Manager.Commands.CommandArgs;
029import be.Balor.Manager.Exceptions.PlayerNotFound;
030import be.Balor.Manager.Permissions.ActionNotPermitedException;
031import be.Balor.Player.ACPlayer;
032import be.Balor.Player.PlayerManager;
033import be.Balor.Tools.Utils;
034import be.Balor.bukkit.AdminCmd.ACPluginManager;
035
036import com.google.common.base.Joiner;
037
038/**
039 * @author Lathanael (aka Philippe Leipold)
040 * 
041 */
042public class Search extends PlayerCommand {
043
044        public Search() {
045                cmdName = "bal_search";
046                permNode = "admincmd.player.search";
047        }
048
049        /*
050         * (non-Javadoc)
051         * 
052         * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
053         * CommandSender, be.Balor.Manager.Commands.CommandArgs)
054         */
055        @Override
056        public void execute(final CommandSender sender, final CommandArgs args)
057                        throws PlayerNotFound, ActionNotPermitedException {
058                String ip = null;
059                if (args.hasFlag('i')) {
060                        ip = args.getValueFlag('i');
061                } else if (args.hasFlag('p')) {
062                        ip = Utils.getACPlayerParam(sender, args, permNode)
063                                        .getInformation("last-ip").getString();
064                }
065
066                if (ip == null || ip.equals("") || ip.length() == 0) {
067                        return;
068                }
069                final Player[] onPlayers = ACPluginManager.getServer()
070                                .getOnlinePlayers();
071                final List<ACPlayer> exPlayers = PlayerManager.getInstance()
072                                .getExistingPlayers();
073                final TreeSet<String> players = new TreeSet<String>();
074                final TreeSet<String> playersOld = new TreeSet<String>();
075                final String on = "[ON] ", off = "[OFF] ";
076                InetAddress ipAdress;
077                for (final Player p : onPlayers) {
078                        ipAdress = p.getAddress().getAddress();
079                        if (ipAdress != null && ipAdress.toString().substring(1).equals(ip)) {
080                                players.add(on + Utils.getPlayerName(p));
081                                playersOld.add(p.getName());
082                        }
083                }
084                String ip2;
085                for (final ACPlayer p : exPlayers) {
086                        ip2 = p.getInformation("last-ip").getString();
087                        if (ip2 != null && ip2.contains(ip)
088                                        && !playersOld.contains(p.getName())) {
089                                players.add(off + p.getName());
090                        }
091                }
092                final String found = Joiner.on(", ").join(players);
093                sender.sendMessage(found);
094                return;
095
096        }
097
098        /*
099         * (non-Javadoc)
100         * 
101         * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
102         */
103        @Override
104        public boolean argsCheck(final String... args) {
105                return args != null && args.length >= 1;
106        }
107}
