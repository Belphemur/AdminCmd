017    package be.Balor.Manager.Commands;
018    
019    import org.bukkit.command.CommandSender;
020    import org.bukkit.command.PluginCommand;
021    import org.bukkit.permissions.Permission;
022    import org.bukkit.permissions.PermissionDefault;
023    
024    import be.Balor.Manager.Exceptions.CommandAlreadyExist;
025    import be.Balor.Manager.Exceptions.CommandNotFound;
026    import be.Balor.Manager.Exceptions.PlayerNotFound;
027    import be.Balor.Manager.Permissions.PermChild;
028    import be.Balor.Manager.Permissions.PermParent;
029    import be.Balor.Manager.Permissions.PermissionManager;
030    import be.Balor.bukkit.AdminCmd.ACPluginManager;
031    import be.Balor.bukkit.AdminCmd.AbstractAdminCmdPlugin;
032    
033    /**
034     * @author Balor (aka Antoine Aflalo)
035     * 
036     */
037    public abstract class CoreCommand {
038            protected String permNode = "";
039            protected String cmdName = "";
040            protected Permission bukkitPerm = null;
041            protected PermissionDefault bukkitDefault = PermissionDefault.OP;
042            protected boolean other = false;
043            protected PluginCommand pluginCommand;
044            protected final AbstractAdminCmdPlugin plugin;
045            protected PermParent permParent;
046    
047            /**
048             * Constructor of CoreCommand
049             * 
050             * @param name
051             *            name of the command (in the plugin.yml)
052             * @param perm
053             *            permission needed by the player to execute the command
054             */
055            public CoreCommand(final String name, final String perm) {
056                    this.permNode = perm;
057                    this.cmdName = name;
058                    this.plugin = ACPluginManager.getCorePlugin();
059            }
060    
061            public CoreCommand() {
062                    this.plugin = ACPluginManager.getCorePlugin();
063            }
064    
065            /**
066             * Constructor of CoreCommand
067             * 
068             * @param name
069             *            name of the command (in the plugin.yml)
070             * @param perm
071             *            permission needed by the player to execute the command
072             * @param plugin
073             *            name of the AdminCmd plugin that the command belong to.
074             */
075            public CoreCommand(final String name, final String perm, final String plugin) {
076                    this.permNode = perm;
077                    this.cmdName = name;
078                    this.plugin = ACPluginManager.getPluginInstance(plugin);
079            }
080    
081            /**
082             * Constructor of CoreCommand
083             * 
084             * @param name
085             *            name of the command (in the plugin.yml)
086             * @param perm
087             *            permission needed by the player to execute the command
088             * @param plugin
089             *            name of the AdminCmd plugin that the command belong to.
090             * @param parent
091             *            PermParent used to register the permission of the command
092             */
093            public CoreCommand(final String name, final String perm, final String plugin,
094                            final PermParent parent) {
095                    this.permNode = perm;
096                    this.cmdName = name;
097                    this.plugin = ACPluginManager.getPluginInstance(plugin);
098                    this.permParent = parent;
099            }
100    
101            /**
102             * Execute the command
103             * 
104             * @param sender
105             *            sender of the command
106             * @param args
107             *            arguments to be processed by the command
108             * 
109             * @throws PlayerNotFound
110             *             the target player of the command is not found
111             */
112            public abstract void execute(CommandSender sender, CommandArgs args) throws PlayerNotFound;
113    
114            /**
115             * Check if the command can be executed
116             * 
117             * @param args
118             * @return
119             */
120            public abstract boolean argsCheck(String... args);
121    
122            /**
123             * Check if the sender have the permission for the command.
124             * 
125             * @param sender
126             * @return
127             */
128            public boolean permissionCheck(final CommandSender sender) {
129                    if (permNode != null && !permNode.isEmpty())
130                            return PermissionManager.hasPerm(sender, bukkitPerm);
131                    else
132                            return true;
133            }
134    
135            /**
136             * @return the cmdName
137             */
138            public String getCmdName() {
139                    return cmdName;
140            }
141    
142            /**
143             * @return the permNode
144             */
145            public String getPermNode() {
146                    return permNode;
147            }
148    
149            /**
150             * Register the bukkit Permission
151             */
152            public void registerBukkitPerm() {
153                    if (permNode != null && !permNode.isEmpty()) {
154                            if (permParent != null) {
155                                    final PermChild child = new PermChild(permNode, bukkitDefault);
156                                    permParent.addChild(child);
157                                    bukkitPerm = child.getBukkitPerm();
158                                    if (other)
159                                            permParent.addChild(new PermChild(permNode + ".other", bukkitDefault));
160    
161                                    return;
162                            }
163                            bukkitPerm = plugin.getPermissionLinker().addPermChild(permNode, bukkitDefault);
164                            if (other)
165                                    plugin.getPermissionLinker().addPermChild(permNode + ".other", bukkitDefault);
166                    }
167            }
168    
169            /**
170             * Check if the command is usable on other people than the CommandSender
171             * 
172             * @return
173             */
174            public boolean toOther() {
175                    return other;
176            }
177    
178            /**
179             * Initialize the bukkit plugin command
180             * 
181             * @param plugin
182             * @throws CommandNotFound
183             * @throws CommandAlreadyExist
184             */
185            public void initializeCommand() throws CommandNotFound, CommandAlreadyExist {
186                    if ((pluginCommand = plugin.getCommand(cmdName)) == null)
187                            throw new CommandNotFound(cmdName + " is not loaded in bukkit. Command deactivated");
188    
189                    if (pluginCommand.getAliases().isEmpty())
190                            throw new CommandAlreadyExist(cmdName
191                                            + " has all his alias already registered. Command deactivated");
192            }
193    
194            /**
195             * @return the pluginCommand
196             */
197            public PluginCommand getPluginCommand() {
198                    return pluginCommand;
199            }
200    
201            /**
202             * @return the plugin
203             */
204            public AbstractAdminCmdPlugin getPlugin() {
205                    return plugin;
206            }
207    
208            /*
209             * (non-Javadoc)
210             * 
211             * @see java.lang.Object#toString()
212             */
213            @Override
214            public String toString() {
215                    return "CoreCommand [permNode=" + permNode + ", cmdName=" + cmdName + ", plugin=" + plugin
216                                    + ", permParent=" + permParent + "]";
217            }
218    
219    }
