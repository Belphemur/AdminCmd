017package be.Balor.Manager.Commands.Server;
018
019import java.util.ArrayList;
020import java.util.HashMap;
021
022import org.bukkit.Material;
023import org.bukkit.command.CommandSender;
024
025import be.Balor.Manager.Commands.CommandArgs;
026import be.Balor.Manager.Exceptions.PlayerNotFound;
027import be.Balor.Manager.Permissions.ActionNotPermitedException;
028import be.Balor.Tools.MaterialContainer;
029import be.Balor.Tools.Utils;
030import be.Balor.bukkit.AdminCmd.ACHelper;
031
032/**
033 * @author Balor (aka Antoine Aflalo)
034 * 
035 */
036public class ReplaceBlock extends ServerCommand {
037
038        /**
039         * 
040         */
041        public ReplaceBlock() {
042                permNode = "admincmd.server.replace";
043                cmdName = "bal_replace";
044        }
045
046        /*
047         * (non-Javadoc)
048         * 
049         * @see
050         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
051         * java.lang.String[])
052         */
053        @Override
054        public void execute(final CommandSender sender, final CommandArgs args)
055                        throws ActionNotPermitedException, PlayerNotFound {
056                final MaterialContainer mc = ACHelper.getInstance().checkMaterial(
057                                sender, args.getString(0));
058                if (mc.isNull()) {
059                        return;
060                }
061                final ArrayList<Material> mats = new ArrayList<Material>();
062                mats.add(mc.getMaterial());
063                if (mc.getMaterial().equals(Material.LAVA)) {
064                        mats.add(Material.STATIONARY_LAVA);
065                } else if (mc.getMaterial().equals(Material.WATER)) {
066                        mats.add(Material.STATIONARY_WATER);
067                }
068                final Integer count = Utils.replaceBlockByAir(sender, args, mats, 10);
069                if (count == null) {
070                        return;
071                }
072                final HashMap<String, String> replace = new HashMap<String, String>();
073                replace.put("nb", String.valueOf(count));
074                replace.put("mat", mc.getMaterial().toString());
075                Utils.sI18n(sender, "replaced", replace);
076        }
077
078        /*
079         * (non-Javadoc)
080         * 
081         * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
082         */
083        @Override
084        public boolean argsCheck(final String... args) {
085                return args != null && args.length >= 1;
086        }
087
088}
