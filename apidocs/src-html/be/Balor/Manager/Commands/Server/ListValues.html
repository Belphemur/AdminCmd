017    package be.Balor.Manager.Commands.Server;
018    
019    import java.util.Arrays;
020    import java.util.List;
021    
022    import org.bukkit.ChatColor;
023    import org.bukkit.command.CommandSender;
024    
025    import be.Balor.Manager.Commands.CommandArgs;
026    import be.Balor.Player.ACPlayer;
027    import be.Balor.Tools.Type;
028    import be.Balor.Tools.Utils;
029    
030    /**
031     * @author Balor (aka Antoine Aflalo)
032     * 
033     */
034    public class ListValues extends ServerCommand {
035    
036            /**
037             * 
038             */
039            public ListValues() {
040                    permNode = "admincmd.server.list";
041                    cmdName = "bal_list";
042            }
043    
044            /*
045             * (non-Javadoc)
046             * 
047             * @see be.Balor.Manager.ACCommand#execute(org.bukkit.command.CommandSender,
048             * java.lang.String[])
049             */
050            @Override
051            public void execute(final CommandSender sender, final CommandArgs args) {
052                    if (args.length == 0) {
053                            sender.sendMessage(ChatColor.DARK_AQUA + "Possibles Types :");
054                            sender.sendMessage(Arrays.toString(Type.values()));
055                            return;
056                    }
057                    String arg = "";
058                    for (final String str : args)
059                            arg += str + " ";
060                    arg = arg.trim();
061                    if (Type.matchType(arg) == null) {
062                            Utils.sI18n(sender, "emptyList");
063                            return;
064                    }
065                    final List<ACPlayer> list = ACPlayer.getPlayers(arg);
066                    if (list != null) {
067                            sender.sendMessage(ChatColor.AQUA + Type.matchType(arg).display() + ChatColor.WHITE
068                                            + " (" + list.size() + ") " + ChatColor.AQUA + ":");
069                            String buffer = "";
070                            for (final ACPlayer value : list)
071                                    buffer += value.getName() + ", ";
072                            if (!buffer.equals("")) {
073                                    if (buffer.endsWith(", "))
074                                            buffer = buffer.substring(0, buffer.lastIndexOf(","));
075                                    sender.sendMessage(buffer);
076                            }
077                    } else
078                            Utils.sI18n(sender, "emptyList");
079    
080            }
081    
082            /*
083             * (non-Javadoc)
084             * 
085             * @see be.Balor.Manager.ACCommand#argsCheck(java.lang.String[])
086             */
087            @Override
088            public boolean argsCheck(final String... args) {
089                    return args != null;
090            }
091    
092    }
