017    package be.Balor.Manager.Commands.Mob;
018    
019    import java.util.HashMap;
020    
021    import org.bukkit.Material;
022    import org.bukkit.block.Block;
023    import org.bukkit.block.CreatureSpawner;
024    import org.bukkit.command.CommandSender;
025    import org.bukkit.entity.EntityType;
026    import org.bukkit.entity.Player;
027    
028    import be.Balor.Manager.Commands.CommandArgs;
029    import be.Balor.Tools.Utils;
030    import be.Balor.bukkit.AdminCmd.ACPluginManager;
031    
032    /**
033     * @author Lathanael (aka Philippe Leipold)
034     * 
035     */
036    public class ChangeMobSpawner extends MobCommand {
037    
038            /**
039             *
040             */
041            public ChangeMobSpawner() {
042                    permNode = "admincmd.mob.spawner";
043                    cmdName = "bal_changespawner";
044            }
045    
046            /*
047             * (non-Javadoc)
048             * 
049             * @see
050             * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
051             * java.lang.String[])
052             */
053            @Override
054            public void execute(final CommandSender sender, final CommandArgs args) {
055                    if (Utils.isPlayer(sender)) {
056                            final HashMap<String, String> replace = new HashMap<String, String>();
057                            final Player player = (Player) sender;
058                            final Block block = player.getTargetBlock(null, 100);
059                            if (block.getTypeId() != Material.MOB_SPAWNER.getId()) {
060                                    player.sendMessage("Not a Mob Spawner");
061                                    return;
062                            }
063                            final CreatureSpawner spawner = (CreatureSpawner) block.getState();
064                            if (args.hasFlag('m')) {
065                                    final String name = args.getString(0);
066                                    if (name == null) {
067                                            return;
068                                    }
069                                    final EntityType type = EntityType.fromName(name);
070                                    if (type == null) {
071                                            replace.put("mob", args.getString(0));
072                                            Utils.sI18n(sender, "errorMob", replace);
073                                    }
074    
075                                    ACPluginManager.scheduleSyncTask(new Runnable() {
076                                            @Override
077                                            public void run() {
078                                                    spawner.setCreatureTypeByName(type.getName());
079    
080                                            }
081                                    });
082                                    replace.put("type", args.getString(0));
083                                    Utils.sI18n(sender, "spawnerSetType", replace);
084                            } else if (args.hasFlag('d')) {
085                                    int delay;
086                                    try {
087                                            delay = args.getInt(0);
088                                    } catch (final Exception e) {
089                                            Utils.sI18n(sender, "spawnerNaN");
090                                            return;
091                                    }
092                                    final int fDelay = delay;
093                                    ACPluginManager.scheduleSyncTask(new Runnable() {
094                                            @Override
095                                            public void run() {
096                                                    spawner.setDelay(fDelay);
097    
098                                            }
099                                    });
100    
101                                    replace.put("delay", String.valueOf(args.getInt(0)));
102                                    Utils.sI18n(sender, "spawnerSetDelay", replace);
103                            } else if (args.hasFlag('g')) {
104                                    final int delay = spawner.getDelay();
105                                    final String type = spawner.getCreatureTypeName();
106                                    replace.put("mob", type);
107                                    replace.put("delay", String.valueOf(delay));
108                                    Utils.sI18n(sender, "spawnerGetData", replace);
109                            }
110                    }
111            }
112    
113            /*
114             * (non-Javadoc)
115             * 
116             * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
117             */
118            @Override
119            public boolean argsCheck(final String... args) {
120                    return args != null;
121            }
122    }
