017package be.Balor.Manager.Commands.Mob;

019import java.util.HashMap;
020
021import org.bukkit.Location;
022import org.bukkit.command.CommandSender;
023import org.bukkit.entity.EntityType;
024import org.bukkit.entity.Player;
025
026import be.Balor.Manager.Commands.CommandArgs;
027import be.Balor.Manager.Exceptions.PlayerNotFound;
028import be.Balor.Manager.Permissions.ActionNotPermitedException;
029import be.Balor.Tools.Utils;
030import be.Balor.bukkit.AdminCmd.ACPluginManager;
031
032/**
033 * @author Balor (aka Antoine Aflalo)
034 * 
035 */
036public class SpawnMob extends MobCommand {
037
038        /**
039         *
040         */
041        public SpawnMob() {
042                permNode = "admincmd.mob.spawn";
043                cmdName = "bal_mob";
044                other = true;
045        }
046
047        /*
048         * (non-Javadoc)
049         * 
050         * @see
051         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
052         * java.lang.String[])
053         */
054        @Override
055        public void execute(final CommandSender sender, final CommandArgs args)
056                        throws ActionNotPermitedException, PlayerNotFound {
057                if (Utils.isPlayer(sender)) {
058                        final HashMap<String, String> replace = new HashMap<String, String>();
059                        final String name = args.getString(0);
060                        replace.put("mob", name);
061                        int nbTaped;
062                        int distance = 0;
063                        try {
064                                nbTaped = args.getInt(1);
065                        } catch (final Exception e) {
066                                nbTaped = 1;
067                        }
068                        try {
069                                distance = args.getInt(2);
070                        } catch (final Exception e) {
071                                distance = 0;
072                        }
073
074                        Player temp = Utils.getUser(sender, args, permNode, 3, false);
075                        if (temp == null) {
076                                temp = ((Player) sender);
077                        }
078
079                        final Player player = temp;
080                        Location loc;
081                        if (distance == 0 && player.equals(sender)) {
082                                loc = player.getTargetBlock(null, 100).getLocation()
083                                                .add(0, 1, 0);
084                        } else if (distance == 0) {
085                                final Location playerLoc = player.getLocation();
086                                loc = playerLoc.add(
087                                                playerLoc
088                                                                .getDirection()
089                                                                .normalize()
090                                                                .multiply(2)
091                                                                .toLocation(player.getWorld(),
092                                                                                playerLoc.getYaw(),
093                                                                                playerLoc.getPitch())).add(0, 1D, 0);
094                        } else {
095                                final Location playerLoc = player.getLocation();
096                                loc = playerLoc.add(
097                                                playerLoc
098                                                                .getDirection()
099                                                                .normalize()
100                                                                .multiply(distance)
101                                                                .toLocation(player.getWorld(),
102                                                                                playerLoc.getYaw(),
103                                                                                playerLoc.getPitch())).add(0, 1D, 0);
104                        }
105                        EntityType ct = null;
106                        if (name.contains(":")) {
107                                final String[] creatures = name.split(":");
108                                // TODO: Remove this if-case as soon as bukkit fixes the
109                                // spelling error
110                                if (creatures[0].equalsIgnoreCase("Ocelot")) {
111                                        creatures[0] = "Ozelot";
112                                }
113                                ct = EntityType.fromName(creatures[0]);
114                                // TODO: Remove this if-case as soon as bukkit fixes the
115                                // spelling error
116                                if (creatures[1].equalsIgnoreCase("Ocelot")) {
117                                        creatures[1] = "Ozelot";
118                                }
119                                final EntityType ct2 = EntityType.fromName(creatures[1]);
120                                if (ct == null) {
121                                        replace.put("mob", creatures[0]);
122                                        Utils.sI18n(sender, "errorMob", replace);
123                                }
124                                if (ct2 == null) {
125                                        replace.put("mob", creatures[1]);
126                                        Utils.sI18n(sender, "errorMob", replace);
127                                }
128                                if (ct == null || ct2 == null) {
129                                        return;
130                                }
131                                ACPluginManager.scheduleSyncTask(new PassengerMob(loc, nbTaped,
132                                                ct, ct2, player, sender));
133                        } else {
134                                ct = EntityType.fromName(name);
135                                if (ct == null) {
136                                        Utils.sI18n(sender, "errorMob", replace);
137                                        return;
138                                }
139
140                                ACPluginManager.scheduleSyncTask(new NormalMob(loc, nbTaped,
141                                                ct, player, sender));
142                        }
143                }
144
145        }
146
147        /*
148         * (non-Javadoc)
149         * 
150         * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
151         */
152        @Override
153        public boolean argsCheck(final String... args) {
154                return args != null && args.length >= 1;
155        }
156
157        protected class NormalMob implements Runnable {
158                protected Location loc;
159                protected int nb;
160                protected EntityType ct;
161                protected Player player;
162                protected CommandSender sender;
163
164                /**
165                 *
166                 */
167                public NormalMob(final Location loc, final int nb, final EntityType ct,
168                                final Player player, final CommandSender sender) {
169                        this.loc = loc;
170                        this.nb = nb;
171                        this.ct = ct;
172                        this.player = player;
173                        this.sender = sender;
174                }
175
176                @Override
177                public void run() {
178                        final HashMap<String, String> replace = new HashMap<String, String>();
179                        replace.put("mob", ct.getName());
180                        for (int i = 0; i < nb; i++) {
181                                loc.getWorld().spawnEntity(loc, ct);
182                        }
183
184                        replace.put("nb", String.valueOf(nb));
185                        if (player.equals(sender)) {
186                                Utils.sI18n(player, "spawnMob", replace);
187                        } else {
188                                replace.put("player", Utils.getPlayerName((Player) sender));
189                                Utils.sI18n(player, "spawnMobOther", replace);
190                        }
191                }
192        }
193
194        protected class PassengerMob extends NormalMob {
195                protected EntityType passenger;
196
197                /**
198                 * @param loc
199                 * @param nb
200                 * @param ct
201                 * @param player
202                 */
203                public PassengerMob(final Location loc, final int nb,
204                                final EntityType mount, final EntityType passenger,
205                                final Player player, final CommandSender sender) {
206                        super(loc, nb, mount, player, sender);
207                        this.passenger = passenger;
208                }
209
210                @Override
211                public void run() {
212                        final HashMap<String, String> replace = new HashMap<String, String>();
213                        replace.put("mob", ct.getName() + "-" + passenger.getName());
214                        for (int i = 0; i < nb; i++) {
215                                loc.getWorld()
216                                                .spawnEntity(loc, ct)
217                                                .setPassenger(
218                                                                loc.getWorld().spawnEntity(loc, passenger));
219                        }
220
221                        replace.put("nb", String.valueOf(nb));
222                        if (player.equals(sender)) {
223                                Utils.sI18n(player, "spawnMob", replace);
224                        } else {
225                                replace.put("player", Utils.getPlayerName((Player) sender));
226                                Utils.sI18n(player, "spawnMobOther", replace);
227                        }
228                }
229
230        }
