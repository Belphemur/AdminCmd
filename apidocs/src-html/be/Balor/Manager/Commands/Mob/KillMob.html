060        /*
061         * (non-Javadoc)
062         * 
063         * @see
064         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
065         * java.lang.String[])
066         */
067        @Override
068        public void execute(final CommandSender sender, final CommandArgs args)
069                        throws ActionNotPermitedException, PlayerNotFound {
070                final HashMap<String, String> replace = new HashMap<String, String>();
071                final List<String> types = new ArrayList<String>();
072                Integer range = 25;
073                if (args.hasFlag('r')) {
074                        range = Integer.parseInt(args.getValueFlag('r'));
075                        range *= range;
076                }
077                final String worldString = args.getValueFlag('w');
078                if (args.length >= 1) {
079                        for (final String type : args) {
080                                types.add(type);
081                        }
082                } else {
083                        types.add("all");
084                }
085                replace.put("type", Joiner.on(", ").join(types));
086
087                final List<World> worldList = new ArrayList<World>();
088
089                if (Utils.isPlayer(sender, false)) {
090                        World w = ((Player) sender).getWorld();
091                        if (worldString != null) {
092                                w = getWorld(worldString);
093                        }
094                        worldList.add(w);
095                        replace.put("worlds", w.getName());
096                        Utils.sI18n(sender, "killMob", replace);
097                } else {
098                        if (worldString != null) {
099                                worldList.add(getWorld(worldString));
100                        } else {
101                                String worlds = "";
102                                for (final World world : sender.getServer().getWorlds()) {
103                                        worldList.add(world);
104                                        worlds += world.getName() + ", ";
105                                }
106                                if (!worlds.equals("")) {
107                                        if (worlds.endsWith(", ")) {
108                                                worlds = worlds.substring(0, worlds.lastIndexOf(","));
109                                        }
110                                        replace.put("worlds", worlds);
111                                        Utils.sI18n(sender, "killMob", replace);
112                                }
113                        }
114                }
115                final CommandSender finalSender = sender;
116                final Integer finalRange = range;
117                ACPluginManager.getScheduler().runTaskAsynchronously(
118                                ACPluginManager.getCorePlugin(), new Runnable() {
119                                        @Override
120                                        public void run() {
121                                                killMobs(worldList, types, finalSender, finalRange);
122                                        }
123                                });
124
125        }
126
127        private World getWorld(final String name) {
128                final World w = Bukkit.getWorld(name);
129                if (w == null) {
130                        throw new WorldNotLoaded("The World " + name + " is not loaded");
131                }
132                return w;
133        }

145        @SuppressWarnings("unchecked")
146        private void killMobs(final List<World> worlds, final List<String> types,
147                        final CommandSender sender, final Integer range) {
148                int mobKilled = 0;
149                final List<Class<? extends Entity>> classes = new ArrayList<Class<? extends Entity>>();
150                if (!types.contains("all")) {
151                        for (final String type : types) {
152                                if (type.equalsIgnoreCase("monsters")) {
153                                        classes.add(Monster.class);
154                                        classes.add(Snowman.class);
155                                        classes.add(ComplexLivingEntity.class);
156                                } else if (type.equalsIgnoreCase("animals")) {
157                                        classes.add(Animals.class);
158                                } else {
159                                        EntityType ct = null;
160                                        ct = EntityType.fromName(type);
161                                        if (ct == null) {
162                                                final HashMap<String, String> replace = new HashMap<String, String>();
163                                                replace.put("mob", type);
164                                                Utils.sI18n(sender, "errorMob", replace);
165                                                continue;
166                                        }
167                                        classes.add(ct.getEntityClass());
168                                }
169                        }
170                } else {
171                        classes.add(Monster.class);
172                        classes.add(Animals.class);
173                        classes.add(ComplexLivingEntity.class);
174                        classes.add(Snowman.class);
175                }
176
177                final Class<Entity>[] array = (Class<Entity>[]) Array.newInstance(
178                                Entity.class.getClass(), classes.size());
179                for (final World w : worlds) {
180                        for (final Entity m : w
181                                        .getEntitiesByClasses(classes.toArray(array))) {
182                                if (killEntity(m, sender, range)) {
183                                        mobKilled++;
184                                }
185                        }
186                }
187
188                Utils.sI18n(sender, "killedMobs", "nbKilled", String.valueOf(mobKilled));
189        }
190
191        private boolean killEntity(final Entity e, final CommandSender sender,
192                        final Integer range) {
193                if (!checkKillCondition(e, sender, range)) {
194                        return false;
195                }
196                final Object entity = MinecraftReflection.getHandle(e);
197                final MethodHandler die = new MethodHandler(entity.getClass(), "die");
198                die.invoke(entity);
199                return true;
200        }
201
202        /**
203         * Check if the entity can be killed with the giving conditions
204         * 
205         * @param toCheck
206         *            entity to be checked
207         * @param sender
208         *            sender of the command
209         * @param range
210         *            distance between the entity and the player accepted for a
211         *            kill.
212         * @return
213         */
214        private boolean checkKillCondition(final Entity toCheck,
215                        final CommandSender sender, final Integer range) {
216                boolean result = true;
217                if (range != null && Utils.isPlayer(sender, false)) {
218                        result = toCheck.getLocation().distanceSquared(
219                                        ((Player) sender).getLocation()) <= range;
220                }
221                return result;
222        }
