040    /**
041     * @author Balor (aka Antoine Aflalo)
042     * 
043     */
044    public class KillMob extends MobCommand {
045    
046            /**
047             * 
048             */
049            public KillMob() {
050                    permNode = "admincmd.mob.kill";
051                    cmdName = "bal_killmob";
052            }
053    
054            /*
055             * (non-Javadoc)
056             * 
057             * @see
058             * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
059             * java.lang.String[])
060             */
061            @Override
062            public void execute(final CommandSender sender, final CommandArgs args) {
063                    final HashMap<String, String> replace = new HashMap<String, String>();
064                    String type = "all";
065                    if (args.length >= 1)
066                            type = args.getString(0);
067                    replace.put("type", type);
068                    final String worldString = args.getValueFlag('w');
069                    final List<World> worldList = new ArrayList<World>();
070    
071                    if (Utils.isPlayer(sender, false)) {
072                            World w = ((Player) sender).getWorld();
073                            if (worldString != null)
074                                    w = getWorld(worldString);
075                            worldList.add(w);
076                            replace.put("worlds", w.getName());
077                            Utils.sI18n(sender, "killMob", replace);
078                    } else {
079                            if (worldString != null)
080                                    worldList.add(getWorld(worldString));
081                            else {
082                                    String worlds = "";
083                                    for (final World world : sender.getServer().getWorlds()) {
084                                            worldList.add(world);
085                                            worlds += world.getName() + ", ";
086                                    }
087                                    if (!worlds.equals("")) {
088                                            if (worlds.endsWith(", "))
089                                                    worlds = worlds.substring(0, worlds.lastIndexOf(","));
090                                            replace.put("worlds", worlds);
091                                            Utils.sI18n(sender, "killMob", replace);
092                                    }
093                            }
094                    }
095                    final String finalType = type;
096                    final CommandSender finalSender = sender;
097                    ACPluginManager.getScheduler().scheduleAsyncDelayedTask(ACPluginManager.getCorePlugin(),
098                                    new Runnable() {
099                                            @Override
100                                            public void run() {
101                                                    killMobs(worldList, finalType, finalSender);
102                                            }
103                                    });
104    
105            }
106    
107            private World getWorld(final String name) {
108                    final World w = Bukkit.getWorld(name);
109                    if (w == null)
110                            throw new WorldNotLoaded("The World " + name + " is not loaded");
111                    return w;
112            }
113    
114            /*
115             * (non-Javadoc)
116             * 
117             * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
118             */
119            @Override
120            public boolean argsCheck(final String... args) {
121                    return args != null;
122            }
123    
124            private void killMobs(final List<World> worlds, final String type, final CommandSender sender) {
125                    int mobKilled = 0;
126                    if (type.equalsIgnoreCase("all")) {
127                            for (final World w : worlds)
128                                    for (final LivingEntity m : w.getLivingEntities()) {
129                                            if (m instanceof HumanEntity) {
130                                                    continue;
131                                            }
132                                            final Entity entity = ((CraftLivingEntity) m).getHandle();
133                                            entity.die();
134                                            mobKilled++;
135    
136                                    }
137                    } else {
138                            EntityType ct = null;
139                            ct = EntityType.fromName(type);
140                            if (ct == null) {
141                                    final HashMap<String, String> replace = new HashMap<String, String>();
142                                    replace.put("mob", type);
143                                    Utils.sI18n(sender, "errorMob", replace);
144                                    return;
145                            }
146                            for (final World w : worlds)
147                                    for (final org.bukkit.entity.Entity m : w.getEntitiesByClasses(ct.getEntityClass())) {
148                                            final Entity entity = ((CraftEntity) m).getHandle();
149                                            entity.die();
150                                            mobKilled++;
151                                    }
152                    }
153                    Utils.sI18n(sender, "killedMobs", "nbKilled", String.valueOf(mobKilled));
154            }
155    }
