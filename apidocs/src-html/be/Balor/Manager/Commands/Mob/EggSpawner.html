028    import be.Balor.Tools.Type;
029    import be.Balor.Tools.Utils;
030    import be.Balor.Tools.Debug.ACLogger;
031    import be.Balor.Tools.Egg.EggPermissionManager;
032    import be.Balor.Tools.Egg.EggType;
033    import be.Balor.Tools.Egg.Exceptions.DontHaveThePermissionException;
034    import be.Balor.Tools.Egg.Exceptions.ExceptionType;
035    import be.Balor.Tools.Egg.Exceptions.ParameterMissingException;
036    import be.Balor.Tools.Egg.Exceptions.ProcessingArgsException;
037    import be.Balor.Tools.Egg.Types.NormalEgg;
038    
039    import com.google.common.base.Joiner;
040    
041    /**
042     * @author Balor (aka Antoine Aflalo)
043     * 
044     */
045    public class EggSpawner extends MobCommand {
046    
047            /**
048             *
049             */
050            public EggSpawner() {
051                    this.cmdName = "bal_egg";
052            }
053    
054            /*
055             * (non-Javadoc)
056             * 
057             * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
058             * CommandSender, be.Balor.Manager.Commands.CommandArgs)
059             */
060            @Override
061            public void execute(final CommandSender sender, final CommandArgs args) {
062                    if (!Utils.isPlayer(sender))
063                            return;
064                    final Player player = (Player) sender;
065                    final ACPlayer acp = ACPlayer.getPlayer(player);
066                    EggType<?> egg = null;
067                    try {
068                            egg = EggType.createEggType(player, args);
069                    } catch (final ParameterMissingException e) {
070                            if (e.getParam() == 'E') {
071                                    final String list = Joiner.on(", ").skipNulls()
072                                                    .join(EggPermissionManager.INSTANCE.getEggTypeNames(player));
073                                    sender.sendMessage(e.getMessage());
074                                    sender.sendMessage(ChatColor.GOLD + "Egg List : ");
075                                    sender.sendMessage(ChatColor.YELLOW + list);
076                            } else {
077                                    Utils.sI18n(sender, "paramMissing", "param", String.valueOf(e.getParam()));
078                                    sender.sendMessage(e.getMessage());
079                            }
080                            return;
081                    } catch (final ProcessingArgsException e) {
082                            if (e.getType().equals(ExceptionType.NO_CLASS))
083                                    Utils.sI18n(sender, "eggDontExists", "egg", e.getMessage());
084                            else if (e.getType().equals(ExceptionType.DONT_EXISTS))
085                                    Utils.sI18n(sender, "entityDontExists", "entity", e.getMessage());
086                            else if (e.getType().equals(ExceptionType.CUSTOM)) {
087                                    final Map<String, String> replace = new HashMap<String, String>();
088                                    replace.put("egg", args.getValueFlag('E'));
089                                    replace.put("error", e.getMessage());
090                                    Utils.sI18n(sender, "eggCustomError", replace);
091                            } else
092                                    ACLogger.severe("Problem with an Egg Type : " + e.getMessage(), e);
093                            return;
094                    } catch (final DontHaveThePermissionException e) {
095                            sender.sendMessage(e.getMessage());
096                            return;
097                    } catch (final NullPointerException e) {
098                            if (args.hasFlag('E')) {
099                                    final String list = Joiner.on(", ").skipNulls()
100                                                    .join(EggPermissionManager.INSTANCE.getEggTypeNames(player));
101                                    sender.sendMessage(e.getMessage());
102                                    sender.sendMessage(ChatColor.GOLD + "Egg List : ");
103                                    sender.sendMessage(ChatColor.YELLOW + list);
104                            } else {
105                                    Utils.sI18n(sender, "eggNoParamGiven");
106                                    sender.sendMessage(e.getMessage());
107                            }
108                            return;
109                    }
110                    if (egg instanceof NormalEgg) {
111                            acp.removePower(Type.EGG);
112                            Utils.sI18n(sender, "eggNormal");
113                    } else {
114                            acp.setPower(Type.EGG, egg);
115                            Utils.sI18n(sender, "eggEnabled", "egg", egg.toString());
116                    }
117    
118            }
119    
120            /*
121             * (non-Javadoc)
122             * 
123             * @see
124             * be.Balor.Manager.Commands.CoreCommand#permissionCheck(org.bukkit.command
125             * .CommandSender)
126             */
127            @Override
128            public boolean permissionCheck(final CommandSender sender) {
129                    return true;
130            }
131    
132            /*
133             * (non-Javadoc)
134             * 
135             * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
136             */
137            @Override
138            public boolean argsCheck(final String... args) {
139                    return args != null;
140            }
141    
142    }
