017package be.Balor.Manager.Commands.Mob;
018
019import java.util.HashMap;
020import java.util.Map;
021
022import org.bukkit.ChatColor;
023import org.bukkit.command.CommandSender;
024import org.bukkit.entity.Player;
025
026import be.Balor.Manager.Commands.CommandArgs;
027import be.Balor.Manager.Exceptions.PlayerNotFound;
028import be.Balor.Manager.Permissions.ActionNotPermitedException;
029import be.Balor.Player.ACPlayer;
030import be.Balor.Tools.Type;
031import be.Balor.Tools.Utils;
032import be.Balor.Tools.Debug.ACLogger;
033import be.Balor.Tools.Egg.EggPermissionManager;
034import be.Balor.Tools.Egg.EggType;
035import be.Balor.Tools.Egg.Exceptions.DontHaveThePermissionException;
036import be.Balor.Tools.Egg.Exceptions.ExceptionType;
037import be.Balor.Tools.Egg.Exceptions.ParameterMissingException;
038import be.Balor.Tools.Egg.Exceptions.ProcessingArgsException;
039import be.Balor.Tools.Egg.Types.NormalEgg;
040
041import com.google.common.base.Joiner;
042
043/**
044 * @author Balor (aka Antoine Aflalo)
045 * 
046 */
047public class EggSpawner extends MobCommand {
048
049        /**
050         *
051         */
052        public EggSpawner() {
053                this.cmdName = "bal_egg";
054        }
055
056        /*
057         * (non-Javadoc)
058         * 
059         * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
060         * CommandSender, be.Balor.Manager.Commands.CommandArgs)
061         */
062        @Override
063        public void execute(final CommandSender sender, final CommandArgs args)
064                        throws ActionNotPermitedException, PlayerNotFound {
065                if (!Utils.isPlayer(sender)) {
066                        return;
067                }
068                final Player player = (Player) sender;
069                final ACPlayer acp = ACPlayer.getPlayer(player);
070                EggType<?> egg = null;
071                try {
072                        egg = EggType.createEggType(player, args);
073                } catch (final ParameterMissingException e) {
074                        if (e.getParam() == 'e') {
075                                final String list = Joiner
076                                                .on(", ")
077                                                .skipNulls()
078                                                .join(EggPermissionManager.INSTANCE
079                                                                .getEggTypeNames(player));
080                                sender.sendMessage(e.getMessage());
081                                sender.sendMessage(ChatColor.GOLD + "Egg List : ");
082                                sender.sendMessage(ChatColor.YELLOW + list);
083                        } else {
084                                Utils.sI18n(sender, "paramMissing", "param",
085                                                String.valueOf(e.getParam()));
086                                sender.sendMessage(e.getMessage());
087                        }
088                        return;
089                } catch (final ProcessingArgsException e) {
090                        if (e.getType().equals(ExceptionType.NO_CLASS)) {
091                                Utils.sI18n(sender, "eggDontExists", "egg", e.getMessage());
092                        } else if (e.getType().equals(ExceptionType.DONT_EXISTS)) {
093                                Utils.sI18n(sender, "entityDontExists", "entity",
094                                                e.getMessage());
095                        } else if (e.getType().equals(ExceptionType.CUSTOM)) {
096                                final Map<String, String> replace = new HashMap<String, String>();
097                                replace.put("egg", args.getValueFlag('t'));
098                                replace.put("error", e.getMessage());
099                                Utils.sI18n(sender, "eggCustomError", replace);
100                        } else {
101                                ACLogger.severe("Problem with an Egg Type : " + e.getMessage(),
102                                                e);
103                        }
104                        return;
105                } catch (final DontHaveThePermissionException e) {
106                        sender.sendMessage(e.getMessage());
107                        return;
108                } catch (final NullPointerException e) {
109                        if (args.hasFlag('e')) {
110                                final String list = Joiner
111                                                .on(", ")
112                                                .skipNulls()
113                                                .join(EggPermissionManager.INSTANCE
114                                                                .getEggTypeNames(player));
115                                sender.sendMessage(ChatColor.GOLD + "Egg List : ");
116                                sender.sendMessage(ChatColor.YELLOW + list);
117                        } else {
118                                Utils.sI18n(sender, "eggNoParamGiven");
119                                sender.sendMessage(e.getMessage());
120                        }
121                        return;
122                }
123                if (egg instanceof NormalEgg) {
124                        acp.removePower(Type.EGG);
125                        Utils.sI18n(sender, "eggNormal");
126                } else {
127                        acp.setPower(Type.EGG, egg);
128                        Utils.sI18n(sender, "eggEnabled", "egg", egg.toString());
129                }
130
131        }
132
133        /*
134         * (non-Javadoc)
135         * 
136         * @see
137         * be.Balor.Manager.Commands.CoreCommand#permissionCheck(org.bukkit.command
138         * .CommandSender)
139         */
140        @Override
141        public boolean permissionCheck(final CommandSender sender) {
142                return true;
143        }
144
145        /*
146         * (non-Javadoc)
147         * 
148         * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
149         */
150        @Override
151        public boolean argsCheck(final String... args) {
152                return args != null;
153        }
154
155}
