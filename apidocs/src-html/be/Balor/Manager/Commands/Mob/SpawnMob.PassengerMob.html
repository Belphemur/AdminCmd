018    
019    import java.util.HashMap;
020    
021    import org.bukkit.Location;
022    import org.bukkit.command.CommandSender;
023    import org.bukkit.entity.EntityType;
024    import org.bukkit.entity.Player;
025

030    /**
031     * @author Balor (aka Antoine Aflalo)
032     * 
033     */
034    public class SpawnMob extends MobCommand {
035    
036            /**
037             *
038             */
039            public SpawnMob() {
040                    permNode = "admincmd.mob.spawn";
041                    cmdName = "bal_mob";
042                    other = true;
043            }
044    
045            /*
046             * (non-Javadoc)
047             * 
048             * @see
049             * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
050             * java.lang.String[])
051             */
052            @Override
053            public void execute(final CommandSender sender, final CommandArgs args) {
054                    if (Utils.isPlayer(sender)) {
055                            final HashMap<String, String> replace = new HashMap<String, String>();
056                            final String name = args.getString(0);
057                            replace.put("mob", name);
058                            int nbTaped;
059                            int distance = 0;
060                            try {
061                                    nbTaped = args.getInt(1);
062                            } catch (final Exception e) {
063                                    nbTaped = 1;
064                            }
065                            try {
066                                    distance = args.getInt(2);
067                            } catch (final Exception e) {
068                                    distance = 0;
069                            }
070    
071                            Player temp = Utils.getUser(sender, args, permNode, 3, false);
072                            if (temp == null)
073                                    temp = ((Player) sender);
074    
075                            final Player player = temp;
076                            Location loc;
077                            if (distance == 0 && player.equals(sender))
078                                    loc = player.getTargetBlock(null, 100).getLocation().add(0, 1, 0);
079                            else if (distance == 0) {
080                                    final Location playerLoc = player.getLocation();
081                                    loc = playerLoc.add(
082                                                    playerLoc
083                                                                    .getDirection()
084                                                                    .normalize()
085                                                                    .multiply(2)
086                                                                    .toLocation(player.getWorld(), playerLoc.getYaw(),
087                                                                                    playerLoc.getPitch())).add(0, 1D, 0);
088                            } else {
089                                    final Location playerLoc = player.getLocation();
090                                    loc = playerLoc.add(
091                                                    playerLoc
092                                                                    .getDirection()
093                                                                    .normalize()
094                                                                    .multiply(distance)
095                                                                    .toLocation(player.getWorld(), playerLoc.getYaw(),
096                                                                                    playerLoc.getPitch())).add(0, 1D, 0);
097                            }
098                            EntityType ct = null;
099                            if (name.contains(":")) {
100                                    final String[] creatures = name.split(":");
101                                    ct = EntityType.fromName(creatures[0]);
102                                    final EntityType ct2 = EntityType.fromName(creatures[1]);
103                                    if (ct == null) {
104                                            replace.put("mob", creatures[0]);
105                                            Utils.sI18n(sender, "errorMob", replace);
106                                    }
107                                    if (ct2 == null) {
108                                            replace.put("mob", creatures[1]);
109                                            Utils.sI18n(sender, "errorMob", replace);
110                                    }
111                                    if (ct == null || ct2 == null)
112                                            return;
113                                    ACPluginManager.scheduleSyncTask(new PassengerMob(loc, nbTaped, ct, ct2, player,
114                                                    sender));
115                            } else {
116                                    ct = EntityType.fromName(name);
117                                    if (ct == null) {
118                                            Utils.sI18n(sender, "errorMob", replace);
119                                            return;
120                                    }
121    
122                                    ACPluginManager.scheduleSyncTask(new NormalMob(loc, nbTaped, ct, player, sender));
123                            }
124                    }
125    
126            }
127    
128            /*
129             * (non-Javadoc)
130             * 
131             * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
132             */
133            @Override
134            public boolean argsCheck(final String... args) {
135                    return args != null && args.length >= 1;
136            }
137    
138            protected class NormalMob implements Runnable {
139                    protected Location loc;
140                    protected int nb;
141                    protected EntityType ct;
142                    protected Player player;
143                    protected CommandSender sender;
144    
145                    /**
146                     *
147                     */
148                    public NormalMob(final Location loc, final int nb, final EntityType ct,
149                                    final Player player, final CommandSender sender) {
150                            this.loc = loc;
151                            this.nb = nb;
152                            this.ct = ct;
153                            this.player = player;
154                            this.sender = sender;
155                    }
156    
157                    @Override
158                    public void run() {
159                            final HashMap<String, String> replace = new HashMap<String, String>();
160                            replace.put("mob", ct.getName());
161                            for (int i = 0; i < nb; i++) {
162                                    loc.getWorld().spawnCreature(loc, ct);
163                            }
164                            replace.put("nb", String.valueOf(nb));
165                            if (player.equals(sender))
166                                    Utils.sI18n(player, "spawnMob", replace);
167                            else {
168                                    replace.put("player", Utils.getPlayerName((Player) sender));
169                                    Utils.sI18n(player, "spawnMobOther", replace);
170                            }
171                    }
172            }
173    
174            protected class PassengerMob extends NormalMob {
175                    protected EntityType passenger;
176    
177                    /**
178                     * @param loc
179                     * @param nb
180                     * @param ct
181                     * @param player
182                     */
183                    public PassengerMob(final Location loc, final int nb, final EntityType mount,
184                                    final EntityType passenger, final Player player, final CommandSender sender) {
185                            super(loc, nb, mount, player, sender);
186                            this.passenger = passenger;
187                    }
188    
189                    @Override
190                    public void run() {
191                            final HashMap<String, String> replace = new HashMap<String, String>();
192                            replace.put("mob", ct.getName() + "-" + passenger.getName());
193                            for (int i = 0; i < nb; i++) {
194                                    loc.getWorld().spawnCreature(loc, ct)
195                                                    .setPassenger(loc.getWorld().spawnCreature(loc, passenger));
196                            }
197                            replace.put("nb", String.valueOf(nb));
198                            if (player.equals(sender))
199                                    Utils.sI18n(player, "spawnMob", replace);
200                            else {
201                                    replace.put("player", Utils.getPlayerName((Player) sender));
202                                    Utils.sI18n(player, "spawnMobOther", replace);
203                            }
204                    }
205    
206            }
