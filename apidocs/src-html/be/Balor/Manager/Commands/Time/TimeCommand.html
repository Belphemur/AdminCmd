017package be.Balor.Manager.Commands.Time;
018
019import java.util.HashMap;
020
021import org.bukkit.Bukkit;
022import org.bukkit.World;
023import org.bukkit.command.CommandSender;
024import org.bukkit.entity.Player;
025
026import be.Balor.Manager.Commands.CoreCommand;
027import be.Balor.Tools.Type;
028import be.Balor.Tools.Utils;
029import be.Balor.Tools.Threads.SetTimeTask;
030import be.Balor.World.ACWorld;
031import be.Balor.bukkit.AdminCmd.ACPluginManager;
032
033/**
034 * @author Balor (aka Antoine Aflalo)
035 * 
036 */
037public abstract class TimeCommand extends CoreCommand {
038        /**
039 * 
040 */
041        public TimeCommand() {
042                super();
043                this.permParent = plugin.getPermissionLinker().getPermParent(
044                                "admincmd.time.*");
045        }
046
047        /**
048         * @param string
049         * @param string2
050         */
051        public TimeCommand(final String cmd, final String permNode) {
052                super(cmd, permNode);
053                this.permParent = plugin.getPermissionLinker().getPermParent(
054                                "admincmd.time.*");
055        }
056
057        public static boolean timeSet(final CommandSender sender, final String time) {
058                return timeSet(sender, time, null);
059        }
060
061        // all functions return if they handled the command
062        // false then means to show the default handle
063        // ! make sure the player variable IS a player!
064        // set world time to a new value
065        public static boolean timeSet(final CommandSender sender,
066                        final String time, final String world) {
067                if (Utils.isPlayer(sender, false) && world == null) {
068                        final Player p = (Player) sender;
069                        setTime(sender, p.getWorld(), time);
070                } else if (world != null) {
071                        final World w = sender.getServer().getWorld(world);
072                        if (w == null) {
073                                final HashMap<String, String> replace = new HashMap<String, String>();
074                                replace.put("world", world);
075                                Utils.sI18n(sender, "worldNotFound", replace);
076                                return true;
077                        }
078                        setTime(sender, w, time);
079                } else {
080                        for (final World w : sender.getServer().getWorlds()) {
081                                setTime(sender, w, time);
082                        }
083                }
084                return true;
085
086        }
087
088        private static void setTime(final CommandSender sender, final World w,
089                        final String arg) {
090                final HashMap<String, String> replace = new HashMap<String, String>();
091                replace.put("type", arg);
092                replace.put("world", w.getName());
093                if (ACWorld.getWorld(w).getInformation(Type.TIME_FROZEN.toString())
094                                .isNull()) {
095                        if (arg.equalsIgnoreCase("pause")) {
096                                pauseTime(w);
097                        } else {
098                                final long newtime = calculNewTime(w, arg);
099                                ACPluginManager.scheduleSyncTask(new SetTimeTask(w, newtime));
100                        }
101                        Utils.sI18n(sender, "timeSet", replace);
102                } else if (arg.equalsIgnoreCase("unpause")) {
103                        unPauseTime(w);
104                        Utils.sI18n(sender, "timeSet", replace);
105                } else {
106                        Utils.sI18n(sender, "timePaused", "world", w.getName());
107                }
108
109        }
110
111        /**
112         * @param w
113         */
114        private static void pauseTime(final World w) {
115                ACWorld.getWorld(w).setInformation(Type.TIME_FROZEN.toString(),
116                                w.getTime());
117                ACPluginManager.scheduleSyncTask(new Runnable() {
118                        @Override
119                        public void run() {
120                                for (final Player p : Bukkit.getOnlinePlayers()) {
121                                        p.setPlayerTime(w.getTime(), false);
122                                }
123
124                        }
125                });
126        }
127
128        /**
129         * @param w
130         */
131        private static void unPauseTime(final World w) {
132                ACWorld.getWorld(w).removeInformation(Type.TIME_FROZEN.toString());
133                ACPluginManager.scheduleSyncTask(new Runnable() {
134
135                        @Override
136                        public void run() {
137                                for (final Player p : Bukkit.getOnlinePlayers()) {
138                                        p.resetPlayerTime();
139                                }
140
141                        }
142                });
143        }
144
145        /**
146         * Calcul the new time to set for the wanted world.
147         * 
148         * @param w
149         *            word
150         * @param arg
151         *            keyword or time to set
152         * @return the newtime
153         */
154        public static long calculNewTime(final World w, final String arg) {
155                final long curtime = w.getTime();
156                if (arg.equalsIgnoreCase("normal")) {
157                        return curtime;
158                }
159                long newtime = curtime - (curtime % 24000);
160                if (arg.equalsIgnoreCase("day")) {
161                        newtime += 0;
162                } else if (arg.equalsIgnoreCase("night")) {
163                        newtime += 14000;
164                } else if (arg.equalsIgnoreCase("dusk")) {
165                        newtime += 12500;
166                } else if (arg.equalsIgnoreCase("dawn")) {
167                        newtime += 23000;
168                } else {
169                        newtime = Long.parseLong(arg);
170
171                }
172                return newtime;
173        }
174}
