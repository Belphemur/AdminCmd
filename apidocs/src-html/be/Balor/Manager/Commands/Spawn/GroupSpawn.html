019package be.Balor.Manager.Commands.Spawn;
020
021import java.util.HashMap;
022
023import org.bukkit.ChatColor;
024import org.bukkit.Location;
025import org.bukkit.command.CommandSender;
026import org.bukkit.entity.Player;
027
028import be.Balor.Manager.Commands.CommandArgs;
029import be.Balor.Manager.Exceptions.PlayerNotFound;
030import be.Balor.Manager.Permissions.PermissionManager;
031import be.Balor.Tools.Utils;
032import be.Balor.Tools.Help.String.ACMinecraftFontWidthCalculator;
033import be.Balor.Tools.Threads.TeleportTask;
034import be.Balor.World.ACWorld;
035import be.Balor.bukkit.AdminCmd.ACHelper;
036import be.Balor.bukkit.AdminCmd.ACPluginManager;
037import be.Balor.bukkit.AdminCmd.LocaleHelper;
038
039/**
040 * @author Lathanael (aka Philippe Leipold)
041 * 
042 */
043public class GroupSpawn extends SpawnCommand {
044
045        /**
046         *
047         */
048        public GroupSpawn() {
049                permNode = "admincmd.spawn.gtp";
050                cmdName = "bal_groupspawn";
051        }
052
053        /*
054         * (non-Javadoc)
055         * 
056         * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
057         * CommandSender, be.Balor.Manager.Commands.CommandArgs)
058         */
059        @Override
060        public void execute(final CommandSender sender, final CommandArgs args)
061                        throws PlayerNotFound {
062                if (Utils.isPlayer(sender)) {
063                        final HashMap<String, String> replace = new HashMap<String, String>();
064                        final Player p = (Player) sender;
065                        if (args.hasFlag('l')) {
066                                p.sendMessage(ChatColor.GREEN + "Possible Group Names are:");
067                                String msg = "";
068                                for (final String gName : ACHelper.getInstance().getGroupList()) {
069                                        msg += gName + ", ";
070                                        if (msg.length() >= ACMinecraftFontWidthCalculator.chatwidth) {
071                                                sender.sendMessage(ChatColor.AQUA + msg);
072                                                msg = "";
073                                        }
074                                }
075                                if (!msg.equals("")) {
076                                        if (msg.endsWith(", ")) {
077                                                msg = msg.substring(0, msg.lastIndexOf(","));
078                                        }
079                                        sender.sendMessage(ChatColor.AQUA + msg);
080                                }
081                                return;
082                        }
083                        if (args.length < 1) {
084                                final ACWorld w = ACWorld.getWorld(p.getWorld());
085                                if (w == null) {
086                                        return;
087                                }
088                                Location loc = null;
089                                for (final String gName : ACHelper.getInstance().getGroupList()) {
090                                        final String lcGname = gName.toLowerCase();
091                                        if (PermissionManager.hasPerm(p, "admincmd.respawn."
092                                                        + lcGname)) {
093                                                loc = w.getGroupSpawn(lcGname);
094                                                replace.put("groupName", lcGname);
095                                                break;
096                                        }
097                                }
098                                if (loc != null) {
099                                        ACPluginManager.scheduleSyncTask(new TeleportTask(p, loc));
100                                        LocaleHelper.GROUP_SPAWN.sendLocale(p, replace);
101                                        return;
102                                } else {
103                                        LocaleHelper.NO_GROUP_SPAWN.sendLocale(p, replace);
104                                        return;
105                                }
106                        } else if (args.length == 1) {
107                                final String gName = args.getString(0).toLowerCase();
108                                if (!ACHelper.getInstance().getGroupList().contains(gName)) {
109                                        replace.put("groupName", gName);
110                                        LocaleHelper.NO_SUCH_GROUP.sendLocale(p, replace);
111                                        return;
112                                }
113                                final ACWorld w = ACWorld.getWorld(p.getWorld().getName());
114                                if (w == null) {
115                                        return;
116                                }
117                                Location loc = null;
118                                if (PermissionManager.hasPerm(p, "admincmd.respawn." + gName,
119                                                true)
120                                                || PermissionManager.hasPerm(p,
121                                                                "admincmd.respawn.admin")) {
122                                        loc = w.getGroupSpawn(gName);
123                                        replace.put("groupName", gName);
124                                        if (loc == null) {
125                                                LocaleHelper.NO_GROUP_SPAWN.sendLocale(p, replace);
126                                                return;
127                                        }
128                                        ACPluginManager.scheduleSyncTask(new TeleportTask(p, loc));
129                                        LocaleHelper.GROUP_SPAWN.sendLocale(p, replace);
130                                }
131                        } else if (args.length >= 2) {
132                                final String gName = args.getString(0).toLowerCase();
133                                final ACWorld w = ACWorld.getWorld(args.getString(1));
134                                if (!ACHelper.getInstance().getGroupList().contains(gName)) {
135                                        replace.put("groupName", gName);
136                                        LocaleHelper.NO_SUCH_GROUP.sendLocale(p, replace);
137                                        return;
138                                }
139                                if (w == null) {
140                                        return;
141                                }
142                                Location loc = null;
143                                if (PermissionManager.hasPerm(p, "admincmd.respawn." + gName,
144                                                true)
145                                                || PermissionManager.hasPerm(p,
146                                                                "admincmd.respawn.admin")) {
147                                        loc = w.getGroupSpawn(gName);
148                                        replace.put("groupName", gName);
149                                        if (loc == null) {
150                                                LocaleHelper.NO_GROUP_SPAWN.sendLocale(p, replace);
151                                                return;
152                                        }
153                                        ACPluginManager.scheduleSyncTask(new TeleportTask(p, loc));
154                                        LocaleHelper.GROUP_SPAWN.sendLocale(p, replace);
155                                }
156                        }
157                }
158        }
159
160        /*
161         * (non-Javadoc)
162         * 
163         * @see be.Balor.Manager.Commands.CoreCommand#argsCheck(java.lang.String[])
164         */
165        @Override
166        public boolean argsCheck(final String... args) {
167                return args != null;
168        }
169
170}
