017package be.Balor.Manager.Commands.Spawn;
018
019import static be.Balor.Tools.Utils.sendMessage;
020
021import java.util.HashMap;
022import java.util.Map;
023
024import org.bukkit.Bukkit;
025import org.bukkit.World;
026import org.bukkit.command.CommandSender;
027import org.bukkit.entity.Player;
028
029import be.Balor.Manager.Commands.CommandArgs;
030import be.Balor.Manager.Exceptions.PlayerNotFound;
031import be.Balor.Manager.Permissions.ActionNotPermitedException;
032import be.Balor.Manager.Permissions.PermissionManager;
033import be.Balor.Tools.SimplifiedLocation;
034import be.Balor.Tools.Utils;
035import be.Balor.World.ACWorld;
036import be.Balor.bukkit.AdminCmd.ACHelper;
037import be.Balor.bukkit.AdminCmd.ACPluginManager;
038import be.Balor.bukkit.AdminCmd.ConfigEnum;
039import be.Balor.bukkit.AdminCmd.LocaleHelper;
040
041/**
042 * @authors Balor, Lathanael
043 * 
044 */
045public class Spawn extends SpawnCommand {
046
047        /**
048         *
049         */
050        public Spawn() {
051                permNode = "admincmd.spawn.tp";
052                cmdName = "bal_spawn";
053        }
054
055        /*
056         * (non-Javadoc)
057         * 
058         * @see
059         * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
060         * java.lang.String[])
061         */
062        @Override
063        public void execute(final CommandSender sender, final CommandArgs args)
064                        throws ActionNotPermitedException, PlayerNotFound {
065                if (args.length >= 1 && Utils.isPlayer(sender, true)) {
066                        final ACWorld w = ACWorld.getWorld(args.getString(0));
067                        final Player target = (Player) sender;
068                        if (!target.getWorld().equals(w.getHandle())
069                                        && !PermissionManager.hasPerm(sender, "admincmd.spawn.tp."
070                                                        + w.getName().toLowerCase())) {
071                                return;
072                        }
073                        ACPluginManager.getScheduler().scheduleSyncDelayedTask(
074                                        ACHelper.getInstance().getCoreInstance(),
075                                        new DelayedTeleport(target, sender, w),
076                                        ConfigEnum.TP_DELAY.getLong());
077                } else if (Utils.isPlayer(sender, true)) {
078                        final Player target = (Player) sender;
079                        ACPluginManager.getScheduler().scheduleSyncDelayedTask(
080                                        ACHelper.getInstance().getCoreInstance(),
081                                        new DelayedTeleport(target, sender, null),
082                                        ConfigEnum.TP_DELAY.getLong());
083                }
084        }
085
086        /*
087         * (non-Javadoc)
088         * 
089         * @see be.Balor.Manager.Commands.CoreCommand#registerBukkitPerm()
090         */
091        @Override
092        public void registerBukkitPerm() {
093                for (final World world : Bukkit.getWorlds()) {
094                        this.permParent.addChild("admincmd.spawn.tp." + world.getName());
095                }
096                super.registerBukkitPerm();
097        }
098
099        /*
100         * (non-Javadoc)
101         * 
102         * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])
103         */
104        @Override
105        public boolean argsCheck(final String... args) {
106                return args != null;
107        }
108
109        private class DelayedTeleport implements Runnable {
110
111                protected SimplifiedLocation locBefore;
112                protected Player target;
113                protected HashMap<String, String> replace;
114                protected CommandSender sender;
115                protected ACWorld world;
116
117                public DelayedTeleport(final Player target, final CommandSender sender,
118                                final ACWorld world) {
119                        this.target = target;
120                        this.locBefore = new SimplifiedLocation(target.getLocation());
121                        this.sender = sender;
122                        this.world = world;
123                        if (ConfigEnum.TP_DELAY.getLong() > 0) {
124                                final Map<String, String> replace = new HashMap<String, String>();
125                                replace.put(
126                                                "sec",
127                                                String.valueOf(ConfigEnum.TP_DELAY.getLong()
128                                                                / Utils.secInTick));
129                                LocaleHelper.TELEPORT_SOON.sendLocale(target, replace);
130                        }
131                }
132
133                @Override
134                public void run() {
135                        if (!ConfigEnum.CHECKTP.getBoolean()) {
136                                ACHelper.getInstance().spawn((Player) sender, world);
137                                sendMessage(sender, target, "spawn");
138                                return;
139                        }
140
141                        if (locBefore.equals(target.getLocation())) {
142                                ACHelper.getInstance().spawn((Player) sender, world);
143                                sendMessage(sender, target, "spawn");
144                        } else {
145                                replace = new HashMap<String, String>();
146                                replace.put("cmdname", "Warp");
147                                sendMessage(sender, target, "errorMoved", replace);
148                        }
149                }
150        }
151}
