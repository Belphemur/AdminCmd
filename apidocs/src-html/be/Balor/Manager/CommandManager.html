064            private class NormalCommand implements Runnable {
065                    protected final ACCommandContainer acc;
066    
067                    public NormalCommand(final ACCommandContainer acc) {
068                            this.acc = acc;
069                    }
070    
071                    protected void processCmd() throws PlayerNotFound {
072                            acc.processArguments();
073                            acc.execute();
074                    }
075    
076                    /*
077                     * (non-Javadoc)
078                     * 
079                     * @see java.lang.Runnable#run()
080                     */
081                    @Override
082                    public void run() {
083                            try {
084                                    processCmd();
085                            } catch (final ConcurrentModificationException cme) {
086                                    ACPluginManager.getScheduler().scheduleSyncDelayedTask(corePlugin,
087                                                    new SyncCommand(acc));
088                            } catch (final WorldNotLoaded e) {
089                                    String message = e.getMessage();
090                                    if (message == null || message == "") {
091                                            message = "This world is not loaded!";
092                                    }
093                                    final HashMap<String, String> replace = new HashMap<String, String>();
094                                    replace.put("message", message);
095                                    LocaleHelper.WORLD_NOT_LOADED.sendLocale(acc.getSender(), replace);
096                            } catch (final PlayerNotFound e) {
097                                    e.getSender().sendMessage(e.getMessage());
098                            } catch (final Throwable t) {
099                                    ACLogger.severe(acc.debug(), t);
100                                    Utils.broadcastMessage("[AdminCmd] " + acc.debug());
101                            }
102    
103                    }
104    
105                    /*
106                     * (non-Javadoc)
107                     * 
108                     * @see java.lang.Object#toString()
109                     */
110                    @Override
111                    public String toString() {
112                            return "NormalCommand [acc=" + acc + "]";
113                    }
114    
115            }

199            /**
200             * Check if some alias have been disabled for the registered commands
201             */
202            public void checkAlias(final AbstractAdminCmdPlugin plugin) {
203                    if (ConfigEnum.VERBOSE.getBoolean()) {
204                            final HashMap<String, Command> commands = pluginCommands.get(plugin);
205                            if (commands != null)
206                                    for (final String cmdName : commands.keySet()) {
207                                            final Command cmd = commands.get(cmdName);
208                                            if (corePlugin.getCommand(cmd.getName()) != null) {
209                                                    final List<String> aliasesList = new ArrayList<String>(cmd.getAliases());
210                                                    aliasesList.removeAll(corePlugin.getCommand(cmd.getName()).getAliases());
211                                                    aliasesList.removeAll(prioritizedCommands);
212                                                    String aliases = "";
213                                                    for (final String alias : aliasesList)
214                                                            aliases += alias + ", ";
215                                                    if (!aliases.isEmpty() && ConfigEnum.VERBOSE.getBoolean())
216                                                            Logger.getLogger("Minecraft").info(
217                                                                            "[" + corePlugin.getDescription().getName()
218                                                                                            + "] Disabled Alias(es) for " + cmd.getName() + " : "
219                                                                                            + aliases);
220                                            }
221                                    }
222                    }
223            }

225            /**
226             * Check the command if it have alias, prioritized or disabled.
227             * 
228             * @param command
229             * @throws CommandDisabled
230             */
231            private void checkCommand(final CoreCommand command) throws CommandDisabled {
232                    final HashMap<String, Command> commands = pluginCommands.get(command.getPlugin());
233                    if (commands != null)
234                            for (final String alias : commands.get(command.getCmdName()).getAliases()) {
235                                    if (disabledCommands.contains(alias))
236                                            throw new CommandDisabled("Command " + command.getCmdName()
237                                                            + " selected to be disabled in the configuration file.");
238                                    if (prioritizedCommands.contains(alias))
239                                            commandReplacer.put(alias, command);
240                                    if (aliasCommands.containsKey(alias)) {
241                                            for (final String cmd : aliasCommands.get(alias))
242                                                    commandReplacer.put(cmd, command);
243                                    }
244                            }
245            }

247            /**
248             * Used to execute ACCommands
249             * 
250             * @param sender
251             * @param cmd
252             * @param args
253             * @return
254             */
255            private boolean executeCommand(final CommandSender sender, final CoreCommand cmd,
256                            final String[] args) {
257                    ACCommandContainer container = null;
258                    try {
259                            if (!cmd.permissionCheck(sender))
260                                    return true;
261                            if (!cmd.argsCheck(args))
262                                    return false;
263                            container = new ACCommandContainer(sender, cmd, args);
264                            /*
265                             * if (cmd.getCmdName().equals("bal_replace") ||
266                             * cmd.getCmdName().equals("bal_extinguish"))
267                             * corePlugin.getServer().getScheduler()
268                             * .scheduleSyncDelayedTask(corePlugin, new SyncCommand(container));
269                             * else
270                             */
271                            threads.execute(new NormalCommand(container));
272                            if (!cmd.getCmdName().equals("bal_repeat")) {
273                                    if (Utils.isPlayer(sender, false))
274                                            ACPlayer.getPlayer(((Player) sender)).setLastCmd(container);
275                                    else
276                                            ACPlayer.getPlayer("serverConsole").setLastCmd(container);
277                            }
278                            return true;
279                    } catch (final Throwable t) {
280                            ACLogger.severe(container != null ? container.debug() : "The container is null", t);
281                            Utils.broadcastMessage("[AdminCmd] " + container != null ? container.debug()
282                                            : cmd.getCmdName()
283                                                            + " throw an Exception please report the log in a ticket : http://bug.admincmd.com/");
284                            return false;
285                    }
286            }

288            /**
289             * Getting the private field of a another class;
290             * 
291             * @param object
292             * @param field
293             * @return
294             * @throws SecurityException
295             * @throws NoSuchFieldException
296             * @throws IllegalArgumentException
297             * @throws IllegalAccessException
298             */
299            private Object getPrivateField(final Object object, final String field)
300                            throws SecurityException, NoSuchFieldException, IllegalArgumentException,
301                            IllegalAccessException {
302                    final Class<?> clazz = object.getClass();
303                    final Field objectField = clazz.getDeclaredField(field);
304                    objectField.setAccessible(true);
305                    final Object result = objectField.get(object);
306                    objectField.setAccessible(false);
307                    return result;
308            }

310            /**
311             * {@inheritDoc}
312             */
313            @Override
314            public boolean onCommand(final CommandSender sender, final Command command, final String label,
315                            final String[] args) {
316                    CoreCommand cmd = null;
317                    if ((cmd = registeredCommands.get(command)) != null)
318                            return executeCommand(sender, cmd, args);
319                    else
320                            return false;
321            }
322    
323            public boolean processCommandString(final CommandSender sender, final String command) {
324                    final String[] split = command.split("\\s+");
325                    if (split.length == 0)
326                            return false;
327                    final String cmdName = split[0].substring(1).toLowerCase();
328                    final CoreCommand cmd = commandReplacer.get(cmdName);
329                    if (cmd != null) {
330                            if (ConfigEnum.VERBOSE.getBoolean())
331                                    ACLogger.info("Command " + cmdName + " intercepted.");
332                            return executeCommand(sender, cmd, Utils.Arrays_copyOfRange(split, 1, split.length));
333                    }
334                    return false;
335            }

342            public void registerACPlugin(final AbstractAdminCmdPlugin plugin) {
343                    final HashMap<String, Command> commands = new HashMap<String, Command>();
344                    for (final Command cmd : PluginCommandUtil.parse(plugin))
345                            commands.put(cmd.getName(), cmd);
346    
347                    pluginCommands.put(plugin, new HashMap<String, Command>(commands));
348            }

350            /**
351             * Register command
352             * 
353             * @param clazz
354             */
355            public boolean registerCommand(final Class<? extends CoreCommand> clazz) {
356                    CoreCommand command = null;
357                    try {
358                            DebugLog.INSTANCE.info("Begin registering Command " + clazz.getName());
359                            command = clazz.newInstance();
360                            command.initializeCommand();
361                            checkCommand(command);
362                            command.registerBukkitPerm();
363                            command.getPluginCommand().setExecutor(instance);
364                            registeredCommands.put(command.getPluginCommand(), command);
365                    } catch (final InstantiationException e) {
366                            e.printStackTrace();
367                            return false;
368                    } catch (final IllegalAccessException e) {
369                            e.printStackTrace();
370                            return false;
371                    } catch (final CommandDisabled e) {
372                            unRegisterBukkitCommand(command.getPluginCommand());
373                            HelpLister.getInstance().removeHelpEntry(command.getPlugin().getPluginName(),
374                                            command.getCmdName());
375                            if (ConfigEnum.VERBOSE.getBoolean())
376                                    ACLogger.info(e.getMessage());
377                            return false;
378                    } catch (final CommandAlreadyExist e) {
379                            boolean disableCommand = true;
380                            final HashMap<String, Command> commands = pluginCommands.get(command.getPlugin());
381                            if (commands != null)
382                                    for (final String alias : commands.get(command.getCmdName()).getAliases()) {
383                                            if (prioritizedCommands.contains(alias)) {
384                                                    commandReplacer.put(alias, command);
385                                                    disableCommand = false;
386                                            }
387                                            if (aliasCommands.containsKey(alias)) {
388                                                    for (final String cmd : aliasCommands.get(alias))
389                                                            commandReplacer.put(cmd, command);
390                                                    disableCommand = false;
391                                            }
392                                    }
393                            if (disableCommand) {
394                                    unRegisterBukkitCommand(command.getPluginCommand());
395                                    HelpLister.getInstance().removeHelpEntry(command.getPlugin().getPluginName(),
396                                                    command.getCmdName());
397                                    if (ConfigEnum.VERBOSE.getBoolean())
398                                            ACLogger.info(e.getMessage());
399                                    DebugLog.INSTANCE.info("Command Disabled");
400                                    return false;
401                            } else {
402                                    command.registerBukkitPerm();
403                                    command.getPluginCommand().setExecutor(this);
404                                    registeredCommands.put(command.getPluginCommand(), command);
405                                    DebugLog.INSTANCE.info("Command Prioritized but already exists");
406                                    return true;
407                            }
408                    } catch (final CommandException e) {
409                            if (ConfigEnum.VERBOSE.getBoolean())
410                                    Logger.getLogger("Minecraft").info("[AdminCmd] " + e.getMessage());
411                            return false;
412                    }
413                    DebugLog.INSTANCE.info("End registering Command " + clazz.getName());
414                    return true;
415            }

417            /**
418             * @param plugin
419             *            the plugin to set
420             */
421            public void setCorePlugin(final AdminCmd plugin) {
422                    this.corePlugin = plugin;
423                    final ExtendedConfiguration cmds = FileManager.getInstance().getYml("commands");
424                    disabledCommands = cmds.getStringList("disabledCommands", new LinkedList<String>());
425                    prioritizedCommands = cmds.getStringList("prioritizedCommands", new LinkedList<String>());
426                    final ExConfigurationSection alias = cmds.getConfigurationSection("alias");
427                    for (final String cmd : alias.getKeys(false))
428                            aliasCommands.put(cmd,
429                                            new ArrayList<String>(alias.getStringList(cmd, new ArrayList<String>())));
430                    startThreads();
431            }
432    
433            public void startThreads() {
434                    if (!threadsStarted) {
435                            threads.purge();
436                    }
437                    threadsStarted = true;
438            }
439    
440            public void stopAllExecutorThreads() {
441                    threads.shutdown();
442                    threadsStarted = false;
443            }

445            /**
446             * Unregister a command from bukkit.
447             * 
448             * @param cmd
449             */
450            private void unRegisterBukkitCommand(final PluginCommand cmd) {
451                    try {
452                            final Object result = getPrivateField(corePlugin.getServer().getPluginManager(),
453                                            "commandMap");
454                            final SimpleCommandMap commandMap = (SimpleCommandMap) result;
455                            final Object map = getPrivateField(commandMap, "knownCommands");
456                            @SuppressWarnings("unchecked")
457                            final HashMap<String, Command> knownCommands = (HashMap<String, Command>) map;
458                            knownCommands.remove(cmd.getName());
459                            for (final String alias : cmd.getAliases())
460                                    knownCommands.remove(alias);
461                    } catch (final SecurityException e) {
462                            ACLogger.severe("Unregistering command problem", e);
463                    } catch (final IllegalArgumentException e) {
464                            ACLogger.severe("Unregistering command problem", e);
465                    } catch (final NoSuchFieldException e) {
466                            ACLogger.severe("Unregistering command problem", e);
467                    } catch (final IllegalAccessException e) {
468                            ACLogger.severe("Unregistering command problem", e);
469                    } catch (final Exception e) {
470                            ACLogger.severe("Unregistering command problem", e);
471                    }
472            }

474            /**
475             * UnRegister command
476             * 
477             * @param clazz
478             *            command to unregister
479             * @param plugin
480             *            plugin that want the command to be unregister. It has to be
481             *            the same that belong to the command.
482             */
483            public boolean unRegisterCommand(final Class<? extends CoreCommand> clazz,
484                            final AbstractAdminCmdPlugin plugin) {
485                    try {
486                            final CoreCommand command = clazz.newInstance();
487                            if (plugin.equals(command.getPlugin())) {
488                                    try {
489                                            command.initializeCommand();
490                                            final PluginCommand pCmd = command.getPluginCommand();
491                                            registeredCommands.remove(pCmd);
492                                            commandReplacer.remove(command.getCmdName());
493                                            unRegisterBukkitCommand(pCmd);
494                                    } catch (final Exception e) {
495                                            return false;
496                                    }
497                                    return true;
498    
499                            }
500                    } catch (final InstantiationException e) {
501    
502                    } catch (final IllegalAccessException e) {
503    
504                    }
505                    return false;
506    
507            }
