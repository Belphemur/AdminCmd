019package be.Balor.Importer;
020
021import java.io.File;
022import java.io.FileFilter;
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026import java.util.regex.Pattern;
027
028/**
029 * Implements a {@link java.io.FileFilter FileFilter} to return files matching a
030 * given pattern.</br> It also can search all sub-directories by a recursive
031 * implementation.</br> You can also use the Type to only search for DIRS. </p>
032 * Example:
033 * 
034 * <pre>
035 * import be.Balor.Importer.SubDirFileFilter.Type;
036 * import be.Balor.Importer.SubDirFileFilter.PatternFilter;
037 * 
038 * SubDirFileFilter filter = new SubDirFileFilter();
039 * List< File > files = filter.getFiles(new File(""),
040 *                       filter.new PatternFilter(Type.ALL, ".java"),
041 *                       true);
042 * </pre>
043 * 
044 * @author Lathanael (aka Philippe Leipold)
045 */
046public class SubDirFileFilter {
047
048        /**
049         * Returns a List<File> of all Dirs/Files depending on the filter. If
050         * recursive is set to true it will get dirs/files in sub-diretories too.
051         * 
052         * @param basedir
053         * @param filter
054         * @param recursive
055         * @return A list of files and/or directories matching the input pattern
056         */
057        public final List<File> getFiles(final File basedir,
058                        final FileFilter filter, boolean recursive) {
059                List<File> files = new ArrayList<File>();
060                if (basedir != null && basedir.isDirectory()) {
061                        if (recursive)
062                                for (File subdir : basedir.listFiles())
063                                        files.addAll(this.getFiles(subdir, filter, recursive));
064                        files.addAll(Arrays.asList(basedir.listFiles(filter)));
065                }
066                return files;
067        }
068
069        /**
070         * Defines for what type of file should be looked for!
071         */
072        public enum Type implements FileFilter {
073                FILE, DIR, ALL;
074
075                public boolean accept(final File file) {
076                        return file != null
077                                        && (this == ALL || (this == FILE && file.isFile()) || (this == DIR && file
078                                                        .isDirectory()));
079                }
080        }
081
082        /**
083         * Gets a file matching a given suffix pattern like ".java"
084         */
085        public class PatternFilter implements FileFilter {
086                private final Type type;
087                private final String pattern;
088
089                public PatternFilter(final Type type, final String pattern) {
090                        this.type = type;
091                        this.pattern = "^.*" + Pattern.quote(pattern) + "$";
092                }
093
094                public boolean accept(final File file) {
095                        return type.accept(file) && file.getName().matches(pattern);
096                }
097        }
098}
