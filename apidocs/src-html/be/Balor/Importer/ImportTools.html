044public class ImportTools {
045
046        public static String getPluginsFolder(final File file) {
047                final String path = file.getAbsolutePath();
048                return path.substring(0, path.lastIndexOf(File.separator));
049        }
050
051        public static void copyTextFile(final File sourcefile, final File targeFile)
052                        throws IOException {
053                assert sourcefile != null && targeFile != null;
054                final BufferedReader in = new BufferedReader(new UnicodeReader(
055                                new FileInputStream(sourcefile), "UTF-8"));
056                final StringBuffer result = new StringBuffer();
057                String temp;
058                while ((temp = in.readLine()) != null) {
059                        result.append(temp + System.getProperty("line.separator"));
060                }
061                in.close();
062                UnicodeUtil.saveUTF8File(targeFile, result.toString(), false);
063        }
064
065        public static boolean importESHomes(final ExtendedConfiguration userFile,
066                        final String playerName) {
067                Set<String> nodeList;
068                ConfigurationSection home = null, homes = null;
069                Location homeLoc = null;
070                if (userFile == null) {
071                        ACLogger.info("[ERROR] Could not import data of user: "
072                                        + playerName);
073                        return false;
074                }
075                homes = userFile.getConfigurationSection("homes");
076                if (homes != null) {
077                        nodeList = homes.getKeys(false);
078                        for (final String name : nodeList) {
079                                home = homes.getConfigurationSection(name);
080                                final ACWorld w = ACWorld.getWorld(home.getString("world"));
081                                try {
082                                        homeLoc = buildLocation(home, w);
083                                } catch (final Exception e) {
084                                        ACLogger.info("[ERROR] Could not import homes of user: "
085                                                        + playerName);
086                                        DebugLog.INSTANCE.log(Level.WARNING,
087                                                        "[ERROR] Could not import homes of user: "
088                                                                        + playerName, e);
089                                        return false;
090                                }
091                                ACPlayer.getPlayer(playerName).setHome(name, homeLoc);
092                        }
093                } else {
094                        if (ConfigEnum.VERBOSE.getBoolean()) {
095                                ACLogger.info("User " + playerName + " did not have any homes.");
096                        }
097                        return false;
098                }
099                return false;
100        }
101
102        public static Location buildLocation(final ConfigurationSection section,
103                        final ACWorld world) {
104                Location loc = null;
105                assert section != null && world != null;
106                double x = 0D, y = 0D, z = 0D;
107                float pitch = 0F, yaw = 0F;
108                x = section.getDouble("x");
109                y = section.getDouble("y");
110                z = section.getDouble("z");
111                yaw = Float.parseFloat(section.getString("yaw"));
112                pitch = Float.parseFloat(section.getString("pitch"));
113                loc = new Location(world.getHandle(), x, y, z, yaw, pitch);
114                return loc;
115        }
116
117        public static boolean importESLastLocation(
118                        final ExtendedConfiguration uerFile, final String playerName) {
119                Location lastLoc = null;
120                ConfigurationSection iLoc = null;
121                iLoc = uerFile.getConfigurationSection("lastlocation");
122                if (iLoc == null) {
123                        DebugLog.INSTANCE
124                                        .info("Can find the lastLocation in the configuration file for "
125                                                        + playerName + " in file " + uerFile);
126                        return false;
127                }
128                final ACWorld w = ACWorld.getWorld(iLoc.getString("world"));
129                try {
130                        lastLoc = buildLocation(iLoc, w);
131                } catch (final Exception e) {
132                        ACLogger.info("[ERROR] Could not import the last location of user: "
133                                        + playerName);
134                        DebugLog.INSTANCE.log(Level.WARNING,
135                                        "[ERROR] Could not import last location of user: "
136                                                        + playerName, e);
137                        return false;
138                }
139                ACPlayer.getPlayer(playerName).setLastLocation(lastLoc);
140                return true;
141        }
