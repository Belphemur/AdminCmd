017package be.Balor.OpenInv;
018
019import java.lang.reflect.InvocationHandler;
020import java.lang.reflect.InvocationTargetException;
021import java.lang.reflect.Method;
022import java.lang.reflect.Proxy;
023
024import org.bukkit.entity.Player;
025
026import be.Balor.Tools.Compatibility.MinecraftReflection;
027import be.Balor.Tools.Compatibility.Reflect.FieldUtils;
028import be.Balor.Tools.Compatibility.Reflect.MethodHandler;
029
030/**
031 * @author Antoine
032 * 
033 */
034public class PlayerInventoryProxy implements InvocationHandler {
035
036        protected final Object obj;
037        protected final Player proprietary;
038        private final Object[] extra = new Object[5];
039        private final int size;
040
041        /**
042         * @param obj2
043         */
044        protected PlayerInventoryProxy(final Player prop, final Object obj) {
045                this.proprietary = prop;
046                this.obj = obj;
047                final Object playerHandle = MinecraftReflection.getHandle(proprietary);
048                final Object inventory = FieldUtils.getField(playerHandle, "inventory");
049                final Object[] armor = FieldUtils.getField(inventory, "armor");
050                final Object[] items = FieldUtils.getField(inventory, "items");
051                size = armor.length + items.length + extra.length;
052                FieldUtils.setField(this.obj, "armor", armor);
053                FieldUtils.setField(this.obj, "items", items);
054        }
055
056        public static Object newInstance(final Player prop, final Object obj) {
057                if (!MinecraftReflection.getPlayerInventoryClass().isAssignableFrom(
058                                obj.getClass())) {
059                        throw new RuntimeException(
060                                        "The object must be of the type of PlayerInventory");
061                }
062                return Proxy.newProxyInstance(obj.getClass().getClassLoader(), obj
063                                .getClass().getInterfaces(),
064                                new PlayerInventoryProxy(prop, obj));
065        }
066
067        /*
068         * (non-Javadoc)
069         * 
070         * @see java.lang.reflect.InvocationHandler#invoke(java.lang.Object,
071         * java.lang.reflect.Method, java.lang.Object[])
072         */
073        @Override
074        public Object invoke(final Object proxy, final Method method,
075                        final Object[] args) throws Throwable {
076                final String methodName = method.getName();
077                if (methodName.equals("onClose")) {
078                        this.onClose(args[0]);
079                        return null;
080                } else if (methodName.equals("getContents")) {
081                        return getContents();
082                } else if (methodName.equals("getSize")) {
083                        return getSize();
084                } else if (methodName.equals("a_")) {
085                        return a_();
086                } else if (methodName.equals("getName")) {
087                        return getName();
088                } else if (methodName.equals("getItem")) {
089                        return getItem((Integer) args[0]);
090                } else if (methodName.equals("splitStack")) {
091                        return splitStack((Integer) args[0], (Integer) args[1]);
092                } else if (methodName.equals("splitWithoutUpdate")) {
093                        return splitWithoutUpdate((Integer) args[0]);
094                } else if (methodName.equals("setItem")) {
095                        setItem((Integer) args[0], args[1]);
096                        return null;
097                } else {
098                        return method.invoke(obj, args);
099                }
100        }
101
102        protected Object[] getItems() {
103                return FieldUtils.getField(obj, "items");
104        }
105
106        protected Object[] getArmor() {
107                return FieldUtils.getField(obj, "armor");
108        }
109
110        /*
111         * (non-Javadoc)
112         * 
113         * @see
114         * net.minecraft.server.PlayerInventory#onClose(org.bukkit.craftbukkit.entity
115         * .CraftHumanEntity)
116         */
117        private void onClose(final Object who) throws IllegalAccessException,
118                        IllegalArgumentException, InvocationTargetException {
119                final MethodHandler superOnClose = new MethodHandler(obj.getClass(),
120                                "onClose", MinecraftReflection.getCraftHumanEntityClass());
121                superOnClose.invoke(obj, who);
122                checkCloseEvent();
123        }
124
125        protected void checkCloseEvent() {
126                final Object transactions = FieldUtils.getField(obj, "transaction");
127                final MethodHandler isEmpty = new MethodHandler(
128                                transactions.getClass(), "isEmpty");
129                final boolean empty = isEmpty.invoke(transactions);
130                if (empty && !proprietary.isOnline()) {
131                        InventoryManager.INSTANCE.closeOfflineInv(proprietary);
132                }
133        }
134
135        protected Object[] getContents() {
136                final Object[] C = new Object[getSize()];
137                final Object[] items = getItems();
138                final Object[] armor = getArmor();
139                System.arraycopy(items, 0, C, 0, items.length);
140                System.arraycopy(armor, 0, C, items.length, armor.length);
141                return C;
142        }
143
144        /*
145         * (non-Javadoc)
146         * 
147         * @see net.minecraft.server.PlayerInventory#getSize()
148         */
149        protected int getSize() {
150                return size;
151        }
152
153        protected boolean a_() {
154                return true;
155        }
156
157        protected String getName() {
158                if (proprietary.getName().length() > 16) {
159                        return proprietary.getName().substring(0, 16);
160                }
161                return proprietary.getName();
162        }
163
164        private int getReversedItemSlotNum(final int i) {
165                if (i >= 27) {
166                        return i - 27;
167                } else {
168                        return i + 9;
169                }
170        }
171
172        private int getReversedArmorSlotNum(final int i) {
173                if (i == 0) {
174                        return 3;
175                }
176                if (i == 1) {
177                        return 2;
178                }
179                if (i == 2) {
180                        return 1;
181                }
182                if (i == 3) {
183                        return 0;
184                } else {
185                        return i;
186                }
187        }
188
189        protected Object getItem(int i) {
190                Object[] is = getItems();
191
192                if (i >= is.length) {
193                        i -= is.length;
194                        is = getArmor();
195                } else {
196                        i = getReversedItemSlotNum(i);
197                }
198
199                if (i >= is.length) {
200                        i -= is.length;
201                        is = this.extra;
202                } else if (is == getArmor()) {
203                        i = getReversedArmorSlotNum(i);
204                }
205
206                return is[i];
207        }
208
209        private int getCount(final Object itemstack) {
210                return FieldUtils.getField(itemstack, "count");
211        }
212
213        protected Object splitStack(int i, final int j) {
214                Object[] is = getItems();
215
216                if (i >= is.length) {
217                        i -= is.length;
218                        is = getArmor();
219                } else {
220                        i = getReversedItemSlotNum(i);
221                }
222
223                if (i >= is.length) {
224                        i -= is.length;
225                        is = this.extra;
226                } else if (is == getArmor()) {
227                        i = getReversedArmorSlotNum(i);
228                }
229
230                if (is[i] != null) {
231                        Object itemstack;
232
233                        if (getCount(is[i]) <= j) {
234                                itemstack = is[i];
235                                is[i] = null;
236                                return itemstack;
237                        } else {
238                                final MethodHandler a = new MethodHandler(is[i].getClass(),
239                                                "a", int.class);
240                                itemstack = a.invoke(is[i], j);
241                                if (getCount(is[i]) == 0) {
242                                        is[i] = null;
243                                }
244
245                                return itemstack;
246                        }
247                } else {
248                        return null;
249                }
250        }
251
252        protected Object splitWithoutUpdate(int i) {
253                Object[] is = getItems();
254
255                if (i >= is.length) {
256                        i -= is.length;
257                        is = getArmor();
258                } else {
259                        i = getReversedItemSlotNum(i);
260                }
261
262                if (i >= is.length) {
263                        i -= is.length;
264                        is = this.extra;
265                } else if (is == getArmor()) {
266                        i = getReversedArmorSlotNum(i);
267                }
268
269                if (is[i] != null) {
270                        final Object itemstack = is[i];
271
272                        is[i] = null;
273                        return itemstack;
274                } else {
275                        return null;
276                }
277        }
278
279        protected void setItem(int i, final Object itemstack) {
280                Object[] is = getItems();
281
282                if (i >= is.length) {
283                        i -= is.length;
284                        is = getArmor();
285                } else {
286                        i = getReversedItemSlotNum(i);
287                }
288
289                if (i >= is.length) {
290                        i -= is.length;
291                        is = this.extra;
292                } else if (is == getArmor()) {
293                        i = getReversedArmorSlotNum(i);
294                }
295                is[i] = itemstack;
296        }
