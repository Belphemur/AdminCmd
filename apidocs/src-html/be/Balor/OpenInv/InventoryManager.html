046public class InventoryManager {
047        public static InventoryManager INSTANCE;

057        public static void createInstance() {
058                if (INSTANCE == null) {
059                        INSTANCE = new InventoryManager();
060                }
061        }
062
063        public void onQuit(final Player p) {
064                final String name = p.getName();
065                final Object inv = replacedInv.get(name);
066                if (inv == null) {
067                        return;
068                }
069                final MethodHandler getViewers = new MethodHandler(inv.getClass(),
070                                "getViewers");
071                final Object viewers = getViewers.invoke(inv);
072                final MethodHandler isEmpty = new MethodHandler(viewers.getClass(),
073                                "isEmpty");
074                final boolean empty = isEmpty.invoke(viewers);
075                if (empty) {
076                        replacedInv.remove(name);
077                }
078        }
079
080        void closeOfflineInv(final Player p) {
081                onQuit(p);
082                offlineInv.remove(p.getName());
083                p.saveData();
084                DebugLog.INSTANCE.info("Saving Offline Inventory of " + p.getName());
085        }
086
087        public void onJoin(final Player p) {
088                final String name = p.getName();
089                Object inv = offlineInv.get(name);
090                if (inv == null) {
091                        inv = replacedInv.get(name);
092                }
093                if (inv == null) {
094                        return;
095                }
096                final PlayerInventoryProxy proxy = (PlayerInventoryProxy) Proxy
097                                .getInvocationHandler(inv);
098                final Object inventory = MinecraftReflection.getInventory(p);
099                FieldUtils.setField(inventory, "armor", proxy.getArmor());
100                FieldUtils.setField(inventory, "items", proxy.getItems());
101
102        }
103
104        /**
105         * Open the inventory of an offline player
106         * 
107         * @param sender
108         * @param name
109         * @throws PlayerNotFound
110         * @author lishd {@link https
111         *         ://github.com/lishd/OpenInv/blob/master/src/lishid
112         *         /openinv/commands/OpenInvPluginCommand.java}
113         */
114        public void openOfflineInv(final Player sender, final String name,
115                        final String world) throws PlayerNotFound {
116                Player target = null;
117                final HashMap<String, String> replace = new HashMap<String, String>();
118                replace.put("player", name);
119                // Offline inv here...
120                // See if the player has data files
121
122                // Find the player folder
123                ACWorld acworld = null;
124                try {
125                        acworld = ACWorld.getWorld(world);
126                } catch (final WorldNotLoaded e) {
127                        replace.put("message", world);
128                        LocaleHelper.WORLD_NOT_LOADED.sendLocale(sender, replace);
129                        return;
130                }
131                final File playerfolder = new File(acworld.getHandle()
132                                .getWorldFolder(), "players");
133                if (!playerfolder.exists()) {
134                        throw new PlayerNotFound(Utils.I18n("playerNotFound", replace),
135                                        sender);
136                }
137
138                final String playername = matchUser(
139                                Arrays.asList(DatFilter.INSTANCE.listRecursively(playerfolder)),
140                                name);
141                if (playername == null) {
142                        throw new PlayerNotFound(Utils.I18n("playerNotFound", replace),
143                                        sender);
144                }
145
146                // Create an entity to load the player data
147                final Object entity = NMSBuilder.buildEntityPlayer(playername);
148                if (entity == null) {
149                        target = null;
150                } else {
151                        target = MinecraftReflection.getBukkitEntity(entity);
152                }
153                if (target != null) {
154                        target.loadData();
155                } else {
156                        throw new PlayerNotFound(Utils.I18n("playerNotFound", replace),
157                                        sender);
158                }
159                if (Utils.checkImmunity(sender, target)) {
160                        openInv(sender, target, true);
161                } else {
162                        Utils.sI18n(sender, "insufficientLvl");
163                }
164        }
165
166        /**
167         * Open the inventory of the connected player
168         * 
169         * @param sender
170         *            the user who'll see the inventory
171         * @param target
172         *            player to have his inventory opened
173         */
174        public void openInv(final Player sender, final Player target) {
175                openInv(sender, target, false);
176        }
177
178        private void openInv(final Player sender, final Player target,
179                        final boolean offline) {
180                final Object inventory = getInventory(target, offline);
181                final Object eh = MinecraftReflection.getHandle(sender);
182                final MethodHandler openContainer = new MethodHandler(
183                                MinecraftReflection.getEntityPlayerClass(), "openContainer",
184                                MinecraftReflection.getIInventoryClass());
185                openContainer.invoke(eh, inventory);
186        }
187
188        private Object getInventory(final Player player, final boolean offline) {
189                Object inventory = replacedInv.get(player.getName());
190                if (inventory == null) {
191                        final Object playerInv = NMSBuilder.buildPlayerInventory(player);
192                        if (offline) {
193                                inventory = OfflinePlayerInventoryProxy.newInstance(player,
194                                                playerInv);
195                                offlineInv.put(player.getName(), inventory);
196                        } else {
197                                inventory = PlayerInventoryProxy.newInstance(player, playerInv);
198                                replacedInv.put(player.getName(), inventory);
199                        }
200                }
201                return inventory;
202
203        }
204
205        private String matchUser(final Collection<File> container,
206                        final String search) {
207                String found = null;
208                if (search == null) {
209                        return found;
210                }
211                final String lowerSearch = search.toLowerCase();
212                int delta = Integer.MAX_VALUE;
213                for (final File file : container) {
214                        final String filename = file.getName();
215                        final String str = filename.substring(0, filename.length() - 4);
216                        if (!str.toLowerCase().startsWith(lowerSearch)) {
217                                continue;
218                        }
219                        final int curDelta = str.length() - lowerSearch.length();
220                        if (curDelta < delta) {
221                                found = str;
222                                delta = curDelta;
223                        }
224                        if (curDelta == 0) {
225                                break;
226                        }
227
228                }
229                return found;
230
231        }
