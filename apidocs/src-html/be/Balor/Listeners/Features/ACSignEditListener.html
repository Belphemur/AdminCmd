017package be.Balor.Listeners.Features;
018
019import java.util.HashMap;
020import java.util.Map;
021
022import org.bukkit.Bukkit;
023import org.bukkit.Location;
024import org.bukkit.Material;
025import org.bukkit.block.Block;
026import org.bukkit.block.BlockState;
027import org.bukkit.block.Sign;
028import org.bukkit.entity.Player;
029import org.bukkit.event.EventHandler;
030import org.bukkit.event.EventPriority;
031import org.bukkit.event.Listener;
032import org.bukkit.event.block.BlockPlaceEvent;
033import org.bukkit.event.block.SignChangeEvent;
034
035import be.Balor.Listeners.Events.ACSignChangeEvent;
036import be.Balor.Manager.Permissions.PermissionManager;
037import be.Balor.Tools.SignEditor;
038
039/**
040 * @author Balor (aka Antoine Aflalo)
041 * 
042 */
043public class ACSignEditListener implements Listener {
044
045        private final Map<Location, SignEditor> signEditor = new HashMap<Location, SignEditor>();
046
047        @EventHandler(ignoreCancelled = true)
048        public void onSignChange(final SignChangeEvent event) {
049                if (event instanceof ACSignChangeEvent) {
050                        return;
051                }
052                final Location signLoc = event.getBlock().getLocation();
053                final SignEditor saved = signEditor.get(signLoc);
054                if (saved == null) {
055                        return;
056                }
057                final Player player = event.getPlayer();
058                final ACSignChangeEvent signChange = new ACSignChangeEvent(
059                                saved.getUpdatedSignBlk(), player, event.getLines());
060                Bukkit.getPluginManager().callEvent(signChange);
061                if (signChange.isCancelled()) {
062                        signEditor.remove(signLoc);
063                        return;
064                }
065                saved.updateSign(signChange.getLines());
066                saved.removeEditorSign(player);
067                signEditor.remove(signLoc);
068                event.setCancelled(true);
069
070        }
071
072        @EventHandler(priority = EventPriority.HIGHEST, ignoreCancelled = true)
073        public void signPlace(final BlockPlaceEvent event) {
074                final Block block = event.getBlockPlaced();
075                if (block.getType() != Material.WALL_SIGN
076                                && block.getType() != Material.SIGN_POST) {
077                        return;
078                }
079                final Block blockAgainst = event.getBlockAgainst();
080                final BlockState blockAgainstState = blockAgainst.getState();
081                if (!(blockAgainstState instanceof Sign)) {
082                        return;
083                }
084                if (PermissionManager.hasPerm(event.getPlayer(),
085                                "admincmd.spec.signedit")) {
086                        signEditor.put(block.getLocation(), new SignEditor(block,
087                                        blockAgainst));
088                }
089
090        }
091}
