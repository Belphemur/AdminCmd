017package be.Balor.Listeners.Features;
018
019import java.util.List;
020
021import org.bukkit.World;
022import org.bukkit.entity.Entity;
023import org.bukkit.entity.HumanEntity;
024import org.bukkit.entity.LivingEntity;
025import org.bukkit.event.EventHandler;
026import org.bukkit.event.EventPriority;
027import org.bukkit.event.Listener;
028import org.bukkit.event.entity.CreatureSpawnEvent;
029
030import be.Balor.Manager.Exceptions.WorldNotLoaded;
031import be.Balor.Tools.Type;
032import be.Balor.World.ACWorld;
033
034/**
035 * @author Balor (aka Antoine Aflalo)
036 * 
037 */
038public class ACCreatureSpawnListener implements Listener {
039        @EventHandler(priority = EventPriority.HIGHEST, ignoreCancelled = true)
040        public void onCreatureSpawn(final CreatureSpawnEvent event) {
041                final Entity e = event.getEntity();
042                if (e instanceof HumanEntity) {
043                        return;
044                }
045                final World world = e.getWorld();
046                final ACWorld acWorld;
047                try {
048                        acWorld = ACWorld.getWorld(world);
049                } catch (final WorldNotLoaded e2) {
050                        return;
051                }
052
053                final List<LivingEntity> livEntities = world.getLivingEntities();
054                Integer limit = acWorld.getInformation(Type.MOB_LIMIT.toString())
055                                .getInt(-1);
056                if (limit != -1) {
057                        if ((livEntities.size() - world.getPlayers().size()) >= limit) {
058                                event.setCancelled(true);
059                                return;
060                        }
061                }
062
063                final Class<? extends Entity> entityClass = e.getClass();
064                final String entityName = entityClass.getSimpleName();
065                limit = acWorld.getMobLimit(entityName);
066                if (limit == -1) {
067                        return;
068                }
069                final int count = world.getEntitiesByClass(entityClass).size();
070                if (count >= limit) {
071                        event.setCancelled(true);
072                }
073
074        }
075}
