017package be.Balor.Listeners.Features;
018
019import java.net.InetAddress;
020import java.util.HashMap;
021import java.util.concurrent.ConcurrentMap;
022
023import org.bukkit.Bukkit;
024import org.bukkit.entity.Player;
025import org.bukkit.event.EventHandler;
026import org.bukkit.event.Listener;
027import org.bukkit.event.player.PlayerEvent;
028import org.bukkit.event.player.PlayerJoinEvent;
029import org.bukkit.event.player.PlayerKickEvent;
030import org.bukkit.event.player.PlayerQuitEvent;
031
032import be.Balor.Tools.Utils;
033import be.Balor.Tools.Debug.ACLogger;
034import be.Balor.bukkit.AdminCmd.ACPluginManager;
035import be.Balor.bukkit.AdminCmd.LocaleHelper;
036
037import com.google.common.collect.MapMaker;
038
039/**
040 * @author Balor (aka Antoine Aflalo)
041 * 
042 */
043public class ACIpCheckListener implements Listener {
044        private final ConcurrentMap<String, Player> ips = new MapMaker().makeMap();
045
046        @EventHandler
047        public void onJoin(final PlayerJoinEvent event) {
048                ACPluginManager.runTaskLaterAsynchronously(new Runnable() {
049
050                        @Override
051                        public void run() {
052                                final Player p = event.getPlayer();
053                                final InetAddress address = p.getAddress().getAddress();
054                                final HashMap<String, String> replace = new HashMap<String, String>();
055                                final Player sameIP = addIP(p, address);
056                                if (sameIP != null) {
057                                        replace.put("player", Utils.getPlayerName(p));
058                                        replace.put("player2", Utils.getPlayerName(sameIP));
059                                        replace.put("ip", address.toString().substring(1));
060                                        broadcastIP(replace);
061                                }
062
063                        }
064                });
065
066        }
067
068        @EventHandler
069        public void onQuit(final PlayerQuitEvent event) {
070                removePlayer(event);
071        }
072
073        @EventHandler
074        public void onKick(final PlayerKickEvent event) {
075                removePlayer(event);
076        }
077
078        /**
079         * Broadcasts a message if someone joined from the same IP to all players
080         * with a permission and to the console.
081         * 
082         * @param message
083         *            - The message
084         * @param players
085         *            - List of all players with the permission
086         * @author Lathanael (aka Philippe Leipold)
087         */
088        private void broadcastIP(final HashMap<String, String> replace) {
089                final String message = LocaleHelper.IP_BROADCAST.getLocale(replace);
090                if (message == null) {
091                        return;
092                }
093                ACPluginManager.scheduleSyncTask(new Runnable() {
094
095                        @Override
096                        public void run() {
097                                Bukkit.getServer().broadcast(message,
098                                                "admincmd.spec.ipbroadcast");
099                        }
100                });
101
102                ACLogger.info(message);
103        }
104
105        private void removePlayer(final PlayerEvent event) {
106                final Player player = event.getPlayer();
107                updateIP(player, player.getAddress().getAddress());
108        }
109
110        /**
111         * If a player quits loop through the online players and see if a second
112         * player uses the same IP and replace the quitting player with the online
113         * player. Otherwise remove the player from the list!
114         * 
115         * @param quits
116         * @param address
117         */
118        private void updateIP(final Player quits, final InetAddress address) {
119                ACPluginManager.runTaskLaterAsynchronously(new Runnable() {
120                        @Override
121                        public void run() {
122                                for (final Player p : Utils.getOnlinePlayers()) {
123                                        if (p.equals(quits)) {
124                                                continue;
125                                        }
126                                        if (p.getAddress().getAddress().equals(address)) {
127                                                ips.replace(address.toString().substring(1), p);
128                                                return;
129                                        }
130                                }
131                                ips.remove(address.toString().substring(1));
132
133                        }
134                });
135
136        }
137
138        /**
139         * Adds a player to the ip list if his ip is not already in.
140         * 
141         * @param player
142         * @param address
143         * @return The player who also uses this IP or null if there was none
144         *         previously
145         */
146        private Player addIP(final Player player, final InetAddress address) {
147                final Player p = ips.putIfAbsent(address.toString().substring(1),
148                                player);
149                return p;
150        }
151
152        /**
153         * Checks if an IP is already in use
154         * 
155         * @param address
156         * @return {@code true} if an IP is found, {@code false} otherwise.
157         */
158        public boolean ipInUse(final InetAddress address) {
159                if (ips.containsKey(address.toString().substring(1))) {
160                        return true;
161                }
162                return false;
163        }
164
165}
