049/**
050 * @author Lathanael (aka Philippe Leipold)
051 * 
052 */
053public class ACDeathListener implements Listener {
054
055        private final boolean noMessage;
056
057        public ACDeathListener(final boolean noMessage) {
058                this.noMessage = noMessage;
059        }
060
061        @EventHandler(priority = EventPriority.NORMAL, ignoreCancelled = true)
062        public void onPlayerDeath(final EntityDeathEvent event) {
063                if (!(event instanceof PlayerDeathEvent)) {
064                        return;
065                }
066                final PlayerDeathEvent e = (PlayerDeathEvent) event;
067                if (noMessage) {
068                        e.setDeathMessage(null);
069                        return;
070                }
071                final HashMap<String, String> replace = new HashMap<String, String>();
072                replace.put("player", Utils.getPlayerName(e.getEntity()));
073                String message = null;
074                final EntityDamageEvent cause = e.getEntity().getLastDamageCause();
075                try {
076                        if (cause instanceof EntityDamageByEntityEvent) {
077                                message = getMessage(cause);
078                        } else {
079                                switch (cause.getCause()) {
080                                case CONTACT:
081                                        message = ACHelper.getInstance().getDeathMessage("contact");
082                                        break;
083                                case DROWNING:
084                                        message = ACHelper.getInstance()
085                                                        .getDeathMessage("drowning");
086                                        break;
087                                case FALL:
088                                        message = ACHelper.getInstance().getDeathMessage("falling");
089                                        break;
090                                case FIRE_TICK:
091                                case FIRE:
092                                        message = ACHelper.getInstance().getDeathMessage("fire");
093                                        break;
094                                case STARVATION:
095                                        message = ACHelper.getInstance()
096                                                        .getDeathMessage("starving");
097                                        break;
098                                case SUFFOCATION:
099                                        message = ACHelper.getInstance().getDeathMessage(
100                                                        "suffocation");
101                                        break;
102                                case VOID:
103                                        message = ACHelper.getInstance().getDeathMessage("void");
104                                        break;
105                                case LAVA:
106                                        message = ACHelper.getInstance().getDeathMessage("lava");
107                                        break;
108                                case POISON:
109                                        message = ACHelper.getInstance().getDeathMessage("poison");
110                                        break;
111                                case LIGHTNING:
112                                        message = ACHelper.getInstance().getDeathMessage(
113                                                        "lightning");
114                                        break;
115                                case MAGIC:
116                                        message = ACHelper.getInstance().getDeathMessage("magic");
117                                        break;
118                                case SUICIDE:
119                                        message = ACHelper.getInstance().getDeathMessage("suicide");
120                                        break;
121                                case ENTITY_EXPLOSION:
122                                        message = ACHelper.getInstance().getDeathMessage("creeper");
123                                        break;
124                                case BLOCK_EXPLOSION:
125                                        message = ACHelper.getInstance().getDeathMessage("tnt");
126                                        break;
127                                case CUSTOM:
128                                        message = ACHelper.getInstance().getDeathMessage("custom");
129                                        break;
130                                default:
131                                        message = ACHelper.getInstance().getDeathMessage("default");
132                                        break;
133                                }
134                        }
135                } catch (final NullPointerException ex) {
136                        return;
137                }
138                if (message == null) {
139                        return;
140                }
141                replace.put("msg", message);
142                e.setDeathMessage(LocaleHelper.DEATH_MSG.getLocale(replace));
143                // Utils.broadcastMessage(LocaleHelper.DEATH_MSG.getLocale(replace));
144        }
145
146        private String getMessage(final EntityDamageEvent e) {
147                final Entity damager = ((EntityDamageByEntityEvent) e).getDamager();
148                if (damager instanceof Wolf) {
149                        return ACHelper.getInstance().getDeathMessage("wolf");
150                } else if (damager instanceof Player) {
151                        return ACHelper.getInstance().getDeathMessage("player");
152                } else if (damager instanceof Skeleton) {
153                        return ACHelper.getInstance().getDeathMessage("skeleton");
154                } else if (damager instanceof TNTPrimed) {
155                        return ACHelper.getInstance().getDeathMessage("TNTPrimed");
156                } else if (damager instanceof Zombie) {
157                        if (damager instanceof PigZombie) {
158                                return ACHelper.getInstance().getDeathMessage("zombiepig");
159                        }
160                        return ACHelper.getInstance().getDeathMessage("zombie");
161                } else if (damager instanceof Projectile) {
162                        if (damager instanceof Arrow) {
163                                if (((Arrow) damager).getShooter() == null) {
164                                        return ACHelper.getInstance().getDeathMessage("dispenser");
165                                } else if (((Arrow) damager).getShooter() instanceof Player) {
166                                        return ACHelper.getInstance().getDeathMessage("playerBow")
167                                                        + Utils.getPlayerName(((Player) ((Arrow) damager)
168                                                                        .getShooter()));
169                                } else if (((Projectile) damager).getShooter() instanceof Skeleton) {
170                                        return ACHelper.getInstance().getDeathMessage("skeleton");
171                                }
172                        } else if (damager instanceof Fireball) {
173                                if (((Fireball) damager).getShooter() instanceof Blaze) {
174                                        return ACHelper.getInstance().getDeathMessage("blaze");
175                                } else if (((Fireball) damager).getShooter() instanceof Ghast) {
176                                        return ACHelper.getInstance().getDeathMessage("ghast");
177                                }
178                        } else if (damager instanceof ThrownPotion) {
179                                if (((ThrownPotion) damager).getShooter() instanceof Player) {
180                                        return ACHelper.getInstance().getDeathMessage("potion");
181                                }
182                        }
183                } else if (damager instanceof LivingEntity) {
184                        return (ACHelper.getInstance().getDeathMessage("mob") + damager
185                                        .getType().getName());
186                }
187                return ACHelper.getInstance().getDeathMessage("default");
188        }
189}
