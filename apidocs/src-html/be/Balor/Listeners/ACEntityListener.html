017    package be.Balor.Listeners;
018    
019    import org.bukkit.entity.Player;
020    import org.bukkit.event.EventHandler;
021    import org.bukkit.event.EventPriority;
022    import org.bukkit.event.Listener;
023    import org.bukkit.event.entity.EntityDeathEvent;
024    import org.bukkit.event.entity.EntityTargetEvent;
025    
026    import be.Balor.Manager.Permissions.PermissionManager;
027    import be.Balor.Player.ACPlayer;
028    import belgium.Balor.Workers.InvisibleWorker;
029    
030    /**
031     * @author Balor (aka Antoine Aflalo)
032     * 
033     */
034    public class ACEntityListener implements Listener {
035    
036            @EventHandler
037            public void onEntityDeath(final EntityDeathEvent event) {
038                    if (!(event.getEntity() instanceof Player))
039                            return;
040                    final Player player = (Player) event.getEntity();
041                    ACPlayer.getPlayer(player).setLastLocation(player.getLocation());
042            }
043    
044            @EventHandler(priority = EventPriority.HIGH, ignoreCancelled = true)
045            public void onEntityTarget(final EntityTargetEvent event) {
046                    final Player p = (Player) event.getTarget();
047                    if (!(p instanceof Player))
048                            return;
049                    if (InvisibleWorker.getInstance().hasInvisiblePowers(p.getName())
050                                    && PermissionManager.hasPerm(p, "admincmd.invisible.notatarget", false))
051                            event.setCancelled(true);
052            }
053    }
