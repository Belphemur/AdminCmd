017package be.Balor.Listeners.Events;
018
019import org.bukkit.entity.Player;
020import org.bukkit.event.Cancellable;
021import org.bukkit.event.Event;
022import org.bukkit.event.HandlerList;
023
024/**
025 * Prevent that the player will go AFK, if cancelled, the player won't go AFK.
026 * 
027 * @author Antoine
028 * 
029 */
030public class ACGoAFKEvent extends Event implements Cancellable {
031        public enum Reason {
032                AUTO, DECIDED;
033        }
034
035        private static final HandlerList handlers = new HandlerList();
036        private final Reason reason;
037        private String message;
038        private final Player player;
039        private boolean cancelled = false;
040
041        /**
042         * @param who
043         * @param reason
044         * @param message
045         */
046        public ACGoAFKEvent(final Player who, final Reason reason,
047                        final String message) {
048                super(true);
049                this.player = who;
050                this.reason = reason;
051                this.message = message;
052        }
053
054        /*
055         * (non-Javadoc)
056         * 
057         * @see org.bukkit.event.Event#getHandlers()
058         */
059        @Override
060        public HandlerList getHandlers() {
061                return handlers;
062        }
063
064        /**
065         * @return the reason of the AFK
066         */
067        public Reason getReason() {
068                return reason;
069        }
070
071        /**
072         * @return the message
073         */
074        public String getMessage() {
075                return message;
076        }
077
078        /**
079         * @param message
080         *            the message to set
081         */
082        public void setMessage(final String message) {
083                this.message = message;
084        }
085
086        /**
087         * @return the player
088         */
089        public Player getPlayer() {
090                return player;
091        }
092
093        /*
094         * (non-Javadoc)
095         * 
096         * @see org.bukkit.event.Cancellable#isCancelled()
097         */
098        @Override
099        public boolean isCancelled() {
100                return cancelled;
101        }
102
103        /*
104         * (non-Javadoc)
105         * 
106         * @see org.bukkit.event.Cancellable#setCancelled(boolean)
107         */
108        @Override
109        public void setCancelled(final boolean cancel) {
110                this.cancelled = cancel;
111
112        }
113
114}
