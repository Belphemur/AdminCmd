017    package be.Balor.Listeners;
018    
019    import org.bukkit.block.Block;
020    import org.bukkit.entity.Player;
021    import org.bukkit.event.EventHandler;
022    import org.bukkit.event.Listener;
023    import org.bukkit.event.block.BlockPlaceEvent;
024    
025    import be.Balor.Tools.MaterialContainer;
026    import be.Balor.bukkit.AdminCmd.ACHelper;
027    
028    /**
029     * @author Balor (aka Antoine Aflalo)
030     * 
031     */
032    public class ACBlockListener implements Listener {
033            /**
034             * @author Lathanael (aka Philippe Leipold)
035             * 
036             */
037            @EventHandler(ignoreCancelled = true)
038            public void onBlockPlace(final BlockPlaceEvent event) {
039                    final Player player = event.getPlayer();
040                    final Block block = event.getBlock();
041                    final MaterialContainer mat = ACHelper.getInstance().checkMaterial(player,
042                                    String.valueOf(block.getTypeId()));
043                    if (!ACHelper.getInstance().inBlackListBlock(player, mat))
044                            return;
045                    event.setCancelled(true);
046            }
047    }
