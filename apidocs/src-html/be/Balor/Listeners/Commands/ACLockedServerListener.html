017package be.Balor.Listeners.Commands;
018
019import org.bukkit.event.EventHandler;
020import org.bukkit.event.EventPriority;
021import org.bukkit.event.Listener;
022import org.bukkit.event.player.PlayerLoginEvent;
023import org.bukkit.event.player.PlayerLoginEvent.Result;
024
025import be.Balor.Manager.Permissions.PermissionManager;
026import be.Balor.Tools.Utils;
027import be.Balor.bukkit.AdminCmd.ACHelper;
028
029/**
030 * @author Balor (aka Antoine Aflalo)
031 * 
032 */
033public class ACLockedServerListener implements Listener {
034        @EventHandler(priority = EventPriority.HIGHEST)
035        public void onPlayerLogin(final PlayerLoginEvent event) {
036                if (!event.getResult().equals(Result.ALLOWED)) {
037                        return;
038                }
039                if (ACHelper.getInstance().isServerLocked()
040                                && !PermissionManager.hasPerm(event.getPlayer(),
041                                                "admincmd.server.lockdown", false)) {
042                        event.disallow(Result.KICK_OTHER, Utils.I18n("serverLockMessage"));
043                }
044        }
045}
