017    package be.Balor.Listeners.Commands;
018    
019    import org.bukkit.event.EventHandler;
020    import org.bukkit.event.Listener;
021    import org.bukkit.event.block.BlockDamageEvent;
022    import org.bukkit.inventory.ItemStack;
023    
024    import be.Balor.Player.ACPlayer;
025    import be.Balor.Tools.Type;
026    import be.Balor.bukkit.AdminCmd.ConfigEnum;
027    
028    /**
029     * @author Balor (aka Antoine Aflalo)
030     * 
031     */
032    public class ACSuperBreaker implements Listener {
033            @EventHandler(ignoreCancelled = true)
034            public void onBlockDamage(final BlockDamageEvent event) {
035                    final ACPlayer player = ACPlayer.getPlayer(event.getPlayer());
036                    if (!player.hasPower(Type.SUPER_BREAKER))
037                            return;
038                    final ItemStack itemInHand = event.getItemInHand();
039                    if (itemInHand != null && itemInHand.getTypeId() == ConfigEnum.SB_ITEM.getInt()) {
040                            event.setInstaBreak(true);
041                            itemInHand.setDurability((short) 0);
042                    }
043            }
044    }
