017package be.Balor.Listeners.Commands;
018
019import org.bukkit.Location;
020import org.bukkit.entity.Fireball;
021import org.bukkit.entity.Player;
022import org.bukkit.event.EventHandler;
023import org.bukkit.event.Listener;
024import org.bukkit.event.block.Action;
025import org.bukkit.event.player.PlayerInteractEvent;
026
027import be.Balor.Player.ACPlayer;
028import be.Balor.Tools.Type;
029
030/**
031 * @author Balor (aka Antoine Aflalo)
032 * 
033 */
034public class ACFireballListener implements Listener {
035        @EventHandler
036        public void onPlayerInteract(final PlayerInteractEvent event) {
037                if ((event.getAction() != Action.LEFT_CLICK_BLOCK)
038                                && (event.getAction() != Action.LEFT_CLICK_AIR)) {
039                        return;
040                }
041                final Player player = event.getPlayer();
042                final ACPlayer acPlayer = ACPlayer.getPlayer(player);
043                Float power;
044                if ((power = acPlayer.getPower(Type.FIREBALL).getFloat(0)) == 0) {
045                        return;
046                }
047                final Location playerLoc = player.getLocation();
048                final Location fbLocation = playerLoc.add(
049                                playerLoc
050                                                .getDirection()
051                                                .normalize()
052                                                .multiply(2)
053                                                .toLocation(player.getWorld(), playerLoc.getYaw(),
054                                                                playerLoc.getPitch())).add(0, 1D, 0);
055                final Fireball f = player.getWorld().spawn(fbLocation, Fireball.class);
056                f.setYield(power);
057                f.setShooter(player);
058
059        }
060}
