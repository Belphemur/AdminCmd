017    package be.Balor.Listeners.Commands;
018    
019    import org.bukkit.Location;
020    import org.bukkit.entity.Fireball;
021    import org.bukkit.entity.Player;
022    import org.bukkit.event.EventHandler;
023    import org.bukkit.event.Listener;
024    import org.bukkit.event.block.Action;
025    import org.bukkit.event.player.PlayerInteractEvent;
026    
027    import be.Balor.Player.ACPlayer;
028    import be.Balor.Tools.Type;
029    
030    /**
031     * @author Balor (aka Antoine Aflalo)
032     * 
033     */
034    public class ACFireballListener implements Listener {
035            @EventHandler
036            public void onPlayerInteract(final PlayerInteractEvent event) {
037                    if ((event.getAction() != Action.LEFT_CLICK_BLOCK)
038                                    && (event.getAction() != Action.LEFT_CLICK_AIR))
039                            return;
040                    final Player player = event.getPlayer();
041                    final ACPlayer acPlayer = ACPlayer.getPlayer(player);
042                    Float power;
043                    if ((power = acPlayer.getPower(Type.FIREBALL).getFloat(0)) == 0)
044                            return;
045                    final Location playerLoc = player.getLocation();
046                    final Location fbLocation = playerLoc.add(
047                                    playerLoc.getDirection().normalize().multiply(2)
048                                                    .toLocation(player.getWorld(), playerLoc.getYaw(), playerLoc.getPitch()))
049                                    .add(0, 1D, 0);
050                    final Fireball f = player.getWorld().spawn(fbLocation, Fireball.class);
051                    f.setYield(power);
052                    f.setShooter(player);
053    
054            }
055    }
