017    package be.Balor.Listeners;
018    
019    import org.bukkit.event.EventHandler;
020    import org.bukkit.event.Listener;
021    import org.bukkit.event.block.SignChangeEvent;
022    
023    import be.Balor.Manager.Permissions.PermissionManager;
024    import be.Balor.Tools.Utils;
025    
026    /**
027     * @author Balor (aka Antoine Aflalo)
028     * 
029     */
030    public class ACColorSignListener implements Listener {
031            @EventHandler(ignoreCancelled = true)
032            public void onSignChange(final SignChangeEvent event) {
033                    String parsed = null;
034                    String line;
035                    if (Utils.signExtention && (line = event.getLine(0)) != null && line.endsWith("Sign]"))
036                            return;
037                    for (int i = 0; i < 4; i++) {
038                            line = event.getLine(i);
039                            if (line == null || (line != null && line.isEmpty()))
040                                    continue;
041                            if (!Utils.regexColorParser.matcher(line).find())
042                                    continue;
043                            if (!PermissionManager.hasPerm(event.getPlayer(), "admincmd.coloredsign.create"))
044                                    return;
045                            parsed = Utils.colorParser(line);
046                            if (parsed != null)
047                                    event.setLine(i, parsed);
048    
049                    }
050            }
051    }
