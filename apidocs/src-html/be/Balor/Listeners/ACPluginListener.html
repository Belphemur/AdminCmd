022import org.bukkit.event.EventHandler;

029

033import be.Balor.Tools.Utils;

037import be.Balor.bukkit.AdminCmd.ConfigEnum;
038import belgium.Balor.Workers.InvisibleWorker;
039
040import com.nijikokun.bukkit.Permissions.Permissions;
041import com.platymuus.bukkit.permissions.PermissionsPlugin;
042
043import de.diddiz.LogBlock.LogBlock;
044
045/**
046 * @author Balor (aka Antoine Aflalo)
047 * 
048 */
049public class ACPluginListener implements Listener {
050
051        @EventHandler(priority = EventPriority.MONITOR)
052        public void onPluginDisable(final PluginDisableEvent event) {
053                ACPluginManager.unRegisterACPlugin(event.getPlugin());
054        }
055
056        @EventHandler(priority = EventPriority.MONITOR)
057        public void onPluginEnable(final PluginEnableEvent event) {
058                if (!PermissionManager.isPermissionsExSet()) {
059                        final Plugin Permissions = ACPluginManager.getServer()
060                                        .getPluginManager().getPlugin("PermissionsEx");
061                        if (Permissions != null) {
062                                if (Permissions.isEnabled()) {
063                                        PermissionManager.setPEX(PermissionsEx
064                                                        .getPermissionManager());
065                                }
066                        }
067                }
068                if (!PermissionManager.isGroupManagerSet()) {
069                        final Plugin GMplugin = ACPluginManager.getServer()
070                                        .getPluginManager().getPlugin("GroupManager");
071                        if (GMplugin != null) {
072                                if (GMplugin.isEnabled()) {
073                                        PermissionManager.setGroupManager(GMplugin);
074                                }
075                        }
076                }
077                if (!PermissionManager.isYetiPermissionsSet()) {
078                        final Plugin Permissions = ACPluginManager.getServer()
079                                        .getPluginManager().getPlugin("Permissions");
080                        if (Permissions != null) {
081                                if (Permissions.isEnabled()) {
082                                        PermissionManager
083                                                        .setYetiPermissions(((Permissions) Permissions)
084                                                                        .getHandler());
085                                }
086                        }
087                }
088                if (!PermissionManager.isbPermissionsSet()) {
089                        final Plugin plugin = ACPluginManager.getServer()
090                                        .getPluginManager().getPlugin("bPermissions");
091                        if (plugin != null) {
092                                String version = plugin.getDescription().getVersion();
093                                version = version.replace(".", "");
094                                final int ver = Integer.parseInt(version);
095                                if (ver < 285) {
096                                        ACLogger.info("You are using bPermissions v"
097                                                        + plugin.getDescription().getVersion()
098                                                        + ". This is an outdated version, permission support for bPermission will be disabled.");
099                                        return;
100                                }
101                                PermissionManager.setbPermissions();
102                        }
103                }
104                if (!PermissionManager.isPermissionsBukkitSet()) {
105                        final Plugin plugin = ACPluginManager.getServer()
106                                        .getPluginManager().getPlugin("PermissionsBukkit");
107                        if (plugin != null) {
108                                PermissionManager
109                                                .setPermissionsBukkit((PermissionsPlugin) plugin);
110                        }
111                }
112                if (Utils.oddItem == null) {
113                        final Plugin items = ACPluginManager.getServer().getPluginManager()
114                                        .getPlugin("OddItem");
115                        if (items != null && items.isEnabled()) {
116                                Utils.oddItem = (OddItemBase) items;
117                                ACLogger.info("Successfully linked with OddItem");
118                        }
119                }
120                if (!SuperPermissions.isApiSet()) {
121                        final Plugin mChatPlugin = ACPluginManager.getServer()
122                                        .getPluginManager().getPlugin("mChatSuite");
123                        if (mChatPlugin != null && mChatPlugin.isEnabled()) {
124                                SuperPermissions.setmChatapi((mChatSuite) mChatPlugin);
125                                Utils.mChatPresent = true;
126                                ACLogger.info("Successfully linked with mChatSuite");
127                        }
128                }
129                if (Utils.logBlock == null) {
130                        final Plugin plugin = ACPluginManager.getServer()
131                                        .getPluginManager().getPlugin("LogBlock");
132                        if (plugin != null && plugin.isEnabled()) {
133                                Utils.setLogBlock(((LogBlock) plugin).getConsumer());
134                                ACLogger.info("Successfully linked with LogBlock");
135                        }
136                }
137                if (InvisibleWorker.dynmapAPI == null) {
138                        final Plugin plugin = ACPluginManager.getServer()
139                                        .getPluginManager().getPlugin("dynmap");
140                        if (plugin != null && plugin.isEnabled()) {
141                                InvisibleWorker.dynmapAPI = (DynmapAPI) plugin;
142                                ACLogger.info("Successfully linked with Dynmap");
143                        }
144                }
145                if (ConfigEnum.H_ALLPLUGIN.getBoolean()) {
146                        for (final Plugin plugin : event.getPlugin().getServer()
147                                        .getPluginManager().getPlugins()) {
148                                HelpLister.getInstance().addPlugin(plugin);
149                        }
150                }
151                if (!Utils.signExtention) {
152                        final Plugin plugin = ACPluginManager.getServer()
153                                        .getPluginManager().getPlugin("SignExtensions");
154                        if (plugin != null) {
155                                Utils.signExtention = true;
156                        }
157                }
158        }
159}
