028public enum Type {
029        FLY(Category.SUPER_POWER),
030        FLY_OLD(Category.SUPER_POWER),
031        VULCAN(Category.SUPER_POWER),
032        GOD(Category.SUPER_POWER),
033        THOR(Category.SUPER_POWER),
034        BANNED(Category.SANCTION),
035        KICKED(Category.SANCTION),
036        FIREBALL(Category.SUPER_POWER),
037        SPYMSG(Category.OTHER),
038        FROZEN(Category.SANCTION),
039        MUTED(Category.SANCTION),
040        MUTED_COMMAND(Category.SANCTION),
041        MOB_LIMIT(Category.WORLD),
042        NO_PICKUP(Category.SUPER_POWER),
043        WEATHER_FROZEN(Category.WORLD),
044        REPEAT_CMD(Category.OTHER),
045        TIME_FROZEN(Category.WORLD),
046        TP_REQUEST(Category.OTHER),
047        TP_AT_SEE(Category.SUPER_POWER),
048        SUPER_BREAKER(Category.SUPER_POWER),
049        INVISIBLE(Category.SUPER_POWER),
050        FAKEQUIT(Category.SUPER_POWER),
051        ETERNAL(Category.SUPER_POWER),
052        NO_DROP(Category.SUPER_POWER),
053        EGG(Category.SUPER_POWER),
054        CUSTOM(Category.MISC);
055
056        private static final Map<String, Type> lookupName = new HashMap<String, Type>();
057
058        @Override
059        public String toString() {
060                final String s = super.toString();
061                return s.toLowerCase();
062        }
063
064        public String display() {
065                final String s = super.toString();
066                return s.substring(0, 1)
067                                + s.substring(1).toLowerCase().replaceAll("_", " ");
068        }
069
070        private final Category category;
071
072        private Type(final Category category) {
073                this.category = category;
074        }
075
076        public int id() {
077                return ordinal() ^ getClass().getName().hashCode();
078        }
079
080        /**
081         * Attempts to match the Type with the given name. This is a match lookup;
082         * names will be converted to uppercase, then stripped of special characters
083         * in an attempt to format it like the enum
084         * 
085         * @param name
086         *            Name of the type to get
087         * @return Type if found, or null
088         */
089        public static Type matchType(final String name) {
090                Type result = null;
091
092                String filtered = name.toUpperCase();
093                filtered = filtered.replaceAll("\\s+", "_").replaceAll("\\W", "");
094                result = lookupName.get(filtered);
095
096                return result;
097        }
098
099        public static void addCustomPower(final String name) {
100                lookupName.put(name, CUSTOM);
101        }
102
103        static {
104                for (final Type type : values()) {
105                        lookupName.put(type.name(), type);
106                }
107        }
108
109        /**
110         * Gets the Category assigned to this type
111         * 
112         * @return Category of this Type
113         */
114        public Category getCategory() {
115                return category;
116        }

151        public enum ArmorPart {
152                BOOTS(0, 301), LEGS(1, 300), CHEST(2, 299), HEAD(3, 298);
153                private final int placeInInventory;
154                private final List<Integer> possibleId = new ArrayList<Integer>();
155                private static final int nbEquipment = 5;
156                private static Map<String, ArmorPart> perName = new HashMap<String, Type.ArmorPart>();
157                static {
158                        for (final ArmorPart part : ArmorPart.values()) {
159                                perName.put(part.name().toLowerCase(), part);
160                        }
161                }
162
163                public static ArmorPart getByName(final String name) {
164                        return perName.get(name.toLowerCase());
165                }
166
167                /**
168                 * @param placeInInventory
169                 */
170                private ArmorPart(final int placeInInventory, final int firstPossible) {
171                        this.placeInInventory = placeInInventory;
172                        for (int i = 0; i < nbEquipment; i++) {
173                                possibleId.add((i * 4) + firstPossible);
174                        }
175                }
176
177                /**
178                 * @return the placeInInventory
179                 */
180                public int getPlaceInInventory() {
181                        return placeInInventory;
182                }
183
184                @Override
185                public String toString() {
186                        return super.toString().toLowerCase();
187                }
188
189                /**
190                 * Check if the id is valid for that ArmorPart
191                 * 
192                 * @param toCheck
193                 * @return
194                 */
195                public boolean isValid(final int toCheck) {
196                        return possibleId.contains(toCheck);
197                }
198        }
