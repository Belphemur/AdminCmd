017    package be.Balor.Tools;
018    
019    import java.util.HashMap;
020    import java.util.Map;
021    
022    import org.bukkit.Material;
023    import org.bukkit.enchantments.Enchantment;
024    import org.bukkit.inventory.ItemStack;
025    
026    /**
027     * @author Balor (aka Antoine Aflalo)
028     * 
029     */
030    public class MaterialContainer implements Comparable<MaterialContainer> {
031            private Material material = null;
032            private short dmg = 0;
033            private int amount = 1;
034            private final Map<Enchantment, Integer> enchantments;
035    
036            public MaterialContainer(final ItemStack is) {
037                    material = is.getType();
038                    dmg = is.getDurability();
039                    this.enchantments = is.getEnchantments();
040            }
041    
042            public MaterialContainer(final String mat) {
043                    String[] info = new String[2];
044                    if (mat.contains(":"))
045                            info = mat.split(":");
046                    else {
047                            info[0] = mat;
048                            info[1] = "0";
049                    }
050                    parseMat(info[0]);
051                    parseDmg(info[1]);
052                    this.enchantments = new HashMap<Enchantment, Integer>();
053            }
054    
055            /**
056             * @param amount
057             *            the amount to set
058             */
059            public void setAmount(final int amount) {
060                    this.amount = amount;
061            }
062    
063            public MaterialContainer(final String mat, final String damage) {
064                    parseMat(mat);
065                    parseDmg(damage);
066                    this.enchantments = new HashMap<Enchantment, Integer>();
067            }
068    
069            public MaterialContainer() {
070                    this.enchantments = new HashMap<Enchantment, Integer>();
071            }
072    
073            private void parseMat(final String material) {
074                    Material m = null;
075                    try {
076                            final int id = Integer.parseInt(material);
077                            m = Material.getMaterial(id);
078                    } catch (final NumberFormatException e) {
079                            m = Material.matchMaterial(material);
080                    }
081                    this.material = m;
082            }
083    
084            private void parseDmg(final String damage) {
085                    short d = 0;
086                    try {
087                            d = Short.parseShort(damage);
088                    } catch (final NumberFormatException e) {
089                    }
090                    this.dmg = d;
091            }
092    
093            public boolean isNull() {
094                    return material == null;
095            }
096    
097            /**
098             * Transform the MaterialContainer to an ItemStack
099             * 
100             * @param amount
101             * @return
102             */
103    
104            public ItemStack getItemStack() {
105                    final ItemStack toReturn = new ItemStack(material, amount, dmg);
106                    toReturn.addUnsafeEnchantments(enchantments);
107                    return toReturn;
108            }
109    
110            @Override
111            public String toString() {
112                    if (material != null)
113                            return material.getId() + ":" + dmg;
114                    else
115                            return "";
116            }
117    
118            public String display() {
119                    if (material != null)
120                            return material + ":" + dmg;
121                    else
122                            return "";
123            }
124    
125            /**
126             * @return the material
127             */
128            public Material getMaterial() {
129                    return material;
130            }
131    
132            /**
133             * @return the amount
134             */
135            public int getAmount() {
136                    return amount;
137            }
138    
139            /**
140             * @return the dmg
141             */
142            public short getDmg() {
143                    return dmg;
144            }
145    
146            /**
147             * @return the enchantments
148             */
149            public Map<Enchantment, Integer> getEnchantments() {
150                    return enchantments;
151            }
152    
153            /**
154             * @param material
155             *            the material to set
156             */
157            public void setMaterial(final Material material) {
158                    this.material = material;
159            }
160    
161            /**
162             * @param dmg
163             *            the dmg to set
164             */
165            public void setDmg(final short dmg) {
166                    this.dmg = dmg;
167            }
168    
169            /*
170             * (non-Javadoc)
171             * 
172             * @see java.lang.Comparable#compareTo(java.lang.Object)
173             */
174            @Override
175            public int compareTo(final MaterialContainer o) {
176                    if (material.equals(o.getMaterial()))
177                            return dmg - o.getDmg();
178                    return material.compareTo(o.getMaterial());
179            }
180    
181            /*
182             * (non-Javadoc)
183             * 
184             * @see java.lang.Object#hashCode()
185             */
186            @Override
187            public int hashCode() {
188                    final int prime = 31;
189                    int result = 1;
190                    result = prime * result + dmg;
191                    result = prime * result + ((material == null) ? 0 : material.hashCode());
192                    return result;
193            }
194    
195            /*
196             * (non-Javadoc)
197             * 
198             * @see java.lang.Object#equals(java.lang.Object)
199             */
200            @Override
201            public boolean equals(final Object obj) {
202                    if (this == obj)
203                            return true;
204                    if (obj == null)
205                            return false;
206                    if (!(obj instanceof MaterialContainer))
207                            return false;
208                    final MaterialContainer other = (MaterialContainer) obj;
209                    if (dmg != other.dmg)
210                            return false;
211                    if (material != other.material)
212                            return false;
213                    return true;
214            }
215    
216    }
