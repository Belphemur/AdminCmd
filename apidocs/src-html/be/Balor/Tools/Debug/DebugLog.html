017package be.Balor.Tools.Debug;
018
019import java.io.File;
020import java.io.IOException;
021import java.util.Stack;
022import java.util.logging.FileHandler;
023import java.util.logging.Handler;
024import java.util.logging.Level;
025import java.util.logging.Logger;
026
027import com.google.common.io.Files;
028
029/**
030 * @author Balor (aka Antoine Aflalo)
031 * 
032 */
033public class DebugLog {
034        private static class DebugInfo {
035                private final String blockMsg;
036                private final Long timeStamp;
037
038                /**
039                 * @param blockMsg
040                 * @param timeStamp
041                 */
042                public DebugInfo(final String blockMsg) {
043                        this.blockMsg = blockMsg;
044                        this.timeStamp = System.currentTimeMillis();
045                }
046
047                /**
048                 * @return the blockMsg
049                 */
050                public String getBlockMsg() {
051                        return blockMsg;
052                }
053
054                /**
055                 * @return the timeStamp
056                 */
057                public Long getElapsedTime() {
058                        return System.currentTimeMillis() - timeStamp;
059                }
060
061        }
062
063        public static final Logger INSTANCE = Logger.getLogger("AdminCmd");
064        public static final String BEGIN_PREFIX = "[BEGIN] ";
065        public static final String END_PREFIX = "[END] ";
066        private static final Stack<DebugInfo> debugInfos = new Stack<DebugInfo>();
067        static {
068                INSTANCE.setUseParentHandlers(false);
069                INSTANCE.setLevel(Level.ALL);
070        }
071
072        public static void setFile(final String path) {
073                FileHandler fh;
074                try {
075                        // This block configure the logger with handler and formatter
076                        final File file = new File(path + File.separator + "debug.log");
077                        if (file.exists()) {
078                                file.delete();
079                        } else {
080                                Files.createParentDirs(file);
081                        }
082
083                        fh = new FileHandler(file.getPath(), true);
084                        INSTANCE.addHandler(fh);
085                        fh.setFormatter(new LogFormatter());
086
087                        // the following statement is used to log any messages
088                        INSTANCE.info("Logger created");
089
090                } catch (final SecurityException e) {
091                        e.printStackTrace();
092                } catch (final IOException e) {
093                        e.printStackTrace();
094                }
095        }
096
097        public static void stopLogging() {
098                for (final Handler h : INSTANCE.getHandlers()) {
099                        h.close();
100                        INSTANCE.removeHandler(h);
101                }
102        }
103
104        /**
105         * Begin logging of a block of code
106         * 
107         * @param msg
108         */
109        public static void beginInfo(final String msg) {
110                debugInfos.add(new DebugInfo(msg));
111                INSTANCE.info(getSpaces() + BEGIN_PREFIX + msg);
112
113        }
114
115        /**
116         * @return
117         */
118        private static String getSpaces() {
119                final StringBuffer space = new StringBuffer();
120                for (int i = 0; i < debugInfos.size(); i++) {
121                        space.append("\t");
122                }
123                return space.toString();
124        }
125
126        /**
127         * End logging a block of code
128         */
129        public static void endInfo() {
130                try {
131                        final String spaces = getSpaces();
132                        final DebugInfo info = debugInfos.pop();
133                        INSTANCE.info(spaces + END_PREFIX + info.getBlockMsg() + " => "
134                                        + info.getElapsedTime() + " milisec");
135                } catch (final Exception e) {
136                }
137
138        }
139}
