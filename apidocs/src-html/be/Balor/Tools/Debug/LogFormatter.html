020    package be.Balor.Tools.Debug;
021    
022    import java.text.DateFormat;
023    import java.text.MessageFormat;
024    import java.text.SimpleDateFormat;
025    import java.util.Date;
026    import java.util.logging.Formatter;
027    import java.util.logging.LogManager;
028    import java.util.logging.LogRecord;
029    
030    /**
031     * A {@link Formatter} that may be customised in a {@code logging.properties}
032     * file. The syntax of the property
033     * {@code com.thinktankmaths.logging.TerseFormatter.format} specifies the
034     * output. A newline will be appended to the string and the following special
035     * characters will be expanded (case sensitive):-
036     * <ul>
037     * <li>{@code %m} - message</li>
038     * <li>{@code %L} - log level</li>
039     * <li>{@code %n} - name of the logger</li>
040     * <li>{@code %t} - a timestamp (in ISO-8601 "yyyy-MM-dd HH:mm:ss Z" format)</li>
041     * <li>{@code %M} - source method name (if available, otherwise "?")</li>
042     * <li>{@code %c} - source class name (if available, otherwise "?")</li>
043     * <li>{@code %C} - source simple class name (if available, otherwise "?")</li>
044     * <li>{@code %T} - thread ID</li>
045     * <li>{@code %S} - StackTrace if exeption thrown</li>
046     * </ul>
047     * The default format is {@value #DEFAULT_FORMAT}. Curly brace characters are
048     * not allowed.
049     * 
050     * @author Samuel Halliday
051     */
052    public class LogFormatter extends Formatter {
053            private static final String DEFAULT_FORMAT = "%t [%L] *%C.%M* : %m [%T] %S";
054    
055            private final MessageFormat messageFormat;
056    
057            private final DateFormat dateFormat = new SimpleDateFormat("HH:mm:ss");
058    
059            /** */
060            public LogFormatter() {
061                    super();
062    
063                    // load the format from logging.properties
064                    final String propName = getClass().getName() + ".format";
065                    String format = LogManager.getLogManager().getProperty(propName);
066                    if (format == null || format.trim().length() == 0)
067                            format = DEFAULT_FORMAT;
068                    if (format.contains("{") || format.contains("}"))
069                            throw new IllegalArgumentException("curly braces not allowed");
070    
071                    // convert it into the MessageFormat form
072                    format = format.replace("%L", "{0}").replace("%m", "{1}").replace("%M", "{2}")
073                                    .replace("%t", "{3}").replace("%c", "{4}").replace("%T", "{5}")
074                                    .replace("%n", "{6}").replace("%C", "{7}").replace("%S", "{8}")
075                                    + "\n";
076    
077                    messageFormat = new MessageFormat(format);
078            }
079    
080            @Override
081            public String format(final LogRecord record) {
082                    final String[] arguments = new String[9];
083                    // %L
084                    arguments[0] = record.getLevel().toString();
085                    arguments[1] = record.getMessage();
086                    // sometimes the message is empty, but there is a throwable
087                    if (arguments[1] == null || arguments[1].length() == 0) {
088                            final Throwable thrown = record.getThrown();
089                            if (thrown != null) {
090                                    arguments[1] = thrown.getMessage();
091                            }
092                    }
093                    // %m
094                    arguments[1] = record.getMessage();
095                    // %M
096                    if (record.getSourceMethodName() != null) {
097                            arguments[2] = record.getSourceMethodName();
098                    } else {
099                            arguments[2] = "?";
100                    }
101                    // %t
102                    final Date date = new Date(record.getMillis());
103                    synchronized (dateFormat) {
104                            arguments[3] = dateFormat.format(date);
105                    }
106                    // %c
107                    if (record.getSourceClassName() != null) {
108                            arguments[4] = record.getSourceClassName();
109                    } else {
110                            arguments[4] = "?";
111                    }
112                    // %T
113                    arguments[5] = Integer.valueOf(record.getThreadID()).toString();
114                    // %n
115                    arguments[6] = record.getLoggerName();
116                    // %C
117                    final int start = arguments[4].lastIndexOf(".") + 1;
118                    if (start > 0 && start < arguments[4].length()) {
119                            arguments[7] = arguments[4].substring(start);
120                    } else {
121                            arguments[7] = arguments[4];
122                    }
123                    // %S
124                    if (record.getThrown() == null)
125                            arguments[8] = "";
126                    else {
127                            if (record.getMessage() != null)
128                                    arguments[1] += " " + record.getThrown().toString();
129                            else
130                                    arguments[1] = record.getThrown().toString();
131                            String stackTrace = "\n";
132                            for (final StackTraceElement st : record.getThrown().getStackTrace()) {
133                                    stackTrace += "\t" + st.toString() + "\n";
134                            }
135                            arguments[8] = stackTrace;
136                    }
137                    synchronized (messageFormat) {
138                            return messageFormat.format(arguments);
139                    }
140            }
141    }
