020package be.Balor.Tools.Debug;
021
022import java.text.DateFormat;
023import java.text.MessageFormat;
024import java.text.SimpleDateFormat;
025import java.util.Date;
026import java.util.logging.Formatter;
027import java.util.logging.LogManager;
028import java.util.logging.LogRecord;
029
030/**
031 * A {@link Formatter} that may be customised in a {@code logging.properties}
032 * file. The syntax of the property
033 * {@code com.thinktankmaths.logging.TerseFormatter.format} specifies the
034 * output. A newline will be appended to the string and the following special
035 * characters will be expanded (case sensitive):-
036 * <ul>
037 * <li>{@code %m} - message</li>
038 * <li>{@code %L} - log level</li>
039 * <li>{@code %n} - name of the logger</li>
040 * <li>{@code %t} - a timestamp (in ISO-8601 "yyyy-MM-dd HH:mm:ss Z" format)</li>
041 * <li>{@code %M} - source method name (if available, otherwise "?")</li>
042 * <li>{@code %c} - source class name (if available, otherwise "?")</li>
043 * <li>{@code %C} - source simple class name (if available, otherwise "?")</li>
044 * <li>{@code %T} - thread ID</li>
045 * <li>{@code %S} - StackTrace if exeption thrown</li>
046 * </ul>
047 * The default format is {@value #DEFAULT_FORMAT}. Curly brace characters are
048 * not allowed.
049 * 
050 * @author Samuel Halliday
051 */
052public class LogFormatter extends Formatter {
053        private static final String DEFAULT_FORMAT = "%t [%L] *%C.%M* : %m [%T] %S";
054
055        private final MessageFormat messageFormat;
056
057        private final DateFormat dateFormat = new SimpleDateFormat("HH:mm:ss.SSS");
058
059        /** */
060        public LogFormatter() {
061                super();
062
063                // load the format from logging.properties
064                final String propName = getClass().getName() + ".format";
065                String format = LogManager.getLogManager().getProperty(propName);
066                if (format == null || format.trim().length() == 0) {
067                        format = DEFAULT_FORMAT;
068                }
069                if (format.contains("{") || format.contains("}")) {
070                        throw new IllegalArgumentException("curly braces not allowed");
071                }
072
073                // convert it into the MessageFormat form
074                format = format.replace("%L", "{0}").replace("%m", "{1}")
075                                .replace("%M", "{2}").replace("%t", "{3}").replace("%c", "{4}")
076                                .replace("%T", "{5}").replace("%n", "{6}").replace("%C", "{7}")
077                                .replace("%S", "{8}")
078                                + "\n";
079
080                messageFormat = new MessageFormat(format);
081        }
082
083        @Override
084        public String format(final LogRecord record) {
085                final String[] arguments = new String[9];
086                // %L
087                arguments[0] = record.getLevel().toString();
088                arguments[1] = record.getMessage();
089                // sometimes the message is empty, but there is a throwable
090                if (arguments[1] == null || arguments[1].length() == 0) {
091                        final Throwable thrown = record.getThrown();
092                        if (thrown != null) {
093                                arguments[1] = thrown.getMessage();
094                        }
095                }
096                // %m
097                arguments[1] = record.getMessage();
098                // %M
099                if (record.getSourceMethodName() != null) {
100                        arguments[2] = record.getSourceMethodName();
101                } else {
102                        arguments[2] = "?";
103                }
104                // %t
105                final Date date = new Date(record.getMillis());
106                synchronized (dateFormat) {
107                        arguments[3] = dateFormat.format(date);
108                }
109                // %c
110                if (record.getSourceClassName() != null) {
111                        arguments[4] = record.getSourceClassName();
112                } else {
113                        arguments[4] = "?";
114                }
115                // %T
116                arguments[5] = Integer.valueOf(record.getThreadID()).toString();
117                // %n
118                arguments[6] = record.getLoggerName();
119                // %C
120                final int start = arguments[4].lastIndexOf(".") + 1;
121                if (start > 0 && start < arguments[4].length()) {
122                        arguments[7] = arguments[4].substring(start);
123                } else {
124                        arguments[7] = arguments[4];
125                }
126                // %S
127                if (record.getThrown() == null) {
128                        arguments[8] = "";
129                } else {
130                        if (record.getMessage() != null) {
131                                arguments[1] += " " + record.getThrown().toString();
132                        } else {
133                                arguments[1] = record.getThrown().toString();
134                        }
135                        String stackTrace = "\n";
136                        for (final StackTraceElement st : record.getThrown()
137                                        .getStackTrace()) {
138                                stackTrace += "\t" + st.toString() + "\n";
139                        }
140                        arguments[8] = stackTrace;
141                }
142                synchronized (messageFormat) {
143                        return messageFormat.format(arguments);
144                }
145        }
146}
