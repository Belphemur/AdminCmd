043            REPEAT_CMD(Category.OTHER),
044            TIME_FREEZED(Category.WORLD),

055            private static final Map<String, Type> lookupName = new HashMap<String, Type>();
056    
057            @Override
058            public String toString() {
059                    final String s = super.toString();
060                    return s.toLowerCase();
061            }
062    
063            public String display() {
064                    final String s = super.toString();
065                    return s.substring(0, 1) + s.substring(1).toLowerCase().replaceAll("_", " ");
066            }
067    
068            private final Category category;
069    
070            private Type(final Category category) {
071                    this.category = category;
072            }
073    
074            public int id() {
075                    return ordinal() ^ getClass().getName().hashCode();
076            }
077    
078            /**
079             * Attempts to match the Type with the given name. This is a match lookup;
080             * names will be converted to uppercase, then stripped of special characters
081             * in an attempt to format it like the enum
082             * 
083             * @param name
084             *            Name of the type to get
085             * @return Type if found, or null
086             */
087            public static Type matchType(final String name) {
088                    Type result = null;
089    
090                    String filtered = name.toUpperCase();
091                    filtered = filtered.replaceAll("\\s+", "_").replaceAll("\\W", "");
092                    result = lookupName.get(filtered);
093    
094                    return result;
095            }
096    
097            public static void addCustomPower(final String name) {
098                    lookupName.put(name, CUSTOM);
099            }
100    
101            static {
102                    for (final Type type : values()) {
103                            lookupName.put(type.name(), type);
104                    }
105            }
106    
107            /**
108             * Gets the Category assigned to this type
109             * 
110             * @return Category of this Type
111             */
112            public Category getCategory() {
113                    return category;
114            }

159            public enum ArmorPart {
160                    BOOTS(0, 301),
161                    LEGS(1, 300),
162                    CHEST(2, 299),
163                    HEAD(3, 298);
164                    private final int placeInInventory;
165                    private final List<Integer> possibleId = new ArrayList<Integer>();
166                    private static final int nbEquipment = 5;
167    
168                    /**
169                     * @param placeInInventory
170                     */
171                    private ArmorPart(final int placeInInventory, final int firstPossible) {
172                            this.placeInInventory = placeInInventory;
173                            for (int i = 0; i < nbEquipment; i++)
174                                    possibleId.add((i * 4) + firstPossible);
175                    }
176    
177                    /**
178                     * @return the placeInInventory
179                     */
180                    public int getPlaceInInventory() {
181                            return placeInInventory;
182                    }
183    
184                    @Override
185                    public String toString() {
186                            return super.toString().toLowerCase();
187                    }
188    
189                    /**
190                     * Check if the id is valid for that ArmorPart
191                     * 
192                     * @param toCheck
193                     * @return
194                     */
195                    public boolean isValid(final int toCheck) {
196                            return possibleId.contains(toCheck);
197                    }
198            }
