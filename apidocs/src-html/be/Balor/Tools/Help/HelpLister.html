017package be.Balor.Tools.Help;
018
019import java.util.ArrayList;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Set;
023
024import org.bukkit.ChatColor;
025import org.bukkit.command.CommandSender;
026import org.bukkit.plugin.Plugin;
027
028import be.Balor.Tools.Debug.DebugLog;
029import be.Balor.Tools.Help.String.ACMinecraftFontWidthCalculator;
030import be.Balor.Tools.Help.String.Str;
031import be.Balor.bukkit.AdminCmd.ConfigEnum;
032
033/**
034 * @author Balor (aka Antoine Aflalo)
035 * 
036 */
037public class HelpLister {
038        private static HelpLister instance = null;
039        private final HashMap<String, HelpList> plugins = new HashMap<String, HelpList>();
040        private final List<String> noCmds = new ArrayList<String>();
041
042        private HelpLister() {
043        }
044
045        /**
046         * @return the instance
047         */
048        public static HelpLister getInstance() {
049                if (instance == null) {
050                        instance = new HelpLister();
051                }
052                return instance;
053        }
054
055        public static void killInstance() {
056                instance = null;
057        }
058
059        /**
060         * Add a plugin to the lister
061         * 
062         * @param plugin
063         */
064        public void addPlugin(final Plugin plugin) {
065                final String pName = plugin.getDescription().getName();
066                if (!plugins.containsKey(pName) && !noCmds.contains(pName)) {
067                        try {
068                                plugins.put(pName, new HelpList(plugin));
069                        } catch (final IllegalArgumentException e) {
070                                noCmds.add(pName);
071                        }
072                }
073
074        }
075
076        /**
077         * 
078         * @return the list of all registered plugin
079         */
080        public Set<String> getPluginList() {
081                return plugins.keySet();
082        }
083
084        /**
085         * Add a new helpEntry for the wanted plugin. If the plugin is not found,
086         * add it in the database.
087         * 
088         * @param command
089         *            command
090         * @param description
091         *            description of the command
092         * @param plugin
093         *            plugin that is attached to the command
094         * @param permissions
095         *            list of permissions to execute the command.
096         * @param cmdName
097         *            true name of the command
098         */
099        public void addHelpEntry(final String command, final String description,
100                        final String detailedDesc, final String plugin,
101                        final List<String> permissions, final String cmdName) {
102                HelpList help = plugins.get(plugin);
103                if (help == null) {
104                        help = new HelpList(plugin);
105                        plugins.put(plugin, help);
106                }
107                help.addEntry(new HelpEntry(command, description, detailedDesc,
108                                permissions, cmdName));
109        }
110
111        public boolean removeHelpEntry(final String plugin, final String commandName) {
112                final HelpList help = plugins.get(plugin);
113                if (help == null) {
114                        DebugLog.INSTANCE.severe("Plugin " + plugin + " not found.");
115                        return false;
116                }
117                return help.removeEntry(commandName);
118        }
119
120        /**
121         * Send the help for the given plugin.
122         * 
123         * @param plugin
124         *            name of the plugin
125         * @param page
126         *            number of the page
127         * @param sender
128         *            the sender of the command
129         * @return
130         */
131        public boolean sendHelpPage(final String plugin, final int page,
132                        final CommandSender sender) {
133                final HelpList help = matchPlugin(plugin);
134                if (help == null) {
135                        return false;
136                }
137                final List<String> toDisplay = help.getPage(page, sender);
138                for (final String send : toDisplay) {
139                        for (final String l : send.split("\n")) {
140                                sender.sendMessage(ChatColor.getByChar(ConfigEnum.H_C_NORMAL
141                                                .getString()) + l);
142                        }
143                }
144                return true;
145
146        }
147
148        private HelpList matchPlugin(final String plugin) {
149                HelpList help = plugins.get(plugin);
150                if (help == null) {
151                        final String keyFound = Str.matchString(plugins.keySet(), plugin);
152                        if (keyFound == null) {
153                                return null;
154                        }
155                        help = plugins.get(keyFound);
156                }
157                return help;
158
159        }
160
161        /**
162         * Send the help of the given command to the command sender.
163         * 
164         * @param pluginName
165         *            name of the plugin where to search for the command. If
166         *            <b>NULL</b> search in every plugins.
167         * @param command
168         *            command to look for.
169         * @param sender
170         *            sender of the command.
171         * @return true if the command is found, else if not found.
172         */
173        public boolean sendHelpCmd(final String pluginName, final String command,
174                        final CommandSender sender, final boolean detailed) {
175                List<HelpEntry> chat = null;
176                boolean found = false;
177                if (pluginName == null) {
178                        for (final HelpList plugin : plugins.values()) {
179                                chat = plugin.getCommandMatch(command, sender);
180                                if (chat.isEmpty()) {
181                                        continue;
182                                }
183                                displayHelpMessage(chat, plugin.getPluginName(), sender,
184                                                detailed);
185                                found = true;
186                        }
187                        return found;
188                } else {
189                        final HelpList plugin = matchPlugin(pluginName);
190                        if (plugin == null) {
191                                return false;
192                        }
193                        chat = plugin.getCommandMatch(command, sender);
194                        if (chat.isEmpty()) {
195                                return false;
196                        }
197                        displayHelpMessage(chat, plugin.getPluginName(), sender, detailed);
198                }
199                return true;
200        }
201
202        public boolean sendHelpCmd(final String pluginName, final String command,
203                        final CommandSender sender) {
204                return sendHelpCmd(pluginName, command, sender, false);
205        }
206
207        /**
208         * Display the help for a specific cmdname
209         * 
210         * @param sender
211         * @param plugin
212         * @param cmdName
213         * @return
214         */
215        public boolean displayExactCommandHelp(final CommandSender sender,
216                        final String plugin, final String cmdName) {
217                final HelpList list = plugins.get(plugin);
218                if (list == null) {
219                        return false;
220                }
221                final HelpEntry entry = list.getExactCommand(cmdName);
222                if (entry == null) {
223                        return false;
224                }
225                final String chat = entry.chatString(false);
226                for (final String l : chat.split("\n")) {
227                        sender.sendMessage(l);
228                }
229                return true;
230
231        }
232
233        private void displayHelpMessage(final List<HelpEntry> list,
234                        final String pluginName, final CommandSender sender,
235                        final boolean detailed) {
236                sender.sendMessage(ChatColor.AQUA
237                                + ACMinecraftFontWidthCalculator.strPadCenterChat(
238                                                ChatColor.DARK_GREEN + " " + pluginName + " "
239                                                                + ChatColor.AQUA, '=') + "\n");
240                if (detailed) {
241                        final HelpEntry entry = list.get(0);
242                        final String chat = entry.chatString(detailed);
243                        for (final String l : chat.split("\n")) {
244                                sender.sendMessage(ChatColor.getByChar(ConfigEnum.H_C_NORMAL
245                                                .getString()) + l);
246                        }
247                        return;
248                }
249                for (final HelpEntry entry : list) {
250                        final String chat = entry.chatString(detailed);
251                        for (final String l : chat.split("\n")) {
252                                sender.sendMessage(ChatColor.getByChar(ConfigEnum.H_C_NORMAL
253                                                .getString()) + l);
254                        }
255                }
256        }
257}
