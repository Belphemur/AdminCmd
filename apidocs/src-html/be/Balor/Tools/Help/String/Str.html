033        public static String argStr(final String[] s, final String sep,
034                        final int start) {
035                String ret = "";
036                if (s != null) {
037                        for (int i = start; i < s.length; ++i) {
038                                ret += s[i];
039                                if (i + 1 < s.length) {
040                                        ret += sep;
041                                }
042                        }
043                }
044                return ret;
045        }

276        public static String[] removePortionOfArray(final String[] array,
277                        final int from, final int to, final String replace) {
278                final String[] newArray = new String[from + array.length - to
279                                - (replace == null ? 1 : 0)];
280                System.arraycopy(array, 0, newArray, 0, from);
281                if (replace != null) {
282                        newArray[from] = replace;
283                }
284                System.arraycopy(array, to + (replace == null ? 0 : 1), newArray, from
285                                + (replace == null ? 0 : 1), array.length - to - 1);
286                return newArray;
287        }

289        public static String[] removeCaseOfArray(final String[] array,
290                        final int index) {
291                final String[] newArray = new String[array.length - 1];
292                for (int i = 0; i < index; i++) {
293                        newArray[i] = array[i];
294                }
295                for (int j = index + 1; j < array.length; j++) {
296                        newArray[j - 1] = array[j];
297                }
298                return newArray;
299        }

309        /**
310         * Search for the given string in the list and return it.
311         * 
312         * @param container
313         * @param search
314         * @return
315         */
316        public static String matchString(final Collection<String> container,
317                        final String search) {
318                String found = null;
319                if (search == null) {
320                        return found;
321                }
322                final String lowerSearch = search.toLowerCase();
323                int delta = Integer.MAX_VALUE;
324                for (final String str : container) {
325                        if (str.toLowerCase().startsWith(lowerSearch)) {
326                                final int curDelta = str.length() - lowerSearch.length();
327                                if (curDelta < delta) {
328                                        found = str;
329                                        delta = curDelta;
330                                }
331                                if (curDelta == 0) {
332                                        break;
333                                }
334                        }
335                }
336                return found;
337
338        }
