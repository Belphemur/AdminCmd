034                    String ret = "";
035                    if (s != null) {
036                            for (int i = start; i < s.length; ++i) {
037                                    ret += s[i];
038                                    if (i + 1 < s.length) {
039                                            ret += sep;
040                                    }
041                            }
042                    }
043                    return ret;

274            public static String[] removePortionOfArray(final String[] array, final int from, final int to,
275                            final String replace) {
276                    final String[] newArray = new String[from + array.length - to - (replace == null ? 1 : 0)];
277                    System.arraycopy(array, 0, newArray, 0, from);
278                    if (replace != null)
279                            newArray[from] = replace;
280                    System.arraycopy(array, to + (replace == null ? 0 : 1), newArray, from
281                                    + (replace == null ? 0 : 1), array.length - to - 1);
282                    return newArray;
283            }

285            public static String[] removeCaseOfArray(final String[] array, final int index) {
286                    final String[] newArray = new String[array.length - 1];
287                    for (int i = 0; i < index; i++)
288                            newArray[i] = array[i];
289                    for (int j = index + 1; j < array.length; j++)
290                            newArray[j - 1] = array[j];
291                    return newArray;
292            }

302            /**
303             * Search for the given string in the list and return it.
304             * 
305             * @param container
306             * @param search
307             * @return
308             */
309            public static String matchString(final Collection<String> container, final String search) {
310                    String found = null;
311                    if (search == null)
312                            return found;
313                    final String lowerSearch = search.toLowerCase();
314                    int delta = Integer.MAX_VALUE;
315                    for (final String str : container) {
316                            if (str.toLowerCase().startsWith(lowerSearch)) {
317                                    final int curDelta = str.length() - lowerSearch.length();
318                                    if (curDelta < delta) {
319                                            found = str;
320                                            delta = curDelta;
321                                    }
322                                    if (curDelta == 0)
323                                            break;
324                            }
325                    }
326                    return found;
327    
328            }
