001package be.Balor.Tools.Help;
002
003import java.io.File;
004import java.io.FileInputStream;
005import java.io.IOException;
006import java.util.ArrayList;
007import java.util.HashMap;
008import java.util.List;
009import java.util.Map;
010import java.util.logging.Logger;
011
012import org.yaml.snakeyaml.Yaml;
013import org.yaml.snakeyaml.constructor.SafeConstructor;
014import org.yaml.snakeyaml.reader.UnicodeReader;
015
016import be.Balor.Tools.Files.Filters.YmlFilter;
017import be.Balor.Tools.Help.String.Str;
018import be.Balor.bukkit.AdminCmd.ConfigEnum;
019
020public class HelpLoader {
021        private static Logger HelpLogger = Logger.getLogger("Minecraft");
022
023        @SuppressWarnings({"unchecked", "rawtypes"})
024        public static void load(final File dataFolder) {
025                final File helpFolder = new File(dataFolder, "HelpFiles");
026                if (!helpFolder.exists()) {
027                        helpFolder.mkdirs();
028                } else if (helpFolder.isDirectory()) {
029                        int count = 0;
030                        // File files[] = helpFolder.listFiles(new YmlFilter());
031                        final File files[] = YmlFilter.INSTANCE.listRecursively(helpFolder,
032                                        1);
033                        final String folder = helpFolder.getAbsolutePath() + File.separator;
034                        if (files == null) {
035                                return;
036                        }
037                        final HashMap<String, Integer> filesLoaded = new HashMap<String, Integer>();
038                        for (final File insideFile : files) {
039                                String fileName, fn = insideFile.getAbsolutePath();
040                                if (fn.length() > folder.length()) {
041                                        fn = fn.substring(folder.length());
042                                        // if is a directory, use that as the name instead
043                                        if (fn.contains(File.separator)) {
044                                                fileName = fn.substring(0, fn.indexOf(File.separator));
045                                        } else {
046                                                fileName = insideFile.getName().replaceFirst(".yml$",
047                                                                "");
048                                        }
049                                } else {
050                                        fileName = insideFile.getName().replaceFirst(".yml$", "");
051                                }
052                                final Yaml yaml = new Yaml(new SafeConstructor());
053                                Map<String, Object> root = null;
054                                FileInputStream input = null;
055                                try {
056                                        input = new FileInputStream(insideFile);
057                                        root = (Map<String, Object>) yaml.load(new UnicodeReader(
058                                                        input));
059                                        if (root == null || root.isEmpty()) {
060                                                System.out.println("The file " + fn + " is empty");
061                                                continue;
062                                        }
063                                } catch (final Exception ex) {
064                                        // HelpLogger.severe("Error!", ex);
065                                        String err = Str.getStackStr(ex);
066                                        final String er = ex.getStackTrace()[0].toString();
067                                        if (err.contains(er)) {
068                                                err = err.substring(0, err.indexOf(er));
069                                        }
070                                        if (err.contains("at")) {
071                                                err = err.substring(0, err.lastIndexOf("at"));
072                                                err = err.substring(0, err.lastIndexOf("\n"));
073                                        }
074                                        HelpLogger.severe("Error loading " + fn + "\n" + err); // ex.getMessage()
075                                } finally {
076                                        if (input != null) {
077                                                try {
078                                                        input.close();
079                                                } catch (final IOException ex) {}
080                                        }
081                                }
082                                if (root != null) {
083                                        int num = 0;
084                                        for (final String helpKey : root.keySet()) {
085                                                final Map<String, Object> helpNode = (Map<String, Object>) root
086                                                                .get(helpKey);
087
088                                                if (!helpNode.containsKey("command")) {
089                                                        HelpLogger.warning("Help entry node \"" + helpKey
090                                                                        + "\" is missing a command name in " + fn);
091                                                        continue;
092                                                }
093                                                if (!helpNode.containsKey("description")) {
094                                                        HelpLogger.warning("Help entry node \"" + helpKey
095                                                                        + "\" is missing a description in " + fn);
096                                                        continue;
097                                                }
098                                                final String command = helpNode.get("command")
099                                                                .toString();
100                                                String description = "";
101                                                try {
102                                                        description = helpNode.get("description")
103                                                                        .toString();
104                                                } catch (final NullPointerException e) {
105
106                                                }
107                                                String detailedDescription = "";
108                                                try {
109                                                        detailedDescription = helpNode.get("detailed")
110                                                                        .toString();
111                                                } catch (final NullPointerException e) {
112
113                                                }
114                                                final String commandName = helpNode
115                                                                .containsKey("cmdname") ? helpNode.get(
116                                                                "cmdname").toString() : helpKey;
117                                                final String plugin = helpNode.containsKey("plugin")
118                                                                ? helpNode.get("plugin").toString()
119                                                                : fileName;
120                                                final ArrayList<String> permissions = new ArrayList<String>();
121
122                                                if (helpNode.containsKey("permissions")) {
123                                                        if (helpNode.get("permissions") instanceof List) {
124                                                                for (final Object permission : (List) helpNode
125                                                                                .get("permissions")) {
126                                                                        permissions.add(permission.toString());
127                                                                }
128                                                        } else {
129                                                                permissions.add(helpNode.get("permissions")
130                                                                                .toString());
131                                                        }
132                                                }
133                                                HelpLister.getInstance().addHelpEntry(command,
134                                                                description, detailedDescription, plugin,
135                                                                permissions, commandName);
136                                                ++num;
137                                                ++count;
138                                        }
139                                        if (filesLoaded.containsKey(fileName)) {
140                                                filesLoaded.put(fileName, filesLoaded.get(fileName)
141                                                                + num);
142                                        } else {
143                                                filesLoaded.put(fileName, num);
144                                        }
145                                        // filesLoaded += fileName + String.format("(%d), ", num);
146                                }
147                        }
148                        String loaded = "";
149                        for (final String f : filesLoaded.keySet()) { // Arrays.sort()
150                                loaded += String.format("%s(%d), ", f, filesLoaded.get(f));
151                        }
152                        // HelpLogger.info(count + " extra help entries loaded" +
153                        // (filesLoaded.length()>2 ? " from files: " +
154                        // filesLoaded.replaceFirst(", $", "") : ""));
155                        if (ConfigEnum.VERBOSE.getBoolean()) {
156                                HelpLogger
157                                                .info("[AdminCmd] "
158                                                                + count
159                                                                + " extra help entries loaded"
160                                                                + (loaded.length() > 2 ? " from files: "
161                                                                                + loaded.substring(0,
162                                                                                                loaded.length() - 2) : ""));
163                        }
164                } else {
165                        HelpLogger.warning("Error: ExtraHelp is a file");
166                }
167        }
168}
