001    package be.Balor.Tools.Help;
002    
003    import java.io.File;
004    import java.io.FileInputStream;
005    import java.io.IOException;
006    import java.util.ArrayList;
007    import java.util.HashMap;
008    import java.util.List;
009    import java.util.Map;
010    import java.util.logging.Logger;
011    
012    import org.yaml.snakeyaml.Yaml;
013    import org.yaml.snakeyaml.constructor.SafeConstructor;
014    import org.yaml.snakeyaml.reader.UnicodeReader;
015    
016    import be.Balor.Tools.Files.YmlFilter;
017    import be.Balor.Tools.Help.String.Str;
018    import be.Balor.bukkit.AdminCmd.ConfigEnum;
019    
020    public class HelpLoader {
021            private static Logger HelpLogger = Logger.getLogger("Minecraft");
022    
023            @SuppressWarnings({ "unchecked", "rawtypes" })
024            public static void load(final File dataFolder) {
025                    final File helpFolder = new File(dataFolder, "HelpFiles");
026                    if (!helpFolder.exists()) {
027                            helpFolder.mkdirs();
028                    } else if (helpFolder.isDirectory()) {
029                            int count = 0;
030                            // File files[] = helpFolder.listFiles(new YmlFilter());
031                            final File files[] = YmlFilter.listRecursively(helpFolder, 1);
032                            final String folder = helpFolder.getAbsolutePath() + File.separator;
033                            if (files == null) {
034                                    return;
035                            }
036                            final HashMap<String, Integer> filesLoaded = new HashMap<String, Integer>();
037                            for (final File insideFile : files) {
038                                    String fileName, fn = insideFile.getAbsolutePath();
039                                    if (fn.length() > folder.length()) {
040                                            fn = fn.substring(folder.length());
041                                            // if is a directory, use that as the name instead
042                                            if (fn.contains(File.separator)) {
043                                                    fileName = fn.substring(0, fn.indexOf(File.separator));
044                                            } else {
045                                                    fileName = insideFile.getName().replaceFirst(".yml$", "");
046                                            }
047                                    } else {
048                                            fileName = insideFile.getName().replaceFirst(".yml$", "");
049                                    }
050                                    final Yaml yaml = new Yaml(new SafeConstructor());
051                                    Map<String, Object> root = null;
052                                    FileInputStream input = null;
053                                    try {
054                                            input = new FileInputStream(insideFile);
055                                            root = (Map<String, Object>) yaml.load(new UnicodeReader(input));
056                                            if (root == null || root.isEmpty()) {
057                                                    System.out.println("The file " + fn + " is empty");
058                                                    continue;
059                                            }
060                                    } catch (final Exception ex) {
061                                            // HelpLogger.severe("Error!", ex);
062                                            String err = Str.getStackStr(ex);
063                                            final String er = ex.getStackTrace()[0].toString();
064                                            if (err.contains(er)) {
065                                                    err = err.substring(0, err.indexOf(er));
066                                            }
067                                            if (err.contains("at")) {
068                                                    err = err.substring(0, err.lastIndexOf("at"));
069                                                    err = err.substring(0, err.lastIndexOf("\n"));
070                                            }
071                                            HelpLogger.severe("Error loading " + fn + "\n" + err); // ex.getMessage()
072                                    } finally {
073                                            if (input != null) {
074                                                    try {
075                                                            input.close();
076                                                    } catch (final IOException ex) {
077                                                    }
078                                            }
079                                    }
080                                    if (root != null) {
081                                            int num = 0;
082                                            for (final String helpKey : root.keySet()) {
083                                                    final Map<String, Object> helpNode = (Map<String, Object>) root
084                                                                    .get(helpKey);
085    
086                                                    if (!helpNode.containsKey("command")) {
087                                                            HelpLogger.warning("Help entry node \"" + helpKey
088                                                                            + "\" is missing a command name in " + fn);
089                                                            continue;
090                                                    }
091                                                    if (!helpNode.containsKey("description")) {
092                                                            HelpLogger.warning("Help entry node \"" + helpKey
093                                                                            + "\" is missing a description in " + fn);
094                                                            continue;
095                                                    }
096                                                    final String command = helpNode.get("command").toString();
097                                                    String description = "";
098                                                    try {
099                                                            description = helpNode.get("description").toString();
100                                                    } catch (final NullPointerException e) {
101    
102                                                    }
103                                                    String detailedDescription = "";
104                                                    try {
105                                                            detailedDescription = helpNode.get("detailed").toString();
106                                                    } catch (final NullPointerException e) {
107    
108                                                    }
109                                                    final String commandName = helpNode.containsKey("cmdname") ? helpNode.get(
110                                                                    "cmdname").toString() : helpKey;
111                                                    final String plugin = helpNode.containsKey("plugin") ? helpNode.get(
112                                                                    "plugin").toString() : fileName;
113                                                    final ArrayList<String> permissions = new ArrayList<String>();
114    
115                                                    if (helpNode.containsKey("permissions")) {
116                                                            if (helpNode.get("permissions") instanceof List) {
117                                                                    for (final Object permission : (List) helpNode.get("permissions")) {
118                                                                            permissions.add(permission.toString());
119                                                                    }
120                                                            } else {
121                                                                    permissions.add(helpNode.get("permissions").toString());
122                                                            }
123                                                    }
124                                                    HelpLister.getInstance().addHelpEntry(command, description,
125                                                                    detailedDescription, plugin, permissions, commandName);
126                                                    ++num;
127                                                    ++count;
128                                            }
129                                            if (filesLoaded.containsKey(fileName)) {
130                                                    filesLoaded.put(fileName, filesLoaded.get(fileName) + num);
131                                            } else {
132                                                    filesLoaded.put(fileName, num);
133                                            }
134                                            // filesLoaded += fileName + String.format("(%d), ", num);
135                                    }
136                            }
137                            String loaded = "";
138                            for (final String f : filesLoaded.keySet()) { // Arrays.sort()
139                                    loaded += String.format("%s(%d), ", f, filesLoaded.get(f));
140                            }
141                            // HelpLogger.info(count + " extra help entries loaded" +
142                            // (filesLoaded.length()>2 ? " from files: " +
143                            // filesLoaded.replaceFirst(", $", "") : ""));
144                            if (ConfigEnum.VERBOSE.getBoolean())
145                                    HelpLogger.info("[AdminCmd] "
146                                                    + count
147                                                    + " extra help entries loaded"
148                                                    + (loaded.length() > 2 ? " from files: "
149                                                                    + loaded.substring(0, loaded.length() - 2) : ""));
150                    } else {
151                            HelpLogger.warning("Error: ExtraHelp is a file");
152                    }
153            }
154    }
