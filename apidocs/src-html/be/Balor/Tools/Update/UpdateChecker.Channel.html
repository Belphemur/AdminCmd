017package be.Balor.Tools.Update;

019import java.io.IOException;
020import java.util.HashMap;
021import java.util.Map;
022import java.util.logging.Level;
023
024import javax.xml.parsers.DocumentBuilder;
025import javax.xml.parsers.DocumentBuilderFactory;
026import javax.xml.parsers.ParserConfigurationException;
027
028import org.bukkit.Bukkit;
029import org.w3c.dom.Document;
030import org.w3c.dom.Element;
031import org.w3c.dom.Node;
032import org.w3c.dom.NodeList;
033import org.xml.sax.SAXException;
034
035import be.Balor.Tools.Utils;
036import be.Balor.Tools.Debug.ACLogger;
037import be.Balor.Tools.Debug.DebugLog;
038import be.Balor.bukkit.AdminCmd.ACPluginManager;
039import be.Balor.bukkit.AdminCmd.AdminCmd;
040import be.Balor.bukkit.AdminCmd.LocaleHelper;
041
042/**
043 * @author Antoine
044 * 
045 */
046public class UpdateChecker {
047        public enum Channel {
048                DEV("http://www.admincmd.com/category/dev/feed/") {
049                        @Override
050                        public String toString() {
051                                return "Development Version";
052                        }
053                },
054                STABLE("http://www.admincmd.com/category/stable/feed/") {
055                        @Override
056                        public String toString() {
057                                return "Stable Version";
058                        }
059                };
060                /**
061                 * 
062                 */
063                private final String url;
064
065                private Channel(final String url) {
066                        this.url = url;
067                }
068
069                private Document getRss() throws ParserConfigurationException,
070                                SAXException, IOException {
071                        final DocumentBuilderFactory factory = DocumentBuilderFactory
072                                        .newInstance();
073                        factory.setNamespaceAware(true);
074                        DocumentBuilder builder;
075                        builder = factory.newDocumentBuilder();
076                        return builder.parse(this.url);
077
078                }
079        }
080
081        private Document rss = null;
082
083        private final AdminCmdVersion currentVersion;
084        private final Channel channel;
085
086        private AdminCmdVersion rssVersion;
087
088        public UpdateChecker(final Channel channel, final AdminCmd plugin) {
089                this.channel = channel;
090                currentVersion = new AdminCmdVersion(plugin.getDescription()
091                                .getVersion());
092                try {
093                        rss = this.channel.getRss();
094                        scheduleCheck();
095                } catch (final Exception e) {
096                        DebugLog.INSTANCE.log(Level.SEVERE,
097                                        "Can't get the RSS feed for the update checker", e);
098                }
099
100        }
101
102        /**
103         * 
104         */
105        private void scheduleCheck() {
106                ACPluginManager.getScheduler().runTaskTimerAsynchronously(
107                                ACPluginManager.getCorePlugin(), new Runnable() {
108
109                                        @Override
110                                        public void run() {
111                                                if (checkForUpdate()) {
112                                                        notifyNewVersion();
113                                                }
114
115                                        }
116                                }, 0, 48 * 24 * 3600 * Utils.secInTick);
117
118        }
119
120        /**
121         * Check if there is an update on the RSS flux of the plugin.
122         * 
123         * @return
124         */
125        public boolean checkForUpdate() {
126                try {
127                        final NodeList rssChannel = rss.getElementsByTagName("channel");
128                        final Element elementChannel = (Element) rssChannel.item(0);
129                        final Element rssItem = (Element) elementChannel
130                                        .getElementsByTagName("item").item(0);
131                        final Node title = rssItem.getElementsByTagName("title").item(0);
132                        final Node node = title;
133                        rssVersion = new AdminCmdVersion(node.getTextContent());
134                        DebugLog.INSTANCE.fine("This version : " + currentVersion);
135                        DebugLog.INSTANCE.fine("Distant version found : " + rssVersion);
136                        return rssVersion.isNewerThan(currentVersion);
137
138                } catch (final Exception e) {
139                        DebugLog.INSTANCE.log(Level.WARNING,
140                                        "Problem while checking the RSS flux", e);
141                        return false;
142                }
143        }
144
145        private void notifyNewVersion() {
146                ACPluginManager.scheduleSyncTask(new Runnable() {
147
148                        @Override
149                        public void run() {
150                                final Map<String, String> replace = new HashMap<String, String>();
151                                replace.put("version", rssVersion.getVersion());
152                                final String message = LocaleHelper.NEW_VERSION
153                                                .getLocale(replace);
154                                ACLogger.warning(message);
155                                Bukkit.getServer().broadcast(message,
156                                                "admincmd.spec.versionbcast");
157                        }
158                });
159
160        }
161
162}
