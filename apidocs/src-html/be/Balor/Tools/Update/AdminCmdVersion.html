017package be.Balor.Tools.Update;
018
019import java.text.ParseException;
020import java.text.SimpleDateFormat;
021import java.util.Date;
022import java.util.regex.Matcher;
023import java.util.regex.Pattern;
024import java.util.regex.PatternSyntaxException;
025
026import com.google.common.collect.ComparisonChain;
027
028/**
029 * @author Antoine
030 * 
031 */
032public class AdminCmdVersion implements Comparable<AdminCmdVersion> {
033        private final static Pattern admincmdVersion = Pattern
034                        .compile("([\\d]*)\\.([\\d]*)\\.([\\d]*)(-SNAPSHOT | )\\(BUILD (.*)\\)");
035        private int major, minor, build;
036        private boolean dev = false;
037        private Date buildDate;
038        private final String version;
039
040        /**
041         * 
042         */
043        public AdminCmdVersion(final String version) {
044                this.version = version;
045                try {
046
047                        final Matcher regexMatcher = admincmdVersion.matcher(version);
048                        if (regexMatcher.find()) {
049                                major = Integer.parseInt(regexMatcher.group(1));
050                                minor = Integer.parseInt(regexMatcher.group(2));
051                                build = Integer.parseInt(regexMatcher.group(3));
052                                if (regexMatcher.group(4).equalsIgnoreCase("-SNAPSHOT ")) {
053                                        dev = true;
054                                }
055                                buildDate = new SimpleDateFormat("dd.MM.yyyy '@' HH:mm:ss")
056                                                .parse(regexMatcher.group(5));
057                        }
058
059                } catch (final PatternSyntaxException ex) {
060                        ex.printStackTrace();
061                } catch (final ParseException e) {
062                        // TODO Auto-generated catch block
063                        e.printStackTrace();
064                }
065        }
066
067        /**
068         * Check if this version is newer than an other one
069         * 
070         * @param version
071         * @return true if newer.
072         */
073        public boolean isNewerThan(final AdminCmdVersion version) {
074                return compareTo(version) == 1;
075        }
076
077        /*
078         * (non-Javadoc)
079         * 
080         * @see java.lang.Object#hashCode()
081         */
082        @Override
083        public int hashCode() {
084                final int prime = 31;
085                int result = 1;
086                result = prime * result + build;
087                result = prime * result
088                                + ((buildDate == null) ? 0 : buildDate.hashCode());
089                result = prime * result + (dev ? 1231 : 1237);
090                result = prime * result + major;
091                result = prime * result + minor;
092                return result;
093        }
094
095        /*
096         * (non-Javadoc)
097         * 
098         * @see java.lang.Object#equals(java.lang.Object)
099         */
100        @Override
101        public boolean equals(final Object obj) {
102                if (this == obj) {
103                        return true;
104                }
105                if (obj == null) {
106                        return false;
107                }
108                if (getClass() != obj.getClass()) {
109                        return false;
110                }
111                final AdminCmdVersion other = (AdminCmdVersion) obj;
112                if (build != other.build) {
113                        return false;
114                }
115                if (buildDate == null) {
116                        if (other.buildDate != null) {
117                                return false;
118                        }
119                } else if (!buildDate.equals(other.buildDate)) {
120                        return false;
121                }
122                if (dev != other.dev) {
123                        return false;
124                }
125                if (major != other.major) {
126                        return false;
127                }
128                if (minor != other.minor) {
129                        return false;
130                }
131                return true;
132        }
133
134        /*
135         * (non-Javadoc)
136         * 
137         * @see java.lang.Object#toString()
138         */
139        @Override
140        public String toString() {
141                return String
142                                .format("AdminCmdVersion [major=%s, minor=%s, build=%s, dev=%s, buildDate=%s]",
143                                                major, minor, build, dev, buildDate);
144        }
145
146        /**
147         * @return the version
148         */
149        public String getVersion() {
150                return version;
151        }
152
153        /*
154         * (non-Javadoc)
155         * 
156         * @see java.lang.Comparable#compareTo(java.lang.Object)
157         */
158        @Override
159        public int compareTo(final AdminCmdVersion o) {
160                if (o == null) {
161                        return 1;
162                }
163
164                return ComparisonChain.start().compare(major, o.major)
165                                .compare(minor, o.minor).compare(build, o.build)
166                                .compare(buildDate, o.buildDate).result();
167        }
168
169}
