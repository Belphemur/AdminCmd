017    package be.Balor.Tools;
018    
019    import org.bukkit.Location;
020    import org.bukkit.World;
021    
022    /**
023     * @author Balor (aka Antoine Aflalo)
024     * 
025     */
026    public class SimplifiedLocation extends Location {
027    
028            /**
029             * @param world
030             * @param x
031             * @param y
032             * @param z
033             */
034            public SimplifiedLocation(final World world, final double x, final double y, final double z) {
035                    super(world, x, y, z);
036            }
037    
038            public SimplifiedLocation(final Location loc) {
039                    super(loc.getWorld(), loc.getX(), loc.getY(), loc.getZ());
040            }
041    
042            /*
043             * (non-Javadoc)
044             * 
045             * @see org.bukkit.Location#equals(java.lang.Object)
046             */
047            @Override
048            public boolean equals(final Object obj) {
049                    if (!(obj instanceof Location)) {
050                            return false;
051                    }
052    
053                    final Location other = (Location) obj;
054                    return other.getBlockX() == this.getBlockX() && other.getBlockY() == this.getBlockY()
055                                    && other.getBlockZ() == this.getBlockZ();
056            }
057    
058    }
