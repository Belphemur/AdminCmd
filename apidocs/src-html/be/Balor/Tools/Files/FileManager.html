104            public String getTextFile(final String fileName) {
105                    final StringBuffer result = new StringBuffer();
106                    try {
107                            final File fileDir = getInnerFile(fileName);
108                            final BufferedReader in = new BufferedReader(new InputStreamReader(new FileInputStream(
109                                            fileDir), "UTF8"));
110                            String temp;
111                            while ((temp = in.readLine()) != null) {
112                                    result.append(temp + "\n");
113                            }
114                            in.close();
115                    } catch (final UnsupportedEncodingException e) {
116                            // TODO: Better debug code here
117                            ACLogger.Log(Level.SEVERE, e.getMessage(), e);
118                    } catch (final IOException e) {
119                            // TODO: Better debug code here
120                            ACLogger.Log(Level.SEVERE, e.getMessage(), e);
121                    } catch (final Exception e) {
122                            // TODO: Better debug code here
123                            ACLogger.Log(Level.SEVERE, e.getMessage(), e);
124                    }
125                    if (result.length() == 0)
126                            return null;
127                    else
128                            return result.toString().trim();
129            }

188            public File getFile(final String directory, final String filename, final boolean create) {
189                    if (lastFile != null && lastDirectory.equals(directory == null ? "" : directory)
190                                    && lastFilename.equals(filename))
191                            return lastFile;
192                    File file = null;
193                    if (directory != null) {
194                            final File directoryFile = new File(this.pathFile, directory);
195                            if (!directoryFile.exists() && create) {
196                                    directoryFile.mkdir();
197                            }
198                            file = new File(directoryFile, filename);
199                    } else
200                            file = new File(pathFile, filename);
201    
202                    if (!file.exists() && create) {
203    
204                            try {
205                                    file.createNewFile();
206                            } catch (final IOException ex) {
207                                    System.out.println("cannot create file " + file.getPath());
208                            }
209                    }
210                    lastFile = file;
211                    lastDirectory = directory == null ? "" : directory;
212                    lastFilename = filename;
213                    return file;
214            }

299            public File getInnerFile(final String filename, final String directory, final boolean replace) {
300                    final File file;
301                    if (directory != null) {
302                            final File directoryFile = new File(this.pathFile, directory);
303                            if (!directoryFile.exists()) {
304                                    directoryFile.mkdirs();
305                            }
306                            file = new File(directoryFile, filename);
307                    } else
308                            file = new File(pathFile, filename);
309                    if (file.exists() && replace) {
310                            BufferedReader reader = null;
311    
312                            try {
313                                    reader = new BufferedReader(new FileReader(file));
314                            } catch (final FileNotFoundException e) {
315                            }
316                            try {
317                                    final String version = reader.readLine();
318                                    final String versioncheck = version.substring(10);
319                                    if (!versioncheck.equals(fileVersion)) {
320                                            reader.close();
321                                            file.delete();
322                                            DebugLog.INSTANCE.info("Delete file : " + file);
323                                    } else
324                                            return file;
325                            } catch (final IOException e) {
326                                    file.delete();
327                            }
328    
329                            try {
330                                    reader.close();
331                            } catch (final IOException e) {
332                            }
333    
334                    }
335                    if (!file.exists()) {
336                            final InputStream res = this.getClass().getResourceAsStream("/" + filename);
337                            FileWriter tx = null;
338                            try {
339                                    tx = new FileWriter(file);
340                                    for (int i = 0; (i = res.read()) > 0;) {
341                                            tx.write(i);
342                                    }
343                                    tx.flush();
344                            } catch (final IOException ex) {
345                                    ex.printStackTrace();
346                                    return file;
347                            } finally {
348                                    try {
349                                            res.close();
350                                    } catch (final Exception ex) {
351                                    }
352                                    try {
353                                            if (tx != null) {
354                                                    tx.close();
355                                            }
356                                    } catch (final Exception ex) {
357                                    }
358                            }
359                    }
360                    return file;
361            }

364                    final HashMap<String, MaterialContainer> result = new HashMap<String, MaterialContainer>();
365                    final ExtendedConfiguration conf = getYml("Alias");
366                    final List<String> aliasList = conf.getStringList("alias", new ArrayList<String>());
367                    final List<String> idList = conf.getStringList("ids", new ArrayList<String>());
368                    int i = 0;
369                    try {
370                            final CSVReader csv = new CSVReader(new FileReader(
371                                            getInnerFile("items.csv", null, true)));
372                            String[] alias;
373                            while ((alias = csv.readNext()) != null) {
374                                    try {
375                                            result.put(alias[0], new MaterialContainer(alias[1], alias[2]));
376                                    } catch (final ArrayIndexOutOfBoundsException e) {
377                                            try {
378                                                    result.put(alias[0], new MaterialContainer(alias[1]));
379                                            } catch (final ArrayIndexOutOfBoundsException e2) {
380                                            }
381    
382                                    }
383    
384                            }
385    
386                    } catch (final FileNotFoundException e) {
387                            e.printStackTrace();
388                    } catch (final IOException e) {
389                            e.printStackTrace();
390                    }
391                    for (final String alias : aliasList) {
392                            result.put(alias, new MaterialContainer(idList.get(i)));
393                            i++;
394                    }
395                    return result;

426             * @param directory
427             * @return
428             */
429            @Override
430            public Location getLocation(final String property, final String filename, final String directory)
431                            throws WorldNotLoaded {
432                    final ExtendedConfiguration conf = getYml(filename, directory);
433                    if (conf.get(property + ".world") == null) {
434                            final Location loc = parseLocation(property, conf);
435                            if (loc != null)
436                                    writeLocation(loc, property, filename, directory);
437                            return loc;
438                    } else {
439                            final World w = ACPluginManager.getServer().getWorld(
440                                            conf.getString(property + ".world"));
441                            if (w != null)
442                                    return new Location(w, conf.getDouble(property + ".x", 0), conf.getDouble(property
443                                                    + ".y", 0), conf.getDouble(property + ".z", 0), Float.parseFloat(conf
444                                                    .getString(property + ".yaw")), Float.parseFloat(conf.getString(property
445                                                    + ".pitch")));
446                            else
447                                    throw new WorldNotLoaded(conf.getString(property + ".world"));
448    
449                    }
450            }

486            /**
487             * Parse String to create a location
488             * 
489             * @param property
490             * @param conf
491             * @return
492             */
493            private Location parseLocation(final String property, final ExtendedConfiguration conf) {
494                    final String toParse = conf.getString(property, null);
495                    if (toParse == null)
496                            return null;
497                    if (toParse.isEmpty())
498                            return null;
499                    final Double coords[] = new Double[3];
500                    final Float direction[] = new Float[2];
501                    final String[] infos = toParse.split(";");
502                    for (int i = 0; i < coords.length; i++)
503                            try {
504                                    coords[i] = Double.parseDouble(infos[i]);
505                            } catch (final NumberFormatException e) {
506                                    return null;
507                            }
508                    for (int i = 3; i < infos.length - 1; i++)
509                            try {
510                                    direction[i - 3] = Float.parseFloat(infos[i]);
511                            } catch (final NumberFormatException e) {
512                                    return null;
513                            }
514                    return new Location(ACPluginManager.getServer().getWorld(infos[5]), coords[0], coords[1],
515                                    coords[2], direction[0], direction[1]);
516            }

538            @Override
539            public Map<String, BannedPlayer> loadBan() {
540                    Map<String, BannedPlayer> result = new HashMap<String, BannedPlayer>();
541                    final ExtendedConfiguration conf = getYml("banned");
542                    if (conf.get("bans") != null) {
543                            final ConfigurationSection node = conf.getConfigurationSection("bans");
544                            for (final String key : node.getKeys(false))
545                                    result.put(key, (BannedPlayer) node.get(key));
546    
547                    }
548                    if (ConfigEnum.IMPORT_BAN_TXT.getBoolean()) {
549                            result = importBannedPlayerTXT(result);
550                            ConfigEnum.IMPORT_BAN_TXT.setValue(false);
551                    }
552                    return result;
553            }

555            private Map<String, BannedPlayer> importBannedPlayerTXT(final Map<String, BannedPlayer> result) {
556                    final Set<OfflinePlayer> banned = ACPluginManager.getServer().getBannedPlayers();
557                    final Iterator<OfflinePlayer> it = banned.iterator();
558                    while (it.hasNext()) {
559                            final OfflinePlayer op = it.next();
560                            final String name = op.getName();
561                            final BannedPlayer bp = new BannedPlayer(name, "Import from banned-players.txt");
562                            if (!result.containsKey(name))
563                                    result.put(name, bp);
564                    }
565                    return result;
566            }

590                            ConfigurationSection armorItems = null;

595                                    parents = kitNode.getStringList("parents");
596                            } catch (final NullPointerException e) {
597                                    DebugLog.INSTANCE.warning("Problem with kit " + kitName);
598                                    continue;
599                            }
600    
601                            if (kitItems != null)
602                                    for (final String item : kitItems.getKeys(false)) {
603                                            try {
604                                                    final MaterialContainer m = Utils.checkMaterial(item);
605                                                    m.setAmount(kitItems.getInt(item, 1));
606                                                    if (!m.isNull())
607                                                            items.add(m);
608                                            } catch (final InvalidInputException e) {
609                                                    DebugLog.INSTANCE.log(Level.WARNING, "Problem with kit : " + item, e);
610                                            }
611                                    }
612                            delay = kitNode.getInt("delay", 0);
613                            /*
614                             * Old convertor code, not used anymore TODO: CLEAN IT. } else {
615                             * kitNode.addDefault("items", new HashMap<String, Object>());
616                             * kitItems = kitNode.getConfigurationSection("items"); for (String
617                             * item : kitNode.getKeys(false)) { if (item.equals("items"))
618                             * continue; MaterialContainer m = Utils.checkMaterial(item); int
619                             * amount = kitNode.getInt(item, 1); m.setAmount(amount); if
620                             * (!m.isNull()) { items.add(m); kitItems.set(item, amount);
621                             * kitNode.set(item, null); } } kitNode.set("delay", 0); convert =
622                             * true; }
623                             */
624                            if (armorItems != null) {
625                                    for (final ArmorPart part : ArmorPart.values()) {
626                                            final String partId = armorItems.getString(part.toString());
627                                            if (partId == null)
628                                                    continue;
629                                            try {
630                                                    final MaterialContainer m = Utils.checkMaterial(partId);
631                                                    if (!m.isNull())
632                                                            armor.put(part, m);
633                                            } catch (final InvalidInputException e) {
634                                                    DebugLog.INSTANCE.log(Level.WARNING, "Problem with kit : " + partId, e);
635                                            }
636                                    }
637                                    result.put(kitName, new ArmoredKitInstance(kitName, delay,
638                                                    new ArrayList<MaterialContainer>(items),
639                                                    new EnumMap<Type.ArmorPart, MaterialContainer>(armor)));
640                            } else
641                                    result.put(kitName, new KitInstance(kitName, delay,
642                                                    new ArrayList<MaterialContainer>(items)));
643    
644                            if (parents != null)
645                                    kitParents.put(kitName, parents);
646                            else
647                                    ACLogger.info(kitName + " has no parents");
648    
649                            items.clear();
650                            armor.clear();
651                    }
652                    for (final Entry<String, List<String>> entry : kitParents.entrySet()) {
653                            KitInstance kit = result.get(entry.getKey());
654                            for (final String parent : entry.getValue()) {
655                                    final KitInstance parentKit = result.get(parent);
656                                    if (parentKit == null)
657                                            continue;
658                                    if (parentKit instanceof ArmoredKitInstance && !(kit instanceof ArmoredKitInstance)) {
659                                            kit = new ArmoredKitInstance(kit);
660                                            result.put(kit.getName(), kit);
661                                    }
662                                    kit.addParent(parentKit);
663                            }
664                    }
665    
666                    /*
667                     * if (convert) try { kits.save(); } catch (IOException e) { }
668                     */
669                    return result;
670            }
