104         * Get a txt-file and return its content in a String
105         * 
106         * @param fileName
107         *            - The name of the file to be loaded
108         * @return The content of the file
109         */
110        public String getTextFile(final String fileName) {
111                final StringBuffer result = new StringBuffer();
112                try {
113                        final File fileDir = getInnerFile(fileName);
114                        final BufferedReader in = new BufferedReader(new UnicodeReader(
115                                        new FileInputStream(fileDir), "UTF-8"));
116                        String temp;
117                        while ((temp = in.readLine()) != null) {
118                                result.append(temp + "\n");
119                        }
120                        in.close();
121                } catch (final UnsupportedEncodingException e) {
122                        // TODO: Better debug code here
123                        ACLogger.Log(Level.SEVERE, e.getMessage(), e);
124                } catch (final IOException e) {
125                        // TODO: Better debug code here
126                        ACLogger.Log(Level.SEVERE, e.getMessage(), e);
127                } catch (final Exception e) {
128                        // TODO: Better debug code here
129                        ACLogger.Log(Level.SEVERE, e.getMessage(), e);
130                }
131                if (result.length() == 0) {
132                        return null;
133                } else {
134                        return result.toString().trim();
135                }
136        }

199        public File getFile(final String directory, final String filename,
200                        final boolean create) {
201                if (lastFile != null
202                                && lastDirectory.equals(directory == null ? "" : directory)
203                                && lastFilename.equals(filename)) {
204                        return lastFile;
205                }
206                File file = null;
207                if (directory != null) {
208                        final File directoryFile = new File(this.pathFile, directory);
209                        if (!directoryFile.exists() && create) {
210                                directoryFile.mkdir();
211                        }
212                        file = new File(directoryFile, filename);
213                } else {
214                        file = new File(pathFile, filename);
215                }
216
217                if (!file.exists() && create) {
218
219                        try {
220                                file.createNewFile();
221                        } catch (final IOException ex) {
222                                System.out.println("cannot create file " + file.getPath());
223                        }
224                }
225                lastFile = file;
226                lastDirectory = directory == null ? "" : directory;
227                lastFilename = filename;
228                return file;
229        }

307        public File getInnerFile(final String filename, final String directory,
308                        final boolean replace) {
309                final File file;
310                if (directory != null) {
311                        final File directoryFile = new File(this.pathFile, directory);
312                        if (!directoryFile.exists()) {
313                                directoryFile.mkdirs();
314                        }
315                        file = new File(directoryFile, filename);
316                } else {
317                        file = new File(pathFile, filename);
318                }
319                if (file.exists() && replace) {
320                        BufferedReader reader = null;
321
322                        try {
323                                reader = new BufferedReader(new UnicodeReader(
324                                                new FileInputStream(file), "UTF-8"));
325                        } catch (final FileNotFoundException e) {
326                        }
327                        try {
328                                final String version = reader.readLine();
329                                final String versioncheck = version.substring(10);
330                                if (!versioncheck.equals(fileVersion)) {
331                                        reader.close();
332                                        file.delete();
333                                        DebugLog.INSTANCE.info("Delete file : " + file);
334                                } else {
335                                        return file;
336                                }
337                        } catch (final IOException e) {
338                                file.delete();
339                        }
340
341                        try {
342                                reader.close();
343                        } catch (final IOException e) {
344                        }
345
346                }
347                if (!file.exists()) {
348                        try {
349                                UnicodeUtil.saveUTF8File(file, this.getClass()
350                                                .getResourceAsStream("/" + filename), false);
351                        } catch (final IOException e) {
352                                ACLogger.severe("Can't copy the inner file " + filename
353                                                + " to " + file, e);
354                        }
355                }
356                return file;
357        }

359        public HashMap<String, MaterialContainer> getAlias() {
360                final HashMap<String, MaterialContainer> result = new HashMap<String, MaterialContainer>();
361                final ExtendedConfiguration conf = getYml("Alias");
362                final List<String> aliasList = conf.getStringList("alias",
363                                new ArrayList<String>());
364                final List<String> idList = conf.getStringList("ids",
365                                new ArrayList<String>());
366                int i = 0;
367                CSVReader csv = null;
368                try {
369                        csv = new CSVReader(new FileReader(getInnerFile("items.csv", null,
370                                        true)));
371                        String[] alias;
372                        while ((alias = csv.readNext()) != null) {
373                                try {
374                                        result.put(alias[0], new MaterialContainer(alias[1],
375                                                        alias[2]));
376                                } catch (final ArrayIndexOutOfBoundsException e) {
377                                        try {
378                                                result.put(alias[0], new MaterialContainer(alias[1]));
379                                        } catch (final ArrayIndexOutOfBoundsException e2) {
380                                        }
381
382                                }
383
384                        }
385
386                } catch (final FileNotFoundException e) {
387                        e.printStackTrace();
388                } catch (final IOException e) {
389                        e.printStackTrace();
390                } finally {
391                        if (csv != null) {
392                                try {
393                                        csv.close();
394                                } catch (final IOException e) {
395                                        e.printStackTrace();
396                                }
397                        }
398                }
399                for (final String alias : aliasList) {
400                        result.put(alias, new MaterialContainer(idList.get(i)));
401                        i++;
402                }

433         * @param filename
434         * @param directory
435         * @return
436         */
437        @Override
438        public Location getLocation(final String property, final String filename,
439                        final String directory) throws WorldNotLoaded {
440                final ExtendedConfiguration conf = getYml(filename, directory);
441                if (conf.get(property + ".world") == null) {
442                        final Location loc = parseLocation(property, conf);
443                        if (loc != null) {
444                                writeLocation(loc, property, filename, directory);
445                        }
446                        return loc;
447                } else {
448                        final World w = ACPluginManager.getServer().getWorld(
449                                        conf.getString(property + ".world"));
450                        if (w != null) {
451                                return new Location(w, conf.getDouble(property + ".x", 0),
452                                                conf.getDouble(property + ".y", 0), conf.getDouble(
453                                                                property + ".z", 0), Float.parseFloat(conf
454                                                                .getString(property + ".yaw")),
455                                                Float.parseFloat(conf.getString(property + ".pitch")));
456                        } else {
457                                throw new WorldNotLoaded(conf.getString(property + ".world"));
458                        }
459
460                }
461        }

500        /**
501         * Parse String to create a location
502         * 
503         * @param property
504         * @param conf
505         * @return
506         */
507        private Location parseLocation(final String property,
508                        final ExtendedConfiguration conf) {
509                final String toParse = conf.getString(property, null);
510                if (toParse == null) {
511                        return null;
512                }
513                if (toParse.isEmpty()) {
514                        return null;
515                }
516                final Double coords[] = new Double[3];
517                final Float direction[] = new Float[2];
518                final String[] infos = toParse.split(";");
519                for (int i = 0; i < coords.length; i++) {
520                        try {
521                                coords[i] = Double.parseDouble(infos[i]);
522                        } catch (final NumberFormatException e) {
523                                return null;
524                        }
525                }
526                for (int i = 3; i < infos.length - 1; i++) {
527                        try {
528                                direction[i - 3] = Float.parseFloat(infos[i]);
529                        } catch (final NumberFormatException e) {
530                                return null;
531                        }
532                }
533                return new Location(ACPluginManager.getServer().getWorld(infos[5]),
534                                coords[0], coords[1], coords[2], direction[0], direction[1]);
535        }

606        private void importBannedPlayerTXT(final Map<String, Ban> result) {
607                final Set<OfflinePlayer> banned = ACPluginManager.getServer()
608                                .getBannedPlayers();
609                final Set<String> ipBanned = ACPluginManager.getServer().getIPBans();
610                final Iterator<OfflinePlayer> it = banned.iterator();
611                while (it.hasNext()) {
612                        final OfflinePlayer op = it.next();
613                        final String name = op.getName();
614                        if (!result.containsKey(name)) {
615                                final BannedPlayer bp = new BannedPlayer(name,
616                                                "Import from banned-players.txt");
617                                result.put(name, bp);
618                        }
619                }
620                for (final String ip : ipBanned) {
621                        if (result.containsKey(ip)) {
622                                continue;
623                        }
624                        result.put(ip, new BannedIP(ip, "Import from banned-ip.txt"));
625                }
626        }

628        /**
629         * Load all the kits
630         * 
631         * @return
632         */
633        public Map<String, KitInstance> loadKits() {
634                final Map<String, KitInstance> result = new LinkedHashMap<String, KitInstance>();
635                final ExtendedConfiguration kits = getYml("kits");
636                final Map<String, List<String>> kitParents = new HashMap<String, List<String>>();
637
638                final ConfigurationSection kitNodes = kits
639                                .getConfigurationSection("kits");
640                if (kitNodes == null) {
641                        ACLogger.severe("A problem happen when wanting to load the kits. Please check your kits.yml file.");
642                        return result;
643                }
644                for (final String kitName : kitNodes.getKeys(false)) {
645                        int delay = 0;
646                        final ConfigurationSection kitNode = kitNodes
647                                        .getConfigurationSection(kitName);
648                        ConfigurationSection kitItems = null;
649                        ConfigurationSection armorItems = null;
650                        List<String> parents = null;
651
652                        try {
653                                kitItems = kitNode.getConfigurationSection("items");
654                                armorItems = kitNode.getConfigurationSection("armor");
655                                parents = kitNode.getStringList("parents");
656                        } catch (final NullPointerException e) {
657                                DebugLog.INSTANCE.warning("Problem with kit " + kitName);
658                                continue;
659                        }
660                        final List<MaterialContainer> items = getItemsFromKit(kitItems);
661                        delay = kitNode.getInt("delay", 0);
662
663                        if (armorItems != null) {
664                                final Map<ArmorPart, MaterialContainer> armor = getArmorFromKit(armorItems);
665                                final ArmoredKitInstance armoredKit = new ArmoredKitInstance(
666                                                kitName, delay,
667                                                new ArrayList<MaterialContainer>(items),
668                                                new EnumMap<Type.ArmorPart, MaterialContainer>(armor));
669                                armoredKit.setColor(kitNode.getString("color", null));
670                                result.put(kitName, armoredKit);
671                        } else {
672                                result.put(kitName, new KitInstance(kitName, delay,
673                                                new ArrayList<MaterialContainer>(items)));
674                        }
675
676                        if (parents != null) {
677                                kitParents.put(kitName, parents);
678                        } else {
679                                ACLogger.info(kitName + " has no parents");
680                        }
681
682                }
683                checkParents(result, kitParents);
684                loadDynamicKits(kits, result);
685                return result;
686        }

720        /**
721         * Check parent to be sure that every kit is correctly related to its parent
722         * 
723         * @param result
724         * @param kitParents
725         */
726        private void checkParents(final Map<String, KitInstance> result,
727                        final Map<String, List<String>> kitParents) {
728                for (final Entry<String, List<String>> entry : kitParents.entrySet()) {
729                        KitInstance kit = result.get(entry.getKey());
730                        for (final String parent : entry.getValue()) {
731                                final KitInstance parentKit = result.get(parent);
732                                if (parentKit == null) {
733                                        continue;
734                                }
735                                if (parentKit instanceof ArmoredKitInstance
736                                                && !(kit instanceof ArmoredKitInstance)) {
737                                        kit = new ArmoredKitInstance(kit);
738                                        result.put(kit.getName(), kit);
739                                }
740                                kit.addParent(parentKit);
741                        }
742                }
743        }

745        /**
746         * @param armorItems
747         * @return
748         */
749        private Map<ArmorPart, MaterialContainer> getArmorFromKit(
750                        final ConfigurationSection armorItems) {
751                final Map<ArmorPart, MaterialContainer> armor = new EnumMap<Type.ArmorPart, MaterialContainer>(
752                                ArmorPart.class);
753
754                for (final ArmorPart part : ArmorPart.values()) {
755                        try {
756                                final MaterialContainer m = createArmorMaterialContainer(
757                                                armorItems, part);
758
759                                if (m == null || m.isNull()) {
760                                        continue;
761                                }
762
763                                armor.put(part, m);
764
765                        } catch (final InvalidInputException e) {
766                                DebugLog.INSTANCE.log(Level.WARNING, "Problem with kit : "
767                                                + part, e);
768                        }
769                }
770                return armor;
771        }

773        /**
774         * @param armorItems
775         * @param part
776         * @return
777         * @throws InvalidInputException
778         */
779        private MaterialContainer createArmorMaterialContainer(
780                        final ConfigurationSection armorItems, final ArmorPart part)
781                        throws InvalidInputException {
782                final ConfigurationSection newTemplate = armorItems
783                                .getConfigurationSection(part.toString());
784                MaterialContainer m;
785                if (newTemplate != null) {
786                        m = Utils.checkMaterial(newTemplate.getString("item"));
787                        m.setDmg((short) newTemplate.getInt("durability"));
788                        for (final String enchant : newTemplate
789                                        .getStringList("echantments")) {
790                                try {
791                                        m.addEnchantment(enchant);
792                                } catch (final Exception e) {
793                                        DebugLog.INSTANCE
794                                                        .log(Level.WARNING,
795                                                                        "Problem with enchantments in ArmoredKit : "
796                                                                                        + part, e);
797                                }
798                        }
799
800                } else {
801                        m = Utils.checkMaterial(armorItems.getString(part.toString()));
802                }
803                return m;
804        }

806        /**
807         * @param kitItems
808         * @return
809         */
810        private List<MaterialContainer> getItemsFromKit(
811                        final ConfigurationSection kitItems) {
812                final List<MaterialContainer> items = new ArrayList<MaterialContainer>();
813                if (kitItems != null) {
814                        for (final String item : kitItems.getKeys(false)) {
815                                try {
816                                        final MaterialContainer m = createMaterialContainer(
817                                                        kitItems, item);
818                                        if (!m.isNull()) {
819                                                items.add(m);
820                                        }
821                                } catch (final InvalidInputException e) {
822                                        DebugLog.INSTANCE.log(Level.WARNING, "Problem with kit : "
823                                                        + item, e);
824                                }
825                        }
826                }
827                return items;
828        }

830        /**
831         * @param kitItems
832         * @param item
833         * @return
834         * @throws InvalidInputException
835         */
836        private MaterialContainer createMaterialContainer(
837                        final ConfigurationSection kitItems, final String item)
838                        throws InvalidInputException {
839                final MaterialContainer m = Utils.checkMaterial(item);
840                final ConfigurationSection newTemplateItem = kitItems
841                                .getConfigurationSection(item);
842                if (newTemplateItem != null) {
843                        m.setAmount(newTemplateItem.getInt("amount", 1));
844                        m.setDmg((short) newTemplateItem.getInt("durability", 0));
845                        for (final String enchant : newTemplateItem
846                                        .getStringList("echantments")) {
847                                try {
848                                        m.addEnchantment(enchant);
849                                } catch (final Exception e) {
850                                        DebugLog.INSTANCE.log(Level.WARNING,
851                                                        "Problem with enchantments in kit : " + item, e);
852                                }
853                        }
854                } else {
855                        m.setAmount(kitItems.getInt(item, 1));
856                }
857                return m;
858        }

867        @Override
868        public void addBan(final IBan player) {
869                final ExtendedConfiguration banFile = getYml("banned");
870                ConfigurationSection bans;
871                if (player instanceof BannedPlayer) {
872                        bans = banFile.addSection("bans");
873                        bans.set(player.getPlayer(), player);
874                } else {
875                        bans = banFile.addSection("IPs");
876                        bans.set(String.valueOf(player.getPlayer().hashCode()), player);
877                }
878
879                try {
880                        banFile.save();
881                } catch (final IOException e) {
882                }
883        }

890        @Override
891        public void unBanPlayer(final IBan ban) {
892                final String player = ban.getPlayer();
893                final Matcher ipv4 = Utils.REGEX_IP_V4.matcher(player);
894                final ExtendedConfiguration banFile = getYml("banned");
895                ConfigurationSection bans;
896                if (ipv4.find()) {
897                        bans = banFile.addSection("IPs");
898                        bans.set(String.valueOf(player.hashCode()), null);
899                } else {
900                        bans = banFile.addSection("bans");
901                        bans.set(player, null);
902                }
903                try {
904                        banFile.save();
905                } catch (final IOException e) {
906                }
907
908        }
