017    package be.Balor.Tools.Files;
018    
019    import java.util.Map;
020    import java.util.Set;
021    
022    import org.bukkit.Location;
023    
024    import be.Balor.Manager.Exceptions.WorldNotLoaded;
025    import be.Balor.Player.BannedPlayer;
026    
027    /**
028     * @author Balor (aka Antoine Aflalo)
029     * 
030     */
031    public interface DataManager {
032            /**
033             * Load the Bans
034             * 
035             * @return
036             */
037            public Map<String, BannedPlayer> loadBan();
038    
039            /**
040             * Added a new ban
041             * 
042             * @param player
043             */
044            public void addBannedPlayer(BannedPlayer player);
045    
046            /**
047             * Unban banned player.
048             * 
049             * @param player
050             */
051            public void unBanPlayer(String player);
052    
053            /**
054             * Return a string List containing all locations names
055             * 
056             * @param user
057             * @param table
058             * @return
059             */
060            public Set<String> getKeys(String info, String user, String table);
061    
062            /**
063             * Remove the given location
064             * 
065             * @param property
066             * @param filename
067             * @param directory
068             */
069            public void removeKey(String property, String filename, String directory);
070    
071            /**
072             * Return the location
073             * 
074             * @param property
075             * @param filename
076             * @param directory
077             * @return
078             */
079            public Location getLocation(String property, String filename, String directory)
080                            throws WorldNotLoaded;
081    
082            /**
083             * Store the location informations
084             * 
085             * @param loc
086             * @param filename
087             * @param directory
088             */
089            public void writeLocation(Location loc, String name, String filename, String directory);
090    
091    }
