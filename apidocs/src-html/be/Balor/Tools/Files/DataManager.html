017package be.Balor.Tools.Files;
018
019import java.util.Map;
020import java.util.Set;
021
022import org.bukkit.Location;
023
024import be.Balor.Manager.Exceptions.WorldNotLoaded;
025import be.Balor.Player.Ban;
026import be.Balor.Player.IBan;
027
028/**
029 * @author Balor (aka Antoine Aflalo)
030 * 
031 */
032public interface DataManager {
033        /**
034         * Load the Bans
035         * 
036         * @return
037         */
038        public Map<String, Ban> loadBan();
039
040        /**
041         * Added a new ban
042         * 
043         * @param player
044         */
045        public void addBan(IBan player);
046
047        /**
048         * Unban banned player.
049         * 
050         * @param ban
051         */
052        public void unBanPlayer(IBan ban);
053
054        /**
055         * Return a string List containing all locations names
056         * 
057         * @param user
058         * @param table
059         * @return
060         */
061        public Set<String> getKeys(String info, String user, String table);
062
063        /**
064         * Remove the given location
065         * 
066         * @param property
067         * @param filename
068         * @param directory
069         */
070        public void removeKey(String property, String filename, String directory);
071
072        /**
073         * Return the location
074         * 
075         * @param property
076         * @param filename
077         * @param directory
078         * @return
079         */
080        public Location getLocation(String property, String filename,
081                        String directory) throws WorldNotLoaded;
082
083        /**
084         * Store the location informations
085         * 
086         * @param loc
087         * @param filename
088         * @param directory
089         */
090        public void writeLocation(Location loc, String name, String filename,
091                        String directory);
092
093}
