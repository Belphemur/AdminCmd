015package be.Balor.Tools.Files.Filters;
016
017import java.io.File;
018import java.io.FilenameFilter;
019import java.util.ArrayList;
020import java.util.Arrays;
021
022/**
023 * @author Balor (aka Antoine Aflalo)
024 * 
025 */
026public abstract class FileFilter implements FilenameFilter {
027        public File[] listRecursively(final File folder) {
028                if (folder != null && folder.isDirectory()) {
029                        return _listRecursively(folder, 5).toArray(new File[0]);
030                }
031                return new File[0];
032        }
033
034        public File[] listRecursively(final File folder, final int depth) {
035                if (folder != null && folder.isDirectory()) {
036                        return _listRecursively(folder, depth).toArray(new File[0]);
037                }
038                return new File[0];
039        }
040
041        private ArrayList<File> _listRecursively(final File folder, final int depth) {
042                final ArrayList<File> files = new ArrayList<File>();
043                if (folder != null && folder.isDirectory()) {
044                        files.addAll(Arrays.asList(folder.listFiles(this)));
045                        if (depth > 0) {// now scan folders
046                                final File folders[] = folder.listFiles(new DirFilter());
047                                if (folders != null) {
048                                        for (final File f : folders) {
049                                                files.addAll(_listRecursively(f, depth - 1));
050                                        }
051                                }
052                        }
053                }
054                return files;
055        }
056
057        public static class DirFilter implements FilenameFilter {
058
059                @Override
060                public boolean accept(final File file, final String name) {
061                        return file.isDirectory();
062                }
063        }
064}
