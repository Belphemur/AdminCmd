017    package be.Balor.Tools.Files;
018    
019    import java.util.ArrayList;
020    import java.util.EnumMap;
021    import java.util.List;
022    import java.util.Map;
023    import java.util.Map.Entry;
024    
025    import org.bukkit.entity.Player;
026    import org.bukkit.inventory.ItemStack;
027    import org.bukkit.inventory.PlayerInventory;
028    
029    import be.Balor.Tools.MaterialContainer;
030    import be.Balor.Tools.Type;
031    import be.Balor.Tools.Type.ArmorPart;
032    import be.Balor.bukkit.AdminCmd.ACPluginManager;
033    import be.Balor.bukkit.AdminCmd.ConfigEnum;
034    
035    /**
036     * @author Balor (aka Antoine Aflalo)
037     * 
038     */
039    public class ArmoredKitInstance extends KitInstance {
040            protected final Map<Type.ArmorPart, MaterialContainer> armor = new EnumMap<Type.ArmorPart, MaterialContainer>(
041                            Type.ArmorPart.class);
042    
043            /**
044             * @param name
045             * @param delay
046             * @param items
047             */
048            public ArmoredKitInstance(final String name, final int delay,
049                            final List<MaterialContainer> items, final Map<Type.ArmorPart, MaterialContainer> armor) {
050                    super(name, delay, items);
051                    for (final Entry<Type.ArmorPart, MaterialContainer> e : armor.entrySet()) {
052                            if (!e.getKey().isValid(e.getValue().getMaterial().getId()))
053                                    continue;
054                            this.armor.put(e.getKey(), e.getValue());
055                    }
056    
057            }
058    
059            public ArmoredKitInstance(final KitInstance kit) {
060                    super(kit.name, kit.delay, kit.items);
061            }
062    
063            /**
064             * Get the wanted part of the armor
065             * 
066             * @param part
067             * @return itemstack containing the part of the armor
068             */
069            public ItemStack getArmorPart(final Type.ArmorPart part) {
070                    final MaterialContainer mat = armor.get(part);
071                    if (mat == null)
072                            return null;
073                    return mat.getItemStack();
074            }
075    
076            /**
077             * Set the player armor using the armor set in the kit
078             * 
079             * @param p
080             *            player to change the armor's inventory.
081             * @return contain the Part of the armor that couldn't be setted.
082             */
083            public ArmorPart[] setPlayerArmorParts(final Player p) {
084                    final List<ArmorPart> armorParts = new ArrayList<Type.ArmorPart>(ArmorPart.values().length);
085                    final ItemStack[] armors = new ItemStack[ArmorPart.values().length];
086                    final PlayerInventory inventory = p.getInventory();
087                    if (ConfigEnum.ARMOR_KIT_OVERRIDE.getBoolean()) {
088                            for (final ArmorPart part : ArmorPart.values()) {
089                                    ItemStack toadd = getArmorPart(part);
090                                    if (toadd == null) {
091                                            toadd = inventory.getItem(inventory.getSize() + part.getPlaceInInventory());
092                                            armorParts.add(part);
093                                    }
094                                    armors[part.getPlaceInInventory()] = toadd;
095                            }
096                    } else {
097                            for (final ArmorPart part : ArmorPart.values()) {
098                                    ItemStack toadd = getArmorPart(part);
099                                    if (inventory.getItem(inventory.getSize() + part.getPlaceInInventory()) != null)
100                                            continue;
101                                    if (toadd == null) {
102                                            toadd = inventory.getItem(inventory.getSize() + part.getPlaceInInventory());
103                                            armorParts.add(part);
104                                    }
105                                    armors[part.getPlaceInInventory()] = toadd;
106                            }
107                    }
108                    ACPluginManager.scheduleSyncTask(new Runnable() {
109    
110                            @Override
111                            public void run() {
112                                    inventory.setArmorContents(armors);
113                            }
114                    });
115                    return armorParts.toArray(new ArmorPart[0]);
116            }
117    
118            /*
119             * (non-Javadoc)
120             * 
121             * @see
122             * be.Balor.Tools.Files.KitInstance#addParent(be.Balor.Tools.Files.KitInstance
123             * )
124             */
125            @Override
126            public void addParent(final KitInstance parent) {
127                    super.addParent(parent);
128                    if (!(parent instanceof ArmoredKitInstance))
129                            return;
130                    final ArmoredKitInstance armoredParent = (ArmoredKitInstance) parent;
131                    for (final ArmorPart part : ArmorPart.values()) {
132                            if (armor.get(part) == null)
133                                    armor.put(part, armoredParent.armor.get(part));
134                    }
135    
136            }
137    }
