001package be.Balor.Tools.Files.Unicode;
002
003import java.io.BufferedReader;
004import java.io.BufferedWriter;
005import java.io.File;
006import java.io.FileOutputStream;
007import java.io.IOException;
008import java.io.InputStream;
009import java.io.OutputStreamWriter;
010import java.io.Reader;
011
012public class UnicodeUtil {
013
014        public static final byte[] UTF8_BOMS = new byte[]{(byte) 0xEF, (byte) 0xBB,
015                        (byte) 0xBF};
016        public static final int BOM_SIZE = 4;
017
018        /**
019         * Save a String to a file in UTF8 with Signature
020         * 
021         * @param file
022         *            file to save
023         * @param data
024         *            data to write
025         * @param append
026         *            append data
027         * @throws IOException
028         *             something goes wrong with the file
029         */
030        public static void saveUTF8File(final File file, final String data,
031                        final boolean append) throws IOException {
032                BufferedWriter bw = null;
033                OutputStreamWriter osw = null;
034
035                final FileOutputStream fos = new FileOutputStream(file, append);
036                try {
037                        // write UTF8 BOM mark if file is empty
038                        if (file.length() < 1) {
039                                fos.write(UTF8_BOMS);
040                        }
041
042                        osw = new OutputStreamWriter(fos, "UTF-8");
043                        bw = new BufferedWriter(osw);
044                        if (data != null) {
045                                bw.write(data);
046                        }
047                } finally {
048                        try {
049                                bw.close();
050                                fos.close();
051                        } catch (final Exception ex) {}
052                }
053        }
054
055        /**
056         * Save the content of an InputStream to a file in UTF8 with Signature
057         * 
058         * @param file
059         *            file to save
060         * @param stream
061         *            InputStream to copy to the file
062         * @param append
063         *            append data
064         * @throws IOException
065         *             something goes wrong with the file
066         */
067        public static void saveUTF8File(final File file, final InputStream stream,
068                        final boolean append) throws IOException {
069                BufferedWriter bw = null;
070                OutputStreamWriter osw = null;
071
072                final FileOutputStream fos = new FileOutputStream(file, append);
073                final Reader reader = new BufferedReader(new UnicodeReader(stream,
074                                "UTF-8"));
075                try {
076                        // write UTF8 BOM mark if file is empty
077                        if (file.length() < 1) {
078                                fos.write(UTF8_BOMS);
079                        }
080                        osw = new OutputStreamWriter(fos, "UTF-8");
081                        bw = new BufferedWriter(osw);
082                        if (reader != null) {
083                                for (int i = 0; (i = reader.read()) > 0;) {
084                                        bw.write(i);
085                                }
086                        }
087                } finally {
088                        try {
089                                bw.close();
090                                fos.close();
091                                reader.close();
092                        } catch (final Exception ex) {}
093                }
094        }
095
096}
