004
005Original pseudocode   : Thomas Weidenfeller
006Implementation tweaked: Aki Nieminen
007
008http://www.unicode.org/unicode/faq/utf_bom.html
009BOMs in byte length ordering:
010  00 00 FE FF    = UTF-32, big-endian
011  FF FE 00 00    = UTF-32, little-endian
012  EF BB BF       = UTF-8,
013  FE FF          = UTF-16, big-endian
014  FF FE          = UTF-16, little-endian
015
016Win2k Notepad:
017  Unicode format = UTF-16LE
018 ***/
019package be.Balor.Tools.Files.Unicode;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.PushbackInputStream;
024
025/**
026 * This inputstream will recognize unicode BOM marks and will skip bytes if
027 * getEncoding() method is called before any of the read(...) methods.
028 * 
029 * Usage pattern: String enc = "ISO-8859-1"; // or NULL to use systemdefault
030 * FileInputStream fis = new FileInputStream(file); UnicodeInputStream uin = new
031 * UnicodeInputStream(fis, enc); enc = uin.getEncoding(); // check and skip
032 * possible BOM bytes InputStreamReader in; if (enc == null) in = new
033 * InputStreamReader(uin); else in = new InputStreamReader(uin, enc);
034 */
035public class UnicodeInputStream extends InputStream {
036        PushbackInputStream internalIn;
037        boolean isInited = false;
038        String defaultEnc;
039        String encoding;
040
041        public UnicodeInputStream(final InputStream in, final String defaultEnc) {
042                internalIn = new PushbackInputStream(in, UnicodeUtil.BOM_SIZE);
043                this.defaultEnc = defaultEnc;
044        }
045
046        public String getDefaultEncoding() {
047                return defaultEnc;
048        }
049
050        public String getEncoding() {
051                if (!isInited) {
052                        try {
053                                init();
054                        } catch (final IOException ex) {
055                                final IllegalStateException ise = new IllegalStateException(
056                                                "Init method failed.");
057                                ise.initCause(ise);
058                                throw ise;
059                        }
060                }
061                return encoding;
062        }
063
064        /**
065         * Read-ahead four bytes and check for BOM marks. Extra bytes are unread
066         * back to the stream, only BOM bytes are skipped.
067         */
068        protected void init() throws IOException {
069                if (isInited) {
070                        return;
071                }
072
073                final byte bom[] = new byte[UnicodeUtil.BOM_SIZE];
074                int n, unread;
075                n = internalIn.read(bom, 0, bom.length);
076
077                if ((bom[0] == (byte) 0x00) && (bom[1] == (byte) 0x00)
078                                && (bom[2] == (byte) 0xFE) && (bom[3] == (byte) 0xFF)) {
079                        encoding = "UTF-32BE";
080                        unread = n - 4;
081                } else if ((bom[0] == (byte) 0xFF) && (bom[1] == (byte) 0xFE)
082                                && (bom[2] == (byte) 0x00) && (bom[3] == (byte) 0x00)) {
083                        encoding = "UTF-32LE";
084                        unread = n - 4;
085                } else if ((bom[0] == (byte) 0xEF) && (bom[1] == (byte) 0xBB)
086                                && (bom[2] == (byte) 0xBF)) {
087                        encoding = "UTF-8";
088                        unread = n - 3;
089                } else if ((bom[0] == (byte) 0xFE) && (bom[1] == (byte) 0xFF)) {
090                        encoding = "UTF-16BE";
091                        unread = n - 2;
092                } else if ((bom[0] == (byte) 0xFF) && (bom[1] == (byte) 0xFE)) {
093                        encoding = "UTF-16LE";
094                        unread = n - 2;
095                } else {
096                        // Unicode BOM mark not found, unread all bytes
097                        encoding = defaultEnc;
098                        unread = n;
099                }
100                // System.out.println("read=" + n + ", unread=" + unread);
101
102                if (unread > 0) {
103                        internalIn.unread(bom, (n - unread), unread);
104                }
105
106                isInited = true;
107        }
108
109        @Override
110        public void close() throws IOException {
111                // init();
112                isInited = true;
113                internalIn.close();
114        }
115
116        @Override
117        public int read() throws IOException {
118                // init();
119                isInited = true;
120                return internalIn.read();
121        }
122}
