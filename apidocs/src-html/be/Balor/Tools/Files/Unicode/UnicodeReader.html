001/**
002 version: 1.1 / 2007-01-25
003 - changed BOM recognition ordering (longer boms first)
004
005 Original pseudocode   : Thomas Weidenfeller
006 Implementation tweaked: Aki Nieminen
007
008 http://www.unicode.org/unicode/faq/utf_bom.html
009 BOMs:
010 00 00 FE FF    = UTF-32, big-endian
011 FF FE 00 00    = UTF-32, little-endian
012 EF BB BF       = UTF-8,
013 FE FF          = UTF-16, big-endian
014 FF FE          = UTF-16, little-endian
015
016 Win2k Notepad:
017 Unicode format = UTF-16LE
018 ***/
019package be.Balor.Tools.Files.Unicode;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.io.InputStreamReader;
024import java.io.PushbackInputStream;
025import java.io.Reader;
026
027/**
028 * Generic unicode textreader, which will use BOM mark to identify the encoding
029 * to be used. If BOM is not found then use a given default or system encoding.
030 */
031public class UnicodeReader extends Reader {
032        PushbackInputStream internalIn;
033        InputStreamReader internalIn2 = null;
034        String defaultEnc;
035
036        /**
037         * 
038         * @param in
039         *            inputstream to be read
040         * @param defaultEnc
041         *            default encoding if stream does not have BOM marker. Give NULL
042         *            to use system-level default.
043         */
044        public UnicodeReader(final InputStream in, final String defaultEnc) {
045                internalIn = new PushbackInputStream(in, UnicodeUtil.BOM_SIZE);
046                this.defaultEnc = defaultEnc;
047        }
048
049        public String getDefaultEncoding() {
050                return defaultEnc;
051        }
052
053        /**
054         * Get stream encoding or NULL if stream is uninitialized. Call init() or
055         * read() method to initialize it.
056         */
057        public String getEncoding() {
058                if (internalIn2 == null) {
059                        return null;
060                }
061                return internalIn2.getEncoding();
062        }
063
064        /**
065         * Read-ahead four bytes and check for BOM marks. Extra bytes are unread
066         * back to the stream, only BOM bytes are skipped.
067         */
068        protected void init() throws IOException {
069                if (internalIn2 != null) {
070                        return;
071                }
072
073                String encoding;
074                final byte bom[] = new byte[UnicodeUtil.BOM_SIZE];
075                int n, unread;
076                n = internalIn.read(bom, 0, bom.length);
077
078                if ((bom[0] == (byte) 0x00) && (bom[1] == (byte) 0x00)
079                                && (bom[2] == (byte) 0xFE) && (bom[3] == (byte) 0xFF)) {
080                        encoding = "UTF-32BE";
081                        unread = n - 4;
082                } else if ((bom[0] == (byte) 0xFF) && (bom[1] == (byte) 0xFE)
083                                && (bom[2] == (byte) 0x00) && (bom[3] == (byte) 0x00)) {
084                        encoding = "UTF-32LE";
085                        unread = n - 4;
086                } else if ((bom[0] == (byte) 0xEF) && (bom[1] == (byte) 0xBB)
087                                && (bom[2] == (byte) 0xBF)) {
088                        encoding = "UTF-8";
089                        unread = n - 3;
090                } else if ((bom[0] == (byte) 0xFE) && (bom[1] == (byte) 0xFF)) {
091                        encoding = "UTF-16BE";
092                        unread = n - 2;
093                } else if ((bom[0] == (byte) 0xFF) && (bom[1] == (byte) 0xFE)) {
094                        encoding = "UTF-16LE";
095                        unread = n - 2;
096                } else {
097                        // Unicode BOM mark not found, unread all bytes
098                        encoding = defaultEnc;
099                        unread = n;
100                }
101                // System.out.println("read=" + n + ", unread=" + unread);
102
103                if (unread > 0) {
104                        internalIn.unread(bom, (n - unread), unread);
105                }
106
107                // Use given encoding
108                if (encoding == null) {
109                        internalIn2 = new InputStreamReader(internalIn);
110                } else {
111                        internalIn2 = new InputStreamReader(internalIn, encoding);
112                }
113        }
114
115        @Override
116        public void close() throws IOException {
117                init();
118                internalIn2.close();
119        }
120
121        @Override
122        public int read(final char[] cbuf, final int off, final int len)
123                        throws IOException {
124                init();
125                return internalIn2.read(cbuf, off, len);
126        }
127
128}
