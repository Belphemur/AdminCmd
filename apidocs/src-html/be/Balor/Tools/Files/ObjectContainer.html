047            /**
048             * Gets a string at a location. This will either return an String or null,
049             * with null meaning that no configuration value exists at that location. If
050             * the object at the particular location is not actually a string, it will
051             * be converted to its string representation.
052             * 
053             * 
054             * @return string or null
055             */
056            public String getString() {
057    
058                    if (obj == null) {
059                            return null;
060                    }

062            }

172            /**
173             * Gets a list of objects at a location. If the list is not defined, null
174             * will be returned. The node must be an actual list.
175             * 
176             * @param path
177             *            path to node (dot notation)
178             * @return boolean or default
179             */
180            @SuppressWarnings("unchecked")
181            public List<Object> getList() {
182                    final Object o = obj;
183    
184                    if (o == null) {
185                            return null;
186                    } else if (o instanceof List) {
187                            return (List<Object>) o;
188                    } else {
189                            return null;
190                    }
191            }
192    
193            /**
194             * Gets a list of strings. Non-valid entries will not be in the list. There
195             * will be no null slots. If the list is not defined, the default will be
196             * returned. 'null' can be passed for the default and an empty list will be
197             * returned instead. If an item in the list is not a string, it will be
198             * converted to a string. The node must be an actual list and not just a
199             * string.
200             * 
201             * @param def
202             *            default value or null for an empty list as default
203             * @return list of strings
204             */
205            public List<String> getStringList(final List<String> def) {
206                    final List<Object> raw = getList();
207    
208                    if (raw == null) {
209                            return def != null ? def : new ArrayList<String>();
210                    }
211    
212                    final List<String> list = new ArrayList<String>();
213    
214                    for (final Object o : raw) {
215                            if (o == null) {
216                                    continue;
217                            }
218    
219                            list.add(o.toString());
220                    }
221    
222                    return list;
223            }

318            /**
319             * Casts a value to an integer. May return null.
320             * 
321             * @param o
322             * @return
323             */
324            private static Integer castInt(final Object o) {
325                    if (o == null) {
326                            return null;
327                    } else if (o instanceof Byte) {
328                            return (int) (Byte) o;
329                    } else if (o instanceof Integer) {
330                            return (Integer) o;
331                    } else if (o instanceof Double) {
332                            return (int) (double) (Double) o;
333                    } else if (o instanceof Float) {
334                            return (int) (float) (Float) o;
335                    } else if (o instanceof Long) {
336                            return (int) (long) (Long) o;
337                    } else {
338                            return null;
339                    }
340            }
341    
342            /**
343             * Casts a value to a double. May return null.
344             * 
345             * @param o
346             * @return
347             */
348            private static Double castDouble(final Object o) {
349                    if (o == null) {
350                            return null;
351                    } else if (o instanceof Float) {
352                            return (double) (Float) o;
353                    } else if (o instanceof Double) {
354                            return (Double) o;
355                    } else if (o instanceof Byte) {
356                            return (double) (Byte) o;
357                    } else if (o instanceof Integer) {
358                            return (double) (Integer) o;
359                    } else if (o instanceof Long) {
360                            return (double) (Long) o;
361                    } else {
362                            return null;
363                    }
364            }

382            /**
383             * Casts a value to a long. May return null.
384             * 
385             * @param o
386             * @return
387             */
388            private static Long castLong(final Object o) {
389                    if (o == null) {
390                            return null;
391                    } else if (o instanceof Long) {
392                            return (Long) o;
393                    } else if (o instanceof Float) {
394                            return Long.parseLong(((Float) o).toString());
395                    } else if (o instanceof Double) {
396                            return Long.parseLong(((Double) o).toString());
397                    } else if (o instanceof Byte) {
398                            return Long.parseLong(((Byte) o).toString());
399                    } else if (o instanceof Integer) {
400                            return Long.parseLong(((Integer) o).toString());
401                    } else {
402                            return null;
403                    }
404            }
405    
406            /**
407             * Casts a value to a float. May return null.
408             * 
409             * @param o
410             * @return
411             */
412            private static Float castFloat(final Object o) {
413                    if (o == null) {
414                            return null;
415                    } else if (o instanceof Float) {
416                            return (Float) o;
417                    } else if (o instanceof Long) {
418                            return Float.parseFloat(((Long) o).toString());
419                    } else if (o instanceof Double) {
420                            return Float.parseFloat(((Double) o).toString());
421                    } else if (o instanceof Byte) {
422                            return Float.parseFloat(((Byte) o).toString());
423                    } else if (o instanceof Integer) {
424                            return Float.parseFloat(((Integer) o).toString());
425                    } else {
426                            return null;
427                    }
