001    package be.Balor.Tools.Files;
002    
003    import java.io.File;
004    import java.io.FilenameFilter;
005    import java.util.ArrayList;
006    import java.util.Arrays;
007    
008    public class YmlFilter implements FilenameFilter {
009    
010            @Override
011            public boolean accept(final File file, final String name) {
012                    if ((name.endsWith(".yml") && !name.endsWith("_orig.yml"))) {
013                            return true;
014                    } else {
015                            return false;
016                    }
017            }
018    
019            public static File[] listRecursively(final File folder) {
020                    if (folder != null && folder.isDirectory()) {
021                            return _listRecursively(folder, 5).toArray(new File[0]);
022                    }
023                    return new File[0];
024            }
025    
026            public static File[] listRecursively(final File folder, final int depth) {
027                    if (folder != null && folder.isDirectory()) {
028                            return _listRecursively(folder, depth).toArray(new File[0]);
029                    }
030                    return new File[0];
031            }
032    
033            private static ArrayList<File> _listRecursively(final File folder, final int depth) {
034                    final ArrayList<File> files = new ArrayList<File>();
035                    if (folder != null && folder.isDirectory()) {
036                            files.addAll(Arrays.asList(folder.listFiles(new YmlFilter())));
037                            if (depth > 0) {// now scan folders
038                                    final File folders[] = folder.listFiles(new DirFilter());
039                                    if (folders != null) {
040                                            for (final File f : folders) {
041                                                    files.addAll(_listRecursively(f, depth - 1));
042                                            }
043                                    }
044                            }
045                    }
046                    return files;
047            }
048    
049            public static class DirFilter implements FilenameFilter {
050    
051                    @Override
052                    public boolean accept(final File file, final String name) {
053                            return file.isDirectory();
054                    }
055            }
056    }
