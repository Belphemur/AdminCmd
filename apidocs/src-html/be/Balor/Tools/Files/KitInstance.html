001    package be.Balor.Tools.Files;
002    
003    import java.util.ArrayList;
004    import java.util.Arrays;
005    import java.util.List;
006    
007    import org.bukkit.inventory.ItemStack;
008    
009    import be.Balor.Manager.Permissions.PermissionLinker;
010    import be.Balor.Tools.MaterialContainer;
011    import be.Balor.bukkit.AdminCmd.ACPluginManager;
012    
013    public class KitInstance {
014    
015            protected final String name;
016            protected int delay = 0;
017            protected final List<MaterialContainer> items;
018            private static final PermissionLinker perm = ACPluginManager.getCorePlugin()
019                            .getPermissionLinker();
020    
021            public KitInstance(final String name, final int delay, final List<MaterialContainer> items) {
022                    this.name = name;
023                    this.delay = delay;
024                    this.items = items;
025                    perm.addPermChild("admincmd.kit." + this.name);
026            }
027    
028            /**
029             * Get ItemStacks for given kit
030             * 
031             * @param kit
032             * @return
033             */
034            public List<ItemStack> getItemStacks() {
035                    final ArrayList<ItemStack> result = new ArrayList<ItemStack>();
036                    try {
037                            // return Utils.oddItem.getItemGroup(kit, -1));
038                    } catch (final Throwable e) {
039                    }
040                    for (final MaterialContainer mc : items)
041                            result.add(mc.getItemStack());
042                    return result;
043            }
044    
045            /**
046             * Adding a parent to the kit
047             * 
048             * @param parent
049             */
050            public void addParent(final KitInstance parent) {
051                    for (final MaterialContainer mc : parent.items) {
052                            final int index = items.indexOf(mc);
053                            if (index == -1)
054                                    items.add(mc);
055                            else {
056                                    final MaterialContainer mat = items.get(index);
057                                    mat.setAmount(mat.getAmount() + mc.getAmount());
058                            }
059                    }
060            }
061    
062            public int getDelay() {
063                    return delay;
064            }
065    
066            /**
067             * @return the name
068             */
069            public String getName() {
070                    return name;
071            }
072    
073            /*
074             * (non-Javadoc)
075             * 
076             * @see java.lang.Object#toString()
077             */
078            @Override
079            public String toString() {
080                    return "KitInstance [name=" + name + ", delay=" + delay + ", items="
081                                    + Arrays.toString(items.toArray()) + "]";
082            }
083    
084    }
