017    package be.Balor.Tools.Blocks;
018    
019    import org.bukkit.Location;
020    import org.bukkit.block.Block;
021    
022    import be.Balor.Tools.Utils;
023    
024    /**
025     * @author Balor (aka Antoine Aflalo)
026     * 
027     */
028    public class LogBlockRemanence extends BlockRemanence {
029            private final String playerName;
030    
031            /**
032             * @param loc
033             */
034            /**
035             * @param loc
036             * @param playerName
037             * @param logblock
038             */
039            public LogBlockRemanence(final Location loc, final String playerName) {
040                    super(loc);
041                    this.playerName = playerName;
042            }
043    
044            @Override
045            public Block returnToThePast() {
046                    final Block result = super.returnToThePast();
047                    Utils.logBlock.queueBlockPlace(playerName, result.getState());
048                    return result;
049            }
050    
051            @Override
052            public void setBlockType(final int type) {
053                    final Block current = loc.getWorld().getBlockAt(loc);
054                    Utils.logBlock.queueBlockBreak(playerName, current.getState());
055                    current.setTypeId(type, true);
056            }
057    
058    }
