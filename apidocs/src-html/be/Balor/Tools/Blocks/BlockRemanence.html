017    package be.Balor.Tools.Blocks;
018    
019    import org.bukkit.Location;
020    import org.bukkit.block.Block;
021    
022    /**
023     * @author Balor (aka Antoine Aflalo)
024     * 
025     */
026    public class BlockRemanence {
027            protected Location loc;
028            private final int oldType;
029            private byte data = 0;
030            private boolean useData;
031    
032            /**
033             *
034             */
035            public BlockRemanence(final Location loc) {
036                    this.loc = loc;
037                    final Block b = loc.getWorld().getBlockAt(loc);
038                    this.oldType = b.getTypeId();
039                    if ((useData = usesData(oldType)))
040                            data = b.getState().getRawData();
041    
042            }
043    
044            public Block returnToThePast() {
045                    final Block b = loc.getWorld().getBlockAt(loc);
046                    if (useData)
047                            b.setTypeIdAndData(oldType, data, true);
048                    else
049                            b.setTypeId(oldType, true);
050                    return b;
051            }
052    
053            public void setBlockType(final int type) {
054                    loc.getWorld().getBlockAt(loc).setTypeId(type, true);
055            }
056    
057            /**
058             * @return the oldType
059             */
060            public int getOldType() {
061                    return oldType;
062            }
063    
064            /**
065             * @return the data
066             */
067            public byte getData() {
068                    return data;
069            }
070
