017package be.Balor.Tools.Blocks;

019import org.bukkit.Location;
020import org.bukkit.block.Block;
021
022/**
023 * @author Balor (aka Antoine Aflalo)
024 * 
025 */
026public class BlockRemanence {
027        protected Location loc;
028        private final int oldType;
029        private byte data = 0;
030        private boolean useData;
031
032        /**
033         *
034         */
035        public BlockRemanence(final Location loc) {
036                this.loc = loc;
037                final Block b = loc.getWorld().getBlockAt(loc);
038                this.oldType = b.getTypeId();
039                if ((useData = usesData(oldType))) {
040                        data = b.getState().getRawData();
041                }
042
043        }
044
045        public Block returnToThePast() {
046                final Block b = loc.getWorld().getBlockAt(loc);
047                if (useData) {
048                        b.setTypeIdAndData(oldType, data, true);
049                } else {
050                        b.setTypeId(oldType, true);
051                }
052                return b;
053        }
054
055        public void setBlockType(final int type) {
056                loc.getWorld().getBlockAt(loc).setTypeId(type, true);
057        }
058
059        /**
060         * @return the oldType
061         */
062        public int getOldType() {
063                return oldType;
064        }
065
066        /**
067         * @return the data
068         */
069        public byte getData() {
070                return data;
071        }
