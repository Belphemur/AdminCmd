110            public Metrics() throws IOException {
111                    // load the config
112                    final File file = new File(CONFIG_FILE);
113                    configuration = YamlConfiguration.loadConfiguration(file);
114    
115                    // add some defaults
116                    configuration.addDefault("opt-out", false);
117                    configuration.addDefault("guid", UUID.randomUUID().toString());
118    
119                    // Do we need to create the file?
120                    if (configuration.get("guid", null) == null) {
121                            configuration.options().header("http://metrics.griefcraft.com").copyDefaults(true);
122                            configuration.save(file);
123                    }
124    
125                    // Load the guid then
126                    guid = configuration.getString("guid");
127            }

129            /**
130             * Construct and create a Graph that can be used to separate specific
131             * plotters to their own graphs on the metrics website. Plotters can be
132             * added to the graph object returned.
133             * 
134             * @param plugin
135             * @param type
136             * @param name
137             * @return Graph object created. Will never return NULL under normal
138             *         circumstances unless bad parameters are given
139             */
140            public Graph createGraph(final Plugin plugin, final Graph.Type type, final String name) {
141                    if (plugin == null || type == null || name == null) {
142                            throw new IllegalArgumentException("All arguments must not be null");
143                    }
144    
145                    // Construct the graph object
146                    final Graph graph = new Graph(type, name);
147    
148                    // Get the graphs for the plugin
149                    final Set<Graph> graphs = getOrCreateGraphs(plugin);
150    
151                    // Now we can add our graph
152                    graphs.add(graph);
153    
154                    // and return back
155                    return graph;
156            }

162             * @param plotter

164            public void addCustomData(final Plugin plugin, final Plotter plotter) {
165                    // The default graph for the plugin
166                    final Graph graph = getOrCreateDefaultGraph(plugin);

169                    graph.addPlotter(plotter);
170    
171                    // Ensure the default graph is included in the submitted graphs
172                    getOrCreateGraphs(plugin).add(graph);
173            }

175            /**
176             * Begin measuring a plugin
177             * 
178             * @param plugin
179             */
180            public void beginMeasuringPlugin(final Plugin plugin) {
181                    // Did we opt out?
182                    if (configuration.getBoolean("opt-out", false)) {
183                            return;
184                    }
185    
186                    // Begin hitting the server with glorious data
187                    plugin.getServer().getScheduler().scheduleAsyncRepeatingTask(plugin, new Runnable() {
188                            private boolean firstPost = true;
189    
190                            @Override
191                            public void run() {
192                                    try {
193                                            // We use the inverse of firstPost because if it is the
194                                            // first time we are posting,
195                                            // it is not a interval ping, so it evaluates to FALSE
196                                            // Each time thereafter it will evaluate to TRUE, i.e PING!
197                                            postPlugin(plugin, !firstPost);
198    
199                                            // After the first post we set firstPost to false
200                                            // Each post thereafter will be a ping
201                                            firstPost = false;
202                                    } catch (final IOException e) {
203                                            System.out.println("[Metrics] " + e.getMessage());
204                                    }
205                            }
206                    }, 0, PING_INTERVAL * 1200);
207            }

209            /**
210             * Generic method that posts a plugin to the metrics website
211             * 
212             * @param plugin
213             */
214            private void postPlugin(final Plugin plugin, final boolean isPing) throws IOException {
215                    // The plugin's description file containg all of the plugin data such as
216                    // name, version, author, etc
217                    final PluginDescriptionFile description = plugin.getDescription();
218    
219                    // The author string, created with description.getAuthors()
220                    // Authors are separated by a comma
221                    String authors = "";
222    
223                    // Add each author to the string
224                    for (final String author : description.getAuthors()) {
225                            authors += author + ", ";
226                    }
227    
228                    // If there were any authors at all, we need to remove the last 2
229                    // characters
230                    // the last 2 characters are the last comma and space
231                    if (!authors.isEmpty()) {
232                            authors = authors.substring(0, authors.length() - 2);
233                    }
234    
235                    // Construct the post data
236                    String data = encode("guid")
237                                    + '='
238                                    + encode(guid)
239                                    + encodeDataPair("authors", authors)
240                                    + encodeDataPair("version", description.getVersion())
241                                    + encodeDataPair("server", Bukkit.getVersion())
242                                    + encodeDataPair("players",
243                                                    Integer.toString(Bukkit.getServer().getOnlinePlayers().length))
244                                    + encodeDataPair("revision", String.valueOf(REVISION));
245    
246                    // If we're pinging, append it
247                    if (isPing) {
248                            data += encodeDataPair("ping", "true");
249                    }
250    
251                    // Add any custom data available for the plugin
252                    final Set<Graph> graphs = getOrCreateGraphs(plugin);
253    
254                    // Acquire a lock on the graphs, which lets us make the assumption we
255                    // also lock everything
256                    // inside of the graph (e.g plotters)
257                    synchronized (graphs) {
258                            final Iterator<Graph> iter = graphs.iterator();
259    
260                            while (iter.hasNext()) {
261                                    final Graph graph = iter.next();
262    
263                                    // Because we have a lock on the graphs set already, it is
264                                    // reasonable to assume
265                                    // that our lock transcends down to the individual plotters in
266                                    // the graphs also.
267                                    // Because our methods are private, no one but us can reasonably
268                                    // access this list
269                                    // without reflection so this is a safe assumption without
270                                    // adding more code.
271                                    for (final Plotter plotter : graph.getPlotters()) {
272                                            // The key name to send to the metrics server
273                                            // The format is C-GRAPHNAME-PLOTTERNAME where separator -
274                                            // is defined at the top
275                                            // Legacy (R4) submitters use the format Custom%s, or
276                                            // CustomPLOTTERNAME
277                                            final String key = String.format("C%s%s%s%s", CUSTOM_DATA_SEPARATOR,
278                                                            graph.getName(), CUSTOM_DATA_SEPARATOR, plotter.getColumnName());
279    
280                                            // The value to send, which for the foreseeable future is
281                                            // just the string
282                                            // value of plotter.getValue()
283                                            final String value = Integer.toString(plotter.getValue());
284    
285                                            // Add it to the http post data :)
286                                            data += encodeDataPair(key, value);
287                                    }
288                            }
289                    }
290    
291                    // Create the url
292                    final URL url = new URL(BASE_URL
293                                    + String.format(REPORT_URL, plugin.getDescription().getName()));
294    
295                    // Connect to the website
296                    URLConnection connection;
297    
298                    // Mineshafter creates a socks proxy, so we can safely bypass it
299                    // It does not reroute POST requests so we need to go around it
300                    if (isMineshafterPresent()) {
301                            connection = url.openConnection(Proxy.NO_PROXY);
302                    } else {
303                            connection = url.openConnection();
304                    }
305    
306                    connection.setDoOutput(true);
307    
308                    // Write the data
309                    final OutputStreamWriter writer = new OutputStreamWriter(connection.getOutputStream());
310                    writer.write(data);
311                    writer.flush();
312    
313                    // Now read the response
314                    final BufferedReader reader = new BufferedReader(new InputStreamReader(
315                                    connection.getInputStream()));
316                    final String response = reader.readLine();
317    
318                    // close resources
319                    writer.close();
320                    reader.close();
321    
322                    if (response.startsWith("ERR")) {
323                            throw new IOException(response); // Throw the exception
324                    } else {
325                            // Is this the first update this hour?
326                            if (response.contains("OK This is your first update this hour")) {
327                                    synchronized (graphs) {
328                                            final Iterator<Graph> iter = graphs.iterator();
329    
330                                            while (iter.hasNext()) {
331                                                    final Graph graph = iter.next();
332    
333                                                    for (final Plotter plotter : graph.getPlotters()) {
334                                                            plotter.reset();
335                                                    }
336                                            }
337                                    }
338                            }
339                    }
340                    // if (response.startsWith("OK")) - We should get "OK" followed by an
341                    // optional description if everything goes right
342            }

344            /**
345             * Get or create the Set of graphs for a specific plugin
346             * 
347             * @param plugin
348             * @return
349             */
350            private Set<Graph> getOrCreateGraphs(final Plugin plugin) {
351                    Set<Graph> theGraphs = graphs.get(plugin);
352    
353                    // Create the Set if it does not already exist
354                    if (theGraphs == null) {
355                            theGraphs = Collections.synchronizedSet(new HashSet<Graph>());
356                            graphs.put(plugin, theGraphs);
357                    }
358    
359                    return theGraphs;
360            }
361    
362            /**
363             * Get the default graph for a plugin and if it does not exist, create one
364             * 
365             * @param plugin
366             * @return
367             */
368            private Graph getOrCreateDefaultGraph(final Plugin plugin) {
369                    Graph graph = defaultGraphs.get(plugin);
370    
371                    // Not yet created :(
372                    if (graph == null) {
373                            graph = new Graph(Graph.Type.Line, "Default");
374                            defaultGraphs.put(plugin, graph);
375                    }
376    
377                    return graph;
378            }

380            /**
381             * Check if mineshafter is present. If it is, we need to bypass it to send
382             * POST requests
383             * 
384             * @return
385             */
386            private boolean isMineshafterPresent() {
387                    try {
388                            Class.forName("mineshafter.MineServer");
389                            return true;
390                    } catch (final Exception e) {
391                            return false;
392                    }
393            }

395            /**
396             * Encode a key/value data pair to be used in a HTTP post request. This
397             * INCLUDES a & so the first key/value pair MUST be included manually, e.g:
398             * <p>
399             * String httpData = encode("guid") + "=" + encode("1234") +
400             * encodeDataPair("authors") + "..";
401             * </p>
402             * 
403             * @param key
404             * @param value
405             * @return
406             */
407            private static String encodeDataPair(final String key, final String value)
408                            throws UnsupportedEncodingException {
409                    return "&" + encode(key) + "=" + encode(value);
410            }
411    
412            /**
413             * Encode text as UTF-8
414             * 
415             * @param text
416             * @return
417             */
418            private static String encode(final String text) throws UnsupportedEncodingException {
419                    return URLEncoder.encode(text, "UTF-8");
420            }

422            /**
423             * Represents a custom graph on the website
424             */
425            public static class Graph {
426    
427                    /**
428                     * The graph's type that will be visible on the website
429                     */
430                    public static enum Type {
431    
432                            /**
433                             * A simple line graph which also includes a scrollable timeline
434                             * viewer to view as little or as much of the data as possible.
435                             */
436                            Line,
437    
438                            /**
439                             * An area graph. This is the same as a line graph except the area
440                             * under the curve is shaded
441                             */
442                            Area,
443    
444                            /**
445                             * A column graph, which is a graph where the data is represented by
446                             * columns on the vertical axis, i.e they go up and down.
447                             */
448                            Column,
449    
450                            /**
451                             * A pie graph. The graph is generated by taking the data for the
452                             * last hour and summing it together. Then the percentage for each
453                             * plotter is calculated via round( (plot / total) * 100, 2 )
454                             */
455                            Pie
456    
457                    }
458    
459                    /**
460                     * What the graph should be plotted as
461                     */
462                    private final Type type;
463    
464                    /**
465                     * The graph's name, alphanumeric and spaces only :) If it does not
466                     * comply to the above when submitted, it is rejected
467                     */
468                    private final String name;
469    
470                    /**
471                     * The set of plotters that are contained within this graph
472                     */
473                    private final Set<Plotter> plotters = new LinkedHashSet<Plotter>();
474    
475                    private Graph(final Type type, final String name) {
476                            this.type = type;
477                            this.name = name;
478                    }
479    
480                    /**
481                     * Gets the graph's name
482                     * 
483                     * @return
484                     */
485                    public String getName() {
486                            return name;
487                    }
488    
489                    /**
490                     * Add a plotter to the graph, which will be used to plot entries
491                     * 
492                     * @param plotter
493                     */
494                    public void addPlotter(final Plotter plotter) {
495                            plotters.add(plotter);
496                    }
497    
498                    /**
499                     * Remove a plotter from the graph
500                     * 
501                     * @param plotter
502                     */
503                    public void removePlotter(final Plotter plotter) {
504                            plotters.remove(plotter);
505                    }
506    
507                    /**
508                     * Gets an <b>unmodifiable</b> set of the plotter objects in the graph
509                     * 
510                     * @return
511                     */
512                    public Set<Plotter> getPlotters() {
513                            return Collections.unmodifiableSet(plotters);
514                    }
515    
516                    @Override
517                    public int hashCode() {
518                            return (type.hashCode() * 17) ^ name.hashCode();
519                    }
520    
521                    @Override
522                    public boolean equals(final Object object) {
523                            if (!(object instanceof Graph)) {
524                                    return false;
525                            }
526    
527                            final Graph graph = (Graph) object;
528                            return graph.type == type && graph.name.equals(name);
529                    }
530    
531            }
532    
533            /**
534             * Removes a custom data plotter for a given plugin
535             * 
536             * @param plugin
537             * @param plotter
538             */
539            public void removeCustomData(final Plugin plugin, final Plotter plotter) {
540                    final Graph graph = getOrCreateDefaultGraph(plugin);
541                    if (graph == null)
542                            return;
543                    graph.removePlotter(plotter);
544            }

546            /**
547             * Interface used to collect custom data for a plugin
548             */
549            public static abstract class Plotter {
550    
551                    /**
552                     * The plot's name
553                     */
554                    private final String name;
555    
556                    /**
557                     * Construct a plotter with the default plot name
558                     */
559                    public Plotter() {
560                            this("Default");
561                    }
562    
563                    /**
564                     * Construct a plotter with a specific plot name
565                     * 
566                     * @param name
567                     */
568                    public Plotter(final String name) {
569                            this.name = name;
570                    }
571    
572                    /**
573                     * Get the current value for the plotted point
574                     * 
575                     * @return
576                     */
577                    public abstract int getValue();
578    
579                    /**
580                     * Get the column name for the plotted point
581                     * 
582                     * @return the plotted point's column name
583                     */
584                    public String getColumnName() {
585                            return name;
586                    }
587    
588                    /**
589                     * Called after the website graphs have been updated
590                     */
591                    public void reset() {
592                    }
593    
594                    @Override
595                    public int hashCode() {
596                            return getColumnName().hashCode() + getValue();
597                    }
598    
599                    @Override
600                    public boolean equals(final Object object) {
601                            if (!(object instanceof Plotter)) {
602                                    return false;
603                            }
604    
605                            final Plotter plotter = (Plotter) object;
606                            return plotter.name.equals(name) && plotter.getValue() == getValue();
607                    }
608    
609            }
