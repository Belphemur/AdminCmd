017package be.Balor.Tools.Converter;
018
019import java.util.List;
020
021import org.bukkit.Bukkit;
022import org.bukkit.World;
023
024import be.Balor.Tools.Debug.ACLogger;
025import be.Balor.World.IWorldFactory;
026import be.Balor.World.WorldConvertTask;
027
028/**
029 * @author Antoine
030 * 
031 */
032public class WorldConverter {
033        private final IWorldFactory oldFactory, newFactory;
034
035        /**
036         * @param oldFactory
037         * @param newFactory
038         */
039        public WorldConverter(final IWorldFactory oldFactory,
040                        final IWorldFactory newFactory) {
041                super();
042                this.oldFactory = oldFactory;
043                this.newFactory = newFactory;
044        }
045
046        public synchronized void convert() {
047                final List<World> worlds = Bukkit.getServer().getWorlds();
048                ACLogger.info("Begin Conversion of " + worlds.size() + " worlds");
049                for (final World world : worlds) {
050                        new WorldConvertTask(oldFactory, newFactory, world).run();
051                }
052        }
053
054}
