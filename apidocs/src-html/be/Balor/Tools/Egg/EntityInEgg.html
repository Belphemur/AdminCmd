017package be.Balor.Tools.Egg;
018
019import java.io.Serializable;
020
021import org.bukkit.entity.Chicken;
022
023/**
024 * @author Balor (aka Antoine Aflalo)
025 * 
026 */
027public class EntityInEgg implements Serializable {
028        /**
029         * 
030         */
031        private static final long serialVersionUID = -2370374531952085516L;
032        private String entityClass;
033        private int nb;
034        private String entityName;
035
036        /**
037         * @param entityClass
038         * @param nb
039         * @param entityName
040         */
041        public EntityInEgg(final String entityClass, final int nb,
042                        final String entityName) {
043                super();
044                this.entityClass = entityClass;
045                this.nb = nb;
046                this.entityName = entityName;
047        }
048
049        /**
050         * 
051         */
052        public EntityInEgg() {
053                super();
054        }
055
056        /**
057         * @return the entityClass
058         */
059        public String getEntityClassName() {
060                return entityClass;
061        }
062
063        /**
064         * @return the nb
065         */
066        public int getNb() {
067                return nb;
068        }
069
070        /**
071         * @return the entityName
072         */
073        public String getEntityName() {
074                return entityName;
075        }
076
077        /**
078         * @param entityClass
079         *            the entityClass to set
080         */
081        public void setEntityClass(final String entityClass) {
082                this.entityClass = entityClass;
083        }
084
085        /**
086         * @param nb
087         *            the nb to set
088         */
089        public void setNb(final int nb) {
090                this.nb = nb;
091        }
092
093        /**
094         * @param entityName
095         *            the entityName to set
096         */
097        public void setEntityName(final String entityName) {
098                this.entityName = entityName;
099        }
100
101        @SuppressWarnings("rawtypes")
102        public Class getEntityClass() {
103                try {
104                        return Class.forName(entityClass);
105                } catch (final ClassNotFoundException e) {
106                        return Chicken.class;
107                }
108        }
109
110        /*
111         * (non-Javadoc)
112         * 
113         * @see java.lang.Object#toString()
114         */
115        @Override
116        public String toString() {
117                return "EntityInEgg [nb=" + nb + ", entity=" + entityName + "]";
118        }
119
120}
