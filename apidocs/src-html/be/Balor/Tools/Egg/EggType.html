017    package be.Balor.Tools.Egg;
018    
019    import java.io.Serializable;
020    
021    import org.bukkit.entity.Player;
022    import org.bukkit.event.player.PlayerEggThrowEvent;
023    import org.bukkit.permissions.Permission;
024    
025    import be.Balor.Manager.Commands.CommandArgs;
026    import be.Balor.Manager.Commands.Mob.EggSpawner;
027    import be.Balor.Manager.Permissions.PermissionManager;
028    import be.Balor.Tools.Utils;
029    import be.Balor.Tools.Egg.Exceptions.DontHaveThePermissionException;
030    import be.Balor.Tools.Egg.Exceptions.ExceptionType;
031    import be.Balor.Tools.Egg.Exceptions.ParameterMissingException;
032    import be.Balor.Tools.Egg.Exceptions.ProcessingArgsException;
033    import be.Balor.bukkit.AdminCmd.LocaleHelper;
034    
035    /**
036     * @author Balor (aka Antoine Aflalo)
037     * 
038     */
039    public abstract class EggType<T> implements Serializable {
040            /**
041             *
042             */
043            private static final long serialVersionUID = 2793422400211176328L;
044    
045            protected T value;
046    
047            private final static ClassLoader eggTypeLoader = new EggTypeClassLoader();
048    
049            /**
050             * This function will be executed by the listener when the player have the
051             * power Egg with this EggType
052             * 
053             * @param event
054             *            triggered when the player throw an egg.
055             */
056            public abstract void onEvent(PlayerEggThrowEvent event);
057    
058            /**
059             * Will be called by the command {@link EggSpawner} to set the value
060             * 
061             * @param sender
062             *            Player that send the command.
063             * @param args
064             *            argument that will be used to set the EggType.
065             * 
066             * @throws ProcessingArgsException
067             *             when there is a problem in the arguments
068             */
069            protected abstract void processArguments(Player sender, CommandArgs args)
070                            throws ProcessingArgsException;
071    
072            /**
073             * Check if the user have the permission to use this Egg
074             * 
075             * @param player
076             * @return
077             * @throws DontHaveThePermissionException
078             *             when the player don't have the permission, with the message
079             *             to display to the user
080             */
081            protected boolean checkPermission(final Player player) throws DontHaveThePermissionException {
082                    final Permission perm = EggPermissionManager.INSTANCE.getPermission(this);
083                    if (!PermissionManager.hasPerm(player, perm, false))
084                            throw new DontHaveThePermissionException(
085                                            Utils.I18n("errorNotPerm", "p", perm.getName()));
086                    return true;
087            }
088    
089            public T getValue() {
090                    return value;
091            }
092    
093            public void setValue(final T value) {
094                    this.value = value;
095            }
096    
097            /**
098             * Will create the Egg that can be assigned later to the player. This method
099             * check if the player has the permission to use that egg and if the
100             * arguments of the command are right.
101             * 
102             * @param player
103             *            That execute the EggSpawn command.
104             * @param args
105             *            Provided by the EggSpawn command.
106             * @return an EggType of the chosen type if every test is passed.
107             * @throws ProcessingArgsException
108             *             if there is a problem while processing the arguments, like a
109             *             Parameter missing, or some other error.
110             * @throws DontHaveThePermissionException
111             *             if the player don't have the permission to use that egg
112             */
113            public static EggType<?> createEggType(final Player player, final CommandArgs args)
114                            throws ProcessingArgsException, DontHaveThePermissionException, NullPointerException {
115                    if (!args.hasFlag('E'))
116                            throw new ParameterMissingException('E', LocaleHelper.EGG_PARAM.getLocale());
117                    EggType<?> eggType;
118                    final String className = args.getValueFlag('E');
119                    try {
120                            eggType = matchEggClass(className);
121                    } catch (final ClassNotFoundException e) {
122                            throw new ProcessingArgsException(ExceptionType.NO_CLASS, className, e);
123                    } catch (final InstantiationException e) {
124                            throw new ProcessingArgsException(ExceptionType.INSTANCE, className, e);
125                    } catch (final IllegalAccessException e) {
126                            throw new ProcessingArgsException(ExceptionType.ILLEGAL_ACCESS, className, e);
127                    } catch (final NullPointerException e) {
128                            throw new NullPointerException();
129                    }
130                    eggType.checkPermission(player);
131                    eggType.processArguments(player, args);
132                    return eggType;
133            }
134    
135            @SuppressWarnings("unchecked")
136            private static EggType<?> matchEggClass(final String name) throws ClassNotFoundException,
137                            InstantiationException, IllegalAccessException, NullPointerException {
138                    final Class<? extends EggType<?>> c = (Class<? extends EggType<?>>) eggTypeLoader
139                                    .loadClass(name);
140                    return c.newInstance();
141            }
142    
143            /*
144             * (non-Javadoc)
145             * 
146             * @see java.lang.Object#toString()
147             */
148            @Override
149            public String toString() {
150                    return getClass().getSimpleName() + " = " + getValue();
151            }
152    
153            /*
154             * (non-Javadoc)
155             * 
156             * @see java.lang.Object#hashCode()
157             */
158            @Override
159            public int hashCode() {
160                    final int prime = 31;
161                    int result = 1;
162                    result = prime * result + ((value == null) ? 0 : value.hashCode());
163                    return result;
164            }
165    
166            /*
167             * (non-Javadoc)
168             * 
169             * @see java.lang.Object#equals(java.lang.Object)
170             */
171            @SuppressWarnings("rawtypes")
172            @Override
173            public boolean equals(final Object obj) {
174                    if (this == obj)
175                            return true;
176                    if (obj == null)
177                            return false;
178                    if (!(obj instanceof EggType))
179                            return false;
180                    final EggType other = (EggType) obj;
181                    if (value == null) {
182                            if (other.value != null)
183                                    return false;
184                    } else if (!value.equals(other.value))
185                            return false;
186                    return true;
187            }
188    
189    }
