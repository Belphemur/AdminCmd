017package be.Balor.Tools.Egg;
018
019import java.io.Serializable;
020
021import org.bukkit.entity.Player;
022import org.bukkit.event.player.PlayerEggThrowEvent;
023import org.bukkit.permissions.Permission;
024
025import be.Balor.Manager.Commands.CommandArgs;
026import be.Balor.Manager.Commands.Mob.EggSpawner;
027import be.Balor.Manager.Permissions.PermissionManager;
028import be.Balor.Tools.Utils;
029import be.Balor.Tools.Egg.Exceptions.DontHaveThePermissionException;
030import be.Balor.Tools.Egg.Exceptions.ExceptionType;
031import be.Balor.Tools.Egg.Exceptions.ParameterMissingException;
032import be.Balor.Tools.Egg.Exceptions.ProcessingArgsException;
033import be.Balor.bukkit.AdminCmd.LocaleHelper;
034
035/**
036 * @author Balor (aka Antoine Aflalo)
037 * 
038 */
039public abstract class EggType<T> implements Serializable {
040        /**
041         *
042         */
043        private static final long serialVersionUID = 2793422400211176328L;
044
045        protected T value;
046
047        private final static ClassLoader eggTypeLoader = new EggTypeClassLoader();
048
049        /**
050         * This function will be executed by the listener when the player have the
051         * power Egg with this EggType
052         * 
053         * @param event
054         *            triggered when the player throw an egg.
055         */
056        public abstract void onEvent(PlayerEggThrowEvent event);
057
058        /**
059         * Will be called by the command {@link EggSpawner} to set the value
060         * 
061         * @param sender
062         *            Player that send the command.
063         * @param args
064         *            argument that will be used to set the EggType.
065         * 
066         * @throws ProcessingArgsException
067         *             when there is a problem in the arguments
068         */
069        protected abstract void processArguments(Player sender, CommandArgs args)
070                        throws ProcessingArgsException;
071
072        /**
073         * Check if the user have the permission to use this Egg
074         * 
075         * @param player
076         * @return
077         * @throws DontHaveThePermissionException
078         *             when the player don't have the permission, with the message
079         *             to display to the user
080         */
081        protected boolean checkPermission(final Player player)
082                        throws DontHaveThePermissionException {
083                final Permission perm = EggPermissionManager.INSTANCE
084                                .getPermission(this);
085                if (!PermissionManager.hasPerm(player, perm, false)) {
086                        throw new DontHaveThePermissionException(Utils.I18n("errorNotPerm",
087                                        "p", perm.getName()));
088                }
089                return true;
090        }
091
092        public T getValue() {
093                return value;
094        }
095
096        public void setValue(final T value) {
097                this.value = value;
098        }
099
100        /**
101         * Will create the Egg that can be assigned later to the player. This method
102         * check if the player has the permission to use that egg and if the
103         * arguments of the command are right.
104         * 
105         * @param player
106         *            That execute the EggSpawn command.
107         * @param args
108         *            Provided by the EggSpawn command.
109         * @return an EggType of the chosen type if every test is passed.
110         * @throws ProcessingArgsException
111         *             if there is a problem while processing the arguments, like a
112         *             Parameter missing, or some other error.
113         * @throws DontHaveThePermissionException
114         *             if the player don't have the permission to use that egg
115         */
116        public static EggType<?> createEggType(final Player player,
117                        final CommandArgs args) throws ProcessingArgsException,
118                        DontHaveThePermissionException, NullPointerException {
119                if (!args.hasFlag('E')) {
120                        throw new ParameterMissingException('E',
121                                        LocaleHelper.EGG_PARAM.getLocale());
122                }
123                EggType<?> eggType;
124                final String className = args.getValueFlag('E');
125                try {
126                        eggType = matchEggClass(className);
127                } catch (final ClassNotFoundException e) {
128                        throw new ProcessingArgsException(ExceptionType.NO_CLASS,
129                                        className, e);
130                } catch (final InstantiationException e) {
131                        throw new ProcessingArgsException(ExceptionType.INSTANCE,
132                                        className, e);
133                } catch (final IllegalAccessException e) {
134                        throw new ProcessingArgsException(ExceptionType.ILLEGAL_ACCESS,
135                                        className, e);
136                } catch (final NullPointerException e) {
137                        throw new NullPointerException();
138                }
139                eggType.checkPermission(player);
140                eggType.processArguments(player, args);
141                return eggType;
142        }
143
144        @SuppressWarnings("unchecked")
145        private static EggType<?> matchEggClass(final String name)
146                        throws ClassNotFoundException, InstantiationException,
147                        IllegalAccessException, NullPointerException {
148                final Class<? extends EggType<?>> c = (Class<? extends EggType<?>>) eggTypeLoader
149                                .loadClass(name);
150                return c.newInstance();
151        }
152
153        /*
154         * (non-Javadoc)
155         * 
156         * @see java.lang.Object#toString()
157         */
158        @Override
159        public String toString() {
160                return getClass().getSimpleName() + " = " + getValue();
161        }
162
163        /*
164         * (non-Javadoc)
165         * 
166         * @see java.lang.Object#hashCode()
167         */
168        @Override
169        public int hashCode() {
170                final int prime = 31;
171                int result = 1;
172                result = prime * result + ((value == null) ? 0 : value.hashCode());
173                return result;
174        }
175
176        /*
177         * (non-Javadoc)
178         * 
179         * @see java.lang.Object#equals(java.lang.Object)
180         */
181        @SuppressWarnings("rawtypes")
182        @Override
183        public boolean equals(final Object obj) {
184                if (this == obj) {
185                        return true;
186                }
187                if (obj == null) {
188                        return false;
189                }
190                if (!(obj instanceof EggType)) {
191                        return false;
192                }
193                final EggType other = (EggType) obj;
194                if (value == null) {
195                        if (other.value != null) {
196                                return false;
197                        }
198                } else if (!value.equals(other.value)) {
199                        return false;
200                }
201                return true;
202        }
203
204}
