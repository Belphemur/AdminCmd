042public class EggTypeClassLoader extends ClassLoader {
043        private static final Map<String, Class<? extends EggType<?>>> classes = new HashMap<String, Class<? extends EggType<?>>>();
044        private static final Map<String, Class<? extends EggType<?>>> classesSimpleName = new HashMap<String, Class<? extends EggType<?>>>();
045        private static final Pattern regex = Pattern.compile("\\w.+Egg");
046
047        /**
048         * Add a Package that containing the EggType class. The Class must have a
049         * name that finish by Egg.
050         * 
051         * @param packageName
052         */
053        @SuppressWarnings("unchecked")
054        public synchronized static void addPackage(final Plugin plugin,
055                        final String packageName) {
056
057                for (final Class<?> clazz : getClassesInPackage(packageName, plugin
058                                .getClass().getClassLoader())) {
059                        Permission perm = null;
060                        if (EggType.class.isAssignableFrom(clazz)) {
061                                if (clazz.isAnnotationPresent(EggPermission.class)) {
062                                        final EggPermission annotation = clazz
063                                                        .getAnnotation(EggPermission.class);
064                                        if (!annotation.permission().isEmpty()) {
065                                                perm = PermissionLinker.addOnTheFly(
066                                                                annotation.permission(),
067                                                                annotation.permissionParent());
068                                        }
069                                } else {
070                                        final String simpleName = clazz.getSimpleName();
071                                        perm = PermissionLinker.addOnTheFly("admincmd.egg."
072                                                        + simpleName.substring(0, simpleName.length() - 3)
073                                                                        .toLowerCase(), "admincmd.egg.*");
074                                }
075                                classes.put(clazz.getName(),
076                                                (Class<? extends EggType<?>>) clazz);
077                                classesSimpleName.put(clazz.getSimpleName(),
078                                                (Class<? extends EggType<?>>) clazz);
079                                EggPermissionManager.INSTANCE.addPermission(
080                                                (Class<? extends EggType<?>>) clazz, perm);
081                        }
082                }
083        }
084
085        private Class<? extends EggType<?>> matchClassName(final String search)
086                        throws ClassNotFoundException {
087                Class<? extends EggType<?>> found = null;
088                final String lowerSearch = search.toLowerCase();
089                int delta = Integer.MAX_VALUE;
090                for (final Entry<String, Class<? extends EggType<?>>> entry : classesSimpleName
091                                .entrySet()) {
092                        final String str = entry.getKey();
093                        if (str.toLowerCase().startsWith(lowerSearch)) {
094                                final int curDelta = str.length() - lowerSearch.length();
095                                if (curDelta < delta) {
096                                        found = entry.getValue();
097                                        delta = curDelta;
098                                }
099                                if (curDelta == 0) {
100                                        break;
101                                }
102                        }
103                }
104                if (found == null) {
105                        throw new ClassNotFoundException("Can't find the class " + search);
106                }
107                return found;
108        }
109
110        /*
111         * (non-Javadoc)
112         * 
113         * @see java.lang.ClassLoader#findClass(java.lang.String)
114         */
115        @Override
116        protected Class<? extends EggType<?>> findClass(final String name)
117                        throws ClassNotFoundException {
118                Class<? extends EggType<?>> clazz = classes.get(name);
119                if (clazz == null) {
120                        clazz = classesSimpleName.get(name);
121                }
122                if (clazz == null) {
123                        clazz = matchClassName(name);
124                }
125                if (clazz == null) {
126                        throw new ClassNotFoundException();
127                }
128                return clazz;
129        }

131        /**
132         * /** Scans all classes accessible from the context class loader which
133         * belong to the given package and subpackages. Adapted from
134         * http://snippets.dzone.com/posts/show/4831 and extended to support use of
135         * JAR files
136         * 
137         * @param packageName
138         *            The base package
139         * 
140         * @param classLoader
141         *            the parent class loader.
142         * @return
143         */
144        private static List<Class<?>> getClassesInPackage(final String packageName,
145                        final ClassLoader classLoader) {
146                try {
147                        assert classLoader != null;
148                        final String path = packageName.replace('.', '/');
149                        final Enumeration<URL> resources = classLoader.getResources(path);
150                        final List<String> dirs = new ArrayList<String>();
151                        while (resources.hasMoreElements()) {
152                                final URL resource = resources.nextElement();
153                                dirs.add(resource.getFile());
154                        }
155                        final SortedSet<String> classes = new TreeSet<String>();
156                        for (final String directory : dirs) {
157                                classes.addAll(findClasses(directory, packageName));
158                        }
159                        final List<Class<?>> classList = new ArrayList<Class<?>>();
160                        for (final String clazz : classes) {
161                                classList.add(Class.forName(clazz, true, classLoader));
162                        }
163                        return classList;
164                } catch (final Exception e) {
165                        e.printStackTrace();
166                        return new ArrayList<Class<?>>();
167                }
168        }
169
170        /**
171         * Recursive method used to find all classes in a given path (directory or
172         * zip file url). Directories are searched recursively. (zip files are
173         * Adapted from http://snippets.dzone.com/posts/show/4831 and extended to
174         * support use of JAR files
175         * 
176         * @param path
177         *            The base directory or url from which to search.
178         * @param packageName
179         *            The package name for classes found inside the base directory
180         * @param regex
181         *            an optional class name pattern. e.g. .*Test
182         * @return The classes
183         */
184        private static SortedSet<String> findClasses(final String path,
185                        final String packageName) throws Exception {
186                final TreeSet<String> classes = new TreeSet<String>();
187                if (path.startsWith("file:") && path.contains("!")) {
188                        final String[] split = path.split("!");
189                        final URL jar = new URL(split[0]);
190                        final ZipInputStream zip = new ZipInputStream(jar.openStream());
191                        ZipEntry entry;
192                        while ((entry = zip.getNextEntry()) != null) {
193                                if (entry.getName().endsWith(".class")) {
194                                        final String className = entry.getName()
195                                                        .replaceAll("[$].*", "").replaceAll("[.]class", "")
196                                                        .replace('/', '.');
197                                        if (className.startsWith(packageName)
198                                                        && (regex == null || regex.matcher(className)
199                                                                        .matches())) {
200                                                classes.add(className);
201                                        }
202                                }
203                        }
204                }
205                final File dir = new File(path);
206                if (!dir.exists()) {
207                        return classes;
208                }
209                final File[] files = dir.listFiles();
210                for (final File file : files) {
211                        if (file.isDirectory()) {
212                                assert !file.getName().contains(".");
213                                classes.addAll(findClasses(file.getAbsolutePath(), packageName
214                                                + "." + file.getName()));
215                        } else if (file.getName().endsWith(".class")) {
216                                final String className = packageName
217                                                + '.'
218                                                + file.getName().substring(0,
219                                                                file.getName().length() - 6);
220                                if (regex == null || regex.matcher(className).matches()) {
221                                        classes.add(className);
222                                }
223                        }
224                }
225                return classes;
226        }
