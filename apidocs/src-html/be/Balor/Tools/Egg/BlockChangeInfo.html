017    package be.Balor.Tools.Egg;
018    
019    import java.io.Serializable;
020    
021    import org.bukkit.Material;
022    
023    /**
024     * @author Balor (aka Antoine Aflalo)
025     * 
026     */
027    public class BlockChangeInfo implements Serializable {
028    
029            private static final long serialVersionUID = 5071733512565352011L;
030            private int blockTypeId;
031            private int radius;
032    
033            /**
034             * @param blockTypeId
035             * @param radius
036             */
037            public BlockChangeInfo(final int blockTypeId, final int radius) {
038                    this.blockTypeId = blockTypeId;
039                    this.radius = radius;
040            }
041    
042            /**
043     * 
044     */
045            public BlockChangeInfo() {
046            }
047    
048            /**
049             * @return the blockTypeId
050             */
051            public int getBlockTypeId() {
052                    return blockTypeId;
053            }
054    
055            /**
056             * @return the radius
057             */
058            public int getRadius() {
059                    return radius;
060            }
061    
062            /**
063             * @param blockTypeId
064             *            the blockTypeId to set
065             */
066            public void setBlockTypeId(final int blockTypeId) {
067                    this.blockTypeId = blockTypeId;
068            }
069    
070            /**
071             * @param radius
072             *            the radius to set
073             */
074            public void setRadius(final int radius) {
075                    this.radius = radius;
076            }
077    
078            /*
079             * (non-Javadoc)
080             * 
081             * @see java.lang.Object#toString()
082             */
083            @Override
084            public String toString() {
085                    return "BlockChangeInfo [b=" + Material.getMaterial(blockTypeId) + ", r=" + radius + "]";
086            }
087    
088    }
