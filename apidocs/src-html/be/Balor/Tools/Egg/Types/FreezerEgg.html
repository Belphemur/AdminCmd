017    package be.Balor.Tools.Egg.Types;
018    
019    import org.bukkit.Material;
020    import org.bukkit.entity.Player;
021    
022    import be.Balor.Manager.Commands.CommandArgs;
023    import be.Balor.Tools.Egg.BlockChangeInfo;
024    import be.Balor.Tools.Egg.Exceptions.ProcessingArgsException;
025    import be.Balor.bukkit.AdminCmd.ConfigEnum;
026    
027    /**
028     * @author Balor (aka Antoine Aflalo)
029     * 
030     */
031    public class FreezerEgg extends BlockEgg {
032    
033            /**
034             * 
035             */
036            private static final long serialVersionUID = 6781269132940660439L;
037    
038            /**
039             * 
040             */
041            public FreezerEgg() {
042                    super(ConfigEnum.DEGG_FREEZE_RADIUS.getInt(), ConfigEnum.MAXEGG_FREEZE_RADIUS.getInt());
043            }
044    
045            /*
046             * (non-Javadoc)
047             * 
048             * @see
049             * be.Balor.Tools.Egg.EggType#processArguments(org.bukkit.entity.Player,
050             * be.Balor.Manager.Commands.CommandArgs)
051             */
052            @Override
053            protected void processArguments(final Player sender, final CommandArgs args)
054                            throws ProcessingArgsException {
055                    final int radius = getRadius(sender, args);
056                    if (radius == -1)
057                            return;
058                    value = new BlockChangeInfo(Material.ICE.getId(), radius);
059    
060            }
061    
062            /*
063             * (non-Javadoc)
064             * 
065             * @see be.Balor.Tools.Egg.Types.BlockEgg#blockTimeOut()
066             */
067            @Override
068            protected int blockTimeOut() {
069                    return ConfigEnum.EGG_FREEZE_TIMEOUT.getInt();
070            }
071    
072            /*
073             * (non-Javadoc)
074             * 
075             * @see java.lang.Object#toString()
076             */
077            @Override
078            public String toString() {
079                    return "FreezerEgg = " + value.getRadius();
080            }
081    
082    }
