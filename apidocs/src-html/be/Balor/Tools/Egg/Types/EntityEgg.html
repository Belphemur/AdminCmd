017    package be.Balor.Tools.Egg.Types;
018    
019    import net.minecraft.server.EntityTypes;
020    
021    import org.bukkit.Location;
022    import org.bukkit.World;
023    import org.bukkit.craftbukkit.CraftWorld;
024    import org.bukkit.entity.Entity;
025    import org.bukkit.entity.Player;
026    import org.bukkit.event.player.PlayerEggThrowEvent;
027    
028    import be.Balor.Manager.Commands.CommandArgs;
029    import be.Balor.Tools.Utils;
030    import be.Balor.Tools.Egg.EggType;
031    import be.Balor.Tools.Egg.EntityInEgg;
032    import be.Balor.Tools.Egg.Exceptions.ExceptionType;
033    import be.Balor.Tools.Egg.Exceptions.ParameterMissingException;
034    import be.Balor.Tools.Egg.Exceptions.ProcessingArgsException;
035    import be.Balor.bukkit.AdminCmd.LocaleHelper;
036    
037    /**
038     * @author Balor (aka Antoine Aflalo)
039     * 
040     */
041    public class EntityEgg extends EggType<EntityInEgg> {
042    
043            /**
044             * 
045             */
046            private static final long serialVersionUID = 805828153137593363L;
047    
048            /*
049             * (non-Javadoc)
050             * 
051             * @see be.Balor.Tools.Egg.EggType#onEvent(org.bukkit.event.player.
052             * PlayerEggThrowEvent)
053             */
054            @SuppressWarnings("unchecked")
055            @Override
056            public void onEvent(final PlayerEggThrowEvent event) {
057                    event.getEgg().remove();
058                    event.setHatching(false);
059                    final World w = event.getEgg().getWorld();
060                    final Location loc = event.getEgg().getLocation();
061    
062                    for (int i = 0; i < value.getNb(); i++)
063                            w.spawn(loc, value.getEntityClass());
064            }
065    
066            /*
067             * (non-Javadoc)
068             * 
069             * @see
070             * be.Balor.Tools.Egg.EggType#processArguments(org.bukkit.entity.Player,
071             * be.Balor.Manager.Commands.CommandArgs)
072             */
073            @Override
074            protected void processArguments(final Player sender, final CommandArgs args)
075                            throws ProcessingArgsException {
076                    final String entityParam = args.getValueFlag('e');
077                    if (entityParam == null)
078                            throw new ParameterMissingException('e', LocaleHelper.EGG_PARAM_ENTITY.getLocale());
079                    final String valFlag = args.getValueFlag('n');
080                    int nbre = 1;
081                    int entityNb = 93;
082                    try {
083                            entityNb = Integer.parseInt(entityParam);
084                    } catch (final NumberFormatException e) {
085                            Utils.sI18n(sender, "NaN", "number", valFlag);
086                            return;
087                    }
088                    if (args.hasFlag('n'))
089                            try {
090                                    nbre = Integer.parseInt(valFlag);
091                                    entityNb = Integer.parseInt(entityParam);
092                            } catch (final NumberFormatException e) {
093                                    Utils.sI18n(sender, "NaN", "number", valFlag);
094                                    return;
095                            }
096                    try {
097                            final Entity entity = EntityTypes.a(entityNb,
098                                            ((CraftWorld) sender.getWorld()).getHandle()).getBukkitEntity();
099                            value = new EntityInEgg(entity.getClass().getName(), nbre, entity.getClass()
100                                            .getSimpleName());
101                    } catch (final NullPointerException e) {
102                            throw new ProcessingArgsException(ExceptionType.DONT_EXISTS, String.valueOf(entityNb));
103                    }
104    
105            }
106    
107    }
