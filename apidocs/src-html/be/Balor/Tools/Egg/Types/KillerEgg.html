017    package be.Balor.Tools.Egg.Types;
018    
019    import java.util.ArrayList;
020    import java.util.List;
021    
022    import net.minecraft.server.DamageSource;
023    import net.minecraft.server.EntityLiving;
024    import net.minecraft.server.EntityPlayer;
025    
026    import org.bukkit.Location;
027    import org.bukkit.World;
028    import org.bukkit.craftbukkit.CraftWorld;
029    import org.bukkit.craftbukkit.entity.CraftPlayer;
030    import org.bukkit.entity.Player;
031    import org.bukkit.event.player.PlayerEggThrowEvent;
032    
033    import be.Balor.Tools.Egg.SimpleRadiusEgg;
034    import be.Balor.bukkit.AdminCmd.ACPluginManager;
035    import be.Balor.bukkit.AdminCmd.ConfigEnum;
036    
037    /**
038     * @author Balor (aka Antoine Aflalo)
039     * 
040     */
041    public class KillerEgg extends SimpleRadiusEgg {
042    
043            /**
044             * 
045             */
046            private static final long serialVersionUID = -7763897329319981939L;
047    
048            /**
049             * 
050             */
051            public KillerEgg() {
052                    super(ConfigEnum.DEGG_KILL_RADIUS.getInt(), ConfigEnum.MAXEGG_KILL_RADIUS.getInt());
053            }
054    
055            /*
056             * (non-Javadoc)
057             * 
058             * @see be.Balor.Tools.Egg.EggType#onEvent(org.bukkit.event.player.
059             * PlayerEggThrowEvent)
060             */
061            @Override
062            public void onEvent(final PlayerEggThrowEvent event) {
063                    event.setHatching(false);
064                    final Location loc = event.getEgg().getLocation();
065                    event.getEgg().remove();
066                    final List<EntityLiving> entities = new ArrayList<EntityLiving>();
067                    final CraftPlayer p = (CraftPlayer) event.getPlayer();
068                    final World w = p.getWorld();
069                    final int radius = value * value;
070                    for (final Object entity : ((CraftWorld) w).getHandle().entityList) {
071                            if (entity instanceof EntityLiving)
072                                    entities.add((EntityLiving) entity);
073                    }
074                    ACPluginManager.getScheduler().scheduleAsyncDelayedTask(ACPluginManager.getCorePlugin(),
075                                    new Runnable() {
076    
077                                            @Override
078                                            public void run() {
079                                                    int count = 0;
080                                                    for (final EntityLiving entity : entities) {
081                                                            if (entity.equals(p.getHandle())) {
082                                                                    continue;
083                                                            }
084                                                            final Location entityLoc = new Location(w, entity.locX, entity.locY,
085                                                                            entity.locZ, entity.yaw, entity.pitch);
086                                                            if (entityLoc.distanceSquared(loc) > radius)
087                                                                    continue;
088                                                            if (entity instanceof EntityPlayer) {
089                                                                    final Player player = (Player) entity.getBukkitEntity();
090                                                                    player.setHealth(0);
091                                                                    count++;
092                                                                    continue;
093                                                            }
094                                                            entity.die(DamageSource.playerAttack(p.getHandle()));
095                                                            entity.die();
096                                                            count++;
097                                                    }
098                                                    p.sendMessage(String.valueOf(count) + " killed.");
099                                            }
100                                    });
101            }
102    
103    }
