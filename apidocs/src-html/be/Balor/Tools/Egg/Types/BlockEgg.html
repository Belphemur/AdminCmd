047    /**
048     * @author Balor (aka Antoine Aflalo)
049     * 
050     */
051    public class BlockEgg extends RadiusEgg<BlockChangeInfo> {
052    
053            /**
054             * 
055             */
056            private static final long serialVersionUID = 1046757929821301999L;
057            private int eggNb = 0;
058            private final Map<Integer, SynchronizedStack<BlockRemanence>> blocksPerEvent = Collections
059                            .synchronizedMap(new HashMap<Integer, SynchronizedStack<BlockRemanence>>());
060    
061            /**
062             * 
063             */
064            public BlockEgg() {
065                    super(ConfigEnum.DEGG_BLOCK_RADIUS.getInt(), ConfigEnum.MAXEGG_BLOCK_RADIUS.getInt());
066            }
067    
068            /**
069             * @param defaultRadius
070             * @param maxRadius
071             */
072            public BlockEgg(final int defaultRadius, final int maxRadius) {
073                    super(defaultRadius, maxRadius);
074            }
075    
076            /*
077             * (non-Javadoc)
078             * 
079             * @see be.Balor.Tools.Egg.EggType#onEvent(org.bukkit.event.player.
080             * PlayerEggThrowEvent)
081             */
082            @Override
083            public void onEvent(final PlayerEggThrowEvent event) {
084                    final int radius = value.getRadius() / 2;
085                    event.getEgg().remove();
086                    event.setHatching(false);
087                    final Location loc = event.getEgg().getLocation();
088                    final SynchronizedStack<BlockRemanence> blocks = new SynchronizedStack<BlockRemanence>();
089                    final World w = loc.getWorld();
090                    if (blockTimeOut() == 0) {
091                            for (int x = loc.getBlockX() - radius; x < loc.getBlockX() + radius; x++)
092                                    for (int z = loc.getBlockZ() - radius; z < loc.getBlockZ() + radius; z++)
093                                            for (int y = loc.getBlockY() - radius; y < loc.getBlockY() + radius; y++) {
094                                                    final Block block = w.getBlockAt(x, y, z);
095                                                    if (block.getTypeId() != Material.AIR.getId()
096                                                                    && block.getTypeId() != Material.SNOW.getId())
097                                                            continue;
098                                                    IBlockRemanenceFactory.FACTORY.createBlockRemanence(
099                                                                    new SimplifiedLocation(w, x, y, z)).setBlockType(
100                                                                    value.getBlockTypeId());
101                                            }
102                            return;
103                    }
104                    final Integer eventId = eggNb++;
105                    for (int x = loc.getBlockX() - radius; x < loc.getBlockX() + radius; x++)
106                            for (int z = loc.getBlockZ() - radius; z < loc.getBlockZ() + radius; z++)
107                                    for (int y = loc.getBlockY() - radius; y < loc.getBlockY() + radius; y++) {
108                                            final int blckId = w.getBlockTypeIdAt(x, y, z);
109                                            if (blckId != Material.AIR.getId() && blckId != Material.SNOW.getId())
110                                                    continue;
111                                            final BlockRemanence blk = IBlockRemanenceFactory.FACTORY
112                                                            .createBlockRemanence(new SimplifiedLocation(w, x, y, z));
113                                            blk.setBlockType(value.getBlockTypeId());
114                                            blocks.add(blk);
115    
116                                    }
117                    blocksPerEvent.put(eventId, blocks);
118                    ACPluginManager.getScheduler().scheduleSyncDelayedTask(ACPluginManager.getCorePlugin(),
119                                    new Runnable() {
120    
121                                            @Override
122                                            public void run() {
123                                                    final SynchronizedStack<BlockRemanence> blocks = blocksPerEvent
124                                                                    .get(eventId);
125                                                    while (!blocks.empty())
126                                                            blocks.pop().returnToThePast();
127                                                    blocksPerEvent.remove(eventId);
128    
129                                            }
130                                    }, blockTimeOut() * Utils.secInTick);
131    
132            }
133    
134            protected int blockTimeOut() {
135                    return ConfigEnum.EGG_BLOCK_TIMEOUT.getInt();
136            }
137    
138            /*
139             * (non-Javadoc)
140             * 
141             * @see
142             * be.Balor.Tools.Egg.EggType#processArguments(org.bukkit.entity.Player,
143             * be.Balor.Manager.Commands.CommandArgs)
144             */
145            @Override
146            protected void processArguments(final Player sender, final CommandArgs args)
147                            throws ProcessingArgsException {
148                    final String block = args.getValueFlag('b');
149                    if (block == null)
150                            throw new ParameterMissingException('b', LocaleHelper.EGG_PARAM_BLOCK.getLocale());
151                    final int radius = getRadius(sender, args);
152                    if (radius == -1)
153                            return;
154                    final MaterialContainer mat = ACHelper.getInstance().checkMaterial(sender, block);
155                    if (mat.isNull()) {
156                            final HashMap<String, String> replace = new HashMap<String, String>();
157                            replace.put("type", LocaleHelper.TYPE_MAT.getLocale());
158                            replace.put("value", block);
159                            throw new ProcessingArgsException(ExceptionType.CUSTOM,
160                                            LocaleHelper.DONT_EXISTS.getLocale(replace));
161                    }
162                    value = new BlockChangeInfo(mat.getMaterial().getId(), radius);
163    
164            }
165    
166    }
