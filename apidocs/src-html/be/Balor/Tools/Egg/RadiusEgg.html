017    package be.Balor.Tools.Egg;
018    
019    import org.bukkit.entity.Player;
020    
021    import be.Balor.Manager.Commands.CommandArgs;
022    import be.Balor.Tools.Utils;
023    
024    /**
025     * @author Balor (aka Antoine Aflalo)
026     * 
027     */
028    public abstract class RadiusEgg<T> extends EggType<T> {
029    
030            /**
031             * 
032             */
033            private static final long serialVersionUID = -8375423945536370931L;
034            private final int defaultRadius;
035            private final int maxRadius;
036    
037            /**
038             * @param defaultRadius
039             * @param maxRadius
040             */
041            public RadiusEgg(final int defaultRadius, final int maxRadius) {
042                    super();
043                    this.defaultRadius = defaultRadius;
044                    this.maxRadius = maxRadius;
045            }
046    
047            protected int getRadius(final Player sender, final CommandArgs args) {
048                    int radius = defaultRadius;
049                    final String valFlag = args.getValueFlag('r');
050                    if (valFlag != null)
051                            try {
052                                    radius = Integer.parseInt(valFlag);
053                            } catch (final NumberFormatException e) {
054                                    Utils.sI18n(sender, "NaN", "number", valFlag);
055                                    return -1;
056                            }
057                    return radius > maxRadius ? maxRadius : radius;
058            }
059    }
