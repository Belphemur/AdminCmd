017package be.Balor.Tools.Configuration;
018
019import java.util.List;
020
021import org.bukkit.configuration.ConfigurationSection;
022
023/**
024 * @author Balor (aka Antoine Aflalo)
025 * 
026 */
027public interface ExConfigurationSection extends ConfigurationSection {
028        /**
029         * Create an entry in the {@link ExConfigurationSection} if it not existing
030         * else return the existing one.
031         * 
032         * @param path
033         *            Path to create/get the entry at.
034         * @return false if the path already exists, true if it's not set.
035         */
036        public boolean add(String path, Object value);
037
038        /**
039         * Create a {@link ExConfigurationSection} if it not existing else return
040         * the existing one.
041         * 
042         * @param path
043         *            Path to create/get the section at.
044         * @return the ConfigurationSection
045         */
046        public ExConfigurationSection addSection(String path);
047
048        @Override
049        public ExConfigurationSection createSection(String path);
050
051        /**
052         * Gets a list of booleans. Non-valid entries will not be in the list. There
053         * will be no null slots. If the list is not defined, the default will be
054         * returned. 'null' can be passed for the default and an empty list will be
055         * returned instead. The node must be an actual list and cannot be just a
056         * boolean,
057         * 
058         * @param path
059         *            path to node (dot notation)
060         * @param def
061         *            default value or null for an empty list as default
062         * @return list of integers
063         */
064        public List<Boolean> getBooleanList(String path, List<Boolean> def);
065
066        @Override
067        public ExConfigurationSection getConfigurationSection(String path);
068
069        /**
070         * Gets a list of doubles. Non-valid entries will not be in the list. There
071         * will be no null slots. If the list is not defined, the default will be
072         * returned. 'null' can be passed for the default and an empty list will be
073         * returned instead. The node must be an actual list and cannot be just a
074         * double.
075         * 
076         * @param path
077         *            path to node (dot notation)
078         * @param def
079         *            default value or null for an empty list as default
080         * @return list of integers
081         */
082        public List<Double> getDoubleList(String path, List<Double> def);
083
084        /**
085         * Gets a list of integers. Non-valid entries will not be in the list. There
086         * will be no null slots. If the list is not defined, the default will be
087         * returned. 'null' can be passed for the default and an empty list will be
088         * returned instead. The node must be an actual list and not just an
089         * integer.
090         * 
091         * @param path
092         *            path to node (dot notation)
093         * @param def
094         *            default value or null for an empty list as default
095         * @return list of integers
096         */
097        public List<Integer> getIntList(String path, List<Integer> def);
098
099        /**
100         * Gets a list of strings. Non-valid entries will not be in the list. There
101         * will be no null slots. If the list is not defined, the default will be
102         * returned. 'null' can be passed for the default and an empty list will be
103         * returned instead. If an item in the list is not a string, it will be
104         * converted to a string. The node must be an actual list and not just a
105         * string.
106         * 
107         * @param path
108         *            path to node (dot notation)
109         * @param def
110         *            default value or null for an empty list as default
111         * @return list of strings
112         */
113        public List<String> getStringList(String path, List<String> def);
114
115        /**
116         * Shortcut to remove an item by setting it null
117         * 
118         * @param path
119         *            Path to remove the entry at.
120         */
121        public void remove(String path);
122
123}
