017package be.Balor.Tools.Configuration;
018
019import java.util.Map;
020
021import org.bukkit.configuration.Configuration;
022import org.bukkit.configuration.ConfigurationSection;
023import org.bukkit.configuration.MemoryConfiguration;
024
025/**
026 * @author Balor (aka Antoine Aflalo)
027 * 
028 */
029public class ExMemoryConfiguration extends ExMemorySection
030                implements
031                        Configuration {
032        protected Configuration defaults;
033        protected ExMemoryConfigurationOption options;
034
035        /**
036         * Creates an empty {@link MemoryConfiguration} with no default values.
037         */
038        public ExMemoryConfiguration() {
039        }
040
041        /**
042         * Creates an empty {@link MemoryConfiguration} using the specified
043         * {@link Configuration} as a source for all default values.
044         * 
045         * @param defaults
046         *            Default value provider
047         * @throws IllegalArgumentException
048         *             Thrown if defaults is null
049         */
050        public ExMemoryConfiguration(final Configuration defaults) {
051                this.defaults = defaults;
052        }
053
054        @Override
055        public void addDefault(final String path, final Object value) {
056                if (path == null) {
057                        throw new IllegalArgumentException("Path may not be null");
058                }
059
060                if (defaults == null) {
061                        defaults = new ExMemoryConfiguration();
062                }
063
064                defaults.set(path, value);
065        }
066
067        @Override
068        public void addDefaults(final Configuration defaults) {
069                if (defaults == null) {
070                        throw new IllegalArgumentException("Defaults may not be null");
071                }
072
073                addDefaults(defaults.getValues(true));
074        }
075
076        @Override
077        public void addDefaults(final Map<String, Object> defaults) {
078                if (defaults == null) {
079                        throw new IllegalArgumentException("Defaults may not be null");
080                }
081
082                for (final Map.Entry<String, Object> entry : defaults.entrySet()) {
083                        addDefault(entry.getKey(), entry.getValue());
084                }
085        }
086
087        @Override
088        public Configuration getDefaults() {
089                return defaults;
090        }
091
092        @Override
093        public ConfigurationSection getParent() {
094                return null;
095        }
096
097        @Override
098        public ExMemoryConfigurationOption options() {
099                if (options == null) {
100                        options = new ExMemoryConfigurationOption(this);
101                }
102
103                return options;
104        }
105
106        @Override
107        public void setDefaults(final Configuration defaults) {
108                if (defaults == null) {
109                        throw new IllegalArgumentException("Defaults may not be null");
110                }
111
112                this.defaults = defaults;
113        }
114}
