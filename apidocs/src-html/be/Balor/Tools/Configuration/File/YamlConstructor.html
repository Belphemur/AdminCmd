017package be.Balor.Tools.Configuration.File;
018
019import java.util.LinkedHashMap;
020import java.util.Map;
021
022import org.bukkit.configuration.serialization.ConfigurationSerialization;
023import org.yaml.snakeyaml.constructor.AbstractConstruct;
024import org.yaml.snakeyaml.constructor.Constructor;
025import org.yaml.snakeyaml.error.YAMLException;
026import org.yaml.snakeyaml.nodes.Node;
027import org.yaml.snakeyaml.nodes.ScalarNode;
028import org.yaml.snakeyaml.nodes.Tag;
029
030import be.Balor.Tools.TpRequest;
031import be.Balor.bukkit.AdminCmd.ACPluginManager;
032
033/**
034 * @author Balor (aka Antoine Aflalo)
035 * 
036 */
037public class YamlConstructor extends Constructor {
038        private class ConstructCustomObject extends ConstructYamlMap {
039                @SuppressWarnings("unchecked")
040                @Override
041                public Object construct(final Node node) {
042                        if (node.isTwoStepsConstruction()) {
043                                throw new YAMLException(
044                                                "Unexpected referential mapping structure. Node: "
045                                                                + node);
046                        }
047
048                        final Map<Object, Object> raw = (Map<Object, Object>) super
049                                        .construct(node);
050
051                        if (raw.containsKey(ConfigurationSerialization.SERIALIZED_TYPE_KEY)) {
052                                final Map<String, Object> typed = new LinkedHashMap<String, Object>(
053                                                raw.size());
054                                for (final Map.Entry<Object, Object> entry : raw.entrySet()) {
055                                        typed.put(entry.getKey().toString(), entry.getValue());
056                                }
057
058                                try {
059                                        return ConfigurationSerialization.deserializeObject(typed);
060                                } catch (final IllegalArgumentException ex) {
061                                        throw new YAMLException("Could not deserialize object", ex);
062                                }
063                        }
064
065                        return raw;
066                }
067
068                @Override
069                public void construct2ndStep(final Node node, final Object object) {
070                        throw new YAMLException(
071                                        "Unexpected referential mapping structure. Node: " + node);
072                }
073        }
074
075        private class ConstructTpRequest extends AbstractConstruct {
076                @Override
077                public Object construct(final Node node) {
078                        final String val = (String) constructScalar((ScalarNode) node);
079                        final String[] split = val.split(";");
080                        return new TpRequest(ACPluginManager.getServer()
081                                        .getPlayer(split[0]), ACPluginManager.getServer()
082                                        .getPlayer(split[1]));
083                }
084        }
085
086        private final ClassLoader pluginClassLoader;
087
088        /**
089         * 
090         */
091        public YamlConstructor(final ClassLoader classLoader) {
092                super();
093                this.yamlConstructors.put(new Tag("!tpRequest"),
094                                new ConstructTpRequest());
095                this.yamlConstructors.put(Tag.MAP, new ConstructCustomObject());
096                this.pluginClassLoader = classLoader;
097        }
098
099        /*
100         * This is a modified version of the Constructor. Rather than using a class
101         * loader to get external classes, they are already predefined above. This
102         * approach works similar to the typeTags structure in the original
103         * constructor, except that class information is pre-populated during
104         * initialization rather than runtime.
105         * 
106         * @see org.yaml.snakeyaml.constructor.Constructor#getClassForName(org.yaml
107         * .snakeyaml.nodes.Node)
108         */
109        @Override
110        protected Class<?> getClassForName(final String name)
111                        throws ClassNotFoundException {
112                return Class.forName(name, true, pluginClassLoader);
113        }
114}
