001    package be.Balor.Tools.Configuration.File;
002    
003    import java.io.BufferedReader;
004    import java.io.File;
005    import java.io.FileInputStream;
006    import java.io.FileNotFoundException;
007    import java.io.FileOutputStream;
008    import java.io.IOException;
009    import java.io.InputStream;
010    import java.io.InputStreamReader;
011    import java.io.OutputStreamWriter;
012    import java.io.Writer;
013    
014    import org.bukkit.configuration.Configuration;
015    import org.bukkit.configuration.InvalidConfigurationException;
016    
017    import be.Balor.Tools.Configuration.ExMemoryConfiguration;
018    import be.Balor.Tools.Debug.ACLogger;
019    
020    import com.google.common.io.Files;
021    
022    /**
023     * This is a base class for all File based implementations of
024     * {@link Configuration}
025     */
026    public abstract class ExFileConfiguration extends ExMemoryConfiguration {
027            protected File file;
028            protected boolean corrupted = false;
029    
030            /**
031             * Creates an empty {@link ExFileConfiguration} with no default values.
032             */
033            public ExFileConfiguration() {
034                    super();
035            }
036    
037            /**
038             * Creates an empty {@link ExFileConfiguration} using the specified
039             * {@link Configuration} as a source for all default values.
040             * 
041             * @param defaults
042             *            Default value provider
043             */
044            public ExFileConfiguration(final Configuration defaults) {
045                    super(defaults);
046            }
047    
048            /**
049             * Compiles the header for this {@link ExFileConfiguration} and returns the
050             * result.
051             * <p>
052             * This will use the header from {@link #options()} ->
053             * {@link ExFileConfigurationOptions#header()}, respecting the rules of
054             * {@link ExFileConfigurationOptions#copyHeader()} if set.
055             * 
056             * @return Compiled header
057             */
058            protected abstract String buildHeader();
059    
060            /*
061             * (non-Javadoc)
062             * 
063             * @see java.lang.Object#equals(java.lang.Object)
064             */
065            @Override
066            public boolean equals(final Object obj) {
067                    if (this == obj)
068                            return true;
069                    if (obj == null)
070                            return false;
071                    if (!(obj instanceof ExFileConfiguration))
072                            return false;
073                    final ExFileConfiguration other = (ExFileConfiguration) obj;
074                    if (file == null) {
075                            if (other.file != null)
076                                    return false;
077                    } else if (!file.equals(other.file))
078                            return false;
079                    return true;
080            }
081    
082            /*
083             * (non-Javadoc)
084             * 
085             * @see java.lang.Object#hashCode()
086             */
087            @Override
088            public int hashCode() {
089                    final int prime = 31;
090                    int result = 1;
091                    result = prime * result + ((file == null) ? 0 : file.hashCode());
092                    return result;
093            }
094    
095            /**
096             * Loads this {@link ExFileConfiguration} from the specified location.
097             * <p>
098             * All the values contained within this configuration will be removed,
099             * leaving only settings and defaults, and the new values will be loaded
100             * from the given file.
101             * <p>
102             * If the file cannot be loaded for any reason, an exception will be thrown.
103             * 
104             * @param file
105             *            File to load from.
106             * @throws FileNotFoundException
107             *             Thrown when the given file cannot be opened.
108             * @throws IOException
109             *             Thrown when the given file cannot be read.
110             * @throws InvalidConfigurationException
111             *             Thrown when the given file is not a valid Configuration.
112             * @throws IllegalArgumentException
113             *             Thrown when file is null.
114             */
115            public void load(final File file) throws FileNotFoundException, IOException,
116                            InvalidConfigurationException {
117                    if (file == null) {
118                            throw new IllegalArgumentException("File cannot be null");
119                    }
120                    this.file = file;
121                    try {
122                            load(new FileInputStream(file));
123                    } catch (final IllegalArgumentException e) {
124                            ACLogger.severe("Problem with File : " + this.file);
125                            ACLogger.severe(e.getLocalizedMessage(), e);
126                    }
127    
128            }
129    
130            /**
131             * Loads this {@link ExFileConfiguration} from the specified stream.
132             * <p>
133             * All the values contained within this configuration will be removed,
134             * leaving only settings and defaults, and the new values will be loaded
135             * from the given stream.
136             * 
137             * @param stream
138             *            Stream to load from
139             * @throws IOException
140             *             Thrown when the given file cannot be read.
141             * @throws InvalidConfigurationException
142             *             Thrown when the given file is not a valid Configuration.
143             * @throws IllegalArgumentException
144             *             Thrown when stream is null.
145             */
146            public void load(final InputStream stream) throws IOException, InvalidConfigurationException {
147                    if (stream == null) {
148                            throw new IllegalArgumentException("Stream cannot be null");
149                    }
150    
151                    final InputStreamReader reader = new InputStreamReader(stream, "UTF8");
152                    final StringBuilder builder = new StringBuilder();
153                    final BufferedReader input = new BufferedReader(reader);
154    
155                    try {
156                            String line;
157    
158                            while ((line = input.readLine()) != null) {
159                                    builder.append(line);
160                                    builder.append('\n');
161                            }
162                    } finally {
163                            input.close();
164                    }
165    
166                    loadFromString(builder.toString());
167            }
168    
169            /**
170             * Loads this {@link ExFileConfiguration} from the specified location.
171             * <p>
172             * All the values contained within this configuration will be removed,
173             * leaving only settings and defaults, and the new values will be loaded
174             * from the given file.
175             * <p>
176             * If the file cannot be loaded for any reason, an exception will be thrown.
177             * 
178             * @param file
179             *            File to load from.
180             * @throws FileNotFoundException
181             *             Thrown when the given file cannot be opened.
182             * @throws IOException
183             *             Thrown when the given file cannot be read.
184             * @throws InvalidConfigurationException
185             *             Thrown when the given file is not a valid Configuration.
186             * @throws IllegalArgumentException
187             *             Thrown when file is null.
188             */
189            public void load(final String file) throws FileNotFoundException, IOException,
190                            InvalidConfigurationException {
191                    if (file == null) {
192                            throw new IllegalArgumentException("File cannot be null");
193                    }
194    
195                    load(new File(file));
196            }
197    
198            /**
199             * Loads this {@link ExFileConfiguration} from the specified string, as
200             * opposed to from file.
201             * <p>
202             * All the values contained within this configuration will be removed,
203             * leaving only settings and defaults, and the new values will be loaded
204             * from the given string.
205             * <p>
206             * If the string is invalid in any way, an exception will be thrown.
207             * 
208             * @param contents
209             *            Contents of a Configuration to load.
210             * @throws InvalidConfigurationException
211             *             Thrown if the specified string is invalid.
212             * @throws IllegalArgumentException
213             *             Thrown if contents is null.
214             */
215            public abstract void loadFromString(String contents) throws InvalidConfigurationException;
216    
217            @Override
218            public ExFileConfigurationOptions options() {
219                    if (options == null) {
220                            options = new ExFileConfigurationOptions(this);
221                    }
222    
223                    return (ExFileConfigurationOptions) options;
224            }
225    
226            /**
227             * Saves this {@link ExFileConfiguration} to the specified location.
228             * <p>
229             * If the file does not exist, it will be created. If already exists, it
230             * will be overwritten. If it cannot be overwritten or created, an exception
231             * will be thrown.
232             * 
233             * @param file
234             *            File to save to.
235             * @throws IOException
236             *             Thrown when the given file cannot be written to for any
237             *             reason.
238             * @throws IllegalArgumentException
239             *             Thrown when file is null.
240             */
241            public void save(final File file) throws IOException {
242                    if (corrupted)
243                            return;
244                    if (file == null) {
245                            throw new IllegalArgumentException("File cannot be null");
246                    }
247                    if (file.getParentFile() != null && !file.getParentFile().exists())
248                            Files.createParentDirs(file);
249                    if (!file.exists())
250                            file.createNewFile();
251    
252                    final String data = saveToString();
253    
254                    final Writer writer = new OutputStreamWriter(new FileOutputStream(file), "UTF-8");
255    
256                    try {
257                            writer.write(data);
258                    } finally {
259                            writer.close();
260                    }
261            }
262    
263            /**
264             * Saves this {@link ExFileConfiguration} to the specified location.
265             * <p>
266             * If the file does not exist, it will be created. If already exists, it
267             * will be overwritten. If it cannot be overwritten or created, an exception
268             * will be thrown.
269             * 
270             * @param file
271             *            File to save to.
272             * @throws IOException
273             *             Thrown when the given file cannot be written to for any
274             *             reason.
275             * @throws IllegalArgumentException
276             *             Thrown when file is null.
277             */
278            public void save(final String file) throws IOException {
279                    if (file == null) {
280                            throw new IllegalArgumentException("File cannot be null");
281                    }
282    
283                    save(new File(file));
284            }
285    
286            /**
287             * Saves this {@link ExFileConfiguration} to a string, and returns it.
288             * 
289             * @return String containing this configuration.
290             */
291            public abstract String saveToString();
292    
293            /*
294             * (non-Javadoc)
295             * 
296             * @see java.lang.Object#toString()
297             */
298            @Override
299            public String toString() {
300                    return "ExFileConfiguration [file=" + file + "]";
301            }
302    }
