001package be.Balor.Tools.Configuration.File;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileInputStream;
006import java.io.FileNotFoundException;
007import java.io.IOException;
008import java.io.InputStream;
009
010import org.bukkit.configuration.Configuration;
011import org.bukkit.configuration.InvalidConfigurationException;
012
013import be.Balor.Tools.Configuration.ExMemoryConfiguration;
014import be.Balor.Tools.Debug.ACLogger;
015import be.Balor.Tools.Files.Unicode.UnicodeReader;
016import be.Balor.Tools.Files.Unicode.UnicodeUtil;
017
018import com.google.common.io.Files;
019
020/**
021 * This is a base class for all File based implementations of
022 * {@link Configuration}
023 */
024public abstract class ExFileConfiguration extends ExMemoryConfiguration {
025        protected File file;
026        protected boolean corrupted = false;
027
028        /**
029         * Creates an empty {@link ExFileConfiguration} with no default values.
030         */
031        public ExFileConfiguration() {
032                super();
033        }
034
035        /**
036         * Creates an empty {@link ExFileConfiguration} using the specified
037         * {@link Configuration} as a source for all default values.
038         * 
039         * @param defaults
040         *            Default value provider
041         */
042        public ExFileConfiguration(final Configuration defaults) {
043                super(defaults);
044        }
045
046        /**
047         * Compiles the header for this {@link ExFileConfiguration} and returns the
048         * result.
049         * <p>
050         * This will use the header from {@link #options()} ->
051         * {@link ExFileConfigurationOptions#header()}, respecting the rules of
052         * {@link ExFileConfigurationOptions#copyHeader()} if set.
053         * 
054         * @return Compiled header
055         */
056        protected abstract String buildHeader();
057
058        /*
059         * (non-Javadoc)
060         * 
061         * @see java.lang.Object#equals(java.lang.Object)
062         */
063        @Override
064        public boolean equals(final Object obj) {
065                if (this == obj) {
066                        return true;
067                }
068                if (obj == null) {
069                        return false;
070                }
071                if (!(obj instanceof ExFileConfiguration)) {
072                        return false;
073                }
074                final ExFileConfiguration other = (ExFileConfiguration) obj;
075                if (file == null) {
076                        if (other.file != null) {
077                                return false;
078                        }
079                } else if (!file.equals(other.file)) {
080                        return false;
081                }
082                return true;
083        }
084
085        /*
086         * (non-Javadoc)
087         * 
088         * @see java.lang.Object#hashCode()
089         */
090        @Override
091        public int hashCode() {
092                final int prime = 31;
093                int result = 1;
094                result = prime * result + ((file == null) ? 0 : file.hashCode());
095                return result;
096        }
097
098        /**
099         * Loads this {@link ExFileConfiguration} from the specified location.
100         * <p>
101         * All the values contained within this configuration will be removed,
102         * leaving only settings and defaults, and the new values will be loaded
103         * from the given file.
104         * <p>
105         * If the file cannot be loaded for any reason, an exception will be thrown.
106         * 
107         * @param file
108         *            File to load from.
109         * @throws FileNotFoundException
110         *             Thrown when the given file cannot be opened.
111         * @throws IOException
112         *             Thrown when the given file cannot be read.
113         * @throws InvalidConfigurationException
114         *             Thrown when the given file is not a valid Configuration.
115         * @throws IllegalArgumentException
116         *             Thrown when file is null.
117         */
118        public void load(final File file) throws FileNotFoundException,
119                        IOException, InvalidConfigurationException {
120                if (file == null) {
121                        throw new IllegalArgumentException("File cannot be null");
122                }
123                this.file = file;
124                try {
125                        load(new FileInputStream(this.file));
126                } catch (final IllegalArgumentException e) {
127                        ACLogger.severe("Problem with File : " + this.file);
128                        ACLogger.severe(e.getLocalizedMessage(), e);
129                }
130
131        }
132
133        /**
134         * Loads this {@link ExFileConfiguration} from the specified stream.
135         * <p>
136         * All the values contained within this configuration will be removed,
137         * leaving only settings and defaults, and the new values will be loaded
138         * from the given stream.
139         * 
140         * @param stream
141         *            Stream to load from
142         * @throws IOException
143         *             Thrown when the given file cannot be read.
144         * @throws InvalidConfigurationException
145         *             Thrown when the given file is not a valid Configuration.
146         * @throws IllegalArgumentException
147         *             Thrown when stream is null.
148         */
149        public void load(final InputStream stream) throws IOException,
150                        InvalidConfigurationException {
151                if (stream == null) {
152                        throw new IllegalArgumentException("Stream cannot be null");
153                }
154                final StringBuilder builder = new StringBuilder();
155                final BufferedReader input = new BufferedReader(new UnicodeReader(
156                                stream, "UTF-8"));
157
158                try {
159                        String line;
160
161                        while ((line = input.readLine()) != null) {
162                                builder.append(line);
163                                builder.append('\n');
164                        }
165                } finally {
166                        input.close();
167                }
168
169                loadFromString(builder.toString());
170        }
171
172        /**
173         * Loads this {@link ExFileConfiguration} from the specified location.
174         * <p>
175         * All the values contained within this configuration will be removed,
176         * leaving only settings and defaults, and the new values will be loaded
177         * from the given file.
178         * <p>
179         * If the file cannot be loaded for any reason, an exception will be thrown.
180         * 
181         * @param file
182         *            File to load from.
183         * @throws FileNotFoundException
184         *             Thrown when the given file cannot be opened.
185         * @throws IOException
186         *             Thrown when the given file cannot be read.
187         * @throws InvalidConfigurationException
188         *             Thrown when the given file is not a valid Configuration.
189         * @throws IllegalArgumentException
190         *             Thrown when file is null.
191         */
192        public void load(final String file) throws FileNotFoundException,
193                        IOException, InvalidConfigurationException {
194                if (file == null) {
195                        throw new IllegalArgumentException("File cannot be null");
196                }
197
198                load(new File(file));
199        }
200
201        /**
202         * Loads this {@link ExFileConfiguration} from the specified string, as
203         * opposed to from file.
204         * <p>
205         * All the values contained within this configuration will be removed,
206         * leaving only settings and defaults, and the new values will be loaded
207         * from the given string.
208         * <p>
209         * If the string is invalid in any way, an exception will be thrown.
210         * 
211         * @param contents
212         *            Contents of a Configuration to load.
213         * @throws InvalidConfigurationException
214         *             Thrown if the specified string is invalid.
215         * @throws IllegalArgumentException
216         *             Thrown if contents is null.
217         */
218        public abstract void loadFromString(String contents)
219                        throws InvalidConfigurationException;
220
221        @Override
222        public ExFileConfigurationOptions options() {
223                if (options == null) {
224                        options = new ExFileConfigurationOptions(this);
225                }
226
227                return (ExFileConfigurationOptions) options;
228        }
229
230        /**
231         * Saves this {@link ExFileConfiguration} to the specified location.
232         * <p>
233         * If the file does not exist, it will be created. If already exists, it
234         * will be overwritten. If it cannot be overwritten or created, an exception
235         * will be thrown.
236         * 
237         * @param file
238         *            File to save to.
239         * @throws IOException
240         *             Thrown when the given file cannot be written to for any
241         *             reason.
242         * @throws IllegalArgumentException
243         *             Thrown when file is null.
244         */
245        public void save(final File file) throws IOException {
246                if (corrupted) {
247                        return;
248                }
249                if (file == null) {
250                        throw new IllegalArgumentException("File cannot be null");
251                }
252                if (file.getParentFile() != null && !file.getParentFile().exists()) {
253                        Files.createParentDirs(file);
254                }
255                if (!file.exists()) {
256                        file.createNewFile();
257                }
258
259                UnicodeUtil.saveUTF8File(file, saveToString(), false);
260        }
261
262        /**
263         * Saves this {@link ExFileConfiguration} to the specified location.
264         * <p>
265         * If the file does not exist, it will be created. If already exists, it
266         * will be overwritten. If it cannot be overwritten or created, an exception
267         * will be thrown.
268         * 
269         * @param file
270         *            File to save to.
271         * @throws IOException
272         *             Thrown when the given file cannot be written to for any
273         *             reason.
274         * @throws IllegalArgumentException
275         *             Thrown when file is null.
276         */
277        public void save(final String file) throws IOException {
278                if (file == null) {
279                        throw new IllegalArgumentException("File cannot be null");
280                }
281
282                save(new File(file));
283        }
284
285        /**
286         * Saves this {@link ExFileConfiguration} to a string, and returns it.
287         * 
288         * @return String containing this configuration.
289         */
290        public abstract String saveToString();
291
292        /*
293         * (non-Javadoc)
294         * 
295         * @see java.lang.Object#toString()
296         */
297        @Override
298        public String toString() {
299                return "ExFileConfiguration [file=" + file + "]";
300        }
301}
