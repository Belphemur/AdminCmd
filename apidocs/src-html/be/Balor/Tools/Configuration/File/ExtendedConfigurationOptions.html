001    package be.Balor.Tools.Configuration.File;
002    
003    import org.bukkit.configuration.file.YamlConfiguration;
004    
005    /**
006     * Various settings for controlling the input and output of a
007     * {@link YamlConfiguration}
008     */
009    public class ExtendedConfigurationOptions extends ExFileConfigurationOptions {
010            private int indent = 2;
011    
012            protected ExtendedConfigurationOptions(final ExtendedConfiguration configuration) {
013                    super(configuration);
014            }
015    
016            @Override
017            public ExtendedConfiguration configuration() {
018                    return (ExtendedConfiguration) super.configuration();
019            }
020    
021            @Override
022            public ExtendedConfigurationOptions copyDefaults(final boolean value) {
023                    super.copyDefaults(value);
024                    return this;
025            }
026    
027            @Override
028            public ExtendedConfigurationOptions copyHeader(final boolean value) {
029                    super.copyHeader(value);
030                    return this;
031            }
032    
033            @Override
034            public ExtendedConfigurationOptions header(final String value) {
035                    super.header(value);
036                    return this;
037            }
038    
039            /**
040             * Gets how much spaces should be used to indent each line.
041             * <p>
042             * The minimum value this may be is 2, and the maximum is 9.
043             * 
044             * @return How much to indent by
045             */
046            public int indent() {
047                    return indent;
048            }
049    
050            /**
051             * Sets how much spaces should be used to indent each line.
052             * <p>
053             * The minimum value this may be is 2, and the maximum is 9.
054             * 
055             * @param value
056             *            New indent
057             * @return This object, for chaining
058             */
059            public ExtendedConfigurationOptions indent(final int value) {
060                    if ((indent < 2) || (value > 9)) {
061                            throw new IllegalArgumentException("Indent must be between 1 and 10 characters");
062                    }
063    
064                    this.indent = value;
065                    return this;
066            }
067    
068            @Override
069            public ExtendedConfigurationOptions pathSeparator(final char value) {
070                    super.pathSeparator(value);
071                    return this;
072            }
073    }
