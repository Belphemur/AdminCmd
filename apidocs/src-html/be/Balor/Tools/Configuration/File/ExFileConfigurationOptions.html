001    package be.Balor.Tools.Configuration.File;
002    
003    import be.Balor.Tools.Configuration.ExMemoryConfiguration;
004    import be.Balor.Tools.Configuration.ExMemoryConfigurationOption;
005    
006    /**
007     * Various settings for controlling the input and output of a
008     * {@link ExFileConfiguration}
009     */
010    public class ExFileConfigurationOptions extends ExMemoryConfigurationOption {
011            private String header = null;
012            private boolean copyHeader = true;
013    
014            protected ExFileConfigurationOptions(final ExMemoryConfiguration configuration) {
015                    super(configuration);
016            }
017    
018            @Override
019            public ExFileConfiguration configuration() {
020                    return (ExFileConfiguration) super.configuration();
021            }
022    
023            @Override
024            public ExFileConfigurationOptions copyDefaults(final boolean value) {
025                    super.copyDefaults(value);
026                    return this;
027            }
028    
029            /**
030             * Gets whether or not the header should be copied from a default source.
031             * <p>
032             * If this is true, if a default {@link ExFileConfiguration} is passed to
033             * {@link ExFileConfiguration#setDefaults(org.bukkit.configuration.Configuration)}
034             * then upon saving it will use the header from that config, instead of the
035             * one provided here.
036             * <p>
037             * If no default is set on the configuration, or the default is not of type
038             * FileConfiguration, or that config has no header ({@link #header()}
039             * returns null) then the header specified in this configuration will be
040             * used.
041             * <p>
042             * Defaults to true.
043             * 
044             * @return Whether or not to copy the header
045             */
046            public boolean copyHeader() {
047                    return copyHeader;
048            }
049    
050            /**
051             * Sets whether or not the header should be copied from a default source.
052             * <p>
053             * If this is true, if a default {@link ExFileConfiguration} is passed to
054             * {@link ExFileConfiguration#setDefaults(org.bukkit.configuration.Configuration)}
055             * then upon saving it will use the header from that config, instead of the
056             * one provided here.
057             * <p>
058             * If no default is set on the configuration, or the default is not of type
059             * FileConfiguration, or that config has no header ({@link #header()}
060             * returns null) then the header specified in this configuration will be
061             * used.
062             * <p>
063             * Defaults to true.
064             * 
065             * @param value
066             *            Whether or not to copy the header
067             * @return This object, for chaining
068             */
069            public ExFileConfigurationOptions copyHeader(final boolean value) {
070                    copyHeader = value;
071    
072                    return this;
073            }
074    
075            /**
076             * Gets the header that will be applied to the top of the saved output.
077             * <p>
078             * This header will be commented out and applied directly at the top of the
079             * generated output of the {@link ExFileConfiguration}. It is not required
080             * to include a newline at the end of the header as it will automatically be
081             * applied, but you may include one if you wish for extra spacing.
082             * <p>
083             * Null is a valid value which will indicate that no header is to be
084             * applied. The default value is null.
085             * 
086             * @return Header
087             */
088            public String header() {
089                    return header;
090            }
091    
092            /**
093             * Sets the header that will be applied to the top of the saved output.
094             * <p>
095             * This header will be commented out and applied directly at the top of the
096             * generated output of the {@link ExFileConfiguration}. It is not required
097             * to include a newline at the end of the header as it will automatically be
098             * applied, but you may include one if you wish for extra spacing.
099             * <p>
100             * Null is a valid value which will indicate that no header is to be
101             * applied. The default value is null.
102             * 
103             * @param value
104             *            New header
105             * @return This object, for chaining
106             */
107            public ExFileConfigurationOptions header(final String value) {
108                    this.header = value;
109                    return this;
110            }
111    
112            @Override
113            public ExFileConfigurationOptions pathSeparator(final char value) {
114                    super.pathSeparator(value);
115                    return this;
116            }
117    }
