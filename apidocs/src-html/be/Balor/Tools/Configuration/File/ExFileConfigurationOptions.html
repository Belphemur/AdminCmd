001package be.Balor.Tools.Configuration.File;
002
003import be.Balor.Tools.Configuration.ExMemoryConfiguration;
004import be.Balor.Tools.Configuration.ExMemoryConfigurationOption;
005
006/**
007 * Various settings for controlling the input and output of a
008 * {@link ExFileConfiguration}
009 */
010public class ExFileConfigurationOptions extends ExMemoryConfigurationOption {
011        private String header = null;
012        private boolean copyHeader = true;
013
014        protected ExFileConfigurationOptions(
015                        final ExMemoryConfiguration configuration) {
016                super(configuration);
017        }
018
019        @Override
020        public ExFileConfiguration configuration() {
021                return (ExFileConfiguration) super.configuration();
022        }
023
024        @Override
025        public ExFileConfigurationOptions copyDefaults(final boolean value) {
026                super.copyDefaults(value);
027                return this;
028        }
029
030        /**
031         * Gets whether or not the header should be copied from a default source.
032         * <p>
033         * If this is true, if a default {@link ExFileConfiguration} is passed to
034         * {@link ExFileConfiguration#setDefaults(org.bukkit.configuration.Configuration)}
035         * then upon saving it will use the header from that config, instead of the
036         * one provided here.
037         * <p>
038         * If no default is set on the configuration, or the default is not of type
039         * FileConfiguration, or that config has no header ({@link #header()}
040         * returns null) then the header specified in this configuration will be
041         * used.
042         * <p>
043         * Defaults to true.
044         * 
045         * @return Whether or not to copy the header
046         */
047        public boolean copyHeader() {
048                return copyHeader;
049        }
050
051        /**
052         * Sets whether or not the header should be copied from a default source.
053         * <p>
054         * If this is true, if a default {@link ExFileConfiguration} is passed to
055         * {@link ExFileConfiguration#setDefaults(org.bukkit.configuration.Configuration)}
056         * then upon saving it will use the header from that config, instead of the
057         * one provided here.
058         * <p>
059         * If no default is set on the configuration, or the default is not of type
060         * FileConfiguration, or that config has no header ({@link #header()}
061         * returns null) then the header specified in this configuration will be
062         * used.
063         * <p>
064         * Defaults to true.
065         * 
066         * @param value
067         *            Whether or not to copy the header
068         * @return This object, for chaining
069         */
070        public ExFileConfigurationOptions copyHeader(final boolean value) {
071                copyHeader = value;
072
073                return this;
074        }
075
076        /**
077         * Gets the header that will be applied to the top of the saved output.
078         * <p>
079         * This header will be commented out and applied directly at the top of the
080         * generated output of the {@link ExFileConfiguration}. It is not required
081         * to include a newline at the end of the header as it will automatically be
082         * applied, but you may include one if you wish for extra spacing.
083         * <p>
084         * Null is a valid value which will indicate that no header is to be
085         * applied. The default value is null.
086         * 
087         * @return Header
088         */
089        public String header() {
090                return header;
091        }
092
093        /**
094         * Sets the header that will be applied to the top of the saved output.
095         * <p>
096         * This header will be commented out and applied directly at the top of the
097         * generated output of the {@link ExFileConfiguration}. It is not required
098         * to include a newline at the end of the header as it will automatically be
099         * applied, but you may include one if you wish for extra spacing.
100         * <p>
101         * Null is a valid value which will indicate that no header is to be
102         * applied. The default value is null.
103         * 
104         * @param value
105         *            New header
106         * @return This object, for chaining
107         */
108        public ExFileConfigurationOptions header(final String value) {
109                this.header = value;
110                return this;
111        }
112
113        @Override
114        public ExFileConfigurationOptions pathSeparator(final char value) {
115                super.pathSeparator(value);
116                return this;
117        }
118}
