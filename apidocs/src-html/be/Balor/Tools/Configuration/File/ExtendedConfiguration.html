054        protected static final String BLANK_CONFIG = "{}\n";

059        /**
060         * Creates a new {@link ExtendedConfiguration}, loading from the given file.
061         * <p>
062         * Any errors loading the Configuration will be logged and then ignored. If
063         * the specified input is not a valid config, a blank config will be
064         * returned.
065         * 
066         * @param file
067         *            Input file
068         * @return Resulting configuration
069         * @throws IllegalArgumentException
070         *             Thrown is file is null
071         */
072        public static ExtendedConfiguration loadConfiguration(final File file) {
073                if (file == null) {
074                        throw new IllegalArgumentException("File cannot be null");
075                }
076
077                final ExtendedConfiguration config = new ExtendedConfiguration();
078
079                try {
080                        config.load(file);
081                } catch (final FileNotFoundException ex) {} catch (final IOException ex) {
082                        Bukkit.getLogger().log(Level.SEVERE, "Cannot load " + file, ex);
083                } catch (final InvalidConfigurationException ex) {
084                        if (ex.getCause() instanceof YAMLException) {
085                                ACLogger.severe("Config file " + file + " isn't valid! "
086                                                + ex.getCause());
087                        } else if ((ex.getCause() == null)
088                                        || (ex.getCause() instanceof ClassCastException)) {
089                                ACLogger.severe("Config file " + file + " isn't valid!");
090                        } else {
091                                ACLogger.severe("Cannot load " + file + ": "
092                                                + ex.getCause().getClass(), ex);
093                        }
094                }
095
096                return config;
097        }

099        public static void setClassLoader(final ClassLoader loader) {
100                yaml = new Yaml(new YamlConstructor(loader), yamlRepresenter,
101                                yamlOptions);
102        }

104        @Override
105        protected String buildHeader() {
106                final String header = options().header();
107
108                if (options().copyHeader()) {
109                        final Configuration def = getDefaults();
110
111                        if ((def != null) && (def instanceof ExFileConfiguration)) {
112                                final ExFileConfiguration filedefaults = (ExFileConfiguration) def;
113                                final String defaultsHeader = filedefaults.buildHeader();
114
115                                if ((defaultsHeader != null) && (defaultsHeader.length() > 0)) {
116                                        return defaultsHeader;
117                                }
118                        }
119                }
120
121                if (header == null) {
122                        return "";
123                }
124
125                final StringBuilder builder = new StringBuilder();
126                final String[] lines = header.split("\r?\n", -1);
127                boolean startedHeader = false;
128
129                for (int i = lines.length - 1; i >= 0; i--) {
130                        builder.insert(0, "\n");
131
132                        if ((startedHeader) || (lines[i].length() != 0)) {
133                                builder.insert(0, lines[i]);
134                                builder.insert(0, COMMENT_PREFIX);
135                                startedHeader = true;
136                        }
137                }
138
139                return builder.toString();
140        }
141
142        protected void convertMapsToSections(final Map<Object, Object> input,
143                        final ConfigurationSection section) {
144                for (final Map.Entry<Object, Object> entry : input.entrySet()) {
145                        final String key = entry.getKey().toString();
146                        final Object value = entry.getValue();
147
148                        if (value instanceof Map<?, ?>) {
149                                convertMapsToSections((Map<Object, Object>) value,
150                                                section.createSection(key));
151                        } else {
152                                section.set(key, value);
153                        }
154                }
155        }

157        /*
158         * (non-Javadoc)
159         * 
160         * @see
161         * org.bukkit.configuration.file.FileConfiguration#loadFromString(java.lang
162         * .String)
163         */
164        @Override
165        public void loadFromString(String contents)
166                        throws InvalidConfigurationException {
167                lock.lock();
168                try {
169
170                        if (contents == null) {
171                                throw new IllegalArgumentException("Contents cannot be null");
172                        }
173
174                        Map<Object, Object> input = null;
175                        contents = contents.replaceAll("\uFFFD", "?").replaceAll("\t",
176                                        "    ");
177                        try {
178                                input = (Map<Object, Object>) yaml.load(contents);
179                        } catch (final ScannerException e) {
180                                if (e.getContextMark() == null) {
181                                        ACLogger.severe(
182                                                        "File : "
183                                                                        + file
184                                                                        + "\n You have to correct the error manualy in the file.",
185                                                        e);
186                                        corrupted = true;
187                                        return;
188                                }
189                                removeLineFromFile(e.getContextMark().getLine());
190                                ACLogger.info("File : " + file + "\n" + e.toString()
191                                                + "\nLINE " + (e.getContextMark().getLine() + 1)
192                                                + " DELETED");
193                                try {
194                                        load(file);
195                                } catch (final FileNotFoundException e1) {} catch (final IOException e1) {}
196                        } catch (final ParserException e) {
197                                ACLogger.severe(
198                                                "File : "
199                                                                + file
200                                                                + "\n You have to correct the error manualy in the file.",
201                                                e);
202                                corrupted = true;
203                                return;
204
205                        } catch (final Throwable ex) {
206                                corrupted = true;
207                                throw new InvalidConfigurationException(
208                                                "Specified contents is not a valid Configuration", ex);
209                        }
210
211                        final int size = (input == null) ? 0 : input.size();
212                        final Map<String, Object> result = new LinkedHashMap<String, Object>(
213                                        size);
214
215                        if (size > 0) {
216                                for (final Map.Entry<Object, Object> entry : input.entrySet()) {
217                                        result.put(entry.getKey().toString(), entry.getValue());
218                                }
219                        }
220
221                        final String header = parseHeader(contents);
222                        if (header.length() > 0) {
223                                options().header(header);
224                        }
225
226                        if (input != null) {
227                                convertMapsToSections(input, this);
228                        }
229                        corrupted = false;
230                } finally {
231                        lock.unlock();
232                }
233
234        }

236        @Override
237        public ExtendedConfigurationOptions options() {
238                if (options == null) {
239                        options = new ExtendedConfigurationOptions(this);
240                }
241
242                return (ExtendedConfigurationOptions) options;
243        }

245        protected String parseHeader(final String input) {
246                final String[] lines = input.split("\r?\n", -1);
247                final StringBuilder result = new StringBuilder();
248                boolean readingHeader = true;
249
250                for (int i = 0; (i < lines.length) && (readingHeader); i++) {
251                        final String line = lines[i];
252
253                        if (line.startsWith(COMMENT_PREFIX)) {
254                                if (i > 0) {
255                                        result.append("\n");
256                                }
257
258                                if (line.length() > COMMENT_PREFIX.length()) {
259                                        result.append(line.substring(COMMENT_PREFIX.length()));
260                                }
261                        } else if (line.length() == 0) {
262                                result.append("\n");
263                        } else {
264                                readingHeader = false;
265                        }
266                }
267
268                return result.toString();
269        }

283        private void removeLineFromFile(final int lineToRemove) {
284                BufferedReader br = null;
285                PrintWriter pw = null;
286                try {
287
288                        final File inFile = file;
289                        // Construct the new file that will later be renamed to the original
290                        // filename.
291                        final File tempFile = File.createTempFile(file.getName(), null);
292                        br = new BufferedReader(
293                                        new UnicodeReader(new FileInputStream(file)));
294                        pw = new PrintWriter(new OutputStreamWriter(new FileOutputStream(
295                                        tempFile), "UTF-8"));
296
297                        String line = null;
298
299                        // Read from the original file and write to the new
300                        // unless content matches data to be removed.
301                        int i = 0;
302                        while ((line = br.readLine()) != null) {
303                                try {
304                                        if (i == lineToRemove) {
305                                                continue;
306                                        }
307                                        pw.println(line);
308                                } finally {
309                                        i++;
310                                }
311                        }
312                        pw.flush();
313                        pw.close();
314                        br.close();
315                        // Delete the original file
316                        if (!inFile.delete()) {
317                                System.out.println("Could not delete file");
318                                return;
319                        }
320
321                        // Rename the new file to the filename the original file had.
322                        if (!tempFile.renameTo(inFile)) {
323                                System.out.println("Could not rename file");
324                        }
325
326                } catch (final FileNotFoundException ex) {
327                        ex.printStackTrace();
328                } catch (final IOException ex) {
329                        ex.printStackTrace();
330                } finally {
331                        if (pw != null) {
332                                pw.close();
333                        }
334                        try {
335                                if (br != null) {
336                                        br.close();
337                                }
338                        } catch (final IOException e) {}
339                }
340        }

356        /*
357         * (non-Javadoc)
358         * 
359         * @see org.bukkit.configuration.file.FileConfiguration#saveToString()
360         */
361        @Override
362        public String saveToString() {
363                lock.lock();
364                String header = "";
365                String dump = "";
366                try {
367                        yamlOptions.setIndent(options().indent());
368                        yamlOptions.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
369                        yamlRepresenter.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
370
371                        header = buildHeader();
372                        dump = yaml.dump(getValues(false));
373
374                        if (dump.equals(BLANK_CONFIG)) {
375                                dump = "";
376                        }
377                } finally {
378                        lock.unlock();
379                }
380                return header + dump;
381        }
