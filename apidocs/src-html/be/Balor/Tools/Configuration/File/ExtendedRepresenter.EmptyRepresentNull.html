017package be.Balor.Tools.Configuration.File;
018
019import java.util.LinkedHashMap;
020import java.util.Map;
021
022import org.bukkit.configuration.ConfigurationSection;
023import org.bukkit.configuration.serialization.ConfigurationSerializable;
024import org.bukkit.configuration.serialization.ConfigurationSerialization;
025import org.yaml.snakeyaml.introspector.Property;
026import org.yaml.snakeyaml.nodes.CollectionNode;
027import org.yaml.snakeyaml.nodes.MappingNode;
028import org.yaml.snakeyaml.nodes.Node;
029import org.yaml.snakeyaml.nodes.NodeTuple;
030import org.yaml.snakeyaml.nodes.SequenceNode;
031import org.yaml.snakeyaml.nodes.Tag;
032import org.yaml.snakeyaml.representer.Represent;
033import org.yaml.snakeyaml.representer.Representer;
034
035import be.Balor.Tools.TpRequest;
036
037/**
038 * @author Balor (aka Antoine Aflalo)
039 * 
040 */
041public class ExtendedRepresenter extends Representer {
042
043        protected class EmptyRepresentNull implements Represent {
044                @Override
045                public Node representData(final Object data) {
046                        return representScalar(Tag.NULL, ""); // Changed "null" to "" so as
047                                                                                                        // to avoid writing nulls
048                }
049        }
050
051        private class RepresentConfigurationSection extends RepresentMap {
052                @Override
053                public Node representData(final Object data) {
054                        return super.representData(((ConfigurationSection) data)
055                                        .getValues(false));
056                }
057        }
058
059        private class RepresentConfigurationSerializable extends RepresentMap {
060                @Override
061                public Node representData(final Object data) {
062                        ConfigurationSerializable serializable = (ConfigurationSerializable) data;
063                        Map<String, Object> values = new LinkedHashMap<String, Object>();
064                        values.put(
065                                        ConfigurationSerialization.SERIALIZED_TYPE_KEY,
066                                        ConfigurationSerialization.getAlias(serializable.getClass()));
067                        values.putAll(serializable.serialize());
068
069                        return super.representData(values);
070                }
071        }
072
073        private class RepresentTpRequest implements Represent {
074                @Override
075                public Node representData(final Object data) {
076                        final TpRequest tpRequest = (TpRequest) data;
077                        final String value = tpRequest.getFrom() + ";" + tpRequest.getTo();
078                        return representScalar(new Tag("!tpRequest"), value);
079                }
080        }
081
082        public ExtendedRepresenter() {
083                super();
084                this.nullRepresenter = new EmptyRepresentNull();
085                this.representers.put(TpRequest.class, new RepresentTpRequest());
086                this.multiRepresenters.put(ConfigurationSection.class,
087                                new RepresentConfigurationSection());
088                this.multiRepresenters.put(ConfigurationSerializable.class,
089                                new RepresentConfigurationSerializable());
090        }
091
092        // Code borrowed from snakeyaml
093        // (http://code.google.com/p/snakeyaml/source/browse/src/test/java/org/yaml/snakeyaml/issues/issue60/SkipBeanTest.java)
094        @Override
095        protected NodeTuple representJavaBeanProperty(final Object javaBean,
096                        final Property property, final Object propertyValue,
097                        final Tag customTag) {
098                final NodeTuple tuple = super.representJavaBeanProperty(javaBean,
099                                property, propertyValue, customTag);
100                final Node valueNode = tuple.getValueNode();
101                if (valueNode instanceof CollectionNode) {
102                        // Removed null check
103                        if (Tag.SEQ.equals(valueNode.getTag())) {
104                                final SequenceNode seq = (SequenceNode) valueNode;
105                                if (seq.getValue().isEmpty()) {
106                                        return null; // skip empty lists
107                                }
108                        }
109                        if (Tag.MAP.equals(valueNode.getTag())) {
110                                final MappingNode seq = (MappingNode) valueNode;
111                                if (seq.getValue().isEmpty()) {
112                                        return null; // skip empty maps
113                                }
114                        }
115                }
116                return tuple;
117        }
118
119}
