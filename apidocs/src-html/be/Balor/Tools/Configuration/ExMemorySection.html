017    package be.Balor.Tools.Configuration;
018    
019    import java.util.List;
020    import java.util.concurrent.locks.Lock;
021    import java.util.concurrent.locks.ReentrantLock;
022    import java.util.regex.Pattern;
023    
024    import org.bukkit.configuration.ConfigurationSection;
025    import org.bukkit.configuration.MemorySection;
026    
027    /**
028     * @author Balor (aka Antoine Aflalo)
029     * 
030     */
031    public class ExMemorySection extends MemorySection implements ExConfigurationSection {
032            protected final Lock lock = new ReentrantLock(true);
033    
034            /**
035             * 
036             */
037            public ExMemorySection() {
038                    super();
039            }
040    
041            /**
042             * @param exMemorySection
043             * @param key
044             */
045            protected ExMemorySection(final ConfigurationSection exMemorySection, final String key) {
046                    super(exMemorySection, key);
047            }
048    
049            @Override
050            public boolean add(final String path, final Object value) {
051                    if (isSet(path))
052                            return false;
053                    set(path, value);
054                    return true;
055            }
056    
057            @Override
058            public ExConfigurationSection addSection(final String path) {
059                    ExConfigurationSection result = getConfigurationSection(path);
060                    if (result == null) {
061                            result = createSection(path);
062                    }
063                    return result;
064            }
065    
066            private long castToLong(final Object value) throws NumberFormatException {
067                    if (value instanceof Long)
068                            return (Long) value;
069                    else if (value instanceof String)
070                            return Long.parseLong((String) value);
071                    else if (value instanceof Integer)
072                            return Long.valueOf(((Integer) value));
073                    throw new NumberFormatException();
074            }
075    
076            /*
077             * (non-Javadoc)
078             * 
079             * @see
080             * org.bukkit.configuration.MemorySection#createSection(java.lang.String)
081             */
082            @Override
083            public ExConfigurationSection createSection(final String path) {
084                    if (path == null) {
085                            throw new IllegalArgumentException("Path cannot be null");
086                    } else if (path.length() == 0) {
087                            throw new IllegalArgumentException("Cannot create section at empty path");
088                    }
089    
090                    final String[] split = path.split(Pattern.quote(Character.toString(getRoot().options()
091                                    .pathSeparator())));
092                    ExConfigurationSection section = this;
093    
094                    for (int i = 0; i < split.length - 1; i++) {
095                            final ExConfigurationSection last = section;
096    
097                            section = getConfigurationSection(split[i]);
098    
099                            if (section == null) {
100                                    section = last.createSection(split[i]);
101                            }
102                    }
103    
104                    final String key = split[split.length - 1];
105    
106                    if (section == this) {
107                            final ExConfigurationSection result = new ExMemorySection(this, key);
108                            map.put(key, result);
109                            return result;
110                    } else {
111                            return section.createSection(key);
112                    }
113            }
114    
115            /*
116             * (non-Javadoc)
117             * 
118             * @see org.bukkit.configuration.MemorySection#get(java.lang.String,
119             * java.lang.Object)
120             */
121            @Override
122            public Object get(final String path, final Object def) {
123                    lock.lock();
124                    Object info;
125                    try {
126                            info = super.get(path, def);
127                            // info = newGet(path, def);
128    
129                    } finally {
130                            lock.unlock();
131                    }
132                    return info;
133            }
134    
135            /*
136             * (non-Javadoc)
137             * 
138             * @see
139             * be.Balor.Tools.Configuration.ExConfigurationSection#getBooleanList(java
140             * .lang.String, java.util.List)
141             */
142            @Override
143            public List<Boolean> getBooleanList(final String path, final List<Boolean> def) {
144                    final List<Boolean> result = super.getBooleanList(path);
145                    if (result == null || (result != null && result.isEmpty()))
146                            return def;
147                    return result;
148            }
149    
150            @Override
151            public ExConfigurationSection getConfigurationSection(final String path) {
152                    if (path == null) {
153                            throw new IllegalArgumentException("Path cannot be null");
154                    }
155    
156                    final Object val = get(path, getDefault(path));
157                    return (val instanceof ExConfigurationSection) ? (ExConfigurationSection) val : null;
158            }
159    
160            /*
161             * (non-Javadoc)
162             * 
163             * @see
164             * be.Balor.Tools.Configuration.ExConfigurationSection#getDoubleList(java
165             * .lang.String, java.util.List)
166             */
167            @Override
168            public List<Double> getDoubleList(final String path, final List<Double> def) {
169                    final List<Double> result = super.getDoubleList(path);
170                    if (result == null || (result != null && result.isEmpty()))
171                            return def;
172                    return result;
173            }
174    
175            /*
176             * (non-Javadoc)
177             * 
178             * @see
179             * be.Balor.Tools.Configuration.ExConfigurationSection#getIntList(java.lang
180             * .String, java.util.List)
181             */
182            @Override
183            public List<Integer> getIntList(final String path, final List<Integer> def) {
184                    final List<Integer> result = getIntegerList(path);
185                    if (result == null || (result != null && result.isEmpty()))
186                            return def;
187                    return result;
188            }
189    
190            @Override
191            public long getLong(final String path, final long def) {
192                    if (path == null) {
193                            throw new IllegalArgumentException("Path cannot be null");
194                    }
195    
196                    final Object val = get(path, def);
197                    long returnVal;
198                    try {
199                            returnVal = castToLong(val);
200                    } catch (final NumberFormatException e) {
201                            returnVal = def;
202                    }
203                    return returnVal;
204            }
205    
206            @Override
207            public String getString(final String path, final String def) {
208                    if (path == null) {
209                            throw new IllegalArgumentException("Path cannot be null");
210                    }
211                    final Object val = get(path, def);
212                    if (val == null)
213                            return def;
214                    return (val instanceof String) ? (String) val : val.toString();
215            }
216    
217            /*
218             * (non-Javadoc)
219             * 
220             * @see
221             * be.Balor.Tools.Configuration.ExConfigurationSection#getStringList(java
222             * .lang.String, java.util.List)
223             */
224            @Override
225            public List<String> getStringList(final String path, final List<String> def) {
226                    final List<String> result = super.getStringList(path);
227                    if (result == null || (result != null && result.isEmpty()))
228                            return def;
229                    return result;
230            }
231    
232            /**
233             * Shortcut to remove an item by setting it null
234             * 
235             * @param path
236             *            Path to remove the entry at.
237             */
238            @Override
239            public void remove(final String path) {
240                    set(path, null);
241            }
242    
243            /*
244             * (non-Javadoc)
245             * 
246             * @see org.bukkit.configuration.MemorySection#set(java.lang.String,
247             * java.lang.Object)
248             */
249            @Override
250            public void set(final String path, final Object value) {
251                    lock.lock();
252                    try {
253                            super.set(path, value);
254                            // newSet(path, value);
255                    } finally {
256                            lock.unlock();
257                    }
258            }
259
