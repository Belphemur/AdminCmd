017package be.Balor.Tools.Configuration;
018
019import java.util.List;
020import java.util.concurrent.locks.Lock;
021import java.util.concurrent.locks.ReentrantLock;
022import java.util.regex.Pattern;
023
024import org.bukkit.configuration.ConfigurationSection;
025import org.bukkit.configuration.MemorySection;
026
027/**
028 * @author Balor (aka Antoine Aflalo)
029 * 
030 */
031public class ExMemorySection extends MemorySection
032                implements
033                        ExConfigurationSection {
034        protected final Lock lock = new ReentrantLock(true);
035
036        /**
037         * 
038         */
039        public ExMemorySection() {
040                super();
041        }
042
043        /**
044         * @param exMemorySection
045         * @param key
046         */
047        protected ExMemorySection(final ConfigurationSection exMemorySection,
048                        final String key) {
049                super(exMemorySection, key);
050        }
051
052        @Override
053        public boolean add(final String path, final Object value) {
054                if (isSet(path)) {
055                        return false;
056                }
057                set(path, value);
058                return true;
059        }
060
061        @Override
062        public ExConfigurationSection addSection(final String path) {
063                ExConfigurationSection result = getConfigurationSection(path);
064                if (result == null) {
065                        result = createSection(path);
066                }
067                return result;
068        }
069
070        private long castToLong(final Object value) throws NumberFormatException {
071                if (value instanceof Long) {
072                        return (Long) value;
073                } else if (value instanceof String) {
074                        return Long.parseLong((String) value);
075                } else if (value instanceof Integer) {
076                        return Long.valueOf(((Integer) value));
077                }
078                throw new NumberFormatException();
079        }
080
081        /*
082         * (non-Javadoc)
083         * 
084         * @see
085         * org.bukkit.configuration.MemorySection#createSection(java.lang.String)
086         */
087        @Override
088        public ExConfigurationSection createSection(final String path) {
089                if (path == null) {
090                        throw new IllegalArgumentException("Path cannot be null");
091                } else if (path.length() == 0) {
092                        throw new IllegalArgumentException(
093                                        "Cannot create section at empty path");
094                }
095
096                final String[] split = path.split(Pattern.quote(Character
097                                .toString(getRoot().options().pathSeparator())));
098                ExConfigurationSection section = this;
099
100                for (int i = 0; i < split.length - 1; i++) {
101                        final ExConfigurationSection last = section;
102
103                        section = getConfigurationSection(split[i]);
104
105                        if (section == null) {
106                                section = last.createSection(split[i]);
107                        }
108                }
109
110                final String key = split[split.length - 1];
111
112                if (section == this) {
113                        final ExConfigurationSection result = new ExMemorySection(this, key);
114                        map.put(key, result);
115                        return result;
116                } else {
117                        return section.createSection(key);
118                }
119        }
120
121        /*
122         * (non-Javadoc)
123         * 
124         * @see org.bukkit.configuration.MemorySection#get(java.lang.String,
125         * java.lang.Object)
126         */
127        @Override
128        public Object get(final String path, final Object def) {
129                lock.lock();
130                Object info;
131                try {
132                        info = super.get(path, def);
133                        // info = newGet(path, def);
134
135                } finally {
136                        lock.unlock();
137                }
138                return info;
139        }
140
141        /*
142         * (non-Javadoc)
143         * 
144         * @see
145         * be.Balor.Tools.Configuration.ExConfigurationSection#getBooleanList(java
146         * .lang.String, java.util.List)
147         */
148        @Override
149        public List<Boolean> getBooleanList(final String path,
150                        final List<Boolean> def) {
151                final List<Boolean> result = super.getBooleanList(path);
152                if (result == null || (result != null && result.isEmpty())) {
153                        return def;
154                }
155                return result;
156        }
157
158        @Override
159        public ExConfigurationSection getConfigurationSection(final String path) {
160                if (path == null) {
161                        throw new IllegalArgumentException("Path cannot be null");
162                }
163
164                final Object val = get(path, getDefault(path));
165                return (val instanceof ExConfigurationSection)
166                                ? (ExConfigurationSection) val
167                                : null;
168        }
169
170        /*
171         * (non-Javadoc)
172         * 
173         * @see
174         * be.Balor.Tools.Configuration.ExConfigurationSection#getDoubleList(java
175         * .lang.String, java.util.List)
176         */
177        @Override
178        public List<Double> getDoubleList(final String path, final List<Double> def) {
179                final List<Double> result = super.getDoubleList(path);
180                if (result == null || (result != null && result.isEmpty())) {
181                        return def;
182                }
183                return result;
184        }
185
186        /*
187         * (non-Javadoc)
188         * 
189         * @see
190         * be.Balor.Tools.Configuration.ExConfigurationSection#getIntList(java.lang
191         * .String, java.util.List)
192         */
193        @Override
194        public List<Integer> getIntList(final String path, final List<Integer> def) {
195                final List<Integer> result = getIntegerList(path);
196                if (result == null || (result != null && result.isEmpty())) {
197                        return def;
198                }
199                return result;
200        }
201
202        @Override
203        public long getLong(final String path, final long def) {
204                if (path == null) {
205                        throw new IllegalArgumentException("Path cannot be null");
206                }
207
208                final Object val = get(path, def);
209                long returnVal;
210                try {
211                        returnVal = castToLong(val);
212                } catch (final NumberFormatException e) {
213                        returnVal = def;
214                }
215                return returnVal;
216        }
217
218        @Override
219        public String getString(final String path, final String def) {
220                if (path == null) {
221                        throw new IllegalArgumentException("Path cannot be null");
222                }
223                final Object val = get(path, def);
224                if (val == null) {
225                        return def;
226                }
227                return (val instanceof String) ? (String) val : val.toString();
228        }
229
230        /*
231         * (non-Javadoc)
232         * 
233         * @see
234         * be.Balor.Tools.Configuration.ExConfigurationSection#getStringList(java
235         * .lang.String, java.util.List)
236         */
237        @Override
238        public List<String> getStringList(final String path, final List<String> def) {
239                final List<String> result = super.getStringList(path);
240                if (result == null || (result != null && result.isEmpty())) {
241                        return def;
242                }
243                return result;
244        }
245
246        /**
247         * Shortcut to remove an item by setting it null
248         * 
249         * @param path
250         *            Path to remove the entry at.
251         */
252        @Override
253        public void remove(final String path) {
254                set(path, null);
255        }
256
257        /*
258         * (non-Javadoc)
259         * 
260         * @see org.bukkit.configuration.MemorySection#set(java.lang.String,
261         * java.lang.Object)
262         */
263        @Override
264        public void set(final String path, final Object value) {
265                lock.lock();
266                try {
267                        super.set(path, value);
268                        // newSet(path, value);
269                } finally {
270                        lock.unlock();
271                }
272        }
