162                                final int length = end - start;
163                                final int copyLength = Math

411                        return toParse;

416         * Because water and lava are fluid, using another algo to "delete"
417         * 
418         * @param block
419         * @param radius
420         * @return
421         */
422        private static Stack<BlockRemanence> drainFluid(final String playername,
423                        final Block block, final int radius) {
424                final Stack<BlockRemanence> blocks = new Stack<BlockRemanence>();
425                final Stack<SimplifiedLocation> processQueue = new Stack<SimplifiedLocation>();
426                BlockRemanence current = null;
427                final World w = block.getWorld();
428                final Location start = block.getLocation();
429                final HashSet<SimplifiedLocation> visited = new HashSet<SimplifiedLocation>();
430                final Stack<BlockRemanence> blocksCache = new Stack<BlockRemanence>();
431                for (int x = block.getX() - 2; x <= block.getX() + 2; x++) {
432                        for (int z = block.getZ() - 2; z <= block.getZ() + 2; z++) {
433                                for (int y = block.getY() - 2; y <= block.getY() + 2; y++) {
434                                        final SimplifiedLocation newPos = new SimplifiedLocation(w,
435                                                        x, y, z);
436                                        if (isFluid(newPos) && !visited.contains(newPos)) {
437                                                visited.add(newPos);
438                                                processQueue.push(newPos);
439                                                current = IBlockRemanenceFactory.FACTORY
440                                                                .createBlockRemanence(newPos);
441                                                blocks.push(current);
442                                                blocksCache.push(current);
443                                                if (blocksCache.size() == MAX_BLOCKS) {
444                                                        ACPluginManager.getScheduler()
445                                                                        .scheduleSyncDelayedTask(
446                                                                                        ACHelper.getInstance()
447                                                                                                        .getCoreInstance(),
448                                                                                        new ReplaceBlockTask(blocksCache));
449                                                }
450                                        }
451
452                                }
453                        }
454                }
455                while (!processQueue.isEmpty()) {
456                        final SimplifiedLocation loc = processQueue.pop();
457                        for (int y = loc.getBlockY() - 1; y <= loc.getBlockY() + 1; y++) {
458                                for (int x = loc.getBlockX() - 1; x <= loc.getBlockX() + 1; x++) {
459                                        for (int z = loc.getBlockZ() - 1; z <= loc.getBlockZ() + 1; z++) {
460                                                final SimplifiedLocation newPos = new SimplifiedLocation(
461                                                                w, x, y, z);
462                                                if (!visited.contains(newPos) && isFluid(newPos)
463                                                                && start.distance(newPos) < radius) {
464                                                        processQueue.push(newPos);
465                                                        current = IBlockRemanenceFactory.FACTORY
466                                                                        .createBlockRemanence(newPos);
467                                                        blocks.push(current);
468                                                        blocksCache.push(current);
469                                                        if (blocksCache.size() == MAX_BLOCKS) {
470                                                                ACPluginManager.getScheduler()
471                                                                                .scheduleSyncDelayedTask(
472                                                                                                ACHelper.getInstance()
473                                                                                                                .getCoreInstance(),
474                                                                                                new ReplaceBlockTask(
475                                                                                                                blocksCache));
476                                                        }
477                                                        visited.add(newPos);
478                                                }
479                                        }
480
481                                }
482                        }
483                }
484                ACPluginManager.getScheduler().scheduleSyncDelayedTask(
485                                ACHelper.getInstance().getCoreInstance(),
486                                new ReplaceBlockTask(blocksCache));
487                return blocks;
488        }

622                final Home result = new Home();
623                if (toParse != null && toParse.contains(":")) {
624                        try {
625                                final String[] split = toParse.split(":");
626                                result.player = split[0];
627                                if (split[1] == null) {
628                                        final HashMap<String, String> replace = new HashMap<String, String>();
629                                        replace.put("arg", "home name after the ':'");
630                                        replace.put("cmdName", "/home");
631                                        LocaleHelper.MISSING_ARG.sendLocale(sender, replace);
632                                        return null;
633                                }
634                                result.home = split[1];
635                        } catch (final ArrayIndexOutOfBoundsException e) {
636                        }
637                        if (isPlayer(sender, false)) {
638                                final Player p = (Player) sender;
639                                if (!p.getName().equals(result.player)
640                                                && !PermissionManager.hasPerm(p, "admincmd.admin.home")) {
641                                        return null;
642                                }
643                        }
644                        return result;
645                }
646                if (!isPlayer(sender)) {
647                        return null;
648                }
649                final Player p = ((Player) sender);
650                result.player = p.getName();
651                if (toParse != null) {
652                        result.home = toParse;
653                } else {
654                        result.home = p.getWorld().getName();
655                }
656
657                return result;
658        }

683                                                        - lowerName.length();
684                                        if (curDelta < delta) {
685                                                found = player;
686                                                delta = curDelta;
687                                        }
688                                }
689                                if (curDelta == 0) {
690                                        break;
691                                }
692                        }
693                }
694                return found;
