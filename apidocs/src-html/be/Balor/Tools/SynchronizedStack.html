017package be.Balor.Tools;
018
019import java.util.Collection;
020import java.util.Stack;
021import java.util.concurrent.locks.Lock;
022import java.util.concurrent.locks.ReentrantLock;
023
024/**
025 * @author Balor (aka Antoine Aflalo)
026 * 
027 */
028public class SynchronizedStack<E> extends Stack<E> {
029        /**
030         * 
031         */
032        private static final long serialVersionUID = 6173963896174648554L;
033        private final Lock lock = new ReentrantLock(true);
034
035        /*
036         * (non-Javadoc)
037         * 
038         * @see java.util.Stack#pop()
039         */
040        @Override
041        public synchronized E pop() {
042                lock.lock();
043                try {
044                        return super.pop();
045                } finally {
046                        lock.unlock();
047                }
048
049        }
050
051        @Override
052        public E push(final E item) {
053                lock.lock();
054                try {
055                        return super.push(item);
056                } finally {
057                        lock.unlock();
058                }
059        };
060
061        /*
062         * (non-Javadoc)
063         * 
064         * @see java.util.Stack#peek()
065         */
066        @Override
067        public synchronized E peek() {
068                lock.lock();
069                try {
070                        return super.peek();
071                } finally {
072                        lock.unlock();
073                }
074        }
075
076        /*
077         * (non-Javadoc)
078         * 
079         * @see java.util.Stack#search(java.lang.Object)
080         */
081        @Override
082        public synchronized int search(final Object o) {
083                lock.lock();
084                try {
085                        return super.search(o);
086                } finally {
087                        lock.unlock();
088                }
089        }
090
091        /*
092         * (non-Javadoc)
093         * 
094         * @see java.util.Stack#empty()
095         */
096        @Override
097        public boolean empty() {
098                lock.lock();
099                try {
100                        return super.empty();
101                } finally {
102                        lock.unlock();
103                }
104        }
105
106        @Override
107        public synchronized boolean add(final E e) {
108                lock.lock();
109                try {
110                        return super.add(e);
111                } finally {
112                        lock.unlock();
113                }
114        }
115
116        /*
117         * (non-Javadoc)
118         * 
119         * @see java.util.Vector#addAll(java.util.Collection)
120         */
121        @Override
122        public synchronized boolean addAll(final Collection<? extends E> c) {
123                lock.lock();
124                try {
125                        return super.addAll(c);
126                } finally {
127                        lock.unlock();
128                }
129
130        }
131
132        /*
133         * (non-Javadoc)
134         * 
135         * @see java.util.Vector#addAll(int, java.util.Collection)
136         */
137        @Override
138        public synchronized boolean addAll(final int index,
139                        final Collection<? extends E> c) {
140                lock.lock();
141                try {
142                        return super.addAll(index, c);
143                } finally {
144                        lock.unlock();
145                }
146        }
147}
