017package be.Balor.Tools;
018
019import java.util.HashMap;
020
021import org.bukkit.entity.Player;
022
023import be.Balor.bukkit.AdminCmd.ACPluginManager;
024import be.Balor.bukkit.AdminCmd.ConfigEnum;
025
026/**
027 * @author Balor (aka Antoine Aflalo)
028 * 
029 */
030public class TpRequest {
031        private final Player from, to;
032        private final long timeOut;
033
034        /**
035         *
036         */
037        public TpRequest(final Player from, final Player to) {
038                this.from = from;
039                this.to = to;
040                timeOut = System.currentTimeMillis()
041                                + (ConfigEnum.TPR_TIMEOUT.getLong() * 60000);
042        }
043
044        public void teleport(final Player sender) {
045                if (System.currentTimeMillis() > timeOut) {
046                        Utils.sI18n(sender, "tpRequestTimeOut");
047                        return;
048                }
049                if (from != null && to != null) {
050                        final String fromName = Utils.getPlayerName(from);
051                        final String toName = Utils.getPlayerName(to);
052                        ACPluginManager.scheduleSyncTask(new Runnable() {
053                                @Override
054                                public void run() {
055                                        from.teleport(to);
056                                        final HashMap<String, String> replace = new HashMap<String, String>();
057                                        replace.put("fromPlayer", fromName);
058                                        replace.put("toPlayer", toName);
059                                        Utils.sI18n(to, "tp", replace);
060                                        Utils.sI18n(from, "tp", replace);
061
062                                }
063                        });
064
065                }
066        }
067
068        /**
069         * @return the from
070         */
071        public String getFrom() {
072                return from == null ? "null" : from.getName();
073        }
074
075        /**
076         * @return the to
077         */
078        public String getTo() {
079                return to == null ? "null" : to.getName();
080        }
081
082        /*
083         * (non-Javadoc)
084         * 
085         * @see java.lang.Object#toString()
086         */
087        @Override
088        public String toString() {
089                return from == null || to == null ? "false" : from.getName() + ":"
090                                + to.getName();
091        }
092}
