026    public class TeleportTask implements Runnable {
027            private final Player player;
028            private final Location loc;
029    
030            /**
031             * @param player
032             * @param loc
033             */
034            public TeleportTask(final Player player, final Location loc) {
035                    super();
036                    this.player = player;
037                    this.loc = loc;
038            }
039    
040            /*
041             * (non-Javadoc)
042             * 
043             * @see java.lang.Runnable#run()
044             */
045            @Override
046            public void run() {
047                    Utils.teleportWithChunkCheck(player, loc);
048            }
049    
050    }
