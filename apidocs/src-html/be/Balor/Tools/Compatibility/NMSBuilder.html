017package be.Balor.Tools.Compatibility;
018
019import java.lang.reflect.Constructor;
020
021import org.bukkit.entity.Player;
022
023import be.Balor.Tools.Compatibility.Reflect.FieldUtils;
024import be.Balor.Tools.Compatibility.Reflect.MethodHandler;
025
026public class NMSBuilder {
027
028        /**
029         * Create a Packet201PlayerInfo
030         * 
031         * @param player
032         *            - Bukkit Player
033         * @param online
034         *            - online or offline
035         * @param ping
036         *            - ping
037         * @return instance of Packet201PlayerInfo
038         */
039        public static Object buildPacket201PlayerInfo(final Player player,
040                        final boolean online, final int ping) {
041                final Object playerHandle = MinecraftReflection.getHandle(player);
042                try {
043                        final String listName = FieldUtils.getField(playerHandle,
044                                        "listName");
045                        final Class<?> packetClass = MinecraftReflection
046                                        .getPacket201PlayerInfoClass();
047                        final Constructor<?> packetConstructor = packetClass
048                                        .getConstructor(String.class, boolean.class, int.class);
049                        return packetConstructor.newInstance(listName, online, ping);
050                } catch (final Exception e) {
051                        throw new RuntimeException("Can't create the wanted packet", e);
052                }
053        }
054
055        /**
056         * Build a NMS PlayerInventory for the given Bukkit Player.
057         * 
058         * @param player
059         *            - bukkit player
060         * @return instance of PlayerInventory
061         */
062        public static Object buildPlayerInventory(final Player player) {
063                final Object playerHandle = MinecraftReflection.getHandle(player);
064                try {
065                        final Class<?> playerInventoryClass = MinecraftReflection
066                                        .getPlayerInventoryClass();
067                        final Constructor<?> invConstructor = playerInventoryClass
068                                        .getConstructor(MinecraftReflection.getEntityHumanClass());
069                        return invConstructor.newInstance(playerHandle);
070                } catch (final Exception e) {
071                        throw new RuntimeException("Can't build PlayerInventory", e);
072                }
073        }
074
075        /**
076         * Build a FAKE entity player
077         * 
078         * @param name
079         *            - name of the player
080         * @return instance NMS of EntityPlayer
081         */
082        public static Object buildEntityPlayer(final String name) {
083                final Object minecraftServer = MinecraftReflection.getMinecraftServer();
084                try {
085                        final MethodHandler getWorldServer = new MethodHandler(
086                                        minecraftServer.getClass(), "getWorldServer", int.class);
087                        final Object worldServer = getWorldServer
088                                        .invoke(minecraftServer, 0);
089                        final Object itemInWorldManager = buildItemInWorldManager(worldServer);
090                        final Class<?> entityPlayer = MinecraftReflection
091                                        .getEntityPlayerClass();
092                        final Constructor<?> ePConstructor = entityPlayer.getConstructor(
093                                        MinecraftReflection.getMinecraftServerClass(),
094                                        MinecraftReflection.getNMSWorldClass(), String.class,
095                                        MinecraftReflection.getItemInWorldManagerClass());
096                        return ePConstructor.newInstance(minecraftServer, worldServer,
097                                        name, itemInWorldManager);
098                } catch (final Exception e) {
099                        throw new RuntimeException("Can't build PlayerInventory", e);
100                }
101        }
102
103        /**
104         * Build a ItemInWorldManager
105         * 
106         * @param worldServer
107         * @return return instance of ItemInWorldManager
108         */
109        public static Object buildItemInWorldManager(final Object worldServer) {
110                if (!MinecraftReflection.getWorldServerClass().isAssignableFrom(
111                                worldServer.getClass())) {
112                        throw new RuntimeException("The constructor need a "
113                                        + MinecraftReflection.getWorldServerClass().getSimpleName());
114                }
115                final Class<?> clazz = MinecraftReflection.getItemInWorldManagerClass();
116                try {
117                        final Constructor<?> constructor = clazz
118                                        .getConstructor(MinecraftReflection.getWorldServerClass());
119                        return constructor.newInstance(worldServer);
120                } catch (final Exception e) {
121                        throw new RuntimeException("Can't build ItemInWorldManager", e);
122                }
123
124        }
125}
