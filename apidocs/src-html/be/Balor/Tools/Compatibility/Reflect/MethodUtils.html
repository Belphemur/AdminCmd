017package be.Balor.Tools.Compatibility.Reflect;
018
019import java.lang.reflect.Method;
020import java.util.Arrays;
021import java.util.HashMap;
022import java.util.Map;
023
024/**
025 * @author Antoine
026 * 
027 */
028public class MethodUtils {
029        private static class MethodKey {
030                private final Class<?> clazz;
031                private final String name;
032                private final Class<?>[] parameterTypes;
033                private final int hashcode;
034
035                /**
036                 * @param clazz
037                 * @param name
038                 * @param parameterTypes
039                 */
040                public MethodKey(final Class<?> clazz, final String name,
041                                final Class<?>... parameterTypes) {
042                        super();
043                        this.clazz = clazz;
044                        this.name = name;
045                        this.parameterTypes = parameterTypes;
046                        this.hashcode = calcHashCode();
047                }
048
049                private int calcHashCode() {
050                        final int prime = 23;
051                        int result = 1;
052                        result = prime * result + ((clazz == null) ? 0 : clazz.hashCode());
053                        result = prime * result + ((name == null) ? 0 : name.hashCode());
054                        result = prime * result + Arrays.hashCode(parameterTypes);
055                        return result;
056                }
057
058                /*
059                 * (non-Javadoc)
060                 * 
061                 * @see java.lang.Object#hashCode()
062                 */
063                @Override
064                public int hashCode() {
065                        return this.hashcode;
066                }
067
068                /*
069                 * (non-Javadoc)
070                 * 
071                 * @see java.lang.Object#equals(java.lang.Object)
072                 */
073                @Override
074                public boolean equals(final Object obj) {
075                        if (this == obj) {
076                                return true;
077                        }
078                        if (obj == null) {
079                                return false;
080                        }
081                        if (getClass() != obj.getClass()) {
082                                return false;
083                        }
084                        final MethodKey other = (MethodKey) obj;
085                        if (clazz == null) {
086                                if (other.clazz != null) {
087                                        return false;
088                                }
089                        } else if (!clazz.equals(other.clazz)) {
090                                return false;
091                        }
092                        if (name == null) {
093                                if (other.name != null) {
094                                        return false;
095                                }
096                        } else if (!name.equals(other.name)) {
097                                return false;
098                        }
099                        if (!Arrays.equals(parameterTypes, other.parameterTypes)) {
100                                return false;
101                        }
102                        return true;
103                }
104
105        }
106
107        private static final Map<MethodKey, Method> cachedMethod = new HashMap<MethodKey, Method>();
108
109        /**
110         * Get the method of the given class, even if it's a private method.
111         * 
112         * @param clazz
113         *            - given class
114         * @param name
115         *            - name of the method
116         * @param parameterTypes
117         *            - parameter type of the method (Class<?>)
118         * @return the wanted method if found
119         * @throws RuntimeException
120         *             if the method don't exist.
121         */
122        public static Method getMethod(final Class<?> clazz, final String name,
123                        final Class<?>... parameterTypes) {
124                Method method;
125                try {
126                        method = getClassMethod(clazz, name, parameterTypes);
127                        return method;
128                } catch (final Exception e) {
129                        throw new RuntimeException("Can't get method " + name + " from "
130                                        + clazz, e);
131                }
132        }
133
134        /**
135         * Invoke the method with the given argument on the given object. Works also
136         * for private method.
137         * 
138         * @param obj
139         *            - given object
140         * @param method
141         *            - given method
142         * @param args
143         *            - given arguments
144         * @return the result of the method.
145         * @throws RuntimeException
146         *             if there is a problem while invoking the method.
147         */
148        @SuppressWarnings("unchecked")
149        public static <T> T invokeMethod(final Object obj, final Method method,
150                        final Object... args) {
151                T result = null;
152                method.setAccessible(true);
153                try {
154                        result = (T) method.invoke(obj, args);
155                } catch (final Exception e) {
156                        throw new RuntimeException("Can't invoke method " + method
157                                        + " from " + obj, e);
158                } finally {
159                        method.setAccessible(false);
160                }
161                return result;
162        }
163
164        private synchronized static Method getClassMethod(final Class<?> clazz,
165                        final String name, final Class<?>... parameterTypes)
166                        throws NoSuchMethodException {
167                final MethodKey key = new MethodKey(clazz, name, parameterTypes);
168                Method classMethod = cachedMethod.get(key);
169                if (classMethod != null) {
170                        return classMethod;
171                }
172                Class<?> copyClass = clazz;
173                try {
174                        classMethod = copyClass.getDeclaredMethod(name, parameterTypes);
175                } catch (final NoSuchMethodException e) {
176                        while (true) {
177                                copyClass = copyClass.getSuperclass();
178                                if (copyClass.equals(Object.class)) {
179                                        throw e;
180                                }
181                                try {
182                                        classMethod = copyClass.getDeclaredMethod(name,
183                                                        parameterTypes);
184                                        break;
185                                } catch (final NoSuchMethodException e1) {
186                                }
187                        }
188                }
189                cachedMethod.put(key, classMethod);
190                return classMethod;
191        }
192}
