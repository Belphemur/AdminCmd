010/*

013import java.io.File;
014import java.io.IOException;
015import java.sql.DriverManager;
016import java.sql.ResultSet;
017import java.sql.SQLException;
018import java.sql.Statement;
019import java.util.logging.Logger;
020
021import lib.SQL.PatPeter.SQLibrary.DatabaseConfig.DatabaseType;
022
023public class SQLite extends Database {
024        public String location;
025        public String name;
026        private final File sqlFile;
027
028        public SQLite(final Logger log, final String prefix, final String name,
029                        final String location) {
030                super(log, prefix, "[SQLite] ");
031                this.name = name;
032                this.location = location;
033                final File folder = new File(this.location);
034                if (this.name.contains("/") || this.name.contains("\\")
035                                || this.name.endsWith(".db")) {
036                        this.writeError("The database name cannot contain: /, \\, or .db",
037                                        true);
038                }
039                if (!folder.exists()) {
040                        folder.mkdir();
041                }
042
043                sqlFile = new File(folder, name + ".db");
044        }
045
046        @Override
047        protected void initialize() throws SQLException {
048                try {
049                        Class.forName("org.sqlite.JDBC");
050                        if (!sqlFile.exists()) {
051                                try {
052                                        sqlFile.createNewFile();
053                                } catch (final IOException e) {
054                                }
055                        }
056                } catch (final ClassNotFoundException e) {
057                        throw new SQLException("Can't load JDBC Driver", e);
058                }
059        }
060
061        @Override
062        public void open() throws SQLException {
063                initialize();
064                this.connection = DriverManager.getConnection("jdbc:sqlite:"
065                                + sqlFile.getAbsolutePath());
066
067        }
068
069        @Override
070        public ResultSet query(final String query) {
071                Statement statement = null;
072                ResultSet result = null;
073
074                try {
075                        synchronized (connection) {
076                                statement = connection.createStatement();
077
078                                switch (this.getStatement(query)) {
079                                case SELECT:
080                                        result = statement.executeQuery(query);
081                                        break;
082
083                                case INSERT:
084                                case UPDATE:
085                                case DELETE:
086                                case CREATE:
087                                case ALTER:
088                                case DROP:
089                                case TRUNCATE:
090                                case RENAME:
091                                case DO:
092                                case REPLACE:
093                                case LOAD:
094                                case HANDLER:
095                                case CALL:
096                                        this.lastUpdate = statement.executeUpdate(query);
097                                        break;
098
099                                default:
100                                        result = statement.executeQuery(query);
101
102                                }
103                        }
104                        return result;
105                } catch (final SQLException e) {
106                        if (e.getMessage().toLowerCase().contains("locking")
107                                        || e.getMessage().toLowerCase().contains("locked")) {
108                                return retry(query);
109                        } else {
110                                this.writeError("SQL exception in query(): " + e.getMessage(),
111                                                false);
112                        }
113
114                }
115                return null;
116        }
117
118        @Override
119        public boolean wipeTable(final String table) {
120                Statement statement = null;
121                String query = null;
122                try {
123                        if (!this.checkTable(table)) {
124                                this.writeError("Table \"" + table
125                                                + "\" in wipeTable() does not exist.", true);
126                                return false;
127                        }
128                        query = "DELETE FROM " + table + ";";
129                        synchronized (connection) {
130                                statement = connection.createStatement();
131                                statement.executeQuery(query);
132                        }
133                        return true;
134                } catch (final SQLException ex) {
135                        if (!(ex.getMessage().toLowerCase().contains("locking") || ex
136                                        .getMessage().toLowerCase().contains("locked"))
137                                        && !ex.toString().contains("not return ResultSet")) {
138                                this.writeError("Error at SQL Wipe Table Query: " + ex, false);
139                        }
140                        return false;
141                }
142        }
143
144        /**
145         * <b>retry</b><br>
146         * <br>
147         * Retries a statement and returns a ResultSet. <br>
148         * <br>
149         * 
150         * @param query
151         *            The SQL query to retry.
152         * 
153         * @return The SQL query result.
154         */
155        private ResultSet retry(final String query) {
156                Statement statement = null;
157                ResultSet result = null;
158
159                try {
160                        synchronized (connection) {
161                                statement = connection.createStatement();
162                                result = statement.executeQuery(query);
163                        }
164                        return result;
165                } catch (final SQLException ex) {
166                        if (ex.getMessage().toLowerCase().contains("locking")
167                                        || ex.getMessage().toLowerCase().contains("locked")) {
168                                this.writeError(
169                                                "Please close your previous ResultSet to run the query: \n\t"
170                                                                + query, false);
171                        } else {
172                                this.writeError("SQL exception in retry(): " + ex.getMessage(),
173                                                false);
174                        }
175                }
176
177                return null;
178        }
179
180        /*
181         * (Non javadoc)
182         * 
183         * @see lib.SQL.PatPeter.SQLibrary.Database#getType()
184         */
185        @Override
186        public DatabaseType getType() {
187                return DatabaseType.SQLITE;
188        }
189}
