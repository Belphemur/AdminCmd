008package lib.SQL.PatPeter.SQLibrary;
009
010import java.sql.DriverManager;
011import java.sql.PreparedStatement;
012import java.sql.ResultSet;
013import java.sql.SQLException;
014import java.sql.Statement;
015import java.util.logging.Logger;
016
017import lib.SQL.PatPeter.SQLibrary.DatabaseConfig.DatabaseType;
018
019public class MySQL extends Database {
020        private String hostname = "localhost";
021        private String portnmbr = "3306";
022        private String username = "minecraft";
023        private String password = "";
024        private String database = "minecraft";
025
026        public MySQL(final Logger log, final String prefix, final String hostname,
027                        final String portnmbr, final String database,
028                        final String username, final String password) {
029                super(log, prefix, "[MySQL] ");
030                this.hostname = hostname;
031                this.portnmbr = portnmbr;
032                this.database = database;
033                this.username = username;
034                this.password = password;
035        }
036
037        @Override
038        protected void initialize() throws SQLException {
039                try {
040                        Class.forName("com.mysql.jdbc.Driver"); // Check that server's Java
041                                                                                                        // has MySQL support.
042                } catch (final ClassNotFoundException e) {
043                        throw new SQLException("Can't load JDBC Driver", e);
044                }
045        }
046
047        @Override
048        public void open() throws SQLException {
049                initialize();
050                String url = "";
051                url = "jdbc:mysql://" + this.hostname + ":" + this.portnmbr + "/"
052                                + this.database;
053                this.connection = DriverManager.getConnection(url, this.username,
054                                this.password);
055        }
056
057        @Override
058        public ResultSet query(final String query) {
059                Statement statement = null;
060                ResultSet result = null;
061                try {
062                        autoReconnect();
063                        synchronized (connection) {
064                                statement = this.connection.createStatement();
065
066                                switch (this.getStatement(query)) {
067                                case SELECT:
068                                        result = statement.executeQuery(query);
069                                        break;
070
071                                case INSERT:
072                                case UPDATE:
073                                case DELETE:
074                                case CREATE:
075                                case ALTER:
076                                case DROP:
077                                case TRUNCATE:
078                                case RENAME:
079                                case DO:
080                                case REPLACE:
081                                case LOAD:
082                                case HANDLER:
083                                case CALL:
084                                        this.lastUpdate = statement.executeUpdate(query);
085                                        break;
086
087                                default:
088                                        result = statement.executeQuery(query);
089                                }
090                        }
091                        return result;
092                } catch (final SQLException e) {
093                        this.writeError("SQL exception in query(): " + e.getMessage(),
094                                        false);
095                }
096                return result;
097        }
098
099        @Override
100        public boolean wipeTable(final String table) {
101                Statement statement = null;
102                String query = null;
103                try {
104                        if (!this.checkTable(table)) {
105                                this.writeError("Table \"" + table
106                                                + "\" in wipeTable() does not exist.", true);
107                                return false;
108                        }
109                        query = "DELETE FROM " + table + ";";
110                        synchronized (connection) {
111                                statement = connection.createStatement();
112                                statement.executeQuery(query);
113                        }
114                        return true;
115                } catch (final SQLException e) {
116                        if (!e.toString().contains("not return ResultSet")) {
117                                return false;
118                        }
119                }
120                return false;
121        }
122
123        /*
124         * (Non javadoc)
125         * 
126         * @see lib.SQL.PatPeter.SQLibrary.Database#getType()
127         */
128        @Override
129        public DatabaseType getType() {
130                return DatabaseType.MYSQL;
131        }
132
133        /*
134         * (non-Javadoc)
135         * 
136         * @see lib.SQL.PatPeter.SQLibrary.Database#prepare(java.lang.String)
137         */
138        @Override
139        public PreparedStatement prepare(final String query) {
140                try {
141                        final PreparedStatement ps;
142                        autoReconnect();
143                        synchronized (connection) {
144                                if (getStatement(query) == Statements.INSERT) {
145                                        ps = connection.prepareStatement(query,
146                                                        PreparedStatement.RETURN_GENERATED_KEYS);
147                                } else {
148                                        ps = connection.prepareStatement(query);
149                                }
150                        }
151                        return ps;
152                } catch (final SQLException e) {
153                        if (!e.toString().contains("not return ResultSet")) {
154                                this.writeError(
155                                                "SQL exception in prepare(): " + e.getMessage(), false);
156                        }
157                }
158                return null;
159        }
160}
