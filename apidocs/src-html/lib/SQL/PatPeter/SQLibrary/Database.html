054        public static void initDb() {
055                if (DATABASE != null) {
056                        DATABASE.close();
057                }
058                final DatabaseConfig config = new DatabaseConfig();
059                Database db;
060                final String dbWrapper = ConfigEnum.DATA_WRAPPER.getString();
061                if (dbWrapper.equalsIgnoreCase("mysql")) {
062                        config.setType(DatabaseType.MYSQL);
063                        try {
064                                config.setParameter(Parameter.HOSTNAME,
065                                                ConfigEnum.MYSQL_HOST.getString());
066                                config.setParameter(Parameter.PASSWORD,
067                                                ConfigEnum.MYSQL_PASS.getString());
068                                config.setParameter(Parameter.USER,
069                                                ConfigEnum.MYSQL_USER.getString());
070                                config.setParameter(Parameter.PORT_NUMBER, "3306");
071                                config.setParameter(Parameter.DATABASE,
072                                                ConfigEnum.MYSQL_DB.getString());
073                        } catch (final NullPointerException e) {
074                        } catch (final InvalidConfigurationException e) {
075                        }
076                } else if (dbWrapper.equalsIgnoreCase("sqlite")) {
077                        config.setType(DatabaseType.SQLITE);
078                        try {
079                                config.setParameter(Parameter.DB_LOCATION, ACHelper
080                                                .getInstance().getCoreInstance().getDataFolder()
081                                                .getAbsolutePath());
082                                config.setParameter(Parameter.DB_NAME, "admincmd");
083
084                        } catch (final NullPointerException e) {
085                        } catch (final InvalidConfigurationException e) {
086                        }
087                }
088                config.setLog(ACHelper.getInstance().getCoreInstance().getLogger());
089                try {
090                        config.setParameter(Parameter.DB_PREFIX, "[AdminCmd Database]");
091                } catch (final NullPointerException e1) {
092
093                } catch (final InvalidConfigurationException e1) {
094
095                }
096
097                try {
098                        if (config.getType() == null) {
099                                db = null;
100                        } else {
101                                db = DatabaseFactory.createDatabase(config);
102                        }
103                } catch (final InvalidConfigurationException e) {
104                        ACLogger.severe("Problem while trying to load the Database", e);
105                        db = null;
106                }
107                DATABASE = db;
108                DebugLog.INSTANCE.info("Database initialization done");
109        }

159        public Database(final Logger log, final String prefix, final String dp) {
160                this.log = log;

167        /**
168         * <b>writeInfo</b><br>
169         * <br>
170         * &nbsp;&nbsp;Writes information to the console. <br>
171         * <br>
172         * 
173         * @param toWrite
174         *            - the <a href=
175         *            "http://download.oracle.com/javase/6/docs/api/java/lang/String.html"
176         *            >String</a> of content to write to the console.
177         */
178        protected void writeInfo(final String toWrite) {
179                if (toWrite != null) {
180                        this.log.info(this.PREFIX + this.DATABASE_PREFIX + toWrite);
181                }
182        }

184        /**
185         * <b>writeError</b><br>
186         * <br>
187         * &nbsp;&nbsp;Writes either errors or warnings to the console. <br>
188         * <br>
189         * 
190         * @param toWrite
191         *            - the <a href=
192         *            "http://download.oracle.com/javase/6/docs/api/java/lang/String.html"
193         *            >String</a> written to the console.
194         * @param severe
195         *            - whether console output should appear as an error or warning.
196         */
197        protected void writeError(final String toWrite, final boolean severe) {
198                if (toWrite != null) {
199                        if (severe) {
200                                this.log.severe(this.PREFIX + this.DATABASE_PREFIX + toWrite);
201                        } else {
202                                this.log.warning(this.PREFIX + this.DATABASE_PREFIX + toWrite);
203                        }
204                }
205        }

287        /**
288         * <b>prepare</b><br>
289         * &nbsp;&nbsp;Prepares to send a query to the database. <br>
290         * <br>
291         * 
292         * @param query
293         *            - the SQL query to prepare to send to the database.
294         * @return the prepared statement.
295         */
296        public PreparedStatement prepare(final String query) {
297                try {
298                        final PreparedStatement ps;
299                        synchronized (connection) {
300                                ps = connection.prepareStatement(query);
301                        }
302                        return ps;
303                } catch (final SQLException e) {
304                        if (!e.toString().contains("not return ResultSet")) {
305                                this.writeError(
306                                                "SQL exception in prepare(): " + e.getMessage(), false);
307                        }
308                }
309                return null;
310        }

312        /**
313         * <b>getStatement</b><br>
314         * &nbsp;&nbsp;Determines the name of the statement and converts it into an
315         * enum. <br>
316         * <br>
317         */
318        protected Statements getStatement(final String query) {
319                final String trimmedQuery = query.trim();
320                if (trimmedQuery.substring(0, 6).equalsIgnoreCase("SELECT")) {
321                        return Statements.SELECT;
322                } else if (trimmedQuery.substring(0, 6).equalsIgnoreCase("INSERT")) {
323                        return Statements.INSERT;
324                } else if (trimmedQuery.substring(0, 6).equalsIgnoreCase("UPDATE")) {
325                        return Statements.UPDATE;
326                } else if (trimmedQuery.substring(0, 6).equalsIgnoreCase("DELETE")) {
327                        return Statements.DELETE;
328                } else if (trimmedQuery.substring(0, 6).equalsIgnoreCase("CREATE")) {
329                        return Statements.CREATE;
330                } else if (trimmedQuery.substring(0, 5).equalsIgnoreCase("ALTER")) {
331                        return Statements.ALTER;
332                } else if (trimmedQuery.substring(0, 4).equalsIgnoreCase("DROP")) {
333                        return Statements.DROP;
334                } else if (trimmedQuery.substring(0, 8).equalsIgnoreCase("TRUNCATE")) {
335                        return Statements.TRUNCATE;
336                } else if (trimmedQuery.substring(0, 6).equalsIgnoreCase("RENAME")) {
337                        return Statements.RENAME;
338                } else if (trimmedQuery.substring(0, 2).equalsIgnoreCase("DO")) {
339                        return Statements.DO;
340                } else if (trimmedQuery.substring(0, 7).equalsIgnoreCase("REPLACE")) {
341                        return Statements.REPLACE;
342                } else if (trimmedQuery.substring(0, 4).equalsIgnoreCase("LOAD")) {
343                        return Statements.LOAD;
344                } else if (trimmedQuery.substring(0, 7).equalsIgnoreCase("HANDLER")) {
345                        return Statements.HANDLER;
346                } else if (trimmedQuery.substring(0, 4).equalsIgnoreCase("CALL")) {
347                        return Statements.CALL;
348                } else {
349                        return Statements.SELECT;
350                }
351        }

353        /**
354         * <b>createTable</b><br>
355         * <br>
356         * &nbsp;&nbsp;Creates a table in the database based on a specified query. <br>
357         * <br>
358         * 
359         * @param query
360         *            - the SQL query for creating a table.
361         * @return the success of the method.
362         */
363        public boolean createTable(final String query) {
364                Statement statement = null;
365                try {
366                        if (query.equals("") || query == null) {
367                                this.writeError(
368                                                "Parameter 'query' empty or null in createTable().",
369                                                true);
370                                return false;
371                        }
372                        synchronized (connection) {
373                                statement = connection.createStatement();
374                                statement.execute(query);
375                        }
376                        return true;
377                } catch (final SQLException ex) {
378                        this.writeError(ex.getMessage(), true);
379                        return false;
380                }
381        }

383        /**
384         * <b>checkTable</b><br>
385         * <br>
386         * &nbsp;&nbsp;Checks a table in a database based on the table's name. <br>
387         * <br>
388         * 
389         * @param table
390         *            - name of the table to check.
391         * @return success of the method.
392         */
393        public boolean checkTable(final String table) {
394                DatabaseMetaData dbm = null;
395                try {
396                        synchronized (connection) {
397                                dbm = this.connection.getMetaData();
398                        }
399
400                        final ResultSet tables = dbm.getTables(null, null, table, null);
401                        if (tables.next()) {
402                                return true;
403                        } else {
404                                return false;
405                        }
406                } catch (final SQLException e) {
407                        this.writeError("Failed to check if table \"" + table
408                                        + "\" exists: " + e.getMessage(), true);
409                        return false;
410                }
411        }

432        /**
433         * Check if the connection is valid
434         * 
435         * @return true if it is
436         */
437        private boolean isConnectionValid() {
438                synchronized (this.connection) {
439                        if (checkConnection()) {
440                                try {
441                                        return this.connection.isValid(3);
442                                } catch (final SQLException e) {
443                                        DebugLog.INSTANCE.log(Level.INFO,
444                                                        "Problem when checking connection state", e);
445                                }
446                        }
447                }
448                return false;
449        }
450
451        /**
452         * Try to reconnect.
453         */
454        private void reconnect() {
455                synchronized (this.connection) {
456                        try {
457                                this.connection.close();
458                        } catch (final SQLException e) {
459                        }
460                        this.connection = null;
461                        try {
462                                open();
463                        } catch (final SQLException e) {
464                                writeError(
465                                                "Problem while reconnection to the database :\n"
466                                                                + e.getMessage(), true);
467                        }
468                }
469        }
470
471        /**
472         * Check if the connection is valid, if not try to reconnect.
473         */
474        public void autoReconnect() {
475                if (!isConnectionValid()) {
476                        reconnect();
477                }
478        }
