017package lib.SQL.PatPeter.SQLibrary;
018
019import java.util.EnumMap;
020import java.util.HashSet;
021import java.util.Map;
022import java.util.Set;
023import java.util.logging.Logger;
024
025import org.bukkit.configuration.InvalidConfigurationException;
026
027/**
028 * @author Balor (aka Antoine Aflalo)
029 * 
030 */
031public class DatabaseConfig {
032        private final Map<Parameter, String> config = new EnumMap<Parameter, String>(
033                        Parameter.class);
034        private DatabaseType type;
035        private Logger log;
036
037        public enum DatabaseType {
038                MYSQL,
039                SQLITE,
040                ALL;
041        }
042
043        public enum Parameter {
044                HOSTNAME(
045                                DatabaseType.MYSQL),
046                USER(
047                                DatabaseType.MYSQL),
048                PASSWORD(
049                                DatabaseType.MYSQL),
050                PORT_NUMBER(
051                                DatabaseType.MYSQL),
052                DATABASE(
053                                DatabaseType.MYSQL),
054                DB_PREFIX(
055                                DatabaseType.ALL),
056                DB_LOCATION(
057                                DatabaseType.SQLITE),
058                DB_NAME(
059                                DatabaseType.SQLITE);
060                private final Set<DatabaseType> dbTypes = new HashSet<DatabaseType>();
061                private static Map<DatabaseType, Integer> count;
062
063                /**
064                 * 
065                 */
066                private Parameter(final DatabaseType... type) {
067                        for (final DatabaseType element : type) {
068                                dbTypes.add(element);
069                                updateCount(element);
070                        }
071
072                }
073
074                public boolean validParam(final DatabaseType toCheck) {
075                        if (dbTypes.contains(DatabaseType.ALL)) {
076                                return true;
077                        }
078                        if (dbTypes.contains(toCheck)) {
079                                return true;
080                        }
081                        return false;
082
083                }
084
085                private static void updateCount(final DatabaseType type) {
086                        if (count == null) {
087                                count = new EnumMap<DatabaseType, Integer>(DatabaseType.class);
088                        }
089                        Integer nb = count.get(type);
090                        if (nb == null) {
091                                nb = 1;
092                        } else {
093                                nb++;
094                        }
095                        count.put(type, nb);
096                }
097
098                public static int getCount(final DatabaseType type) {
099                        final int nb = count.get(DatabaseType.ALL) + count.get(type);
100                        return nb;
101                }
102        }
103
104        /**
105         * @param type
106         *            the type to set
107         */
108        public void setType(final DatabaseType type)
109                        throws IllegalArgumentException {
110                if (type == DatabaseType.ALL) {
111                        throw new IllegalArgumentException(
112                                        "You can't set your database type to ALL");
113                }
114                this.type = type;
115        }
116
117        /**
118         * @param log
119         *            the log to set
120         */
121        public void setLog(final Logger log) {
122                this.log = log;
123        }
124
125        /**
126         * @return the type
127         */
128        public DatabaseType getType() {
129                return type;
130        }
131
132        /**
133         * @return the log
134         */
135        public Logger getLog() {
136                return log;
137        }
138
139        public DatabaseConfig setParameter(final Parameter param, final String value)
140                        throws NullPointerException, InvalidConfigurationException {
141                if (this.type == null) {
142                        throw new NullPointerException(
143                                        "You must set the type of the database first");
144                }
145                if (!param.validParam(type)) {
146                        throw new InvalidConfigurationException(param.toString()
147                                        + " is invalid for a database type of : " + type.toString());
148                }
149                config.put(param, value);
150                return this;
151
152        }
153
154        public String getParameter(final Parameter param) {
155                return config.get(param);
156        }
157
158        public boolean isValid() throws InvalidConfigurationException {
159                if (log == null) {
160                        throw new InvalidConfigurationException(
161                                        "You need to set the logger.");
162                }
163                return config.size() == Parameter.getCount(type);
164        }
165}
