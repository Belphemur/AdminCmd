017    package belgium.Balor.Workers;
018    
019    import java.util.LinkedList;
020    import java.util.concurrent.ConcurrentMap;
021    
022    import net.minecraft.server.EntityPlayer;
023    import net.minecraft.server.Packet20NamedEntitySpawn;
024    import net.minecraft.server.Packet29DestroyEntity;
025    
026    import org.bukkit.craftbukkit.entity.CraftPlayer;
027    import org.bukkit.entity.Player;
028    
029    import be.Balor.Manager.Permissions.PermissionManager;
030    import be.Balor.Tools.UpdateInvisible;
031    import be.Balor.Tools.Utils;
032    import be.Balor.bukkit.AdminCmd.ACHelper;
033    import be.Balor.bukkit.AdminCmd.ACPluginManager;
034    import be.Balor.bukkit.AdminCmd.ConfigEnum;
035    
036    import com.google.common.collect.MapMaker;
037    
038    /**
039     * @author Balor (aka Antoine Aflalo)
040     * 
041     */
042    final public class InvisibleWorker {
043            protected static InvisibleWorker instance = null;
044            private final ConcurrentMap<String, Integer> invisblesWithTaskIds = new MapMaker().makeMap();
045            private long maxRange = 262144;
046            private int tickCheck = 400;
047    
048            /**
049             * 
050             */
051            protected InvisibleWorker() {
052    
053            }
054    
055            /**
056             * @return the instance
057             */
058            public static InvisibleWorker getInstance() {
059                    if (instance == null)
060                            instance = new InvisibleWorker();
061                    return instance;
062            }
063    
064            public static InvisibleWorker createInstance() {
065                    if (instance == null)
066                            instance = new InvisibleWorker();
067                    return instance;
068            }
069    
070            /**
071             * Destroy the instance
072             */
073            public static void killInstance() {
074                    instance = null;
075            }
076    
077            /**
078             * @param maxRange
079             *            the maxRange to set
080             */
081            public void setMaxRange(final long maxRange) {
082                    this.maxRange = maxRange ^ 2;
083            }
084    
085            /**
086             * @param tickCheck
087             *            the tickCheck to set
088             */
089            public void setTickCheck(final int tickCheck) {
090                    this.tickCheck = tickCheck * 20;
091            }
092    
093            /**
094             * return all invisible Players
095             * 
096             * @return
097             */
098            public LinkedList<Player> getAllInvisiblePlayers() {
099                    final LinkedList<Player> result = new LinkedList<Player>();
100                    for (final String p : invisblesWithTaskIds.keySet())
101                            result.add(ACPluginManager.getServer().getPlayerExact(p));
102                    return result;
103            }
104    
105            /**
106             * Adding a player when joining the server after checked if the player have
107             * the invisible power
108             * 
109             * @param toVanish
110             *            player to vanish.
111             */
112            public void onJoinEvent(final Player toVanish) {
113                    final String name = toVanish.getName();
114                    if (!invisblesWithTaskIds.containsKey(name)) {
115                            invisblesWithTaskIds.put(
116                                            name,
117                                            ACPluginManager
118                                                            .getServer()
119                                                            .getScheduler()
120                                                            .scheduleAsyncRepeatingTask(ACHelper.getInstance().getCoreInstance(),
121                                                                            new UpdateInvisible(toVanish), tickCheck / 2, tickCheck));
122                    }
123            }
124    
125            /**
126             * Stop the task that try to make the player invisible when he disconnect
127             * 
128             * @param toReappear
129             */
130            public void onQuitEvent(final Player toReappear) {
131                    final String name = toReappear.getName();
132                    if (invisblesWithTaskIds.containsKey(name)) {
133                            ACPluginManager.getServer().getScheduler().cancelTask(invisblesWithTaskIds.get(name));
134                            invisblesWithTaskIds.remove(name);
135                    }
136            }
137    
138            /**
139             * Make the player reappear
140             * 
141             * @param toReappear
142             */
143            public void reappear(final Player toReappear) {
144                    final String name = toReappear.getName();
145                    if (invisblesWithTaskIds.containsKey(name)) {
146                            ACPluginManager.getServer().getScheduler().cancelTask(invisblesWithTaskIds.get(name));
147                            invisblesWithTaskIds.remove(name);
148    
149                            ACPluginManager
150                                            .getServer()
151                                            .getScheduler()
152                                            .scheduleSyncDelayedTask(ACHelper.getInstance().getCoreInstance(),
153                                                            new Runnable() {
154                                                                    @Override
155                                                                    public void run() {
156                                                                            for (final Player p : Utils.getOnlinePlayers())
157                                                                                    uninvisible(toReappear, p);
158                                                                    }
159                                                            });
160                            if (ConfigEnum.FQINVISIBLE.getBoolean()) {
161                                    Utils.broadcastFakeJoin(toReappear);
162                                    Utils.addPlayerInOnlineList(toReappear);
163                            }
164                    }
165    
166            }
167    
168            /**
169             * Make the player invisible.
170             * 
171             * @param hide
172             * @param hideFrom
173             */
174            public void invisible(final Player hide, final Player hideFrom) {
175                    if (hide == null) {
176                            return;
177                    }
178                    if (hideFrom == null) {
179                            return;
180                    }
181                    if (PermissionManager.hasPerm(hideFrom, "admincmd.invisible.cansee", false))
182                            return;
183                    if (hide.getName().equals(hideFrom.getName()))
184                            return;
185    
186                    if (Utils.getDistanceSquared(hide, hideFrom) > maxRange)
187                            return;
188                    final EntityPlayer craftFrom = ((CraftPlayer) hideFrom).getHandle();
189                    craftFrom.netServerHandler.sendPacket(new Packet29DestroyEntity(hide.getEntityId()));
190    
191            }
192    
193            /**
194             * Make the player unHide
195             * 
196             * @param unHide
197             * @param unHideFrom
198             */
199            private void uninvisible(final Player unHide, final Player unHideFrom) {
200                    if (unHide.equals(unHideFrom))
201                            return;
202    
203                    if (Utils.getDistanceSquared(unHide, unHideFrom) > maxRange)
204                            return;
205    
206                    if (PermissionManager.hasPerm(unHideFrom, "admincmd.invisible.cansee", false))
207                            return;
208                    final EntityPlayer craftFrom = ((CraftPlayer) unHideFrom).getHandle();
209                    final EntityPlayer UnHidePlayer = ((CraftPlayer) unHide).getHandle();
210                    craftFrom.netServerHandler.sendPacket(new Packet29DestroyEntity(unHide.getEntityId()));
211                    craftFrom.netServerHandler.sendPacket(new Packet20NamedEntitySpawn(UnHidePlayer));
212            }
213    
214            /**
215             * Check if the player is invisible
216             * 
217             * @param player
218             * @return
219             */
220            public boolean hasInvisiblePowers(final String player) {
221                    return invisblesWithTaskIds.containsKey(player);
222            }
223    
224            /**
225             * Make the player vanish
226             * 
227             * @param toVanish
228             */
229            public void vanish(final Player toVanish) {
230                    final String name = toVanish.getName();
231                    ACPluginManager
232                                    .getServer()
233                                    .getScheduler()
234                                    .scheduleSyncDelayedTask(ACHelper.getInstance().getCoreInstance(),
235                                                    new UpdateInvisible(toVanish));
236                    if (!invisblesWithTaskIds.containsKey(name)) {
237                            invisblesWithTaskIds.put(
238                                            name,
239                                            ACPluginManager
240                                                            .getServer()
241                                                            .getScheduler()
242                                                            .scheduleAsyncRepeatingTask(ACHelper.getInstance().getCoreInstance(),
243                                                                            new UpdateInvisible(toVanish), tickCheck / 2, tickCheck));
244                    }
245                    if (ConfigEnum.FQINVISIBLE.getBoolean()) {
246                            Utils.broadcastFakeQuit(toVanish);
247                            Utils.removePlayerFromOnlineList(toVanish);
248                    }
249    
250            }
251    
252            /**
253             * return the nb of invisible players
254             * 
255             * @return
256             */
257            public int nbInvisibles() {
258                    return invisblesWithTaskIds.size();
259            }
260
