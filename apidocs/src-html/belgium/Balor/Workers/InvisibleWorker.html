017package belgium.Balor.Workers;
018
019import java.util.Collection;
020import java.util.Collections;
021import java.util.concurrent.ConcurrentMap;
022
023import org.bukkit.entity.Player;
024import org.dynmap.DynmapAPI;
025
026import be.Balor.Manager.Permissions.PermissionManager;
027import be.Balor.Tools.Utils;
028import be.Balor.bukkit.AdminCmd.ACPluginManager;
029import be.Balor.bukkit.AdminCmd.ConfigEnum;
030
031import com.google.common.collect.MapMaker;
032
033/**
034 * @author Balor (aka Antoine Aflalo)
035 * 
036 */
037final public class InvisibleWorker {
038        protected static InvisibleWorker instance = null;
039        private final ConcurrentMap<Player, Object> invisiblesPlayers = new MapMaker()
040                        .makeMap();
041        private static final Object EMPTY = new Object();
042        public static DynmapAPI dynmapAPI;
043
044        /**
045         * 
046         */
047        protected InvisibleWorker() {
048
049        }
050
051        /**
052         * @return the instance
053         */
054        public static InvisibleWorker getInstance() {
055                if (instance == null) {
056                        instance = new InvisibleWorker();
057                }
058                return instance;
059        }
060
061        public static InvisibleWorker createInstance() {
062                if (instance == null) {
063                        instance = new InvisibleWorker();
064                }
065                return instance;
066        }
067
068        /**
069         * Destroy the instance
070         */
071        public static void killInstance() {
072                instance = null;
073        }
074
075        /**
076         * return all invisible Players
077         * 
078         * @return
079         */
080        public Collection<Player> getAllInvisiblePlayers() {
081                return Collections.unmodifiableCollection(invisiblesPlayers.keySet());
082        }
083
084        /**
085         * Stop the task that try to make the player invisible when he disconnect
086         * 
087         * @param toReappear
088         */
089        public void onQuitEvent(final Player toReappear) {
090                invisiblesPlayers.remove(toReappear);
091        }
092
093        /**
094         * Make the player reappear
095         * 
096         * @param toReappear
097         */
098        public void reappear(final Player toReappear) {
099                invisiblesPlayers.remove(toReappear);
100                if (dynmapAPI != null) {
101                        dynmapAPI.setPlayerVisiblity(toReappear, true);
102                }
103                ACPluginManager.getScheduler().runTaskAsynchronously(
104                                ACPluginManager.getCorePlugin(), new Runnable() {
105                                        @Override
106                                        public void run() {
107                                                for (final Player p : Utils.getOnlinePlayers()) {
108                                                        uninvisible(toReappear, p);
109                                                }
110                                        }
111                                });
112                if (ConfigEnum.FQINVISIBLE.getBoolean()) {
113                        Utils.broadcastFakeJoin(toReappear);
114                }
115
116        }
117
118        /**
119         * Make the player invisible.
120         * 
121         * @param hide
122         * @param hideFrom
123         */
124        private void invisible(final Player hide, final Player hideFrom) {
125                if (hide == null) {
126                        return;
127                }
128                if (hideFrom == null) {
129                        return;
130                }
131                if (PermissionManager.hasPerm(hideFrom, "admincmd.invisible.cansee",
132                                false)) {
133                        return;
134                }
135                ACPluginManager.scheduleSyncTask(new Runnable() {
136
137                        @Override
138                        public void run() {
139                                hideFrom.hidePlayer(hide);
140                        }
141                });
142
143        }
144
145        /**
146         * Make the player unHide
147         * 
148         * @param unHide
149         * @param unHideFrom
150         */
151        private void uninvisible(final Player unHide, final Player unHideFrom) {
152                if (PermissionManager.hasPerm(unHideFrom, "admincmd.invisible.cansee",
153                                false)) {
154                        return;
155                }
156                ACPluginManager.scheduleSyncTask(new Runnable() {
157                        @Override
158                        public void run() {
159                                unHideFrom.showPlayer(unHide);
160                        }
161                });
162
163        }
164
165        /**
166         * Check if the player is invisible
167         * 
168         * @param player
169         * @return
170         */
171        public boolean hasInvisiblePowers(final Player player) {
172                if (player == null) {
173                        return false;
174                }
175                return invisiblesPlayers.containsKey(player);
176        }
177
178        /**
179         * Make the player vanish
180         * 
181         * @param toVanish
182         *            player to vanish
183         * @param onJoinEvent
184         *            if it's done on join event.
185         */
186        public void vanish(final Player toVanish, final boolean onJoinEvent) {
187                if (!invisiblesPlayers.containsKey(toVanish)) {
188                        invisiblesPlayers.put(toVanish, EMPTY);
189                        if (dynmapAPI != null) {
190                                dynmapAPI.setPlayerVisiblity(toVanish, false);
191                        }
192                        ACPluginManager.getScheduler().runTaskAsynchronously(
193                                        ACPluginManager.getCorePlugin(), new Runnable() {
194
195                                                @Override
196                                                public void run() {
197                                                        for (final Player p : Utils.getOnlinePlayers()) {
198                                                                invisible(toVanish, p);
199                                                        }
200
201                                                }
202                                        });
203                }
204                if (!onJoinEvent && ConfigEnum.FQINVISIBLE.getBoolean()) {
205                        Utils.broadcastFakeQuit(toVanish);
206                }
207
208        }
209
210        /**
211         * return the nb of invisible players
212         * 
213         * @return
214         */
215        public int nbInvisibles() {
216                return invisiblesPlayers.size();
217        }
218
219        /**
220         * Make all invisible player invisible to the new connected player.
221         * 
222         * @param newPlayer
223         *            new connected player.
224         */
225        public void makeInvisibleToPlayer(final Player newPlayer) {
226                ACPluginManager.getScheduler().runTaskAsynchronously(
227                                ACPluginManager.getCorePlugin(), new Runnable() {
228
229                                        @Override
230                                        public void run() {
231                                                for (final Player inv : invisiblesPlayers.keySet()) {
232                                                        invisible(inv, newPlayer);
233                                                }
234
235                                        }
236                                });
237        }
238
239}
