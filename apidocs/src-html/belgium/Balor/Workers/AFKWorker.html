017    package belgium.Balor.Workers;
018    
019    import java.util.concurrent.ConcurrentMap;
020    
021    import org.bukkit.ChatColor;
022    import org.bukkit.command.CommandSender;
023    import org.bukkit.entity.Player;
024    
025    import be.Balor.Manager.Permissions.PermissionManager;
026    import be.Balor.Player.ACPlayer;
027    import be.Balor.Tools.Type;
028    import be.Balor.Tools.Utils;
029    import be.Balor.bukkit.AdminCmd.ACPluginManager;
030    
031    import com.google.common.collect.MapMaker;
032    
033    /**
034     * @author Balor (aka Antoine Aflalo)
035     * 
036     */
037    final public class AFKWorker {
038            private int afkTime = 60000;
039            private int kickTime = 180000;
040            private final ConcurrentMap<Player, Long> playerTimeStamp = new MapMaker().weakKeys().makeMap();
041            private final ConcurrentMap<Player, Object> playersAfk = new MapMaker().weakKeys().makeMap();
042            private final AfkChecker afkChecker;
043            private final KickChecker kickChecker;
044            private static AFKWorker instance = new AFKWorker();
045    
046            /**
047             * 
048             */
049            private AFKWorker() {
050                    afkChecker = new AfkChecker();
051                    kickChecker = new KickChecker();
052            }
053    
054            /**
055             * @return the instance
056             */
057            public static AFKWorker getInstance() {
058                    return instance;
059            }
060    
061            public static AFKWorker createInstance() {
062                    if (instance == null)
063                            instance = new AFKWorker();
064                    return instance;
065            }
066    
067            /**
068             * destroy the instance.
069             */
070            public static void killInstance() {
071                    instance = null;
072            }
073    
074            /**
075             * @return the afkChecker
076             */
077            public AfkChecker getAfkChecker() {
078                    return afkChecker;
079            }
080    
081            /**
082             * @return the kickChecker
083             */
084            public KickChecker getKickChecker() {
085                    return kickChecker;
086            }
087    
088            /**
089             * @param afkTime
090             *            the afkTime to set
091             */
092            public void setAfkTime(final int afkTime) {
093                    if (afkTime > 0)
094                            this.afkTime = afkTime * 1000;
095            }
096    
097            /**
098             * @param kickTime
099             *            the kickTime to set
100             */
101            public void setKickTime(final int kickTime) {
102                    if (afkTime > 0)
103                            this.kickTime = kickTime * 1000 * 60;
104    
105            }
106    
107            /**
108             * Get the number of afk players
109             * 
110             * @return
111             */
112            public int nbAfk() {
113                    return playersAfk.size();
114            }
115    
116            /**
117             * update a player timeStamp (last time the player moved)
118             * 
119             * @param player
120             * @param timestamp
121             */
122            public void updateTimeStamp(final Player player) {
123                    playerTimeStamp.put(player, System.currentTimeMillis());
124            }
125    
126            /**
127             * Remove the player from the check
128             * 
129             * @param player
130             */
131            public void removePlayer(final Player player) {
132                    playersAfk.remove(player);
133                    playerTimeStamp.remove(player);
134            }
135    
136            /**
137             * Set the player AFK
138             * 
139             * @param p
140             */
141            public void setAfk(final Player p) {
142                    setAfk(p, null);
143            }
144    
145            /**
146             * Set the player AFK with the given msg
147             * 
148             * @param p
149             * @param msg
150             */
151            public void setAfk(final Player p, final String msg) {
152                    if (!InvisibleWorker.getInstance().hasInvisiblePowers(p.getName())
153                                    && !ACPlayer.getPlayer(p).hasPower(Type.FAKEQUIT)) {
154                            String afkString = Utils.I18n("afk", "player", Utils.getPlayerName(p, null));
155                            if (afkString != null)
156                                    afkString += (msg != null ? " : " + ChatColor.GOLD + msg : "");
157                            Utils.broadcastMessage(afkString);
158    
159                    }
160                    if (msg == null || (msg != null && msg.isEmpty()))
161                            playersAfk.put(p, Long.valueOf(System.currentTimeMillis()));
162                    else
163                            playersAfk.put(p, msg);
164                    p.setSleepingIgnored(true);
165            }
166    
167            /**
168             * Send the corresponding afk message to the user
169             * 
170             * @param sender
171             * @param buddy
172             */
173            public void sendAfkMessage(final CommandSender sender, final Player buddy) {
174                    if (InvisibleWorker.getInstance().hasInvisiblePowers(buddy.getName())
175                                    || ACPlayer.getPlayer(buddy.getName()).hasPower(Type.FAKEQUIT))
176                            return;
177                    final Object obj = playersAfk.get(buddy);
178                    if (obj != null) {
179                            Utils.sI18n(sender, "noteAfk", "player", Utils.getPlayerName(buddy, sender));
180                            if (obj instanceof String)
181                                    sender.sendMessage((String) obj);
182                            else if (obj instanceof Long) {
183                                    final Long[] time = Utils.getElapsedTime((Long) obj);
184                                    Utils.sI18n(sender, "idleTime", "mins", time[2].toString());
185                            }
186    
187                    }
188            }
189    
190            /**
191             * Set the player Online
192             * 
193             * @param p
194             */
195            public void setOnline(final Player p) {
196                    if (!InvisibleWorker.getInstance().hasInvisiblePowers(p.getName())
197                                    && !ACPlayer.getPlayer(p.getName()).hasPower(Type.FAKEQUIT)) {
198                            final String online = Utils.I18n("online", "player", Utils.getPlayerName(p, null));
199                            if (online != null)
200                                    Utils.broadcastMessage(online);
201                    }
202                    p.setSleepingIgnored(false);
203                    playersAfk.remove(p);
204            }
205    
206            /**
207             * 
208             * @param p
209             * @return if the player is afk
210             */
211            public boolean isAfk(final Player p) {
212                    return playersAfk.containsKey(p);
213            }
214    
215            private class AfkChecker implements Runnable {
216    
217                    /*
218                     * (non-Javadoc)
219                     * 
220                     * @see java.lang.Runnable#run()
221                     */
222                    @Override
223                    public void run() {
224                            final long now = System.currentTimeMillis();
225                            for (final Player p : Utils.getOnlinePlayers()) {
226                                    final Long timeStamp = playerTimeStamp.get(p);
227                                    if (timeStamp != null && !playersAfk.containsKey(p) && (now - timeStamp) >= afkTime)
228                                            setAfk(p);
229                            }
230    
231                    }
232    
233            }
234    
235            private class KickChecker implements Runnable {
236    
237                    /*
238                     * (non-Javadoc)
239                     * 
240                     * @see java.lang.Runnable#run()
241                     */
242                    @Override
243                    public void run() {
244                            final long now = System.currentTimeMillis();
245                            for (final Player p : playersAfk.keySet()) {
246                                    final Long timeStamp = playerTimeStamp.get(p);
247                                    if (timeStamp != null && (now - timeStamp >= kickTime)
248                                                    && !PermissionManager.hasPerm(p, "admincmd.player.noafkkick")) {
249                                            ACPluginManager.scheduleSyncTask(new Runnable() {
250    
251                                                    @Override
252                                                    public void run() {
253                                                            p.kickPlayer(Utils.I18n("afkKick"));
254                                                            playersAfk.remove(p);
255                                                            playerTimeStamp.remove(p);
256    
257                                                    }
258                                            });
259                                    }
260                            }
261    
262                    }
263    
264            }
265    
266    }
