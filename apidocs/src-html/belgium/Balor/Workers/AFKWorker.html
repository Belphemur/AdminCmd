019import java.util.HashMap;
020import java.util.concurrent.ConcurrentMap;

027import be.Balor.Listeners.Events.ACGoAFKEvent;

030import be.Balor.Manager.Permissions.PermissionManager;
031import be.Balor.Player.ACPlayer;

037
038import com.google.common.collect.MapMaker;
039
040/**
041 * @author Balor (aka Antoine Aflalo)
042 * 
043 */
044final public class AFKWorker {
045        private int afkTime = 60000;
046        private int kickTime = 180000;
047        private final ConcurrentMap<Player, Long> playerTimeStamp = new MapMaker()
048                        .weakKeys().makeMap();
049        private final ConcurrentMap<Player, Object> playersAfk = new MapMaker()
050                        .weakKeys().makeMap();
051        private final AfkChecker afkChecker;
052        private final KickChecker kickChecker;
053        private static AFKWorker instance = new AFKWorker();
054
055        /**
056         *
057         */
058        private AFKWorker() {
059                afkChecker = new AfkChecker();
060                kickChecker = new KickChecker();
061        }
062
063        /**
064         * @return the instance
065         */
066        public static AFKWorker getInstance() {
067                return instance;
068        }
069
070        public static AFKWorker createInstance() {
071                if (instance == null) {
072                        instance = new AFKWorker();
073                }
074                return instance;
075        }
076
077        /**
078         * destroy the instance.
079         */
080        public static void killInstance() {
081                instance = null;
082        }
083
084        /**
085         * @return the afkChecker
086         */
087        public AfkChecker getAfkChecker() {
088                return afkChecker;
089        }
090
091        /**
092         * @return the kickChecker
093         */
094        public KickChecker getKickChecker() {
095                return kickChecker;
096        }
097
098        /**
099         * @param afkTime
100         *            the afkTime to set
101         */
102        public void setAfkTime(final int afkTime) {
103                if (afkTime > 0) {
104                        this.afkTime = afkTime * 1000;
105                }
106        }
107
108        /**
109         * @param kickTime
110         *            the kickTime to set
111         */
112        public void setKickTime(final int kickTime) {
113                if (afkTime > 0) {
114                        this.kickTime = kickTime * 1000 * 60;
115                }
116
117        }
118
119        /**
120         * Get the number of afk players
121         * 
122         * @return
123         */
124        public int nbAfk() {
125                return playersAfk.size();
126        }
127
128        /**
129         * update a player timeStamp (last time the player moved)
130         * 
131         * @param player
132         * @param timestamp
133         */
134        public void updateTimeStamp(final Player player) {
135                playerTimeStamp.put(player, System.currentTimeMillis());
136        }
137
138        /**
139         * Remove the player from the check
140         * 
141         * @param player
142         */
143        public void removePlayer(final Player player) {
144                playersAfk.remove(player);
145                playerTimeStamp.remove(player);
146        }
147
148        /**
149         * Set player afk with the wanted reason
150         * 
151         * @param p
152         * @param reason
153         */
154        public void setAfk(final Player p, final ACGoAFKEvent.Reason reason) {
155                setAfk(p, null, reason);
156        }
157
158        /**
159         * Set the player AFK with the given msg
160         * 
161         * @param p
162         * @param message
163         */
164        public void setAfk(final Player p, final String message,
165                        final ACGoAFKEvent.Reason reason) {
166                final ACGoAFKEvent event = new ACGoAFKEvent(p, reason, message);
167                Bukkit.getPluginManager().callEvent(event);
168                if (event.isCancelled()) {
169                        return;
170                }
171                final String msg = event.getMessage();
172                if (!InvisibleWorker.getInstance().hasInvisiblePowers(p)
173                                && !ACPlayer.getPlayer(p).hasPower(Type.FAKEQUIT)) {
174                        String afkString = Utils.I18n("afk", "player",
175                                        Utils.getPlayerName(p, null));
176                        if (afkString != null) {
177                                afkString += (msg != null ? " : " + ChatColor.GOLD + msg : "");
178                        }
179                        Utils.broadcastMessage(afkString);
180
181                }
182                if (msg == null || (msg != null && msg.isEmpty())) {
183                        playersAfk.put(p, Long.valueOf(System.currentTimeMillis()));
184                } else {
185                        playersAfk.put(p, msg);
186                }
187                p.setSleepingIgnored(true);
188        }
189
190        /**
191         * Send the corresponding afk message to the user
192         * 
193         * @param sender
194         * @param buddy
195         */
196        public void sendAfkMessage(final CommandSender sender, final Player buddy) {
197                if (InvisibleWorker.getInstance().hasInvisiblePowers(buddy)
198                                || ACPlayer.getPlayer(buddy.getName()).hasPower(Type.FAKEQUIT)) {
199                        return;
200                }
201                final Object obj = playersAfk.get(buddy);
202                if (obj != null) {
203                        Utils.sI18n(sender, "noteAfk", "player",
204                                        Utils.getPlayerName(buddy, sender));
205                        if (obj instanceof String) {
206                                sender.sendMessage((String) obj);
207                        } else if (obj instanceof Long) {
208                                final Long[] time = Utils.getElapsedTime((Long) obj);
209                                Utils.sI18n(sender, "idleTime", "mins", time[2].toString());
210                        }
211
212                }
213        }
214
215        /**
216         * Set the player Online
217         * 
218         * @param p
219         */
220        public void setOnline(final Player p) {
221                if (!InvisibleWorker.getInstance().hasInvisiblePowers(p)
222                                && !ACPlayer.getPlayer(p.getName()).hasPower(Type.FAKEQUIT)) {
223                        final String online = Utils.I18n("online", "player",
224                                        Utils.getPlayerName(p, null));
225                        if (online != null) {
226                                Utils.broadcastMessage(online);
227                        }
228                }
229                p.setSleepingIgnored(false);
230                playersAfk.remove(p);
231                Bukkit.getPluginManager().callEvent(new ACReturnedAFKEvent(p));
232        }
233
234        /**
235         * 
236         * @param p
237         * @return if the player is afk
238         */
239        public boolean isAfk(final Player p) {
240                return playersAfk.containsKey(p);
241        }
242
243        private class AfkChecker implements Runnable {
244
245                /*
246                 * (non-Javadoc)
247                 * 
248                 * @see java.lang.Runnable#run()
249                 */
250                @Override
251                public void run() {
252                        final long now = System.currentTimeMillis();
253                        for (final Player p : Utils.getOnlinePlayers()) {
254                                final Long timeStamp = playerTimeStamp.get(p);
255                                if (timeStamp != null && !playersAfk.containsKey(p)
256                                                && (now - timeStamp) >= afkTime) {
257                                        setAfk(p, Reason.AUTO);
258                                }
259                        }
260
261                }
262
263        }
264
265        private class KickChecker implements Runnable {
266
267                /*
268                 * (non-Javadoc)
269                 * 
270                 * @see java.lang.Runnable#run()
271                 */
272                @Override
273                public void run() {
274                        final long now = System.currentTimeMillis();
275                        for (final Player p : playersAfk.keySet()) {
276                                final Long timeStamp = playerTimeStamp.get(p);
277                                if (timeStamp != null
278                                                && (now - timeStamp >= kickTime)
279                                                && !PermissionManager.hasPerm(p,
280                                                                "admincmd.player.noafkkick")) {
281                                        ACPluginManager.scheduleSyncTask(new Runnable() {
282
283                                                @Override
284                                                public void run() {
285                                                        final HashMap<String, String> replace = new HashMap<String, String>();
286                                                        replace.put("player", Utils.getPlayerName(p));
287                                                        p.kickPlayer(Utils.I18n("afkKick"));
288                                                        final String msg = LocaleHelper.AFK_KICK_BCAST
289                                                                        .getLocale(replace);
290                                                        Utils.broadcastMessage(msg);
291                                                        ACLogger.info(msg);
292                                                        playersAfk.remove(p);
293                                                        playerTimeStamp.remove(p);
294
295                                                }
296                                        });
297                                }
298                        }
299
300                }
301
302        }
