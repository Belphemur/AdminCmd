45  	@Before
46  	public void setUp() throws Exception {
47  		ExtendedConfiguration.setClassLoader(this.getClass().getClassLoader());
48  		file = new File("test.yml");
49  		final ExtendedConfiguration conf = ExtendedConfiguration.loadConfiguration(file);
50  		conf.add("test", "blah");
51  		conf.createSection("yatta").set("test", "blah");
52  		conf.save();
53  
54  	}
55  
56  	@Test
57  	public void loadTest() {
58  		final ExtendedConfiguration conf = ExtendedConfiguration.loadConfiguration(file);
59  		assertEquals("blah", conf.get("test"));
60  		assertFalse(conf.add("test", "test"));
61  		assertEquals("blah", conf.get("test"));
62  	}
63  
64  	@Test
65  	public void serializeBukkitTest() throws IOException, InvalidConfigurationException {
66  		final ExtendedConfiguration conf = ExtendedConfiguration.loadConfiguration(file);
67  		final ItemStack test = new ItemStack(Material.WATER, 10);
68  		conf.set("serial.item", test);
69  		conf.save();
70  		conf.reload();
71  		assertEquals(new ItemStack(Material.WATER, 10), conf.get("serial.item"));
72  	}
73  
74  	@Test
75  	public void serializeAdminCmdTest() throws IOException, InvalidConfigurationException {
76  		final ExtendedConfiguration conf = ExtendedConfiguration.loadConfiguration(file);
77  		conf.set("serial.banPlayer", new BannedPlayer("Test", "testing"));
78  		conf.save();
79  		conf.reload();
80  		assertEquals("Test", ((BannedPlayer) conf.get("serial.banPlayer")).getPlayer());
81  	}
82  
83  	@After
84  	public void tearDown() throws Exception {
85  		file.deleteOnExit();
86  	}
