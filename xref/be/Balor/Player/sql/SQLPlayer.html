104 	private void init() {
105 		synchronized (GET_LASTLOC) {
106 			try {
107 				GET_LASTLOC.clearParameters();
108 				GET_LASTLOC.setLong(1, id);
109 				ResultSet rs;
110 				synchronized (GET_LASTLOC.getConnection()) {
111 					rs = GET_LASTLOC.executeQuery();
112 				}
113 				if (rs.next()) {
114 					final String worldName = rs.getString("world");
115 					if (worldName != null && !worldName.isEmpty()) {
116 						World world = Bukkit.getWorld(worldName);
117 						if (world == null) {
118 							world = ACWorld.getWorld(world).getHandle();
119 						}
120 						if (world != null) {
121 							lastLoc = new Location(world, rs.getDouble("x"),
122 									rs.getDouble("y"), rs.getDouble("z"),
123 									rs.getFloat("yaw"), rs.getFloat("pitch"));
124 						} else {
125 							ACLogger.warning("The World " + worldName
126 									+ " is not loaded");
127 						}
128 					}
129 				}
130 			} catch (final SQLException e) {
131 				ACLogger.severe(
132 						"Problem with getting last location from the DB", e);
133 			}
134 		}
135 		synchronized (GET_HOMES) {
136 			try {
137 				GET_HOMES.clearParameters();
138 				GET_HOMES.setLong(1, id);
139 				ResultSet rs;
140 				synchronized (GET_HOMES.getConnection()) {
141 					rs = GET_HOMES.executeQuery();
142 				}
143 				while (rs.next()) {
144 					final String worldName = rs.getString("world");
145 					World world = Bukkit.getWorld(worldName);
146 					if (world == null) {
147 						world = ACWorld.getWorld(world).getHandle();
148 					}
149 					if (world != null) {
150 						homes.put(
151 								rs.getString("name"),
152 								new Location(world, rs.getDouble("x"), rs
153 										.getDouble("y"), rs.getDouble("z"), rs
154 										.getFloat("yaw"), rs.getFloat("pitch")));
155 					} else {
156 						ACLogger.warning("The World " + worldName
157 								+ " is not loaded");
158 					}
159 				}
160 				rs.close();
161 			} catch (final SQLException e) {
162 				ACLogger.severe("Problem with getting homes from the DB", e);
163 			}
164 
165 		}
166 		synchronized (GET_POWERS) {
167 			try {
168 				GET_POWERS.clearParameters();
169 				GET_POWERS.setLong(1, id);
170 				ResultSet rs;
171 				synchronized (GET_POWERS.getConnection()) {
172 					rs = GET_POWERS.executeQuery();
173 				}
174 				while (rs.next()) {
175 					final String powerName = rs.getString("key");
176 					final Type power = Type.matchType(powerName);
177 					synchronized (SQLObjectContainer.yaml) {
178 						if (power == null) {
179 							customPowers.put(powerName, SQLObjectContainer.yaml
180 									.load(rs.getString("info")));
181 						} else {
182 							powers.put(power, SQLObjectContainer.yaml.load(rs
183 									.getString("info")));
184 						}
185 					}
186 				}
187 				rs.close();
188 			} catch (final SQLException e) {
189 				ACLogger.severe("Problem with getting powers from the DB", e);
190 			}
191 
192 		}
193 		synchronized (GET_INFOS) {
194 			try {
195 				GET_INFOS.clearParameters();
196 				GET_INFOS.setLong(1, id);
197 				ResultSet rs;
198 				synchronized (GET_INFOS.getConnection()) {
199 					rs = GET_INFOS.executeQuery();
200 				}
201 				while (rs.next()) {
202 					synchronized (SQLObjectContainer.yaml) {
203 						infos.put(rs.getString("key"), SQLObjectContainer.yaml
204 								.load(rs.getString("info")));
205 					}
206 				}
207 				rs.close();
208 			} catch (final SQLException e) {
209 				ACLogger.severe(
210 						"Problem with getting informations from the DB", e);
211 			}
212 
213 		}
214 		synchronized (GET_KIT_USES) {
215 			try {
216 				GET_KIT_USES.clearParameters();
217 				GET_KIT_USES.setLong(1, id);
218 				ResultSet rs;
219 				synchronized (GET_KIT_USES.getConnection()) {
220 					rs = GET_KIT_USES.executeQuery();
221 				}
222 				while (rs.next()) {
223 					kitUses.put(rs.getString("kit"), rs.getLong("use"));
224 				}
225 				rs.close();
226 			} catch (final SQLException e) {
227 				ACLogger.severe("Problem with getting kit uses from the DB", e);
228 			}
229 		}
230 	}

239 	/***
240 	 * To Schedule the Async task
241 	 */
242 	public static void scheduleAsyncSave() {
243 		if (ACPluginManager.getScheduler().isCurrentlyRunning(prepStmtTaskID)
244 				|| ACPluginManager.getScheduler().isQueued(prepStmtTaskID)) {
245 			return;
246 		}
247 		final int delay = ConfigEnum.E_PST_DELAY.getInt() >= 30 ? ConfigEnum.E_PST_DELAY
248 				.getInt() : 10;
249 		prepStmtTaskID = ACPluginManager
250 				.getScheduler()
251 				.runTaskTimerAsynchronously(
252 						ACHelper.getInstance().getCoreInstance(),
253 						PREP_STMT_TASK, Utils.secInTick * 2 * delay,
254 						Utils.secInTick * delay).getTaskId();
255 		DebugLog.INSTANCE.info("IO Save RepeatingTask created : "
256 				+ prepStmtTaskID);
257 	}
258 
259 	/***
260 	 * To stop the saving task.
261 	 */
262 	public static void stopSavingTask() {
263 		if (!ACPluginManager.getScheduler().isCurrentlyRunning(prepStmtTaskID)
264 				&& !ACPluginManager.getScheduler().isQueued(prepStmtTaskID)) {
265 			return;
266 		}
267 		ACPluginManager.getScheduler().cancelTask(prepStmtTaskID);
268 	}

605 	 * (Non javadoc)
606 	 * 
607 	 * @see be.Balor.Player.ACPlayer#removeAllSuperPower()
608 	 */
609 	@Override
610 	public void removeAllSuperPower() {
611 		boolean found = false;
612 		for (final Type power : powers.keySet()) {
613 			if (power.getCategory() != Type.Category.SUPER_POWER) {
614 				continue;
615 			}
616 			powers.remove(power);
617 			found = true;
618 			if (power != Type.FLY) {
619 				continue;
620 			}
621 			if (handler == null) {
622 				continue;
623 			}
624 			handler.setFlying(false);
625 			handler.setAllowFlight(false);
626 		}
627 		if (found) {
628 			final PreparedStatement deleteSuperPowers = Database.DATABASE
629 					.prepare("delete FROM `ac_powers` WHERE `player_id`=? AND `category`='"
630 							+ Type.Category.SUPER_POWER.name() + "'");
631 			try {
632 				deleteSuperPowers.clearParameters();
633 				deleteSuperPowers.setLong(1, id);
