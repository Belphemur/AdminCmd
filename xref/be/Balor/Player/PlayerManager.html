83  	/***
84  	 * Add a new player
85  	 * 
86  	 * @param player
87  	 */
88  	private synchronized boolean addPlayer(final ACPlayer player) {
89  		final String name = player.getName();
90  		if (name == null) {
91  			throw new NullPointerException();
92  		}
93  
94  		final ACPlayer ref = players.get(name);
95  		if (ref != null) {
96  			return false;
97  		}
98  		players.put(name, player);
99  		return true;
100 	}

160 	public List<ACPlayer> getExistingPlayers() {
161 		final ArrayList<ACPlayer> list = new ArrayList<ACPlayer>();
162 		for (final String name : playerFactory.getExistingPlayers()) {
163 			try {
164 				final ACPlayer player = demandACPlayer(name);
165 				if (!(player instanceof EmptyPlayer)) {
166 					list.add(player);
167 				}
168 			} catch (final Exception e) {
169 				DebugLog.INSTANCE.log(Level.WARNING,
170 						"Problem with instancing ACPlayer : " + name, e);
171 			}
172 		}
173 		return list;
174 	}

212 	synchronized ACPlayer demandACPlayer(final String name) {
213 		if (name == null) {
214 			return getPlayer("serverConsole");
215 		}
216 		ACPlayer result = getPlayer(name);
217 		if (result == null) {
218 			result = playerFactory.createPlayer(name);
219 			addPlayer(result);
220 			result = getPlayer(name);
221 		} else if (result instanceof EmptyPlayer) {
222 			final ACPlayer tmp = playerFactory.createPlayer(name);
223 			if (tmp instanceof EmptyPlayer) {
224 				return result;
225 			}
226 			players.remove(name);
227 			onlinePlayers.remove(result);
228 			result = tmp;
229 			addPlayer(result);
230 			result = getPlayer(name);
231 		}
232 		return result;
233 	}
234 
235 	synchronized ACPlayer demandACPlayer(final Player player) {
236 		if (player == null) {
237 			return getPlayer("serverConsole");
238 		}
239 		final String playerName = player.getName();
240 		ACPlayer result = getPlayer(playerName);
241 		if (result == null) {
242 			result = playerFactory.createPlayer(player);
243 			addPlayer(result);
244 			result = getPlayer(playerName);
245 		} else if (result instanceof EmptyPlayer) {
246 			final ACPlayer tmp = playerFactory.createPlayer(playerName);
247 			if (tmp instanceof EmptyPlayer) {
248 				return result;
249 			}
250 			players.remove(playerName);
251 			onlinePlayers.remove(result);
252 			result = tmp;
253 			addPlayer(result);
254 			result = getPlayer(playerName);
255 		}
256 		return result;
257 	}

259 	/***
260 	 * @param newFactory
261 	 */
262 	synchronized void afterFactoryConversion(final IPlayerFactory newFactory) {
263 		DebugLog.INSTANCE.info("After Conversion launched");
264 		final Map<ACPlayer, Boolean> onlineCopy = new HashMap<ACPlayer, Boolean>();
265 		onlineCopy.putAll(onlinePlayers);
266 		ACPluginManager.scheduleSyncTask(new Runnable() {
267 
268 			@Override
269 			public void run() {
270 				ACLogger.info("Update status of online players : "
271 						+ onlineCopy.size());
272 				for (final ACPlayer p : onlineCopy.keySet()) {
273 					if (!(p instanceof EmptyPlayer)) {
274 						new PlayerConvertTask(newFactory, p).run();
275 					}
276 				}
277 				ACLogger.info("Players conversion Finished");
278 
279 			}
280 		});
281 		this.onlinePlayers.clear();
282 		this.players.clear();
283 		this.playerFactory = newFactory;
284 		DebugLog.INSTANCE.info("Player Conversion Finished (ASYNC)");
285 
286 	}
