106 	/***
107 	 * To Schedule the Async task
108 	 */
109 	public static void scheduleAsyncSave() {
110 		if (ACPluginManager.getScheduler().isCurrentlyRunning(ioStackTaskId)
111 				|| ACPluginManager.getScheduler().isQueued(ioStackTaskId))
112 			return;
113 		final int delay = ConfigEnum.WDELAY.getInt() >= 30 ? ConfigEnum.WDELAY.getInt() : 30;
114 		ioStackTaskId = ACPluginManager.getScheduler().scheduleAsyncRepeatingTask(
115 				ACHelper.getInstance().getCoreInstance(), IOSAVET_TASK, 20 * 60, 20 * delay);
116 		DebugLog.INSTANCE.info("IO Save RepeatingTask created : " + ioStackTaskId);
117 	}
118 
119 	/***
120 	 * To stop the saving task.
121 	 */
122 	public static void stopSavingTask() {
123 		if (!ACPluginManager.getScheduler().isCurrentlyRunning(ioStackTaskId)
124 				&& !ACPluginManager.getScheduler().isQueued(ioStackTaskId))
125 			return;
126 		ACPluginManager.getScheduler().cancelTask(ioStackTaskId);
127 	}

157 	@Override
158 	public Location getHome(final String home) {
159 		final ConfigurationSection homeSection = homes.getConfigurationSection(home);
160 		if (homeSection == null) {
161 			final String found = Str.matchString(homes.getKeys(false), home);
162 			if (found == null)
163 				return null;
164 			return getLocation(homes.getConfigurationSection(found));
165 		} else
166 			return getLocation(homeSection);
167 	}

241 	private Location getLocation(final ConfigurationSection node) throws WorldNotLoaded {
242 		final World w = ACPluginManager.getServer().getWorld(node.getString("world"));
243 		if (w != null)
244 			return new Location(w, node.getDouble("x", 0), node.getDouble("y", 0), node.getDouble(
245 					"z", 0), Float.parseFloat(node.getString("yaw")), Float.parseFloat(node
246 					.getString("pitch")));
247 		else
248 			throw new WorldNotLoaded(node.getString("world"));
249 	}
