52  	/***
53  	 * @param name
54  	 */
55  	public FileWorld(final World world, final String directory) {
56  		super(world);
57  		final File wFile = new File(directory, world.getName() + ".yml");
58  		try {
59  			Files.createParentDirs(wFile);
60  		} catch (final IOException e) {
61  		}
62  		datas = ExtendedConfiguration.loadConfiguration(wFile);
63  
64  		warps = datas.addSection("warps");
65  		informations = datas.addSection("informations");
66  		mobLimits = informations.addSection("mobLimits");
67  		forceSave();
68  	}

86  	@Override
87  	public Location getSpawn() throws WorldNotLoaded {
88  		final Object spawn = datas.get("spawn");
89  		if (spawn == null)
90  			return handler.getSpawnLocation();
91  		else if (spawn instanceof SimpleLocation)
92  			return ((SimpleLocation) spawn).getLocation();
93  		else
94  			return handler.getSpawnLocation();

140 	/*
141 	 * (non-Javadoc)
142 	 * 
143 	 * @see be.Balor.World.ACWorld#getWarp(java.lang.String)
144 	 */
145 	@Override
146 	public Warp getWarp(final String name) throws WorldNotLoaded, IllegalArgumentException {
147 		if (name == null || (name != null && name.isEmpty()))
148 			throw new IllegalArgumentException("Name can't be null or Empty");
149 		Object warp = warps.get(name);
150 		String warpName = name;
151 		if (warp == null) {
152 			warpName = Str.matchString(warps.getKeys(false), name);
153 			if (warpName == null)
154 				return null;
155 			warp = warps.get(warpName);
156 		}
157 
158 		return new Warp(warpName, ((SimpleLocation) warp).getLocation());
159 	}

241 	@Override
242 	public Map<String, String> getInformations() {
243 		final TreeMap<String, String> result = new TreeMap<String, String>();
244 		for (final Entry<String, Object> entry : informations.getValues(false).entrySet())
245 			result.put(entry.getKey(), entry.getValue().toString());
246 		for (final Entry<String, Object> entry : mobLimits.getValues(false).entrySet())
247 			result.put("Limit on " + entry.getKey(), entry.getValue().toString());
248 		return result;
249 	}
