35  public class WorldManager {
36  	private final ConcurrentMap<String, ACWorld> worlds = new MapMaker().makeMap();
37  	private IWorldFactory worldFactory;
38  	private static WorldManager instance = new WorldManager();

53  	/***
54  	 * Add a new world
55  	 * 
56  	 * @param world
57  	 */
58  	private synchronized boolean addWorld(final ACWorld world) {
59  		final String name = world.getName();
60  		if (name == null) {
61  			throw new NullPointerException();
62  		}
63  
64  		final ACWorld ref = worlds.get(name);
65  		if (ref != null)
66  			return false;
67  		worlds.put(name, world);
68  		return true;
69  	}
70  
71  	/***
72  	 * @param worldFactory
73  	 *            the worldFactory to set
74  	 */
75  	public void setWorldFactory(final IWorldFactory worldFactory) {
76  		if (this.worldFactory == null)
77  			this.worldFactory = worldFactory;
78  	}
79  
80  	/***
81  	 * To convert the ACWorld using an another factory
82  	 * 
83  	 * @param factory
84  	 */
85  	public void convertFactory(final IWorldFactory factory) {
86  		final Map<String, ACWorld> newWorlds = new HashMap<String, ACWorld>();
87  		for (final Entry<String, ACWorld> entry : worlds.entrySet()) {
88  			final ACWorld newWorld = factory.createWorld(entry.getKey());
89  			final ACWorld oldWorld = entry.getValue();
90  			newWorld.setDifficulty(oldWorld.getDifficulty());
91  			newWorld.setSpawn(oldWorld.getSpawn());
92  			for (final String mob : oldWorld.getMobLimitList())
93  				newWorld.setMobLimit(mob, oldWorld.getMobLimit(mob));
94  			for (final Entry<String, String> info : oldWorld.getInformations().entrySet())
95  				newWorld.setInformation(info.getKey(), oldWorld.getInformation(info.getKey())
96  						.getObj());
97  			newWorlds.put(newWorld.getName(), newWorld);
98  		}
99  		worlds.clear();
100 		worlds.putAll(newWorlds);
101 		this.worldFactory = factory;
102 	}
103 
104 	ACWorld demandACWorld(final String name) throws WorldNotLoaded {
105 		ACWorld result = worlds.get(name);
106 		if (result == null) {
107 			final String found = Str.matchString(worlds.keySet(), name);
108 			if (found != null)
109 				return worlds.get(found);
110 			result = worldFactory.createWorld(name);
111 			addWorld(result);
112 			result = worlds.get(name);
113 		}
114 		return result;
115 	}
116 
117 	/***
118 	 * Getting all the WarpsName using the world name as prefix
119 	 * 
120 	 * @return
121 	 */
122 	public Set<String> getAllWarpList() {
123 		final Set<String> warps = new HashSet<String>();
124 		for (final ACWorld world : worlds.values())
125 			for (final String warp : world.getWarpList())
126 				warps.add(world.getName() + ":" + warp);
127 		return warps;
128 
129 	}
