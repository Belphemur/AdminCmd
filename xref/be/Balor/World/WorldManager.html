36  public class WorldManager {
37  	private final ConcurrentMap<String, ACWorld> worlds = new MapMaker()
38  			.makeMap();
39  	private IWorldFactory worldFactory;
40  	private static final WorldManager INSTANCE = new WorldManager();
41  
42  	/***
43  	 * 
44  	 */
45  	private WorldManager() {
46  	}
47  
48  	/***
49  	 * @return the instance
50  	 */
51  	public static WorldManager getInstance() {
52  		return INSTANCE;
53  	}
54  
55  	/***
56  	 * Add a new world
57  	 * 
58  	 * @param world
59  	 */
60  	private synchronized boolean addWorld(final ACWorld world) {
61  		final String name = world.getName();
62  		if (name == null) {
63  			throw new NullPointerException();
64  		}
65  
66  		final ACWorld ref = worlds.get(name);
67  		if (ref != null) {
68  			return false;
69  		}
70  		worlds.put(name, world);
71  		return true;
72  	}
73  
74  	/***
75  	 * @param worldFactory
76  	 *            the worldFactory to set
77  	 */
78  	public void setWorldFactory(final IWorldFactory worldFactory) {
79  		if (this.worldFactory == null) {
80  			this.worldFactory = worldFactory;
81  		}
82  	}
83  
84  	/***
85  	 * To convert the ACWorld using an another factory
86  	 * 
87  	 * @param factory
88  	 */
89  	public void convertFactory(final IWorldFactory factory) {
90  		new WorldConverter(worldFactory, factory).convert();
91  		DebugLog.INSTANCE.info("Setting new Factory");
92  		this.worldFactory = factory;
93  	}
94  
95  	ACWorld demandACWorld(final String name) throws WorldNotLoaded {
96  		ACWorld result = worlds.get(name);
97  		if (result == null) {
98  			final String found = Str.matchString(worlds.keySet(), name);
99  			if (found != null) {
100 				return worlds.get(found);
101 			}
102 			result = worldFactory.createWorld(name);
103 			addWorld(result);
104 			result = worlds.get(name);
105 		}
106 		return result;
107 	}
108 
109 	ACWorld demandACWorld(final World world) {
110 		final String name = world.getName();
111 		ACWorld result = worlds.get(name);
112 		if (result == null) {
113 			result = worldFactory.createWorld(world);
114 			addWorld(result);
115 			result = worlds.get(name);
116 		}
117 		return result;
118 	}
119 
120 	/***
121 	 * Getting all the WarpsName using the world name as prefix
122 	 * 
123 	 * @return
124 	 */
125 	public Set<String> getAllWarpList() {
126 		final Set<String> warps = new HashSet<String>();
127 		for (final ACWorld world : worlds.values()) {
128 			for (final String warp : world.getWarpList()) {
129 				warps.add(world.getName() + ":" + warp);
130 			}
131 		}
132 		return warps;
133 
134 	}
