95  		public void run() {
96  			try {
97  				processCmd();
98  				plotters.get(acc.getCommandClass()).increment();
99  			} catch (final ConcurrentModificationException cme) {
100 				ACPluginManager.getScheduler().scheduleSyncDelayedTask(
101 						corePlugin, new SyncCommand(acc));
102 			} catch (final WorldNotLoaded e) {
103 				String message = e.getMessage();
104 				if (message == null || message == "") {
105 					message = "This world is not loaded!";
106 				}
107 				final HashMap<String, String> replace = new HashMap<String, String>();
108 				replace.put("message", message);
109 				LocaleHelper.WORLD_NOT_LOADED.sendLocale(acc.getSender(),
110 						replace);
111 			} catch (final PlayerNotFound e) {
112 				e.getSender().sendMessage(e.getMessage());
113 			} catch (final ActionNotPermitedException e) {
114 				e.sendMessage();
115 			} catch (final NumberFormatException e) {
116 				Utils.sI18n(acc.getSender(), "NaN", "number",
117 						e.getLocalizedMessage());
118 			} catch (final Throwable t) {
119 				ACLogger.severe(acc.debug(), t);
120 				Utils.broadcastMessage("[AdminCmd] " + acc.debug());
121 			}
122 
123 		}

221 	/***
222 	 * Check if some alias have been disabled for the registered commands
223 	 */
224 	public void checkAlias(final AbstractAdminCmdPlugin plugin) {
225 		if (ConfigEnum.VERBOSE.getBoolean()) {
226 			final Map<String, Command> commands = pluginCommands.get(plugin);
227 			if (commands != null) {
228 				for (final String cmdName : commands.keySet()) {
229 					final Command cmd = commands.get(cmdName);
230 					if (corePlugin.getCommand(cmd.getName()) != null) {
231 						final List<String> aliasesList = new ArrayList<String>(
232 								cmd.getAliases());
233 						aliasesList.removeAll(corePlugin.getCommand(
234 								cmd.getName()).getAliases());
235 						aliasesList.removeAll(prioritizedCommands);
236 						String aliases = "";
237 						for (final String alias : aliasesList) {
238 							aliases += alias + ", ";
239 						}
240 						if (!aliases.isEmpty()
241 								&& ConfigEnum.VERBOSE.getBoolean()) {
242 							Logger.getLogger("Minecraft").info(
243 									"[" + corePlugin.getDescription().getName()
244 											+ "] Disabled Alias(es) for "
245 											+ cmd.getName() + " : " + aliases);
246 						}
247 					}
248 				}
249 			}
250 		}
251 	}

253 	/***
254 	 * Check the command if it have alias, prioritized or disabled.
255 	 * 
256 	 * @param command
257 	 * @throws CommandDisabled
258 	 */
259 	private void checkCommand(final CoreCommand command) throws CommandDisabled {
260 		final Map<String, Command> commands = pluginCommands.get(command
261 				.getPlugin());
262 		if (commands != null) {
263 			for (final String alias : commands.get(command.getCmdName())
264 					.getAliases()) {
265 				if (disabledCommands.contains(alias)) {
266 					throw new CommandDisabled(
267 							"Command "
268 									+ command.getCmdName()
269 									+ " selected to be disabled in the configuration file.",
270 							command);
271 				}
272 				if (prioritizedCommands.contains(alias)) {
273 					final CommandAlias cmd = new CommandAlias(
274 							command.getCmdName(), alias, "");
275 					cmd.setCmd(command);
276 					commandsAliasReplacer.put(alias, cmd);
277 				}
278 				final Set<CommandAlias> commandAliases = commandsAlias
279 						.get(alias);
280 				if (commandAliases != null) {
281 					for (final CommandAlias commandAlias : commandAliases) {
282 						commandAlias.setCmd(command);
283 					}
284 				}
285 			}
286 		}

289 	/***
290 	 * Used to execute ACCommands
291 	 * 
292 	 * @param sender
293 	 * @param cmd
294 	 * @param args
295 	 * @return
296 	 */
297 	private boolean executeCommand(final CommandSender sender,
298 			final CoreCommand cmd, final String[] args) {
299 		ACCommandContainer container = null;
300 		try {
301 			if (!cmd.permissionCheck(sender)) {
302 				return true;
303 			}
304 			if (!cmd.argsCheck(args)) {
305 				if (!HelpLister.getInstance().displayExactCommandHelp(sender,
306 						"AdminCmd", cmd.getCmdName())) {
307 					return false;
308 				}
309 				return true;
310 			}
311 			container = new ACCommandContainer(sender, cmd, args);
312 			/*
313 			 * if (cmd.getCmdName().equals("bal_replace") ||
314 			 * cmd.getCmdName().equals("bal_extinguish"))
315 			 * corePlugin.getServer().getScheduler()
316 			 * .scheduleSyncDelayedTask(corePlugin, new SyncCommand(container));
317 			 * else
318 			 */
319 			threads.execute(new NormalCommand(container));
320 			if (!cmd.getCmdName().equals("bal_repeat")) {
321 				if (Utils.isPlayer(sender, false)) {
322 					ACPlayer.getPlayer(((Player) sender)).setLastCmd(container);
323 				} else {
324 					ACPlayer.getPlayer("serverConsole").setLastCmd(container);
325 				}
326 			}
327 			return true;
328 		} catch (final Throwable t) {
329 			ACLogger.severe(container != null ? container.debug()
330 					: "The container is null", t);
331 			Utils.broadcastMessage("[AdminCmd] " + container != null ? container
332 					.debug()
333 					: cmd.getCmdName()
334 							+ " throw an Exception please report the log in a ticket : http://bug.admincmd.com/");
335 			return false;
336 		}
337 	}

353 	/***
354 	 * Execute on the new player the commands found in the Commands.yml.<br />
355 	 * First look as a simple bukkit command, then in the alias defined by the
356 	 * user.
357 	 * 
358 	 * @param player
359 	 *            player that will execute the command.
360 	 */
361 	public void executeFirstJoinCommands(final Player player) {
362 		for (final String command : commandsOnJoin) {
363 			final String[] split = command.split("//s+");
364 			if (split.length == 0) {
365 				continue;
366 			}
367 			final String name = split[0].substring(1).toLowerCase();
368 			final PluginCommand pCmd = Bukkit.getPluginCommand(name);
369 			if (pCmd != null) {
370 				pCmd.execute(player, name,
371 						Utils.Arrays_copyOfRange(split, 1, split.length));
372 			} else {
373 				processCommandString(player, command);
374 			}
375 		}
376 
377 	}

379 	public boolean processCommandString(final CommandSender sender,
380 			final String command) {
381 		final String[] split = command.split("//s+");
382 		if (split.length == 0) {
383 			return false;
384 		}
385 		final String cmdName = split[0].substring(1).toLowerCase();
386 		final CommandAlias cmdAlias = commandsAliasReplacer.get(cmdName);
387 		if (cmdAlias != null) {
388 			if (ConfigEnum.VERBOSE.getBoolean()) {
389 				ACLogger.info("Command " + cmdName + " intercepted for "
390 						+ cmdAlias);
391 			}
392 			final String[] cmdArgsArray = cmdAlias.processArguments(Utils
393 					.Arrays_copyOfRange(split, 1, split.length));
394 			final CoreCommand coreCmd = cmdAlias.getCmd();
395 			if (!coreCmd.argsCheck(cmdArgsArray)) {
396 				if (!HelpLister.getInstance().displayExactCommandHelp(sender,
397 						"AdminCmd", coreCmd.getCmdName())) {
398 					sender.sendMessage(coreCmd.getPluginCommand().getUsage()
399 							.replace("<command>", cmdName));
400 				}
401 				return true;
402 			}
403 			return executeCommand(sender, coreCmd, cmdArgsArray);
404 		}
405 		return false;
406 	}

422 	/***
423 	 * Register command
424 	 * 
425 	 * @param clazz
426 	 */
427 	public boolean registerCommand(final Class<? extends CoreCommand> clazz) {
428 		DebugLog.beginInfo("Register command " + clazz.getName());
429 		try {
430 			registerCommand0(clazz);
431 			final IncrementalPlotter plotter = new ClassPlotter(clazz);
432 			graph.addPlotter(plotter);
433 			plotters.put(clazz, plotter);
434 		} catch (final InstantiationException e) {
435 			e.printStackTrace();
436 			return false;
437 		} catch (final IllegalAccessException e) {
438 			e.printStackTrace();
439 			return false;
440 		} catch (final CommandDisabled e) {
441 			disableCommand(e);
442 			return false;
443 		} catch (final CommandAlreadyExist e) {
444 			return handleExistingCommand(e);
445 		} catch (final CommandException e) {
446 			if (ConfigEnum.VERBOSE.getBoolean()) {
447 				Logger.getLogger("Minecraft").info(
448 						"[AdminCmd] " + e.getMessage());
449 			}
450 			return false;
451 
452 		} finally {
453 			DebugLog.endInfo();
454 		}
455 		return true;
456 	}

481 	private boolean handleExistingCommand(final CommandAlreadyExist e) {
482 		DebugLog.beginInfo("Handle existing command");
483 		try {
484 			final CoreCommand command = e.getCommand();
485 			if (checkCmdStatus(command)) {
486 				disableCommand(e);
487 				DebugLog.INSTANCE.info("Command Disabled");
488 				return false;
489 			} else {
490 				command.registerBukkitPerm();
491 				command.getPluginCommand().setExecutor(this);
492 				registeredCommands.put(command.getPluginCommand(), command);
493 				DebugLog.INSTANCE
494 						.info("Command Prioritized but already exists");
495 				final IncrementalPlotter plotter = new ClassPlotter(
496 						command.getClass());
497 				graph.addPlotter(plotter);
498 				plotters.put(command.getClass(), plotter);
499 				return true;
500 			}
501 		} finally {
502 			DebugLog.endInfo();
503 		}
504 	}

506 	/***
507 	 * Check if the command need to be disabled or is prioritized
508 	 * 
509 	 * @param command
510 	 * @return true if can be disabled.
511 	 */
512 	private boolean checkCmdStatus(final CoreCommand command) {
513 		final Map<String, Command> commands = pluginCommands.get(command
514 				.getPlugin());
515 		if (commands != null) {
516 			for (final String alias : commands.get(command.getCmdName())
517 					.getAliases()) {
518 				if (prioritizedCommands.contains(alias)) {
519 					final CommandAlias cmd = new CommandAlias(
520 							command.getCmdName(), alias, "");
521 					cmd.setCmd(command);
522 					commandsAliasReplacer.put(alias, cmd);
523 					return false;
524 				}
525 				final Set<CommandAlias> commandAliases = commandsAlias
526 						.get(alias);
527 				if (commandAliases != null) {
528 					for (final CommandAlias commandAlias : commandAliases) {
529 						commandAlias.setCmd(command);
530 					}
531 					return false;
532 				}
533 			}
534 		}
535 		return true;
536 	}

557 	/***
558 	 * @param plugin
559 	 *            the plugin to set
560 	 */
561 	public void setCorePlugin(final AdminCmd plugin) {
562 		this.corePlugin = plugin;
563 		final ExtendedConfiguration cmds = FileManager.getInstance().getYml(
564 				"commands");
565 		disabledCommands = cmds.getStringList("disabledCommands",
566 				new LinkedList<String>());
567 		prioritizedCommands = cmds.getStringList("prioritizedCommands",
568 				new LinkedList<String>());
569 		commandsOnJoin.addAll(cmds.getStringList("onNewJoin"));
570 		final ExConfigurationSection aliases = cmds
571 				.getConfigurationSection("aliases");
572 		for (final String command : aliases.getKeys(false)) {
573 			Set<CommandAlias> setAliasCmd = commandsAlias.get(command);
574 			if (setAliasCmd == null) {
575 				setAliasCmd = new HashSet<CommandAlias>();
576 				commandsAlias.put(command, setAliasCmd);
577 			}
578 			final ConfigurationSection aliasSection = aliases
579 					.getConfigurationSection(command);
580 			for (final Entry<String, Object> alias : aliasSection.getValues(
581 					false).entrySet()) {
582 				String params;
583 				final Object value = alias.getValue();
584 				if (value instanceof ConfigurationSection) {
585 					final StringBuffer buffer = new StringBuffer();
586 					for (final Entry<String, Object> entry : ((ConfigurationSection) alias)
587 							.getValues(false).entrySet()) {
588 						buffer.append(entry.getKey()).append(':')
589 								.append(entry.getValue());
590 					}
591 					params = buffer.toString();
592 				} else {
593 					params = value.toString();
594 				}
595 				final CommandAlias commandAlias = new CommandAlias(command,
596 						alias.getKey(), params);
597 				commandsAliasReplacer.put(alias.getKey(), commandAlias);
598 
599 				setAliasCmd.add(commandAlias);
600 			}
601 
602 		}
603 		graph = plugin.getMetrics().createGraph("Commands");
604 		startThreads();
605 	}

619 	/***
620 	 * Unregister a command from bukkit.
621 	 * 
622 	 * @param pCmd
623 	 */
624 	private void unRegisterBukkitCommand(final PluginCommand pCmd) {
625 		final CommandMap commandMap = FieldUtils.getField(corePlugin
626 				.getServer().getPluginManager(), "commandMap");
627 		final HashMap<String, Command> knownCommands = FieldUtils.getField(
628 				commandMap, "knownCommands");
629 		PluginCommand cmd;
630 		final List<String> aliases = new ArrayList<String>();
631 		try {
632 			cmd = (PluginCommand) knownCommands.get(pCmd.getName());
633 			if (pCmd == cmd) {
634 				DebugLog.INSTANCE.info("Remove Command " + pCmd.getName());
635 				knownCommands.remove(pCmd.getName());
636 				aliases.addAll(pCmd.getAliases());
637 				pCmd.unregister(commandMap);
638 			} else {
639 				return;
640 			}
641 		} catch (final ClassCastException e) {
642 			DebugLog.INSTANCE.log(Level.INFO, "Not a Plugin Command", e);
643 		}
644 		for (final String alias : aliases) {
645 			try {
646 				cmd = (PluginCommand) knownCommands.get(alias);
647 				if (pCmd == cmd) {
648 					knownCommands.remove(alias);
649 				}
650 			} catch (final ClassCastException e) {
651 				DebugLog.INSTANCE.log(Level.INFO, "Not a Plugin Command", e);
652 			}
653 		}
654 
655 	}

657 	/***
658 	 * UnRegister command
659 	 * 
660 	 * @param clazz
661 	 *            command to unregister
662 	 * @param plugin
663 	 *            plugin that want the command to be unregister. It has to be
664 	 *            the same that belong to the command.
665 	 */
666 	public boolean unRegisterCommand(final Class<? extends CoreCommand> clazz,
