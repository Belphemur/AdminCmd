41  	/***
42  	 * Add a perm on the fly
43  	 * 
44  	 * @param permNode
45  	 * @param parentNode
46  	 * @return
47  	 */
48  	public static Permission addOnTheFly(final String permNode, final String parentNode) {
49  		Permission child;
50  		if (ACPluginManager.getServer() == null)
51  			return null;
52  		if ((child = ACPluginManager.getServer().getPluginManager().getPermission(permNode)) == null) {
53  			child = new Permission(permNode, PermissionDefault.OP);
54  			ACPluginManager.getServer().getPluginManager().addPermission(child);
55  			if (parentNode.isEmpty())
56  				return child;
57  			Permission parent = ACPluginManager.getServer().getPluginManager()
58  					.getPermission(parentNode);
59  
60  			if (parent == null) {
61  				parent = new Permission(parentNode, PermissionDefault.OP);
62  				ACPluginManager.getServer().getPluginManager().addPermission(parent);
63  			}
64  			child.addParent(parent, true);
65  		}
66  		return child;
67  
68  	}

70  	/***
71  	 * Get the PermissionLinker Note: The PermissionManager may only retain a
72  	 * weak reference to the newly created PermissionLinker. It is important to
73  	 * understand that a previously created PermissionLinker with the given name
74  	 * may be garbage collected at any time if there is no strong reference to
75  	 * the PermissionLinker. In particular, this means that two back-to-back
76  	 * calls like
77  	 * {@code PermissionLinker("MyPermissionLinker").addPermParent(...)} may use
78  	 * different PermissionLinker objects named "MyPermissionLinker" if there is
79  	 * no strong reference to the PermissionLinker named "MyPermissionLinker"
80  	 * elsewhere in the program.
81  	 * 
82  	 * @param name
83  	 * @return
84  	 */
85  	public static synchronized PermissionLinker getPermissionLinker(final String name) {
86  		return PermissionManager.getInstance().demandPermissionLinker(name);
87  	}

108 	/***
109 	 * Add permission child (like myplugin.item.add)
110 	 * 
111 	 * @param permNode
112 	 * @param bukkitDefault
113 	 * @return
114 	 */
115 	public Permission addPermChild(final String permNode, final PermissionDefault bukkitDefault)
116 			throws NullPointerException {
117 		final PermParent parent = matchPermParent(permNode);
118 		final PermChild child = new PermChild(permNode, bukkitDefault);
119 		if (parent == null) {
120 			DebugLog.INSTANCE.info("No Permission Parent found for : " + permNode);
121 			return child.getBukkitPerm();
122 		}
123 		parent.addChild(child);
124 		DebugLog.INSTANCE.info(child + " registered as child of " + parent);
125 		return child.getBukkitPerm();
126 	}

221 	private PermParent matchPermParent(final String search) {
222 		PermParent found = null;
223 		final String lowerSearch = search.toLowerCase();
224 		int delta = Integer.MAX_VALUE;
225 		final Set<PermParent> values = new HashSet<PermParent>();
226 		values.addAll(permissions.values());
227 		values.addAll(childrenPermParents.values());
228 		for (final PermParent perm : values) {
229 			final String str = perm.getCompareName();
230 			if (lowerSearch.toLowerCase().startsWith(str)) {
231 				final int curDelta = lowerSearch.length() - str.length();
232 				if (curDelta < delta) {
233 					found = perm;
234 					delta = curDelta;
235 				}
236 				if (curDelta == 0)
237 					break;
238 			}
239 		}
240 		return found;
241 
242 	}
