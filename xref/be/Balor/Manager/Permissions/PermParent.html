25  public class PermParent extends PermChild {
26  	protected final String compareName;
27  
28  	public PermParent(final String perm) {
29  		this(perm, perm == null ? null : perm.substring(0, perm.length() - 1), PermissionDefault.OP);
30  	}
31  
32  	public PermParent(final String perm, final String compare, final PermissionDefault def) {
33  		super(perm, def);
34  		this.compareName = compare;
35  	}
36  
37  	/***
38  	 * @return the compareName
39  	 */
40  	public String getCompareName() {
41  		return compareName;
42  	}
43  
44  	/***
45  	 * Add a permission Child to the Permission Parent
46  	 * 
47  	 * @param perm
48  	 * @return the PermParent (this)
49  	 */
50  	public PermParent addChild(final PermChild perm) throws IllegalArgumentException {
51  		if (perm.equals(this))
52  			throw new IllegalArgumentException("The Child can't be the parent.");
53  		perm.bukkitPerm.addParent(bukkitPerm, true);
54  		return this;
55  	}
56  
57  	/***
58  	 * Add a permission Child to the Permission Parent
59  	 * 
60  	 * @param perm
61  	 * @return the PermParent (this)
62  	 */
63  	public PermParent addChild(final String perm) {
64  		this.addChild(new PermChild(perm));
65  		return this;
66  	}

72  	 */
73  	@Override
74  	public int hashCode() {
75  		final int prime = 31;
76  		int result = super.hashCode();
77  		result = prime * result + ((compareName == null) ? 0 : compareName.hashCode());
78  		return result;
79  	}
