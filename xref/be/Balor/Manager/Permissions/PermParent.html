35  public class PermParent extends PermChild {
36  	protected final String compareName;
37  	private final Map<String, Boolean> children = new HashMap<String, Boolean>();
38  	private final Set<PermParent> permParentChildren = new HashSet<PermParent>();
39  	private boolean registered = false;
40  
41  	public PermParent(final String perm) {
42  		this(perm, perm == null ? null : perm.substring(0, perm.length() - 1),
43  				PermissionDefault.OP);
44  	}
45  
46  	public PermParent(final String perm, final String compare,
47  			final PermissionDefault def) {
48  		super(perm, def);
49  		this.compareName = compare;
50  
51  	}
52  
53  	/***
54  	 * @return the compareName
55  	 */
56  	public String getCompareName() {
57  		return compareName;
58  	}
59  
60  	/***
61  	 * Add a permission Child to the Permission Parent
62  	 * 
63  	 * @param perm
64  	 * @return the PermParent (this)
65  	 */
66  	public PermParent addChild(final PermChild perm)
67  			throws IllegalArgumentException {
68  		if (perm.equals(this)) {
69  			throw new IllegalArgumentException("The Child can't be the parent.");
70  		}
71  		children.put(perm.getPermName(), true);
72  		if (perm instanceof PermParent) {
73  			permParentChildren.add((PermParent) perm);
74  		}
75  		perm.parent = this;
76  		return this;
77  	}
78  
79  	/***
80  	 * Register the permParent.
81  	 * 
82  	 * @return
83  	 */
84  	public void registerPermission() {
85  		DebugLog.beginInfo("Registering PermParent : " + this.permName);
86  		try {
87  			if (registered) {
88  				return;
89  			}
90  			for (final PermParent perm : permParentChildren) {
91  				perm.registerPermission();
92  			}
93  			Permission bukkitPerm = new Permission(this.permName,
94  					this.permissionDefault, children);
95  			try {
96  				ACPluginManager.getServer().getPluginManager()
97  						.addPermission(bukkitPerm);
98  			} catch (final Exception e) {
99  				DebugLog.INSTANCE
100 						.warning("Trying to register an existing PermParent : "
101 								+ this.permName);
102 				bukkitPerm = Bukkit.getPluginManager().getPermission(permName);
103 				bukkitPerm.getChildren().putAll(children);
104 				bukkitPerm.recalculatePermissibles();
105 			}
106 			registered = true;
107 		} finally {
108 			DebugLog.endInfo();
109 		}
110 	}
111 
112 	/***
113 	 * Add a permission Child to the Permission Parent
114 	 * 
115 	 * @param perm
116 	 * @return the PermParent (this)
117 	 */
118 	public PermParent addChild(final String perm) {
119 		this.addChild(new PermChild(perm));
120 		return this;
121 	}

140 		final int prime = 31;
141 		int result = super.hashCode();
142 		result = prime * result

144 		result = prime * result + (registered ? 1231 : 1237);
145 		return result;
146 	}

155 		if (this == obj) {
156 			return true;
157 		}
158 		if (!super.equals(obj)) {
159 			return false;
160 		}
161 		if (getClass() != obj.getClass()) {
162 			return false;
163 		}
164 		final PermParent other = (PermParent) obj;
165 		if (compareName == null) {
166 			if (other.compareName != null) {
167 				return false;
168 			}
169 		} else if (!compareName.equals(other.compareName)) {
170 			return false;
171 		}
172 		if (registered != other.registered) {
173 			return false;
174 		}
175 		return true;
176 	}
