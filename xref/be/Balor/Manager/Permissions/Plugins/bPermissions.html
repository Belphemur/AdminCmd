38  	/*
39  	 * (non-Javadoc)
40  	 * 
41  	 * @see
42  	 * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
43  	 * .String, org.bukkit.entity.Player)
44  	 */
45  	@Override
46  	public boolean isInGroup(final String groupName, final Player player) {
47  		final String[] groups = ApiLayer.getGroups(player.getWorld().getName(),
48  				CalculableType.USER, player.getName());
49  		if (groups == null)
50  			return false;
51  		if (groups.length == 0)
52  			return false;
53  		for (final String group : groups)
54  			if (group.equalsIgnoreCase(groupName))
55  				return true;
56  		return false;
57  	}

59  	/*
60  	 * (non-Javadoc)
61  	 * 
62  	 * @see
63  	 * be.Balor.Manager.Permissions.AbstractPermission#getUsers(org.java.lang
64  	 * .String)
65  	 */
66  	@Override
67  	public Set<Player> getUsers(final String groupName) throws NoPermissionsPlugin {
68  		final Set<Player> players = new HashSet<Player>();
69  		for (final Player player : Utils.getOnlinePlayers()) {
70  			if (ApiLayer.hasGroup(player.getWorld().getName(), CalculableType.USER,
71  					player.getName(), groupName))
72  				players.add(player);
73  		}
74  		return players;
75  	}

88  			result = ApiLayer.getValue(p.getWorld().getName(), CalculableType.USER, p.getName(),
