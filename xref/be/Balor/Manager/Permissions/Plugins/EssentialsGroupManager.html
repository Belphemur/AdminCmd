23  import org.anjocaido.groupmanager.GroupManager;
24  import org.anjocaido.groupmanager.permissions.AnjoPermissionsHandler;
25  import org.bukkit.entity.Player;
26  import org.bukkit.plugin.Plugin;
27  
28  /***
29   * @author Lathanael (aka Philippe Leipold)
30   * 
31   */
32  public class EssentialsGroupManager extends SuperPermissions {
33  
34  	private final GroupManager groupManager;
35  
36  	public EssentialsGroupManager(final Plugin manager) {
37  		groupManager = (GroupManager) manager;
38  	}
39  
40  	/*
41  	 * (non-Javadoc)
42  	 * 
43  	 * @see
44  	 * be.Balor.Manager.Permissions.AbstractPermission#isInGroup(org.java.lang
45  	 * .String, org.bukkit.entity.Player)
46  	 */
47  	@Override
48  	public boolean isInGroup(final String groupName, final Player player) {
49  		final AnjoPermissionsHandler handler = groupManager.getWorldsHolder().getWorldPermissions(
50  				player);
51  		if (handler == null)
52  			return false;
53  		return handler.inGroup(player.getName(), groupName);
54  	}
55  
56  	/*
57  	 * (non-Javadoc)
58  	 * 
59  	 * @see
60  	 * be.Balor.Manager.Permissions.AbstractPermission#getPrefix(org.bukkit.
61  	 * entity.Player)
62  	 */
63  	@Override
64  	public String getPrefix(final Player base) {
65  		final String prefix = super.getPrefix(base);
66  		if (prefix == null || prefix.isEmpty()) {
67  			final AnjoPermissionsHandler handler = groupManager.getWorldsHolder()
68  					.getWorldPermissions(base);
69  			if (handler == null) {
70  				return "";
71  			}
72  			return handler.getUserPrefix(base.getName());
73  		}
74  		return prefix;
75  	}
76  
77  	/*
78  	 * (non-Javadoc)
79  	 * 
80  	 * @see
81  	 * be.Balor.Manager.Permissions.IPermissionPlugin#getSuffix(org.bukkit.entity
82  	 * .Player)
83  	 */
84  	@Override
85  	public String getSuffix(final Player base) {
86  		final String suffix = super.getSuffix(base);
87  		if (suffix == null || suffix.isEmpty()) {
88  			final AnjoPermissionsHandler handler = groupManager.getWorldsHolder()
89  					.getWorldPermissions(base);
90  			if (handler == null) {
91  				return "";
92  			}
93  			return handler.getUserSuffix(base.getName());
94  		}
95  		return suffix;
96  	}
97  
98  	/*
99  	 * (non-Javadoc)
100 	 * 
101 	 * @see
102 	 * be.Balor.Manager.Permissions.Plugins.SuperPermissions#getGroup(org.bukkit
103 	 * .entity.Player)
104 	 */
105 	@Override
106 	public String getGroup(final Player player) {
107 		final AnjoPermissionsHandler handler = groupManager.getWorldsHolder().getWorldPermissions(
108 				player);
109 		return handler.getGroup(player.getName());
110 	}
