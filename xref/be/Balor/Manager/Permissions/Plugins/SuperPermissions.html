134 	/*
135 	 * (non-Javadoc)
136 	 * 
137 	 * @see
138 	 * be.Balor.Manager.Permissions.AbstractPermission#getPermissionLimit(org
139 	 * .bukkit.entity.Player, java.lang.String)
140 	 */
141 	@Override
142 	public String getPermissionLimit(final Player p, final String limit) {
143 		String result = null;
144 		if (mChatInfo != null)
145 			result = mChatInfo.getInfo(p.getName(), InfoType.USER, p.getWorld().getName(),
146 					"admincmd." + limit);
147 		if (result == null || (result != null && result.isEmpty())) {
148 			final Pattern regex = Pattern
149 					.compile("admincmd//." + limit.toLowerCase() + "//.[0-9]+");
150 			final Set<PermissionAttachmentInfo> perms = new LinkedHashSet<PermissionAttachmentInfo>();
151 			final CountDownLatch countDown = new CountDownLatch(1);
152 			ACPluginManager.scheduleSyncTask(new Runnable() {
153 
154 				@Override
155 				public void run() {
156 					perms.addAll(p.getEffectivePermissions());
157 					countDown.countDown();
158 
159 				}
160 			});
161 			try {
162 				countDown.await();
163 			} catch (final InterruptedException e) {
164 				DebugLog.INSTANCE.log(Level.WARNING,
165 						"Problem to get Permission of the user " + p.getName(), e);
166 			}
167 			int max = Integer.MIN_VALUE;
168 			for (final PermissionAttachmentInfo info : perms) {
169 				final Matcher regexMatcher = regex.matcher(info.getPermission().toLowerCase());
170 				if (!regexMatcher.find())
171 					continue;
172 				final int current = Integer.parseInt(info.getPermission().split("//.")[2]);
173 				if (current < max)
174 					continue;
175 				max = current;
176 
177 			}
178 			if (max != Integer.MIN_VALUE)
179 				return String.valueOf(max);
180 		} else
181 			return result;
182 		return null;
183 	}
