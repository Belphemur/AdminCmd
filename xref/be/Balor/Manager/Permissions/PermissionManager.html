296 	public synchronized boolean addPermissionLinker(final PermissionLinker perm) {
297 		final String name = perm.getName();
298 		if (name == null) {
299 			throw new NullPointerException();
300 		}
301 
302 		final WeakReference<PermissionLinker> ref = permissionLinkers.get(name);
303 		if (ref != null) {
304 			if (ref.get() == null) {
305 				// Hashtable holds stale weak reference
306 				// to a logger which has been GC-ed.
307 				// Allow to register new one.
308 				permissionLinkers.remove(name);
309 			} else {
310 				// We already have a registered logger with the given name.
311 				return false;
312 			}
313 		}
314 		// We're adding a new logger.
315 		// Note that we are creating a weak reference here.
316 		permissionLinkers.put(name, new WeakReference<PermissionLinker>(perm));
317 		return true;
318 	}

320 	PermissionLinker demandPermissionLinker(final String name) {
321 		PermissionLinker result = getPermissionLinker(name);
322 		if (result == null) {
323 			result = new PermissionLinker(name);
324 			addPermissionLinker(result);
325 			result = getPermissionLinker(name);
326 		}
327 		return result;
328 	}
329 
330 	public synchronized PermissionLinker getPermissionLinker(final String name) {
331 		final WeakReference<PermissionLinker> ref = permissionLinkers.get(name);
332 		if (ref == null) {
333 			return null;
334 		}
335 		final PermissionLinker perm = ref.get();
336 		if (perm == null) {
337 			// Hashtable holds stale weak reference
338 			// to a logger which has been GC-ed.
339 			permissionLinkers.remove(name);
340 		}
341 		return perm;
342 	}
