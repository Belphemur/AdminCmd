17  package be.Balor.Manager.Permissions;
18  
19  import org.bukkit.permissions.Permission;
20  import org.bukkit.permissions.PermissionDefault;
21  
22  import be.Balor.bukkit.AdminCmd.ACPluginManager;
23  
24  /***
25   * @author Balor (aka Antoine Aflalo)
26   * 
27   */
28  public class PermChild {
29  	protected Permission bukkitPerm;
30  
31  	public PermChild(final String permName) {
32  		this(permName, PermissionDefault.OP);
33  	}
34  
35  	/***
36  	 * @param permName
37  	 * @param parent
38  	 * @param value
39  	 * @param permDefault
40  	 */
41  	public PermChild(final String permName, final PermissionDefault permDefault) {
42  		if (permName == null)
43  			return;
44  		if (ACPluginManager.getServer() == null)
45  			return;
46  		if ((bukkitPerm = ACPluginManager.getServer().getPluginManager().getPermission(permName)) != null) {
47  			bukkitPerm.setDefault(permDefault);
48  			return;
49  		}
50  		bukkitPerm = new Permission(permName, permDefault);
51  		ACPluginManager.getServer().getPluginManager().addPermission(bukkitPerm);
52  	}
53  
54  	/***
55  	 * @return the permName
56  	 */
57  	public String getPermName() {
58  		return bukkitPerm.getName();
59  	}
60  
61  	/***
62  	 * @return the permDefault
63  	 */
64  	public PermissionDefault getPermDefault() {
65  		return bukkitPerm.getDefault();
66  	}
67  
68  	/***
69  	 * @return the bukkitPerm
70  	 */
71  	public Permission getBukkitPerm() {
72  		return bukkitPerm;
73  	}
74  
75  	/*
76  	 * (non-Javadoc)
77  	 * 
78  	 * @see java.lang.Object#hashCode()
79  	 */
80  	@Override
81  	public int hashCode() {
82  		final int prime = 31;
83  		int result = 1;
84  		result = prime * result + ((bukkitPerm == null) ? 0 : bukkitPerm.hashCode());
85  		return result;
86  	}
87  
88  	/*
89  	 * (non-Javadoc)
90  	 * 
91  	 * @see java.lang.Object#equals(java.lang.Object)
92  	 */
93  	@Override
94  	public boolean equals(final Object obj) {
95  		if (this == obj)
96  			return true;
97  		if (obj == null)
98  			return false;
99  		if (!(obj instanceof PermChild))
100 			return false;
101 		final PermChild other = (PermChild) obj;
102 		if (bukkitPerm == null) {
103 			if (other.bukkitPerm != null)
104 				return false;
105 		} else if (!bukkitPerm.equals(other.bukkitPerm))
106 			return false;
107 		return true;
108 	}
