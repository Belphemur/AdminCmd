19  import java.io.BufferedReader;
20  import java.io.File;
21  import java.io.InputStreamReader;
22  
23  import org.bukkit.command.CommandSender;
24  
25  import be.Balor.Manager.Terminal.TerminalCommand;

31  public class WindowsTerminalCommand extends TerminalCommand {
32  
33  	/***
34  	 * @param commandName
35  	 * @param execution
36  	 * @param args
37  	 * @param workingDir
38  	 */
39  	public WindowsTerminalCommand(final String commandName, final String execution,
40  			final String args, final File workingDir) {
41  		super(commandName, execution, args, workingDir);
42  	}
43  
44  	/***
45  	 * @param commandName
46  	 * @param execution
47  	 * @param args
48  	 * @param workingDir
49  	 */
50  	public WindowsTerminalCommand(final String commandName, final String execution,
51  			final String args, final String workingDir) {
52  		super(commandName, execution, args, workingDir);
53  	}
54  
55  	/*
56  	 * (non-Javadoc)
57  	 * 
58  	 * @see
59  	 * be.Balor.Manager.Terminal.TerminalCommand#execute(org.bukkit.command.
60  	 * CommandSender)
61  	 */
62  	@Override
63  	public void execute(final CommandSender sender) {
64  		try {
65  			ProcessBuilder pb;
66  			if (args != null)
67  				pb = new ProcessBuilder("cmd.exe", "/C", execution, args);
68  			else
69  				pb = new ProcessBuilder("cmd.exe", "/C", execution);
70  			pb.redirectErrorStream(true);
71  			pb.directory(workingDir);
72  			final Process p = pb.start();
73  			final BufferedReader in = new BufferedReader(new InputStreamReader(p.getInputStream()));
74  			String line = null;
75  			while ((line = in.readLine()) != null) {
76  				sender.sendMessage(line);
77  			}
78  		} catch (final Throwable e) {
79  			sender.sendMessage("CMD ERROR : " + e.getMessage());
80  			e.printStackTrace();
81  		}
82  
83  	}
84  
85  }
