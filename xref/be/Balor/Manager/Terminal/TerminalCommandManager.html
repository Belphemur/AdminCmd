58  	/***
59  	 * @param permissionLinker
60  	 *            the perm to set
61  	 */
62  	public void setPerm(final AbstractAdminCmdPlugin plugin) {
63  		this.perm = plugin.getPermissionLinker();
64  		final File scripts = FileManager.getInstance()
65  				.getInnerFile("scripts.yml", "scripts", false);
66  		final File workingDir = scripts.getParentFile();
67  		final ExtendedConfiguration conf = ExtendedConfiguration.loadConfiguration(scripts);
68  		TerminalCommand toAdd;
69  		if (System.getProperty("os.name").contains("Windows"))
70  			for (final String cmdName : conf.getKeys(false)) {
71  				toAdd = new WindowsTerminalCommand(cmdName, conf.getString(cmdName + ".exec"),
72  						conf.getString(cmdName + ".args"), workingDir);
73  				toAdd.setBukkitPerm(perm.addPermChild("admincmd.server.exec." + cmdName));
74  				commands.put(cmdName, toAdd);
75  
76  			}
77  
78  		else
79  
80  			for (final String cmdName : conf.getKeys(false)) {
81  				toAdd = new UnixTerminalCommand(cmdName, conf.getString(cmdName + ".exec"),
82  						conf.getString(cmdName + ".args"), workingDir);
83  				toAdd.setBukkitPerm(perm.addPermChild("admincmd.server.exec." + cmdName));
84  				commands.put(cmdName, toAdd);
85  			}
86  	}

88  	public boolean checkCommand(final String cmdName, final CommandSender sender) {
89  		final TerminalCommand cmd = commands.get(cmdName);
90  		if (cmd == null)
91  			return false;
92  
93  		return cmd.permCheck(sender, false);
94  	}

99  		final File workingDir = scripts.getParentFile();
100 		final ExtendedConfiguration conf = ExtendedConfiguration.loadConfiguration(scripts);

124 	/***
125 	 * Execute the script
126 	 * 
127 	 * @param sender
128 	 * @param cmdName
129 	 * @return
130 	 * @throws CommandNotFound
131 	 */
132 	public boolean execute(final CommandSender sender, final String cmdName, final boolean reload)
133 			throws CommandNotFound {
134 		TerminalCommand cmd = commands.get(cmdName);
135 		if (cmd == null || reload) {
136 			final File scripts = FileManager.getInstance().getInnerFile("scripts.yml", "scripts",
137 					false);
138 			final File workingDir = scripts.getParentFile();
139 			final ExtendedConfiguration conf = ExtendedConfiguration.loadConfiguration(scripts);
140 			if (conf.get(cmdName) == null)
141 				throw new CommandNotFound(cmdName + " is not registered");
142 			if (System.getProperty("os.name").contains("Windows")) {
143 				commands.put(
144 						cmdName,
145 						new WindowsTerminalCommand(cmdName, conf.getString(cmdName + ".exec"), conf
146 								.getString(cmdName + ".args"), workingDir));
147 				cmd = commands.get(cmdName);
148 				cmd.setBukkitPerm(PermissionLinker.addOnTheFly("admincmd.server.exec." + cmdName,
149 						"admincmd.server.exec.*"));
150 			} else {
151 				commands.put(
152 						cmdName,
153 						new UnixTerminalCommand(cmdName, conf.getString(cmdName + ".exec"), conf
154 								.getString(cmdName + ".args"), workingDir));
155 				cmd = commands.get(cmdName);
156 				cmd.setBukkitPerm(PermissionLinker.addOnTheFly("admincmd.server.exec." + cmdName,
157 						"admincmd.server.exec.*"));
158 			}
159 		}
160 		if (!cmd.permCheck(sender, true))
161 			return false;
162 		cmd.execute(sender);
163 		return true;
164 	}
