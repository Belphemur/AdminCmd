217 	/***
218 	 * Looking in every locale file for the locale asked, begin by the
219 	 * principalLocale;
220 	 * 
221 	 * @param key
222 	 * @return
223 	 */
224 	private String getLocale(final String key) {
225 		String result = localesFiles.get(PRIMARY_LOCALE).getString(key);
226 		if (result == null || (result != null && result.isEmpty())) {
227 			for (final Entry<String, ExtendedConfiguration> e : localesFiles
228 					.entrySet()) {
229 				if (e.getKey().equals(PRIMARY_LOCALE)) {
230 					continue;
231 				}
232 				result = e.getValue().getString(key);
233 				if (result != null && !result.isEmpty()) {
234 					break;
235 				}
236 
237 			}
238 		}
239 		return result;
240 
241 	}

243 	private String recursiveReplaceLocale(final String locale,
244 			final Map<String, String> values) {
245 		String ResultString = null;
246 		String result = locale;
247 		try {
248 			Matcher regexMatcher = recursiveLocale.matcher(locale);
249 			while (regexMatcher.find()) {
250 				ResultString = regexMatcher.group(1);
251 				final String recLocale = getLocale(ResultString);
252 				if (recLocale != null) {
253 					result = regexMatcher.replaceFirst(recLocale);
254 				} else {
255 					result = regexMatcher.replaceFirst("");
256 				}
257 				regexMatcher = recursiveLocale.matcher(result);
258 			}
259 			regexMatcher = replaceLocale.matcher(result);
260 			while (regexMatcher.find()) {
261 				ResultString = regexMatcher.group(1);
262 				final String replaceValue = values.get(ResultString);
263 				if (replaceValue != null) {
264 					try {
265 						result = regexMatcher.replaceFirst(Matcher.quoteReplacement(replaceValue));
266 					} catch (final StringIndexOutOfBoundsException e) {
267 						result = regexMatcher.replaceFirst(replaceValue
268 								.replaceAll("//W", ""));
269 					}
270 
271 				} else {
272 					result = regexMatcher.replaceFirst("");
273 				}
274 				regexMatcher = replaceLocale.matcher(result);
275 			}
276 		} catch (final PatternSyntaxException ex) {
277 			// Syntax error in the regular expression
278 		}
279 		// To correct interrogation point (?) problem in the locale file.
280 		final Matcher regexMatcher = buggedLocale.matcher(result);
281 		result = regexMatcher.replaceAll("§$1");
282 		return result;
283 	}

285 	public String get(final String key, final String alias,
286 			final String replaceBy) {
287 		if (noMsg) {
288 			return null;
289 		}
290 		String locale = getLocale(key);
291 		if (locale != null && alias != null) {
292 			final HashMap<String, String> replace = new HashMap<String, String>();
293 			replace.put(alias, replaceBy);
294 			locale = recursiveReplaceLocale(locale, replace);
295 		}
296 
297 		return locale;
298 	}
