62  	public void execute(final CommandSender sender, final CommandArgs args) {
63  		final HashMap<String, String> replace = new HashMap<String, String>();
64  		String type = "all";
65  		if (args.length >= 1)
66  			type = args.getString(0);
67  		replace.put("type", type);
68  		final String worldString = args.getValueFlag('w');
69  		final List<World> worldList = new ArrayList<World>();
70  
71  		if (Utils.isPlayer(sender, false)) {
72  			World w = ((Player) sender).getWorld();
73  			if (worldString != null)
74  				w = getWorld(worldString);
75  			worldList.add(w);
76  			replace.put("worlds", w.getName());
77  			Utils.sI18n(sender, "killMob", replace);
78  		} else {
79  			if (worldString != null)
80  				worldList.add(getWorld(worldString));
81  			else {
82  				String worlds = "";
83  				for (final World world : sender.getServer().getWorlds()) {
84  					worldList.add(world);
85  					worlds += world.getName() + ", ";
86  				}
87  				if (!worlds.equals("")) {
88  					if (worlds.endsWith(", "))
89  						worlds = worlds.substring(0, worlds.lastIndexOf(","));
90  					replace.put("worlds", worlds);
91  					Utils.sI18n(sender, "killMob", replace);
92  				}
93  			}
94  		}
95  		final String finalType = type;
96  		final CommandSender finalSender = sender;
97  		ACPluginManager.getScheduler().scheduleAsyncDelayedTask(ACPluginManager.getCorePlugin(),
98  				new Runnable() {
99  					@Override
100 					public void run() {
101 						killMobs(worldList, finalType, finalSender);
102 					}
103 				});
104 
105 	}

107 	private World getWorld(final String name) {
108 		final World w = Bukkit.getWorld(name);
109 		if (w == null)
110 			throw new WorldNotLoaded("The World " + name + " is not loaded");
111 		return w;
112 	}

124 	private void killMobs(final List<World> worlds, final String type, final CommandSender sender) {
125 		int mobKilled = 0;
126 		if (type.equalsIgnoreCase("all")) {
127 			for (final World w : worlds)
128 				for (final LivingEntity m : w.getLivingEntities()) {
129 					if (m instanceof HumanEntity) {
130 						continue;
131 					}
132 					final Entity entity = ((CraftLivingEntity) m).getHandle();
133 					entity.die();
134 					mobKilled++;
135 
136 				}
137 		} else {
138 			EntityType ct = null;
139 			ct = EntityType.fromName(type);
140 			if (ct == null) {
141 				final HashMap<String, String> replace = new HashMap<String, String>();
142 				replace.put("mob", type);
143 				Utils.sI18n(sender, "errorMob", replace);
144 				return;
145 			}
146 			for (final World w : worlds)
147 				for (final org.bukkit.entity.Entity m : w.getEntitiesByClasses(ct.getEntityClass())) {
148 					final Entity entity = ((CraftEntity) m).getHandle();
149 					entity.die();
150 					mobKilled++;
151 				}
152 		}
153 		Utils.sI18n(sender, "killedMobs", "nbKilled", String.valueOf(mobKilled));
154 	}
