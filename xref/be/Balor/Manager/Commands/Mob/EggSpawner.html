49  	/***
50  	 *
51  	 */
52  	public EggSpawner() {
53  		this.cmdName = "bal_egg";
54  	}
55  
56  	/*
57  	 * (non-Javadoc)
58  	 * 
59  	 * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
60  	 * CommandSender, be.Balor.Manager.Commands.CommandArgs)
61  	 */
62  	@Override
63  	public void execute(final CommandSender sender, final CommandArgs args)
64  			throws ActionNotPermitedException, PlayerNotFound {
65  		if (!Utils.isPlayer(sender)) {
66  			return;
67  		}
68  		final Player player = (Player) sender;
69  		final ACPlayer acp = ACPlayer.getPlayer(player);
70  		EggType<?> egg = null;
71  		try {
72  			egg = EggType.createEggType(player, args);
73  		} catch (final ParameterMissingException e) {
74  			if (e.getParam() == 'e') {
75  				final String list = Joiner
76  						.on(", ")
77  						.skipNulls()
78  						.join(EggPermissionManager.INSTANCE
79  								.getEggTypeNames(player));
80  				sender.sendMessage(e.getMessage());
81  				sender.sendMessage(ChatColor.GOLD + "Egg List : ");
82  				sender.sendMessage(ChatColor.YELLOW + list);
83  			} else {
84  				Utils.sI18n(sender, "paramMissing", "param",
85  						String.valueOf(e.getParam()));
86  				sender.sendMessage(e.getMessage());
87  			}
88  			return;
89  		} catch (final ProcessingArgsException e) {
90  			if (e.getType().equals(ExceptionType.NO_CLASS)) {
91  				Utils.sI18n(sender, "eggDontExists", "egg", e.getMessage());
92  			} else if (e.getType().equals(ExceptionType.DONT_EXISTS)) {
93  				Utils.sI18n(sender, "entityDontExists", "entity",
94  						e.getMessage());
95  			} else if (e.getType().equals(ExceptionType.CUSTOM)) {
96  				final Map<String, String> replace = new HashMap<String, String>();
97  				replace.put("egg", args.getValueFlag('t'));
98  				replace.put("error", e.getMessage());
99  				Utils.sI18n(sender, "eggCustomError", replace);
100 			} else {
101 				ACLogger.severe("Problem with an Egg Type : " + e.getMessage(),
102 						e);
103 			}
104 			return;
105 		} catch (final DontHaveThePermissionException e) {
106 			sender.sendMessage(e.getMessage());
107 			return;
108 		} catch (final NullPointerException e) {
109 			if (args.hasFlag('e')) {
110 				final String list = Joiner
111 						.on(", ")
112 						.skipNulls()
113 						.join(EggPermissionManager.INSTANCE
114 								.getEggTypeNames(player));
115 				sender.sendMessage(ChatColor.GOLD + "Egg List : ");
116 				sender.sendMessage(ChatColor.YELLOW + list);
117 			} else {
118 				Utils.sI18n(sender, "eggNoParamGiven");
119 				sender.sendMessage(e.getMessage());
120 			}
121 			return;
122 		}
123 		if (egg instanceof NormalEgg) {
124 			acp.removePower(Type.EGG);
125 			Utils.sI18n(sender, "eggNormal");
126 		} else {
127 			acp.setPower(Type.EGG, egg);
128 			Utils.sI18n(sender, "eggEnabled", "egg", egg.toString());
129 		}
130 
131 	}
