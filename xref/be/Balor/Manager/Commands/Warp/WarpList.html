39  public class WarpList extends WarpCommand {
40  	private PermChild tpAll;
41  
42  	/***
43  	 * 
44  	 */
45  	public WarpList() {
46  		permNode = "admincmd.warp.tp";
47  		cmdName = "bal_warplist";
48  	}
49  
50  	// TODO: Remove PermWarps a player is not allowed to warp to from the list
51  	/*
52  	 * (non-Javadoc)
53  	 * 
54  	 * @see
55  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
56  	 * java.lang.String[])
57  	 */
58  	@Override
59  	public void execute(final CommandSender sender, final CommandArgs args)
60  			throws ActionNotPermitedException, PlayerNotFound {
61  		if (Utils.isPlayer(sender)) {
62  			final Player p = (Player) sender;
63  			String msg = "";
64  			Set<String> wp;
65  			if (args.hasFlag('a')) {
66  				if (!PermissionManager.hasPerm(sender, tpAll)) {
67  					return;
68  				}
69  				wp = WorldManager.getInstance().getAllWarpList();
70  			} else {
71  				wp = ACWorld.getWorld(p.getWorld()).getWarpList();
72  			}
73  			sender.sendMessage(ChatColor.GOLD + "Warp Point(s) : "
74  					+ ChatColor.WHITE + wp.size());
75  			for (final String name : wp) {
76  				msg += name + ", ";
77  				if (msg.length() >= ACMinecraftFontWidthCalculator.chatwidth) {
78  					sender.sendMessage(msg);
79  					msg = "";
80  				}
81  			}
82  			if (!msg.equals("")) {
83  				if (msg.endsWith(", ")) {
84  					msg = msg.substring(0, msg.lastIndexOf(","));
85  				}
86  				sender.sendMessage(msg);
87  			}
88  		}
89  
90  	}
91
