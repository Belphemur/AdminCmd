55  	/*
56  	 * (non-Javadoc)
57  	 * 
58  	 * @see
59  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
60  	 * java.lang.String[])
61  	 */
62  	@Override
63  	public void execute(final CommandSender sender, final CommandArgs args) {
64  		final Player target = Utils.getUser(sender, args, permNode, 1, true);
65  		if (Utils.isPlayer(sender)) {
66  			final Player p = (Player) sender;
67  			Location loc = null;
68  			if (target != null) {
69  				final HashMap<String, String> replace = new HashMap<String, String>();
70  
71  				if (args.getString(0).contains(":")) {
72  					if (!PermissionManager.hasPerm(sender, tpAll.getBukkitPerm()))
73  						return;
74  					final String[] split = args.getString(0).split(":");
75  					final String world = split[0];
76  					final String warp = split[1];
77  					replace.put("name", world + ":" + warp);
78  					try {
79  						final ACWorld acWorld = ACWorld.getWorld(world);
80  						final Warp warpPoint = acWorld.getWarp(warp);
81  						if (warpPoint == null) {
82  							replace.put("name", args.getString(0));
83  							Utils.sI18n(sender, "errorWarp", replace);
84  							return;
85  						}
86  						loc = warpPoint.loc;
87  						replace.put("name", acWorld.getName() + ":" + warpPoint.name);
88  					} catch (final WorldNotLoaded e) {
89  						Utils.sI18n(sender, "worldNotFound", "world", world);
90  						return;
91  					}
92  				} else {
93  					replace.put("name", args.getString(0));
94  
95  					try {
96  						final Warp warpPoint = ACWorld.getWorld(p.getWorld().getName()).getWarp(
97  								args.getString(0));
98  						if (warpPoint == null) {
99  							replace.put("name", args.getString(0));
100 							Utils.sI18n(sender, "errorWarp", replace);
101 							return;
102 						}
103 						loc = warpPoint.loc;
104 						replace.put("name", warpPoint.name);
105 					} catch (final WorldNotLoaded e) {
106 					}
107 				}
108 				if (loc == null) {
109 					Utils.sI18n(sender, "errorWarp", replace);
110 					return;
111 				} else {
112 					ACPluginManager.getScheduler()
113 							.scheduleSyncDelayedTask(
114 									ACHelper.getInstance().getCoreInstance(),
115 									new DelayedTeleport(target.getLocation(), loc, target, replace,
116 											sender), ConfigEnum.TP_DELAY.getLong());
117 				}
118 			}
119 		}
120 
121 	}

145 	private class DelayedTeleport implements Runnable {
146 
147 		protected Location locBefore, teleportToLoc;
148 		protected Player target;
149 		protected HashMap<String, String> replace;
150 		protected CommandSender sender;
151 
152 		public DelayedTeleport(final Location locBefore, final Location teleportLoc,
153 				final Player target, final HashMap<String, String> replace,
154 				final CommandSender sender) {
155 			this.target = target;
156 			this.locBefore = locBefore;
157 			this.teleportToLoc = teleportLoc;
158 			this.replace = replace;
159 			this.sender = sender;
160 		}
161 
162 		@Override
163 		public void run() {
164 			if (locBefore.equals(target.getLocation()) && ConfigEnum.CHECKTP.getBoolean()) {
165 				Utils.teleportWithChunkCheck(target, teleportToLoc);
166 				sendMessage(sender, target, "tpWarp", replace);
167 			} else if (!ConfigEnum.CHECKTP.getBoolean()) {
168 				Utils.teleportWithChunkCheck(target, teleportToLoc);
169 				sendMessage(sender, target, "tpWarp", replace);
170 			} else {
171 				replace.clear();
172 				replace.put("cmdname", "Warp");
173 				sendMessage(sender, target, "errorMoved", replace);
174 			}
175 		}
176 	}
