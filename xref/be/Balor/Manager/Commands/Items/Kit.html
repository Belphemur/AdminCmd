62  	 * (non-Javadoc)
63  	 * 
64  	 * @see be.Balor.Manager.ACCommand#execute(org.bukkit.command.CommandSender,
65  	 * java.lang.String[])
66  	 */
67  	@Override
68  	public void execute(final CommandSender sender, final CommandArgs args) {
69  		// which material?
70  		Player target;
71  		if (args.length == 0) {
72  			Utils.sI18n(sender, "kitList", "list", ACHelper.getInstance().getKitList(sender));
73  			return;
74  		}
75  		final KitInstance kit = ACHelper.getInstance().getKit(args.getString(0));
76  		if (kit == null) {
77  			Utils.sI18n(sender, "kitNotFound", "kit", args.getString(0));
78  			return;
79  		}
80  
81  		target = Utils.getUser(sender, args, permNode, 1, true);
82  		if (target == null) {
83  			return;
84  		}
85  		if (!PermissionManager.hasPerm(sender, "admincmd.kit." + args.getString(0))) {
86  			return;
87  		}
88  		final ACPlayer actarget = ACPlayer.getPlayer(target);
89  		if (!PermissionManager.hasPerm(sender, "admincmd.item.nodelay", false)) {
90  			final int kitDelay = kit.getDelay();
91  			final long kitLastUse = actarget.getLastKitUse(kit.getName());
92  			if (kitDelay == -1) {
93  				if (kitLastUse == 0)
94  					actarget.updateLastKitUse(kit.getName());
95  				else {
96  					Utils.sI18n(sender, "kitOnce", "kit", kit.getName());
97  					return;
98  				}
99  			} else {
100 				final long nextuse = kitLastUse + kitDelay * 1000L;
101 				final long now = System.currentTimeMillis();
102 				if (now < nextuse) {
103 					final long diff = nextuse - now;
104 					final Long[] timeLeft = Utils.transformToElapsedTime(diff);
105 					final HashMap<String, String> replace = new HashMap<String, String>();
106 					replace.put("d", timeLeft[0].toString());
107 					replace.put("h", timeLeft[1].toString());
108 					replace.put("m", timeLeft[2].toString());
109 					replace.put("s", timeLeft[3].toString());
110 					final String timestamp = (timeLeft[0] > 0 ? (Utils.I18n("days", "d",
111 							timeLeft[0].toString())) : "")
112 							+ (timeLeft[1] > 0 ? (timeLeft[1] + "h ") : "")
113 							+ (timeLeft[2] > 0 ? (timeLeft[2] + "m ") : "")
114 							+ (timeLeft[3] > 0 ? (timeLeft[3] + "s") : "");
115 					Utils.sI18n(sender, "kitDelayNotUp", "delay", timestamp);
116 					return;
117 				}
118 
119 				actarget.updateLastKitUse(kit.getName());
120 			}
121 		}
122 		final HashMap<String, String> replace = new HashMap<String, String>();
123 		replace.put("kit", args.getString(0));
124 		if (Utils.isPlayer(sender, false)) {
125 			if (!target.equals(sender)) {
126 				replace.put("sender", Utils.getPlayerName((Player) sender));
127 				Utils.sI18n(target, "kitOtherPlayer", replace);
128 				replace.remove("sender");
129 				replace.put("target", Utils.getPlayerName(target));
130 				Utils.sI18n(sender, "kitCommandSender", replace);
131 			} else
132 				Utils.sI18n(sender, "kitYourself", replace);
133 		} else {
134 			replace.put("sender", "Server Admin");
135 			Utils.sI18n(target, "kitOtherPlayer", replace);
136 			replace.remove("sender");
137 			replace.put("target", Utils.getPlayerName(target));
138 			Utils.sI18n(sender, "kitCommandSender", replace);
139 		}
140 		final ItemStack[] items = kit.getItemStacks().toArray(new ItemStack[] {});
141 		final Player taskTarget = target;
142 		ACPluginManager.scheduleSyncTask(new Runnable() {
143 			@Override
144 			public void run() {
145 				taskTarget.getInventory().addItem(items);
146 
147 			}
148 		});
149 		if (kit instanceof ArmoredKitInstance)
150 			((ArmoredKitInstance) kit).setPlayerArmorParts(target);
151 
152 	}
