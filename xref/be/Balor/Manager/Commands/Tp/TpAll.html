44  		cmdName = "bal_tpall";

48  	/*
49  	 * (non-Javadoc)
50  	 * 
51  	 * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
52  	 * CommandSender, be.Balor.Manager.Commands.CommandArgs)
53  	 */
54  	@SuppressWarnings("unused")
55  	@Override
56  	public void execute(final CommandSender sender, final CommandArgs args)
57  			throws PlayerNotFound, ActionNotPermitedException {
58  		final Player[] players = ACPluginManager.getServer().getOnlinePlayers();
59  		final HashMap<String, String> replace = new HashMap<String, String>();
60  		String teleporter;
61  		if (Utils.isPlayer(sender, false)) {
62  			teleporter = Utils.getPlayerName((Player) sender);
63  		} else {
64  			teleporter = "Admin";
65  		}
66  		ACWorld world;
67  		if (args.length == 2) {
68  			world = ACWorld.getWorld(args.getString(0));
69  		} else if (args.length >= 4) {
70  			world = ACWorld.getWorld(args.getString(0));
71  		} else {
72  			if (sender instanceof Player) {
73  				world = ACWorld
74  						.getWorld(((Player) sender).getWorld().getName());
75  			} else {
76  				replace.put("arg", "world");
77  				replace.put("cmdName", "/tpall");
78  				LocaleHelper.MISSING_ARG.sendLocale(sender, replace);
79  				return;
80  			}
81  		}
82  		if (world == null) {
83  			replace.put("world", args.getString(0));
84  			Utils.sI18n(sender, "worldNotFound", replace);
85  			return;
86  		}
87  		Location to = null;
88  		if (args.hasFlag('s')) {
89  			// TP to Spawn point
90  			to = world.getSpawn();
91  			replace.put("sender", teleporter);
92  			replace.put("loc", "the Spawn of " + world.getName());
93  			doTeleportAndMessage(players, replace, to);
94  		} else if (args.hasFlag('w')) {
95  			// TP to Warp point
96  			final String value = args.getValueFlag('w');
97  			if (value == null) {
98  				return;
99  			}
100 			final Warp w = world.getWarp(value);
101 			if (w == null) {
102 				replace.clear();
103 				replace.put("name", value);
104 				Utils.sI18n(sender, "errorWarp", replace);
105 				return;
106 			}
107 			to = w.loc;
108 			replace.put("sender", teleporter);
109 			replace.put("loc", w.name);
110 			doTeleportAndMessage(players, replace, to);
111 		} else if (args.hasFlag('p')) {
112 			// TP to Player
113 			final String value = args.getValueFlag('p');
114 			final ACPlayer target = ACPlayer.getPlayer(value);
115 			if (target == null) {
116 				replace.clear();
117 				replace.put("player", value);
118 				Utils.sI18n(sender, "playerNotFound", replace);
119 				return;
120 			}
121 			to = target.getHandler().getLocation();
122 			replace.put("sender", teleporter);
123 			replace.put("loc", Utils.getPlayerName(target.getHandler()));
124 			doTeleportAndMessage(players, replace, to);
125 		} else if (args.hasFlag('l')) {
126 			// TP to Location
127 			String x, y, z;
128 			if (args.length == 3) {
129 				x = args.getString(0);
130 				y = args.getString(1);
131 				z = args.getString(2);
132 			} else if (args.length > 3) {
133 				x = args.getString(1);
134 				y = args.getString(2);
135 				z = args.getString(3);
136 			} else {
137 				replace.clear();
138 				replace.put("cmdName", "/tpall");
139 				replace.put("arg", "Location in x,y,z coordinates");
140 				LocaleHelper.MISSING_ARG.sendLocale(sender, replace);
141 				return;
142 			}
143 			final Double locX = null, locY = null, locZ = null;
144 			parseNumber(sender, x, locX);
145 			parseNumber(sender, y, locY);
146 			parseNumber(sender, z, locZ);
147 			if (locX == null || locY == null || locZ == null) {
148 				return;
149 			}
150 			to = new Location(world.getHandle(), locX, locY, locZ);
151 			replace.put("sender", teleporter);
152 			replace.put("loc", "Location: x=" + x + ", y=" + y + ", z=" + z);
153 			doTeleportAndMessage(players, replace, to);
154 		} else {
155 			// TP to sender if he is a player
156 			if (Utils.isPlayer(sender, false)) {
157 				final Player p = (Player) sender;
158 				to = p.getLocation();
159 				replace.put("sender", teleporter);
160 				replace.put("loc", Utils.getPlayerName((Player) sender));
161 				doTeleportAndMessage(p, players, replace, to);
162 			} else {
163 				replace.put("arg", "world");
164 				replace.put("cmdName", "/tpall");
165 				LocaleHelper.MISSING_ARG.sendLocale(sender, replace);
166 				return;
167 			}
168 		}
169 	}

181 	private void doTeleportAndMessage(final Player[] players,
182 			final HashMap<String, String> replace, final Location loc) {
183 		if (loc == null) {
184 			return;
185 		}
186 		if (players == null) {
187 			return;
188 		}
189 		for (final Player p : players) {
190 			if (p == null) {
191 				continue;
192 			}
193 			p.teleport(loc);
194 			LocaleHelper.TP_ALL.sendLocale(p, replace);
195 		}
196 	}
197 
198 	private void doTeleportAndMessage(final Player sender,
199 			final Player[] players, final HashMap<String, String> replace,
200 			final Location loc) {
201 		if (loc == null) {
202 			return;
203 		}
204 		if (players == null) {
205 			return;
206 		}
207 		for (final Player p : players) {
208 			if (p == null) {
209 				continue;
210 			}
211 			if (p == sender) {
212 				continue;
213 			}
214 			p.teleport(loc);
215 			LocaleHelper.TP_ALL.sendLocale(p, replace);
216 		}
217 	}

219 	private void parseNumber(final CommandSender sender, final String toParse,
220 			Double parsed) {
221 		try {
222 			parsed = (double) Integer.parseInt(toParse);
223 		} catch (final NumberFormatException e) {
224 			Utils.sI18n(sender, "NaN", "number", toParse);
225 			return;
226 		}
227 	}
