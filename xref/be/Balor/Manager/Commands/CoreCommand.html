159 	/***
160 	 * Register the bukkit Permission
161 	 */
162 	public void registerBukkitPerm() {
163 		DebugLog.beginInfo("Registering permission");
164 		try {
165 			if (permNode == null && permNode.isEmpty()) {
166 				return;
167 			}
168 
169 			if (permParent != null) {
170 				DebugLog.beginInfo("Register permission in the permParent");
171 				final PermChild child = new PermChild(permNode, bukkitDefault);
172 				permParent.addChild(child);
173 				permChild = child;
174 				if (other) {
175 					permParent.addChild(new PermChild(permNode + ".other",
176 							bukkitDefault));
177 				}
178 				DebugLog.endInfo();
179 				return;
180 			}
181 			DebugLog.beginInfo("Register permission without a PermParent");
182 			permChild = plugin.getPermissionLinker().addPermChild(permNode,
183 					bukkitDefault);
184 			if (other) {
185 				plugin.getPermissionLinker().addPermChild(permNode + ".other",
186 						bukkitDefault);
187 			}
188 			DebugLog.endInfo();
189 		} finally {
190 			DebugLog.endInfo();
191 		}
192 	}

203 	/***
204 	 * Initialize the bukkit plugin command
205 	 * 
206 	 * @param plugin
207 	 * @throws CommandNotFound
208 	 * @throws CommandAlreadyExist
209 	 */
210 	public void initializeCommand() throws CommandNotFound, CommandAlreadyExist {
211 		DebugLog.beginInfo("Check for bukkit command and status of the command");
212 		try {
213 			if ((pluginCommand = plugin.getCommand(cmdName)) == null) {
214 				throw new CommandNotFound(cmdName
215 						+ " is not loaded in bukkit. Command deactivated", this);
216 			}
217 			DebugLog.beginInfo("Check Alias of the commands");
218 			if (pluginCommand.getAliases().isEmpty()) {
219 				throw new CommandAlreadyExist(
220 						cmdName
221 								+ " has all his alias already registered. Command deactivated",
222 						this);
223 			}
224 		} finally {
225 			DebugLog.endInfo();
226 			DebugLog.endInfo();
227 		}
228 	}
