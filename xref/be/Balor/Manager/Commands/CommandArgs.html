35  	protected final Map<Character, String> valueFlags = new HashMap<Character, String>();

43  	/***
44  		 *
45  		 */
46  	public CommandArgs(final String[] args) {
47  		final List<Integer> argIndexList = new ArrayList<Integer>(args.length);
48  		final List<String> argList = new ArrayList<String>(args.length);
49  		for (int i = 0; i < args.length; ++i) {
50  			String arg = args[i];
51  			if (arg.length() == 0) {
52  				continue;
53  			}
54  
55  			argIndexList.add(i);
56  
57  			switch (arg.charAt(0)) {
58  			case '\'':
59  			case '"':
60  				final StringBuilder build = new StringBuilder();
61  				final char quotedChar = arg.charAt(0);
62  
63  				int endIndex;
64  				for (endIndex = i; endIndex < args.length; ++endIndex) {
65  					final String arg2 = args[endIndex];
66  					if (arg2.charAt(arg2.length() - 1) == quotedChar) {
67  						if (endIndex != i)
68  							build.append(' ');
69  						build.append(arg2.substring(endIndex == i ? 1 : 0, arg2.length() - 1));
70  						break;
71  					} else if (endIndex == i) {
72  						build.append(arg2.substring(1));
73  					} else {
74  						build.append(' ').append(arg2);
75  					}
76  				}
77  
78  				if (endIndex < args.length) {
79  					arg = build.toString();
80  					i = endIndex;
81  				}
82  
83  				// In case there is an empty quoted string
84  				if (arg.length() == 0) {
85  					continue;
86  				}
87  				// else raise exception about hanging quotes?
88  			}
89  			argList.add(arg);
90  		}
91  		// Then flags
92  
93  		final List<Integer> originalArgIndices = new ArrayList<Integer>(argIndexList.size());
94  		this.parsedArgs = new ArrayList<String>(argList.size());
95  
96  		for (int nextArg = 0; nextArg < argList.size();) {
97  			// Fetch argument
98  			final String arg = argList.get(nextArg++);
99  
100 			if (arg.charAt(0) == '-' && arg.length() > 1 && arg.matches("^-[a-zA-Z]+$")) {
101 				for (int i = 1; i < arg.length(); ++i) {
102 					final char flagName = arg.charAt(i);
103 					if (this.valueFlags.containsKey(flagName)) {
104 						continue;
105 					}
106 
107 					if (nextArg >= argList.size())
108 						this.booleanFlags.add(flagName);
109 					else
110 						this.valueFlags.put(flagName, argList.get(nextArg));
111 
112 				}
113 				continue;
114 			}
115 			originalArgIndices.add(argIndexList.get(nextArg - 1));
116 			parsedArgs.add(arg);
117 
118 		}
119 		this.length = parsedArgs.size();
120 	}

186 	/***
187 	 * Get the Value associated with the given flag and remove the flag from the
188 	 * normal arguments.
189 	 * 
190 	 * @param flag
191 	 *            flag to look for.
192 	 * @return null if not found else the value of the flag
193 	 */
194 	public String getValueFlag(final char flag) {
195 		final String result = valueFlags.get(flag);
196 		if (result == null)
197 			return null;
198 		if (parsedArgs.remove(result))
199 			length--;
200 		return result;
201 	}

215 	private String valueFlagsToString() {
216 		final StringBuffer buffer = new StringBuffer();
217 		for (final Entry<Character, String> entry : valueFlags.entrySet()) {
218 			buffer.append(entry.getKey()).append("->").append(entry.getValue()).append(", ");
219 		}
220 		buffer.deleteCharAt(buffer.length() - 1).deleteCharAt(buffer.length() - 1);
221 		return buffer.toString();
222 	}
