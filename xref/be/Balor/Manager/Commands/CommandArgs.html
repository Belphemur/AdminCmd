36  	protected final List<String> parsedArgs;
37  	protected final Map<Character, String> valueFlags = new HashMap<Character, String>();
38  	protected final Set<Character> booleanFlags = new HashSet<Character>();
39  	public int length;

46  	/***
47  		 *
48  		 */
49  	public CommandArgs(final String[] args) {
50  		this.args = args;
51  		final List<Integer> argIndexList = new ArrayList<Integer>(args.length);
52  		final List<String> argList = new ArrayList<String>(args.length);
53  		for (int i = 0; i < args.length; ++i) {
54  			String arg = args[i];
55  			if (arg.length() == 0) {
56  				continue;
57  			}
58  
59  			argIndexList.add(i);
60  
61  			switch (arg.charAt(0)) {
62  			case '\'':
63  			case '"':
64  				final StringBuilder build = new StringBuilder();
65  				final char quotedChar = arg.charAt(0);
66  
67  				int endIndex;
68  				for (endIndex = i; endIndex < args.length; ++endIndex) {
69  					final String arg2 = args[endIndex];
70  					if (arg2.isEmpty()) {
71  						continue;
72  					}
73  					if (arg2.charAt(arg2.length() - 1) == quotedChar) {
74  						if (endIndex != i) {
75  							build.append(' ');
76  						}
77  						if (endIndex == i && arg2.length() == 1) {
78  							continue;
79  						}
80  						build.append(arg2.substring(endIndex == i ? 1 : 0,
81  								arg2.length() - 1));
82  						break;
83  					} else if (endIndex == i) {
84  						build.append(arg2.substring(1));
85  					} else {
86  						build.append(' ').append(arg2);
87  					}
88  				}
89  
90  				if (endIndex < args.length) {
91  					arg = build.toString();
92  					i = endIndex;
93  				}
94  
95  				// In case there is an empty quoted string
96  				if (arg.length() == 0) {
97  					continue;
98  				}
99  				// else raise exception about hanging quotes?
100 			}
101 			argList.add(arg);
102 		}
103 		// Then flags
104 
105 		final List<Integer> originalArgIndices = new ArrayList<Integer>(
106 				argIndexList.size());
107 		this.parsedArgs = new ArrayList<String>(argList.size());
108 
109 		for (int nextArg = 0; nextArg < argList.size();) {
110 			// Fetch argument
111 			final String arg = argList.get(nextArg++);
112 
113 			if (arg.charAt(0) == '-' && arg.length() > 1
114 					&& arg.matches("^-[a-zA-Z]+$")) {
115 				for (int i = 1; i < arg.length(); ++i) {
116 					final char flagName = Character.toLowerCase(arg.charAt(i));
117 					if (this.valueFlags.containsKey(flagName)) {
118 						continue;
119 					}
120 
121 					if (nextArg >= argList.size()) {
122 						this.booleanFlags.add(flagName);
123 					} else {
124 						this.valueFlags.put(flagName, argList.get(nextArg));
125 					}
126 
127 				}
128 				continue;
129 			}
130 			originalArgIndices.add(argIndexList.get(nextArg - 1));
131 			parsedArgs.add(arg);
132 
133 		}
134 		this.length = parsedArgs.size();
135 	}

137 	public String getString(final int index) {
138 		try {
139 			final String result = parsedArgs.get(index);
140 			if (result == null) {
141 				return null;
142 			}
143 			return result;
144 		} catch (final IndexOutOfBoundsException e) {
145 			return null;
146 		}
147 
148 	}

194 	/***
195 	 * Check if the arguments contain the wanted flagF
196 	 * 
197 	 * @param ch
198 	 *            flag to be searched
199 	 * @return true if found.
200 	 */
201 	public boolean hasFlag(final char ch) {
202 		return booleanFlags.contains(Character.toLowerCase(ch))
203 				|| valueFlags.containsKey(Character.toLowerCase(ch));
204 	}

206 	/***
207 	 * Get the Value associated with the given flag and remove the flag from the
208 	 * normal arguments.
209 	 * 
210 	 * @param flag
211 	 *            flag to look for.
212 	 * @return null if not found else the value of the flag
213 	 */
214 	public String getValueFlag(final char flag) {
215 		final String result = valueFlags.get(Character.toLowerCase(flag));
216 		if (result == null) {
217 			return null;
218 		}
219 		if (parsedArgs.remove(result)) {
220 			length--;
221 		}
222 		return result;
223 	}

231 	public String toString() {
232 		return Arrays.toString(parsedArgs.toArray(new String[] {}))
233 				+ " BoolFlags : "
234 				+ Arrays.toString(booleanFlags.toArray(new Character[] {}))
235 				+ " ValFlags : "
236 				+ (valueFlags.isEmpty() ? "[]" : "[" + valueFlagsToString()
237 						+ "]");
238 	}

240 	private String valueFlagsToString() {
241 		final StringBuffer buffer = new StringBuffer();
242 		for (final Entry<Character, String> entry : valueFlags.entrySet()) {
243 			buffer.append(entry.getKey()).append("->").append(entry.getValue())
244 					.append(", ");
245 		}
246 		buffer.deleteCharAt(buffer.length() - 1).deleteCharAt(
247 				buffer.length() - 1);
248 		return buffer.toString();
249 	}

261 	public String concatWithout(final int index) {
262 		final StringBuffer buffer = new StringBuffer();
263 		for (int i = 0; i < args.length; i++) {
264 			if (i == index) {
265 				continue;
266 			}
267 			buffer.append(args[i]).append(" ");
268 		}
269 		return buffer.toString().trim();
270 	}
