36  public class Help extends ServerCommand {
37  
38  	/***
39  	 *
40  	 */
41  	public Help() {
42  		permNode = "admincmd.server.help";
43  		cmdName = "bal_help";
44  	}
45  
46  	/*
47  	 * (non-Javadoc)
48  	 * 
49  	 * @see be.Balor.Manager.ACCommand#execute(org.bukkit.command.CommandSender,
50  	 * java.lang.String[])
51  	 */
52  	@Override
53  	public void execute(final CommandSender sender, final CommandArgs args)
54  			throws ActionNotPermitedException, PlayerNotFound {
55  		if (args.length == 0) {
56  			HelpLister.getInstance().sendHelpPage("AdminCmd", 1, sender);
57  			return;
58  		}
59  		if (args.getString(0).equalsIgnoreCase("list")
60  				|| args.getString(0).equalsIgnoreCase("plugins")) {
61  			String msg = "";
62  			sender.sendMessage(ChatColor.DARK_AQUA
63  					+ ACMinecraftFontWidthCalculator
64  							.strPadCenterChat(ChatColor.WHITE + " Plugins "
65  									+ ChatColor.DARK_AQUA, '-'));
66  			for (final String plugin : HelpLister.getInstance().getPluginList()) {
67  				msg += plugin + ", ";
68  			}
69  			if (!msg.equals("")) {
70  				if (msg.endsWith(", ")) {
71  					msg = msg.substring(0, msg.lastIndexOf(","));
72  				}
73  				sender.sendMessage(msg);
74  			}
75  			return;
76  		}
77  		final String cmd = args.getValueFlag('s');
78  		if (cmd != null) {
79  			final HashMap<String, String> replace = new HashMap<String, String>();
80  			replace.put("type", LocaleHelper.TYPE_CMD.getLocale());
81  			replace.put("value", cmd);
82  			if (!HelpLister.getInstance().sendHelpCmd(args.getString(0), cmd,
83  					sender)) {
84  				LocaleHelper.DONT_EXISTS.sendLocale(sender, replace);
85  			}
86  			return;
87  		}
88  		final String cmd2 = args.getValueFlag('d');
89  		if (cmd2 != null) {
90  			final HashMap<String, String> replace = new HashMap<String, String>();
91  			replace.put("type", LocaleHelper.TYPE_CMD.getLocale());
92  			replace.put("value", cmd2);
93  			if (!HelpLister.getInstance().sendHelpCmd(args.getString(0), cmd2,
94  					sender, true)) {
95  				LocaleHelper.DONT_EXISTS.sendLocale(sender, replace);
96  			}
97  			return;
98  		}
99  		int page = 1;
100 		try {
101 			page = args.getInt(0);
102 			HelpLister.getInstance().sendHelpPage("AdminCmd", page, sender);
103 		} catch (final NumberFormatException e) {
104 			if (args.length == 1) {
105 				if (!HelpLister.getInstance().sendHelpPage(args.getString(0),
106 						1, sender)) {
107 					Utils.sI18n(sender, "pluginNotFound", "plugin",
108 							args.getString(0));
109 				}
110 			} else {
111 				try {
112 					page = args.getInt(1);
113 					if (!HelpLister.getInstance().sendHelpPage(
114 							args.getString(0), page, sender)) {
115 						Utils.sI18n(sender, "pluginNotFound", "plugin",
116 								args.getString(0));
117 					}
118 				} catch (final NumberFormatException e1) {
119 					Utils.sI18n(sender, "NaN", "number", args.getString(1));
120 				}
121 			}
122 		}
123 	}
