35  public class WorldDifficulty extends ServerCommand {
36  
37  	/***
38  	 *
39  	 */
40  	public WorldDifficulty() {
41  		cmdName = "bal_difficulty";
42  		permNode = "admincmd.server.difficulty";
43  	}
44  
45  	/*
46  	 * (non-Javadoc)
47  	 * 
48  	 * @see
49  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
50  	 * java.lang.String[])
51  	 */
52  	@Override
53  	public void execute(final CommandSender sender, final CommandArgs args) {
54  		ACWorld world = null;
55  		boolean worldGiven = false;
56  		int difValue = -1;
57  		try {
58  			try {
59  				difValue = args.getInt(0);
60  			} catch (final NumberFormatException e) {
61  				worldGiven = true;
62  			}
63  			if (args.length >= 1 && worldGiven)
64  				world = ACWorld.getWorld(args.getString(0));
65  			else if (Utils.isPlayer(sender, false))
66  				world = ACWorld.getWorld(((Player) sender).getWorld().getName());
67  			else {
68  				final HashMap<String, String> replace = new HashMap<String, String>();
69  				replace.put("argument", "world");
70  				Utils.sI18n(sender, "errorInsufficientArguments", replace);
71  				return;
72  			}
73  		} catch (final WorldNotLoaded e) {
74  			Utils.sI18n(sender, "worldNotFound");
75  			return;
76  		}
77  
78  		Difficulty toSet = Difficulty.NORMAL;
79  		final Map<String, String> replace = new HashMap<String, String>();
80  		if (args.hasFlag('g')) {
81  			replace.put("world", world.getName());
82  			replace.put("difficulty", world.getDifficulty().toString());
83  			Utils.sI18n(sender, "getDifficulty", replace);
84  		} else if (args.hasFlag('s')) {
85  			if (args.length >= 2)
86  				toSet = Difficulty.getByValue(args.getInt(1));
87  			else if (difValue != -1)
88  				toSet = Difficulty.getByValue(difValue);
89  
90  			if (toSet == null)
91  				toSet = Difficulty.NORMAL;
92  
93  			replace.put("world", world.getName());
94  			replace.put("difficulty", toSet.toString());
95  			world.setDifficulty(toSet);
96  			Utils.sI18n(sender, "setDifficulty", replace);
97  		}
98  		return;
99  	}
