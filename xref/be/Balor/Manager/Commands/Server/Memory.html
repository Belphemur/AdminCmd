100 				|| args.hasFlag('a') || args.hasFlag('n') || args.hasFlag('v') || args.hasFlag('c')
101 				|| args.hasFlag('b')) {
102 			int count = 0;
103 			final HashMap<String, List<Entity>> entityList = new HashMap<String, List<Entity>>(
104 					sender.getServer().getWorlds().size());
105 			final List<World> worlds = sender.getServer().getWorlds();
106 			final Semaphore sema = new Semaphore(0);
107 
108 			ACPluginManager.getScheduler().scheduleSyncDelayedTask(plugin, new Runnable() {
109 				@SuppressWarnings("unchecked")
110 				@Override
111 				public void run() {
112 					for (final World w : worlds) {
113 						final net.minecraft.server.World cWorld = ((CraftWorld) w).getHandle();
114 						synchronized (cWorld.entityList) {
115 							entityList.put(w.getName(), new ArrayList<Entity>(cWorld.entityList));
116 							sema.release();
117 						}
118 
119 					}
120 				}
121 			});
122 			for (final World w : worlds) {
123 				try {
124 					sema.acquire();
125 				} catch (final InterruptedException e) {
126 				}
127 				for (final Entity entity : entityList.get(w.getName())) {
128 					if (dontKill(args, entity))
129 						continue;
130 					entity.die();
131 					count++;
132 				}
133 			}
134 			System.gc();
135 			sender.sendMessage("Freed Entities : " + count);
136 			try {
137 				Thread.sleep(500);
138 			} catch (final InterruptedException e) {
139 			}
140 		}
141 		final long usedMB = (Runtime.getRuntime().totalMemory() - Runtime.getRuntime().freeMemory()) / 1024L / 1024L;
142 		sender.sendMessage(ChatColor.GOLD + "Max Memory : " + ChatColor.WHITE

149 					+ w.getLoadedChunks().length + " chunks, " + w.getEntities().size()

152

155 		if (args.length >= 1)
156 			try {
157 				delay = args.getLong(0);
158 			} catch (final NumberFormatException e) {
159 				final HashMap<String, String> replace = new HashMap<String, String>();
160 				replace.put("number", args.getString(0));
161 				Utils.sI18n(sender, "NaN", replace);
162 				return;
163 			}
164 		if (delay < 20L)
165 			delay = 20L;
166 		final World world = ACPluginManager.getServer().getWorlds().get(0);
167 		ACPluginManager.getScheduler().scheduleSyncDelayedTask(
168 				ACHelper.getInstance().getCoreInstance(),
169 				new CheckTicks(System.nanoTime(), world, world.getFullTime(), sender), delay);
170 	}

197 	private class CheckTicks implements Runnable {
198 
199 		protected long oldNanoTime;
200 		protected long elapsedNanoTime;
201 		protected World world;
202 		protected long start;
203 		protected double ticksPerSecond;
204 		protected long elapsedTicks;
205 		protected CommandSender sender;
206 
207 		public CheckTicks(final long oldNanoTime, final World world, final long start,
208 				final CommandSender sender) {
209 			this.oldNanoTime = oldNanoTime;
210 			this.world = world;
211 			this.start = start;
212 			this.sender = sender;
213 		}
214 
215 		/*
216 		 * (non-Javadoc)
217 		 * 
218 		 * @see java.lang.Runnable#run()
219 		 */
220 		@Override
221 		public void run() {
222 			elapsedNanoTime = System.nanoTime() - oldNanoTime;
223 			elapsedTicks = world.getFullTime() - start;
224 			ticksPerSecond = (elapsedTicks * 1000000000.0) / elapsedNanoTime;
225 			sender.sendMessage("[AdminCmd] TPS: " + ticksPerSecond + " | Ticks elapsed: "
226 					+ elapsedTicks + " | Nano Time:" + elapsedNanoTime);
227 		}
228 	}

230 	private boolean dontKill(final CommandArgs args, final Entity toKill) {
231 		boolean dontKill = true;
232 		if (args.hasFlag('f'))
233 			dontKill = (toKill instanceof EntityHuman || toKill instanceof EntityPainting);
234 		if (args.hasFlag('x'))
235 			dontKill = !(toKill instanceof EntityExperienceOrb);
236 		if (args.hasFlag('i'))
237 			dontKill = !(toKill instanceof EntityItem);
238 		if (args.hasFlag('m'))
239 			dontKill = !(toKill instanceof EntityMonster);
240 		if (args.hasFlag('a'))
241 			dontKill = !(toKill instanceof EntityAnimal);
242 		if (args.hasFlag('n'))
243 			dontKill = !(toKill instanceof EntityVillager);
244 		if (args.hasFlag('c'))
245 			dontKill = !(toKill instanceof EntityMinecart);
246 		if (args.hasFlag('b'))
247 			dontKill = !(toKill instanceof EntityBoat);
248 		if (args.hasFlag('v'))
249 			dontKill = !(toKill instanceof EntityMinecart || toKill instanceof EntityBoat);
250 		return dontKill;
251 	}
