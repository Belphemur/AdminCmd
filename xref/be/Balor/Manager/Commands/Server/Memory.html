88  			return;
89  		}
90  		if (args.hasFlag('v') && !PermissionManager.hasPerm(sender, vehicle)) {
91  			return;
92  		}
93  		if (args.hasFlag('f') || args.hasFlag('x') || args.hasFlag('i')
94  				|| args.hasFlag('m') || args.hasFlag('a') || args.hasFlag('n')
95  				|| args.hasFlag('v') || args.hasFlag('c') || args.hasFlag('b')) {
96  			int count = 0;
97  			final HashMap<String, List<Object>> entityList = new HashMap<String, List<Object>>(
98  					sender.getServer().getWorlds().size());
99  			final List<World> worlds = sender.getServer().getWorlds();
100 			final Semaphore sema = new Semaphore(0);
101 
102 			ACPluginManager.getScheduler().scheduleSyncDelayedTask(plugin,
103 					new Runnable() {
104 						@Override
105 						public void run() {
106 							for (final World w : worlds) {
107 								final Object cWorld = MinecraftReflection
108 										.getHandle(w);
109 								List<Object> wEntityList = null;
110 								try {
111 									wEntityList = FieldUtils.getField(cWorld,
112 											"entityList");
113 								} catch (final Exception e) {
114 									throw new RuntimeException(
115 											"Cannot get entityList from "
116 													+ cWorld, e);
117 								}
118 								synchronized (wEntityList) {
119 									entityList.put(w.getName(),
120 											new ArrayList<Object>(wEntityList));
121 									sema.release();
122 								}
123 
124 							}
125 						}
126 					});
127 			final MethodHandler die = new MethodHandler(
128 					MinecraftReflection.getEntityClass(), "die");
129 			for (final World w : worlds) {
130 				try {
131 					sema.acquire();
132 				} catch (final InterruptedException e) {
133 				}
134 				for (final Object entity : entityList.get(w.getName())) {
135 					if (dontKill(args, entity)) {
136 						continue;
137 					}
138 					die.invoke(entity);
139 					count++;
140 				}
141 			}
142 			System.gc();
143 			sender.sendMessage("Freed Entities : " + count);
144 			try {
145 				Thread.sleep(500);
146 			} catch (final InterruptedException e) {
147 			}
148 		}
149 		final long usedMB = (Runtime.getRuntime().totalMemory() - Runtime
150 				.getRuntime().freeMemory()) / 1024L / 1024L;
151 		sender.sendMessage(ChatColor.GOLD + "Max Memory : " + ChatColor.WHITE
152 				+ Runtime.getRuntime().maxMemory() / 1024L / 1024L + "MB");
153 		sender.sendMessage(ChatColor.DARK_RED + "Used Memory : "
154 				+ ChatColor.WHITE + usedMB + "MB");
155 		sender.sendMessage(ChatColor.DARK_GREEN + "Free Memory : "
156 				+ ChatColor.WHITE + Runtime.getRuntime().freeMemory() / 1024L
157 				/ 1024L + "MB");
158 		for (final World w : sender.getServer().getWorlds()) {
159 			sender.sendMessage(w.getEnvironment() + " \"" + w.getName()
160 					+ "\": " + w.getLoadedChunks().length + " chunks, "
161 					+ w.getEntities().size() + " entities");
162 		}
163 
164 		// Code for TPS from here on
165 		long delay = 40L;
166 		if (args.length >= 1) {
167 			try {
168 				delay = args.getLong(0);
169 			} catch (final NumberFormatException e) {
170 				final HashMap<String, String> replace = new HashMap<String, String>();
171 				replace.put("number", args.getString(0));
172 				Utils.sI18n(sender, "NaN", replace);
173 				return;
174 			}
175 		}
176 		if (delay < 20L) {
177 			delay = 20L;
178 		}
179 		final World world = ACPluginManager.getServer().getWorlds().get(0);
180 		ACPluginManager.getScheduler().scheduleSyncDelayedTask(
181 				ACHelper.getInstance().getCoreInstance(),
182 				new CheckTicks(System.nanoTime(), world, world.getFullTime(),
183 						sender), delay);
184 	}

221 	private class CheckTicks implements Runnable {
222 
223 		protected long oldNanoTime;
224 		protected long elapsedNanoTime;
225 		protected World world;
226 		protected long start;
227 		protected double ticksPerSecond;
228 		protected long elapsedTicks;
229 		protected CommandSender sender;
230 
231 		public CheckTicks(final long oldNanoTime, final World world,
232 				final long start, final CommandSender sender) {
233 			this.oldNanoTime = oldNanoTime;
234 			this.world = world;
235 			this.start = start;
236 			this.sender = sender;
237 		}
238 
239 		/*
240 		 * (non-Javadoc)
241 		 * 
242 		 * @see java.lang.Runnable#run()
243 		 */
244 		@Override
245 		public void run() {
246 			elapsedNanoTime = System.nanoTime() - oldNanoTime;
247 			elapsedTicks = world.getFullTime() - start;
248 			ticksPerSecond = (elapsedTicks * 1000000000.0) / elapsedNanoTime;
249 			sender.sendMessage("[AdminCmd] TPS: " + ticksPerSecond
250 					+ " | Ticks elapsed: " + elapsedTicks + " | Nano Time:"
251 					+ elapsedNanoTime);
252 		}
253 	}

255 	private boolean dontKill(final CommandArgs args, final Object toKill) {
256 		boolean dontKill = true;
257 		if (args.hasFlag('f')) {
258 			dontKill = (MinecraftReflection
259 					.instanceOfNMS(toKill, "EntityHuman") || MinecraftReflection
260 					.instanceOfNMS(toKill, "EntityPainting"));
261 		}
262 		if (args.hasFlag('x')) {
263 			dontKill = !MinecraftReflection.instanceOfNMS(toKill,
264 					"EntityExperienceOrb");
265 		}
266 		if (args.hasFlag('i')) {
267 			dontKill = !MinecraftReflection.instanceOfNMS(toKill, "EntityItem");
268 		}
269 		if (args.hasFlag('m')) {
270 			dontKill = !MinecraftReflection.instanceOfNMS(toKill,
271 					"EntityMonster");
272 		}
273 		if (args.hasFlag('a')) {
274 			dontKill = !MinecraftReflection.instanceOfNMS(toKill,
275 					"EntityAnimal");
276 		}
277 		if (args.hasFlag('n')) {
278 			dontKill = !MinecraftReflection.instanceOfNMS(toKill,
279 					"EntityVillager");
280 		}
281 		if (args.hasFlag('c')) {
282 			dontKill = !MinecraftReflection.instanceOfNMS(toKill,
283 					"EntityMinecart");
284 		}
285 		if (args.hasFlag('b')) {
286 			dontKill = !MinecraftReflection.instanceOfNMS(toKill, "EntityBoat");
287 		}
288 		if (args.hasFlag('v')) {
289 			dontKill = !(MinecraftReflection.instanceOfNMS(toKill,
290 					"EntityMinecart") || MinecraftReflection.instanceOfNMS(
291 					toKill, "EntityBoat"));
292 		}
293 		return dontKill;
