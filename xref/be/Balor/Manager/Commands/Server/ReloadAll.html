31  public class ReloadAll extends ServerCommand {
32  
33  	/***
34  	 * 
35  	 */
36  	public ReloadAll() {
37  		permNode = "admincmd.server.reload";
38  		cmdName = "bal_reloadall";
39  	}
40  
41  	/*
42  	 * (non-Javadoc)
43  	 * 
44  	 * @see
45  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
46  	 * java.lang.String[])
47  	 */
48  	@Override
49  	public void execute(final CommandSender sender, final CommandArgs args) {
50  		ACHelper.getInstance().saveElapsedTime();
51  		final boolean bcast = ConfigEnum.BSRELOAD.getBoolean();
52  		ACPluginManager.scheduleSyncTask(new Runnable() {
53  			@Override
54  			public void run() {
55  				ACPluginManager.getServer().reload();
56  			}
57  		});
58  		if (bcast)
59  			Utils.broadcastMessage(Utils.I18n("serverReload"));
60  		else
61  			Utils.sI18n(sender, "serverReload");
62  	}
63

66  	 * 
67  	 * @see be.Balor.Manager.ACCommands#argsCheck(java.lang.String[])

69  	@Override

72  	}
