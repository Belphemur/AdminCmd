44  	/*
45  	 * (non-Javadoc)
46  	 * 
47  	 * @see
48  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
49  	 * java.lang.String[])
50  	 */
51  	@Override
52  	public void execute(final CommandSender sender, final CommandArgs args)
53  			throws ActionNotPermitedException, PlayerNotFound {
54  		String message = "";
55  		if (args.hasFlag('m')) {
56  			if (PermissionManager.hasPerm(sender, "admincmd.server.set.motd")) {
57  				for (final String msg : args) {
58  					message += msg + " ";
59  				}
60  				message = message.trim();
61  				String result = Utils.colorParser(message);
62  				if (result == null) {
63  					result = message;
64  				}
65  				TextLocale.MOTD.saveContent(result);
66  				Utils.sI18n(sender, "MOTDset", "motd", result);
67  			}
68  		} else if (args.hasFlag('n')) {
69  			if (PermissionManager.hasPerm(sender, "admincmd.server.set.news")) {
70  				for (final String msg : args) {
71  					message += msg + " ";
72  				}
73  				message = message.trim();
74  				String result = Utils.colorParser(message);
75  				if (result == null) {
76  					result = message;
77  				}
78  				TextLocale.NEWS.saveContent(result);
79  				Utils.sI18n(sender, "NEWSset", "news", result);
80  			}
81  		} else if (args.hasFlag('r')) {
82  			if (PermissionManager.hasPerm(sender, "admincmd.server.set.rules")) {
83  				for (final String msg : args) {
84  					message += msg + " ";
85  				}
86  				message = message.trim();
87  				String result = Utils.colorParser(message);
88  				if (result == null) {
89  					result = message;
90  				}
91  				TextLocale.RULES.saveContent(result);
92  				Utils.sI18n(sender, "RulesSet", "rules", result);
93  			}
94  		} else if (args.hasFlag('u')) {
95  			if (PermissionManager.hasPerm(sender, "admincmd.server.set.motd")) {
96  				for (final String msg : args) {
97  					message += msg + " ";
98  				}
99  				message = message.trim();
100 				String result = Utils.colorParser(message);
101 				if (result == null) {
102 					result = message;
103 				}
104 				TextLocale.MOTD_NEW.saveContent(result);
105 				Utils.sI18n(sender, "MOTDset", "motd", result);
106 			}
107 		}
108 	}
