38  public class Mute extends PlayerCommand {
39  	private final PermChild cmdMute;
40  
41  	/***
42  	 *
43  	 */
44  	public Mute() {
45  		permNode = "admincmd.player.mute";
46  		cmdName = "bal_mute";
47  		cmdMute = new PermChild(permNode + ".command");
48  	}

50  	/*
51  	 * (non-Javadoc)
52  	 * 
53  	 * @see
54  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
55  	 * java.lang.String[])
56  	 */
57  	@Override
58  	public void execute(final CommandSender sender, final CommandArgs args) {
59  		final Player player = sender.getServer().getPlayer(args.getString(0));
60  		if (args.hasFlag('c') && !PermissionManager.hasPerm(sender, cmdMute.getBukkitPerm()))
61  			return;
62  		if (player != null) {
63  			final HashMap<String, String> replace = new HashMap<String, String>();
64  			replace.put("player", Utils.getPlayerName(player));
65  			final ACPlayer acp = ACPlayer.getPlayer(player.getName());
66  			if (acp instanceof EmptyPlayer) {
67  				Utils.sI18n(sender, "playerNotFound", replace);
68  				return;
69  			}
70  			if (!Utils.checkImmunity(sender, player)) {
71  				Utils.sI18n(sender, "insufficientLvl");
72  				return;
73  			}
74  			if (args.hasFlag('c')) {
75  				if (!acp.hasPower(Type.MUTED_COMMAND)) {
76  					String msg = "Server Admin";
77  					if (Utils.isPlayer(sender, false))
78  						msg = Utils.getPlayerName((Player) sender);
79  					acp.setPower(Type.MUTED_COMMAND, "Muted(including commands) by " + msg);
80  					if (!player.equals(sender))
81  						Utils.sI18n(sender, "commandMuteEnabledTarget", replace);
82  					if (args.length >= 2) {
83  						Integer tmpMute = null;
84  						try {
85  							tmpMute = args.getInt(args.length - 1);
86  							final String unmute = player.getName();
87  							final CommandSender senderFinal = sender;
88  							ACPluginManager.getScheduler().scheduleAsyncDelayedTask(getPlugin(),
89  									new Runnable() {
90  
91  										@Override
92  										public void run() {
93  											ACPlayer.getPlayer(unmute).removePower(
94  													Type.MUTED_COMMAND);
95  											Utils.sI18n(senderFinal, "commandMuteDisabledTarget",
96  													"player", unmute);
97  										}
98  									}, 20 * 60 * tmpMute);
99  
100 						} catch (final Exception e) {
101 						}
102 						if (tmpMute == null)
103 							Utils.sI18n(player, "commandMuteEnabled");
104 						else
105 							Utils.sI18n(player, "commandTmpMuteEnabled", "minutes",
106 									tmpMute.toString());
107 					}
108 				} else
109 					Utils.sI18n(sender, "alreadyCommandMuted");
110 				return;
111 			}
112 			if (!acp.hasPower(Type.MUTED)) {
113 				String msg = "Server Admin";
114 				if (Utils.isPlayer(sender, false))
115 					msg = Utils.getPlayerName((Player) sender);
116 				acp.setPower(Type.MUTED, "Muted by " + msg);
117 				if (!player.equals(sender))
118 					Utils.sI18n(sender, "muteEnabledTarget", replace);
119 				if (args.length >= 2) {
120 					Integer tmpMute = null;
121 					try {
122 						tmpMute = args.getInt(args.length - 1);
123 						final String unmute = player.getName();
124 						final CommandSender senderFinal = sender;
125 						ACPluginManager.getScheduler().scheduleAsyncDelayedTask(getPlugin(),
126 								new Runnable() {
127 
128 									@Override
129 									public void run() {
130 										ACPlayer.getPlayer(unmute).removePower(Type.MUTED);
131 										Utils.sI18n(senderFinal, "muteDisabledTarget", "player",
132 												unmute);
133 									}
134 								}, 20 * 60 * tmpMute);
135 
136 					} catch (final Exception e) {
137 					}
138 					if (tmpMute == null)
139 						Utils.sI18n(player, "muteEnabled");
140 					else
141 						Utils.sI18n(player, "tmpMuteEnabled", "minutes", tmpMute.toString());
142 				}
143 			} else
144 				Utils.sI18n(sender, "alreadyMuted");
145 
146 		}
147 
148 	}
