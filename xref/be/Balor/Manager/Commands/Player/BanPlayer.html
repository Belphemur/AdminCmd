38  public class BanPlayer extends PlayerCommand {
39  
40  	/***
41  	 *
42  	 */
43  	public BanPlayer() {
44  		permNode = "admincmd.player.ban";
45  		cmdName = "bal_ban";
46  	}
47  
48  	/*
49  	 * (non-Javadoc)
50  	 * 
51  	 * @see
52  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
53  	 * java.lang.String[])
54  	 */
55  	@Override
56  	public void execute(final CommandSender sender, final CommandArgs args) {
57  		final Player toBan = sender.getServer().getPlayer(args.getString(0));
58  		final HashMap<String, String> replace = new HashMap<String, String>();
59  		String message = "";
60  		String banPlayerString;
61  		if (toBan != null) {
62  			banPlayerString = toBan.getName();
63  			if (!Utils.checkImmunity(sender, toBan)) {
64  				Utils.sI18n(sender, "insufficientLvl");
65  				return;
66  			}
67  		} else
68  			banPlayerString = args.getString(0);
69  		Integer tmpBan = null;
70  		if (args.length >= 2) {
71  			if (args.hasFlag('m'))
72  				message = LocaleManager.getInstance().get("kickMessages", args.getValueFlag('m'),
73  						"player", banPlayerString);
74  			if (message == null || (message != null && message.isEmpty())) {
75  				message = "";
76  				for (int i = 1; i < args.length - 1; i++)
77  					message += args.getString(i) + " ";
78  			}
79  			try {
80  				tmpBan = args.getInt(args.length - 1);
81  			} catch (final Exception e) {
82  				if (!args.hasFlag('m'))
83  					message += args.getString(args.length - 1);
84  			}
85  			if (tmpBan != null) {
86  				message += "(Banned for " + tmpBan + " minutes)";
87  				final String unban = banPlayerString;
88  				ACPluginManager.getScheduler().scheduleAsyncDelayedTask(getPlugin(),
89  						new Runnable() {
90  
91  							@Override
92  							public void run() {
93  								ACHelper.getInstance().unBanPlayer(unban);
94  								final String unbanMsg = Utils.I18n("unban", "player", unban);
95  								if (unbanMsg != null)
96  									Utils.broadcastMessage(unbanMsg);
97  							}
98  						}, 20 * 60 * tmpBan);
99  			}
100 		} else {
101 			message = "You have been banned by ";
102 			if (!Utils.isPlayer(sender, false))
103 				message += "Server Admin";
104 			else
105 				message += Utils.getPlayerName((Player) sender);
106 		}
107 		message = message.trim();
108 		replace.put("player", banPlayerString);
109 		replace.put("reason", message);
110 		ACPlayer.getPlayer(toBan).setPower(Type.KICKED);
111 		if (toBan != null) {
112 			final String finalmsg = message;
113 			final Player finalToKick = toBan;
114 			ACPluginManager.scheduleSyncTask(new Runnable() {
115 
116 				@Override
117 				public void run() {
118 					finalToKick.kickPlayer(finalmsg);
119 				}
120 			});
121 		}
122 		if (tmpBan != null)
123 			ACHelper.getInstance().addBannedPlayer(
124 					new TempBannedPlayer(banPlayerString, message, tmpBan * 60 * 1000));
125 		else
126 			ACHelper.getInstance().addBannedPlayer(new BannedPlayer(banPlayerString, message));
127 		Utils.broadcastMessage(Utils.I18n("ban", replace));
128 
129 	}
