69  	@Override
70  	public void execute(final CommandSender sender, final CommandArgs args)
71  			throws ActionNotPermitedException, PlayerNotFound {
72  		final Player toBan = Utils.getPlayer(args.getString(0));
73  		final HashMap<String, String> replace = new HashMap<String, String>();
74  		String message = null;
75  		String banPlayerString;
76  		if (toBan != null) {
77  			banPlayerString = toBan.getName();
78  			if (!Utils.checkImmunity(sender, toBan)) {
79  				Utils.sI18n(sender, "insufficientLvl");
80  				return;
81  			}
82  		} else {
83  			banPlayerString = args.getString(0);
84  		}
85  		Integer tmpBan = null;
86  		if (args.length >= 2) {
87  			if (args.hasFlag('m')) {
88  				message = LocaleManager.getInstance().get("kickMessages",
89  						args.getValueFlag('m'), "player", banPlayerString);
90  			}
91  			tmpBan = checkTempBan(args, banPlayerString, sender);
92  			if (tmpBan != null && tmpBan == -1) {
93  				return;
94  			}
95  		}
96  		message = parseMessage(args, message, tmpBan == null, sender);
97  		replace.put("player", banPlayerString);
98  		replace.put("reason", message);
99  		final IBan ban = getBanType(banPlayerString, tmpBan, message, sender,
100 				replace);
101 
102 		if (ban == null) {
103 			return;
104 		}
105 		ACHelper.getInstance().banPlayer(ban);
106 		if (ban instanceof ITempBan) {
107 			ACPluginManager.getScheduler().runTaskLaterAsynchronously(
108 					getPlugin(), new UnBanTask((ITempBan) ban, true),
109 					Utils.secInTick * 60 * tmpBan);
110 			message += " " + ((ITempBan) ban).getReadableTimeLeft();
111 		}
112 		if (ConfigEnum.ADD_BANNER_IN_BAN.getBoolean()) {
113 			message += " by " + ban.getBanner();
114 		}
115 		if (toBan != null) {
116 			ACPlayer.getPlayer(toBan).setPower(Type.KICKED);
117 			new KickTask(toBan, message).scheduleSync();
118 		}
119 		Utils.broadcastMessage(Utils.I18n("ban", replace));
120 
121 	}

124 	 * @param banPlayerString
125 	 * @param tmpBan
126 	 * @param message
127 	 * @param sender
128 	 * @param replace
129 	 * @return
130 	 */
131 	private IBan getBanType(final String banPlayerString, final Integer tmpBan,
132 			final String message, final CommandSender sender,
133 			final HashMap<String, String> replace) {
134 		final Matcher ipv4 = Utils.REGEX_IP_V4.matcher(banPlayerString);
135 		final Matcher inaccurateIp = Utils.REGEX_INACCURATE_IP_V4
136 				.matcher(banPlayerString);
137 		Ban toDo;
138 		if (tmpBan != null) {
139 			replace.put("reason", message);
140 			if (inaccurateIp.find()) {
141 				if (!ipv4.find()) {
142 					replace.clear();
143 					replace.put("ip", banPlayerString);
144 					LocaleHelper.INACC_IP.sendLocale(sender, replace);
145 					return null;
146 				}
147 				toDo = new TempBannedIP(banPlayerString, message,
148 						tmpBan * 60 * 1000);
149 			} else {
150 				toDo = new TempBannedPlayer(banPlayerString, message,
151 						tmpBan * 60 * 1000);
152 			}
153 
154 		} else {
155 			if (inaccurateIp.find()) {
156 				if (!ipv4.find()) {
157 					replace.clear();
158 					replace.put("ip", banPlayerString);
159 					LocaleHelper.INACC_IP.sendLocale(sender, replace);
160 					return null;
161 				}
162 				toDo = new BannedIP(banPlayerString, message);
163 			} else {
164 				toDo = new BannedPlayer(banPlayerString, message);
165 			}
166 		}
167 		if (!Utils.isPlayer(sender, false)) {
168 			toDo.setBanner("Server Admin");
169 		} else {
170 			toDo.setBanner(ChatColor.stripColor(Utils
171 					.getPlayerName((Player) sender)));
172 		}
173 		return toDo;
174 	}

176 	/***
177 	 * @param args
178 	 * @param message
179 	 * @param b
180 	 * @return
181 	 */
182 	private String parseMessage(final CommandArgs args, String message,
183 			final boolean tempBan, final CommandSender sender) {
184 		if (message == null || (message != null && message.isEmpty())) {
185 			message = "";
186 			if (tempBan) {
187 				for (int i = 1; i < args.length; i++) {
188 					message += args.getString(i) + " ";
189 				}
190 			} else {
191 				for (int i = 1; i < args.length - 1; i++) {
192 					message += args.getString(i) + " ";
193 				}
194 			}
195 		}
196 		message = message.trim();
197 		if (message.isEmpty()) {
198 			message = LocaleHelper.DEFAULT_BAN_MESSAGE.getLocale();
199 		}
200 		return message;
201 	}

209 	private Integer checkTempBan(final CommandArgs args,
210 			final String banPlayerString, final CommandSender sender) {
211 		try {
212 			final int tmpIntTime = Utils.timeParser(args
213 					.getString(args.length - 1));
214 			if (tmpIntTime != -1) {
215 				return tmpIntTime;
216 			}
217 		} catch (final NotANumberException e) {
218 			Utils.sI18n(sender, "NaN", "number",
219 					args.getString(args.length - 1));
220 			return -1;
221 		} catch (final Exception ex) {
222 		}
223 		return null;
224 	}
