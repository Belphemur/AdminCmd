34  public class Experience extends PlayerCommand {
35  
36  	public Experience() {
37  		permNode = "admincmd.player.experience";
38  		cmdName = "bal_exp";
39  		other = true;
40  	}
41  
42  	/*
43  	 * (non-Javadoc)
44  	 * 
45  	 * @see
46  	 * be.Balor.Manager.ACCommands#execute(org.bukkit.command.CommandSender,
47  	 * java.lang.String[])
48  	 */
49  	@Override
50  	public void execute(final CommandSender sender, final CommandArgs args) {
51  		float amount = 0;
52  		Player target = null;
53  		boolean self = false;
54  		if (args.length < 2) {
55  			if (Utils.isPlayer(sender, true)) {
56  				target = (Player) sender;
57  				self = true;
58  				if (!args.hasFlag('t'))
59  					try {
60  						amount = args.getFloat(0);
61  					} catch (final NumberFormatException e) {
62  						final HashMap<String, String> replace = new HashMap<String, String>();
63  						replace.put("number", args.getString(0));
64  						Utils.I18n("NaN", replace);
65  						return;
66  					}
67  			} else
68  				return;
69  		} else {
70  			target = Utils.getPlayer(args.getString(0));
71  			if (!args.hasFlag('t'))
72  				try {
73  					amount = args.getFloat(1);
74  				} catch (final NumberFormatException e) {
75  					final HashMap<String, String> replace = new HashMap<String, String>();
76  					replace.put("number", args.getString(0));
77  					Utils.I18n("NaN", replace);
78  					return;
79  				}
80  		}
81  		if (target == null)
82  			return;
83  		final HashMap<String, String> replace = new HashMap<String, String>();
84  		replace.put("amount", String.valueOf(amount));
85  		final Player taskTarget = target;
86  		final float amountXp = amount;
87  		if (args.hasFlag('d')) {
88  			final Location loc = target.getLocation();
89  			loc.setX(loc.getX() + 2);
90  			ACPluginManager.scheduleSyncTask(new Runnable() {
91  				@Override
92  				public void run() {
93  					taskTarget.getLocation().getWorld().spawn(loc, ExperienceOrb.class)
94  							.setExperience((int) amountXp);
95  				}
96  			});
97  
98  			if (self) {
99  				target.sendMessage(Utils.I18n("expDropped", replace));
100 			} else {
101 				replace.put("target", Utils.getPlayerName(target));
102 				target.sendMessage(Utils.I18n("expDropped", replace));
103 				sender.sendMessage(Utils.I18n("expDroppedTarget", replace));
104 			}
105 		} else if (args.hasFlag('a')) {
106 
107 			ACPluginManager.scheduleSyncTask(new Runnable() {
108 				@Override
109 				public void run() {
110 					taskTarget.giveExp((int) amountXp);
111 				}
112 			});
113 			if (self) {
114 				target.sendMessage(Utils.I18n("expAdded", replace));
115 			} else {
116 				replace.put("target", Utils.getPlayerName(target));
117 				sender.sendMessage(Utils.I18n("expAddedTarget", replace));
118 				target.sendMessage(Utils.I18n("expAdded", replace));
119 			}
120 		} else if (args.hasFlag('p')) {
121 			final float exp = (amount > 1 ? 1 : amount);
122 			ACPluginManager.scheduleSyncTask(new Runnable() {
123 				@Override
124 				public void run() {
125 					taskTarget.setExp(exp);
126 				}
127 			});
128 			replace.put("amount", String.valueOf(exp * 100.0F));
129 			if (self) {
130 				target.sendMessage(Utils.I18n("expProgressionSet", replace));
131 			} else {
132 				replace.put("target", Utils.getPlayerName(target));
133 				target.sendMessage(Utils.I18n("expProgressionSet", replace));
134 				sender.sendMessage(Utils.I18n("expProgressionSetTarget", replace));
135 			}
136 		} else if (args.hasFlag('l')) {
137 			ACPluginManager.scheduleSyncTask(new Runnable() {
138 				@Override
139 				public void run() {
140 					taskTarget.setLevel((int) amountXp);
141 				}
142 			});
143 			if (self) {
144 				target.sendMessage(Utils.I18n("expLevelSet", replace));
145 			} else {
146 				replace.put("target", Utils.getPlayerName(target));
147 				target.sendMessage(Utils.I18n("expLevelSet", replace));
148 				sender.sendMessage(Utils.I18n("expLevelSetTarget", replace));
149 			}
150 		} else if (args.hasFlag('t')) {
151 			replace.put("exp", String.valueOf(target.getTotalExperience()));
152 			if (self) {
153 				sender.sendMessage(Utils.I18n("expTotal", replace));
154 			} else {
155 				replace.put("target", Utils.getPlayerName(target));
156 				sender.sendMessage(Utils.I18n("expTotalTarget", replace));
157 			}
158 		}
159 	}
