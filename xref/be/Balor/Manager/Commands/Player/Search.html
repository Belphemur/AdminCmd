42  public class Search extends PlayerCommand {
43  
44  	public Search() {
45  		cmdName = "bal_search";
46  		permNode = "admincmd.player.search";
47  	}
48  
49  	/*
50  	 * (non-Javadoc)
51  	 * 
52  	 * @see be.Balor.Manager.Commands.CoreCommand#execute(org.bukkit.command.
53  	 * CommandSender, be.Balor.Manager.Commands.CommandArgs)
54  	 */
55  	@Override
56  	public void execute(final CommandSender sender, final CommandArgs args)
57  			throws PlayerNotFound, ActionNotPermitedException {
58  		String ip = null;
59  		if (args.hasFlag('i')) {
60  			ip = args.getValueFlag('i');
61  		} else if (args.hasFlag('p')) {
62  			ip = Utils.getACPlayerParam(sender, args, permNode)
63  					.getInformation("last-ip").getString();
64  		}
65  
66  		if (ip == null || ip.equals("") || ip.length() == 0) {
67  			return;
68  		}
69  		final Player[] onPlayers = ACPluginManager.getServer()
70  				.getOnlinePlayers();
71  		final List<ACPlayer> exPlayers = PlayerManager.getInstance()
72  				.getExistingPlayers();
73  		final TreeSet<String> players = new TreeSet<String>();
74  		final TreeSet<String> playersOld = new TreeSet<String>();
75  		final String on = "[ON] ", off = "[OFF] ";
76  		InetAddress ipAdress;
77  		for (final Player p : onPlayers) {
78  			ipAdress = p.getAddress().getAddress();
79  			if (ipAdress != null && ipAdress.toString().substring(1).equals(ip)) {
80  				players.add(on + Utils.getPlayerName(p));
81  				playersOld.add(p.getName());
82  			}
83  		}
84  		String ip2;
85  		for (final ACPlayer p : exPlayers) {
86  			ip2 = p.getInformation("last-ip").getString();
87  			if (ip2 != null && ip2.contains(ip)
88  					&& !playersOld.contains(p.getName())) {
89  				players.add(off + p.getName());
90  			}
91  		}
92  		final String found = Joiner.on(", ").join(players);
93  		sender.sendMessage(found);
94  		return;
95  
96  	}

102 	 */

104 	public boolean argsCheck(final String... args) {
105 		return args != null && args.length >= 1;
106 	}
