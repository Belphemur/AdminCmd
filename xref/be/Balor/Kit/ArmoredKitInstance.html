53  	 * @param name

55  	 * @param items

57  	public ArmoredKitInstance(final String name, final int delay,
58  			final List<MaterialContainer> items,
59  			final Map<Type.ArmorPart, MaterialContainer> armor) {
60  		super(name, delay, items);
61  		for (final Entry<Type.ArmorPart, MaterialContainer> e : armor
62  				.entrySet()) {
63  			if (!e.getKey().isValid(e.getValue().getMaterial().getId())) {
64  				continue;
65  			}
66  			this.armor.put(e.getKey(), e.getValue());
67  		}
68  
69  	}

71  	/***
72  	 * @param name
73  	 * @param delay
74  	 * @param player
75  	 */
76  	public ArmoredKitInstance(final String name, final int delay,
77  			final Player player) {
78  		super(name, delay, player);
79  		final PlayerInventory inventory = player.getInventory();
80  		ItemStack boots = inventory.getBoots();
81  		if (boots != null) {
82  			this.armor.put(ArmorPart.BOOTS,
83  					new MaterialContainer(boots));
84  		}
85  		ItemStack chestplate = inventory.getChestplate();
86  		if (chestplate != null) {
87  			this.armor.put(ArmorPart.CHEST,
88  					new MaterialContainer(chestplate));
89  		}
90  		ItemStack helmet = inventory.getHelmet();
91  		if (helmet != null) {
92  			this.armor.put(ArmorPart.HEAD,
93  					new MaterialContainer(helmet));
94  		}
95  		ItemStack leggings = inventory.getLeggings();
96  		if (leggings != null) {
97  			this.armor.put(ArmorPart.LEGS,
98  					new MaterialContainer(leggings));
99  		}
100 	}

111 	/***
112 	 * Get the wanted part of the armor
113 	 * 
114 	 * @param part
115 	 * @return itemstack containing the part of the armor
116 	 */
117 	public ItemStack getArmorPart(final Type.ArmorPart part) {
118 		final MaterialContainer mat = armor.get(part);
119 		if (mat == null) {
120 			return null;
121 		}
122 		try {
123 			mat.setColor(color);
124 		} catch (final Exception e) {
125 		}
126 		return mat.getItemStack();
127 	}

129 	/***
130 	 * Set the player armor using the armor set in the kit
131 	 * 
132 	 * @param p
133 	 *            player to change the armor's inventory.
134 	 * @return contain the Part of the armor that couldn't be setted.
135 	 */
136 	public ArmorPart[] setPlayerArmorParts(final Player p) {
137 		final List<ArmorPart> armorParts = new ArrayList<Type.ArmorPart>(
138 				ArmorPart.values().length);
139 		final ItemStack[] armors = new ItemStack[ArmorPart.values().length];
140 		final PlayerInventory inventory = p.getInventory();
141 		if (ConfigEnum.ARMOR_KIT_OVERRIDE.getBoolean()) {
142 			for (final ArmorPart part : ArmorPart.values()) {
143 				ItemStack toadd = getArmorPart(part);
144 				if (toadd == null) {
145 					toadd = inventory.getItem(inventory.getSize()
146 							+ part.getPlaceInInventory());
147 					armorParts.add(part);
148 				}
149 				armors[part.getPlaceInInventory()] = toadd;
150 			}
151 		} else {
152 			for (final ArmorPart part : ArmorPart.values()) {
153 				ItemStack toadd = getArmorPart(part);
154 				if (inventory.getItem(inventory.getSize()
155 						+ part.getPlaceInInventory()) != null) {
156 					continue;
157 				}
158 				if (toadd == null) {
159 					toadd = inventory.getItem(inventory.getSize()
160 							+ part.getPlaceInInventory());
161 					armorParts.add(part);
162 				}
163 				armors[part.getPlaceInInventory()] = toadd;
164 			}
165 		}
166 		ACPluginManager.scheduleSyncTask(new Runnable() {
167 
168 			@Override
169 			public void run() {
170 				inventory.setArmorContents(armors);
171 			}
172 		});
173 		return armorParts.toArray(new ArmorPart[0]);
174 	}

211 	@Override
212 	public Map<String, Object> serialize() {
213 		final Map<String, Object> serialized = super.serialize();
214 		serialized.put("color", color);
215 		final Map<String, MaterialContainer> armor = new LinkedHashMap<String, MaterialContainer>();
216 		for (final Entry<ArmorPart, MaterialContainer> armPart : this.armor
217 				.entrySet()) {
218 			armor.put(armPart.getKey().name().toLowerCase(), armPart.getValue());
219 		}
220 		serialized.put("armor", armor);
221 		return serialized;
222 	}

224 	/***
225 	 * Required method for configuration serialization
226 	 * 
227 	 * @param args
228 	 *            map to deserialize
229 	 * @return deserialized item stack
230 	 * @see ConfigurationSerializable
231 	 */
232 	public static ArmoredKitInstance deserialize(final Map<String, Object> args) {
233 		final KitInstance kit = KitInstance.deserialize(args);
234 		final Map<Type.ArmorPart, MaterialContainer> armor = new LinkedHashMap<Type.ArmorPart, MaterialContainer>();
235 		final Object armorObj = args.get("armor");
236 		if (armorObj instanceof Map<?, ?>) {
237 			for (final Entry<?, ?> entry : ((Map<?, ?>) armorObj).entrySet()) {
238 				armor.put(ArmorPart.getByName(entry.getKey().toString()),
239 						(MaterialContainer) entry.getValue());
240 			}
241 		}
242 		final ArmoredKitInstance armKit = new ArmoredKitInstance(kit, armor);
243 		final Object color = args.get("color");
244 		if (color != null) {
245 			armKit.setColor(color.toString());
246 		}
247 		return armKit;
248 	}
