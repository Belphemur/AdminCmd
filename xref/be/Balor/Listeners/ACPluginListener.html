49  public class ACPluginListener implements Listener {
50  
51  	@EventHandler(priority = EventPriority.MONITOR)
52  	public void onPluginDisable(final PluginDisableEvent event) {
53  		ACPluginManager.unRegisterACPlugin(event.getPlugin());
54  	}
55  
56  	@EventHandler(priority = EventPriority.MONITOR)
57  	public void onPluginEnable(final PluginEnableEvent event) {
58  		if (!PermissionManager.isPermissionsExSet()) {
59  			final Plugin Permissions = ACPluginManager.getServer()
60  					.getPluginManager().getPlugin("PermissionsEx");
61  			if (Permissions != null) {
62  				if (Permissions.isEnabled()) {
63  					PermissionManager.setPEX(PermissionsEx
64  							.getPermissionManager());
65  				}
66  			}
67  		}
68  		if (!PermissionManager.isGroupManagerSet()) {
69  			final Plugin GMplugin = ACPluginManager.getServer()
70  					.getPluginManager().getPlugin("GroupManager");
71  			if (GMplugin != null) {
72  				if (GMplugin.isEnabled()) {
73  					PermissionManager.setGroupManager(GMplugin);
74  				}
75  			}
76  		}
77  		if (!PermissionManager.isYetiPermissionsSet()) {
78  			final Plugin Permissions = ACPluginManager.getServer()
79  					.getPluginManager().getPlugin("Permissions");
80  			if (Permissions != null) {
81  				if (Permissions.isEnabled()) {
82  					PermissionManager
83  							.setYetiPermissions(((Permissions) Permissions)
84  									.getHandler());
85  				}
86  			}
87  		}
88  		if (!PermissionManager.isbPermissionsSet()) {
89  			final Plugin plugin = ACPluginManager.getServer()
90  					.getPluginManager().getPlugin("bPermissions");
91  			if (plugin != null) {
92  				String version = plugin.getDescription().getVersion();
93  				version = version.replace(".", "");
94  				final int ver = Integer.parseInt(version);
95  				if (ver < 285) {
96  					ACLogger.info("You are using bPermissions v"
97  							+ plugin.getDescription().getVersion()
98  							+ ". This is an outdated version, permission support for bPermission will be disabled.");
99  					return;
100 				}
101 				PermissionManager.setbPermissions();
102 			}
103 		}
104 		if (!PermissionManager.isPermissionsBukkitSet()) {
105 			final Plugin plugin = ACPluginManager.getServer()
106 					.getPluginManager().getPlugin("PermissionsBukkit");
107 			if (plugin != null) {
108 				PermissionManager
109 						.setPermissionsBukkit((PermissionsPlugin) plugin);
110 			}
111 		}
112 		if (Utils.oddItem == null) {
113 			final Plugin items = ACPluginManager.getServer().getPluginManager()
114 					.getPlugin("OddItem");
115 			if (items != null && items.isEnabled()) {
116 				Utils.oddItem = (OddItemBase) items;
117 				ACLogger.info("Successfully linked with OddItem");
118 			}
119 		}
120 		if (!SuperPermissions.isApiSet()) {
121 			final Plugin mChatPlugin = ACPluginManager.getServer()
122 					.getPluginManager().getPlugin("mChatSuite");
123 			if (mChatPlugin != null && mChatPlugin.isEnabled()) {
124 				SuperPermissions.setmChatapi((mChatSuite) mChatPlugin);
125 				Utils.mChatPresent = true;
126 				ACLogger.info("Successfully linked with mChatSuite");
127 			}
128 		}
129 		if (Utils.logBlock == null) {
130 			final Plugin plugin = ACPluginManager.getServer()
131 					.getPluginManager().getPlugin("LogBlock");
132 			if (plugin != null && plugin.isEnabled()) {
133 				Utils.setLogBlock(((LogBlock) plugin).getConsumer());
134 				ACLogger.info("Successfully linked with LogBlock");
135 			}
136 		}
137 		if (InvisibleWorker.dynmapAPI == null) {
138 			final Plugin plugin = ACPluginManager.getServer()
139 					.getPluginManager().getPlugin("dynmap");
140 			if (plugin != null && plugin.isEnabled()) {
141 				InvisibleWorker.dynmapAPI = (DynmapAPI) plugin;
142 				ACLogger.info("Successfully linked with Dynmap");
143 			}
144 		}
145 		if (ConfigEnum.H_ALLPLUGIN.getBoolean()) {
146 			for (final Plugin plugin : event.getPlugin().getServer()
147 					.getPluginManager().getPlugins()) {
148 				HelpLister.getInstance().addPlugin(plugin);
149 			}
150 		}
151 		if (!Utils.signExtention) {
152 			final Plugin plugin = ACPluginManager.getServer()
153 					.getPluginManager().getPlugin("SignExtensions");
154 			if (plugin != null) {
155 				Utils.signExtention = true;
156 			}
157 		}
158 	}
159 }
