62  	@EventHandler(priority = EventPriority.LOWEST)
63  	public void onPlayerCommandPreprocess(
64  			final PlayerCommandPreprocessEvent event) {
65  		final Player p = event.getPlayer();
66  		final ACPlayer player = ACPlayer.getPlayer(p);
67  		final String message = event.getMessage();
68  		if (player.hasPower(Type.MUTED_COMMAND)) {
69  			final String[] split = message.split("//s+");
70  			if (split.length != 0) {
71  				if (split[0].contains("/")) {
72  					event.setCancelled(true);
73  					Utils.sI18n(p, "commandMuteEnabled");
74  				}
75  			}
76  		}
77  		if (CommandManager.getInstance().processCommandString(
78  				event.getPlayer(), message)) {
79  			event.setCancelled(true);
80  			event.setMessage("/AdminCmd : " + message);
81  		}
82  	}

95  	@EventHandler(priority = EventPriority.NORMAL)
96  	public void onPlayerJoin(final PlayerJoinEvent event) {
97  		final Player p = event.getPlayer();
98  		final ACPlayer player = PlayerManager.getInstance().setOnline(p);
99  		InvisibleWorker.getInstance().makeInvisibleToPlayer(p);
100 		final InetAddress address = p.getAddress().getAddress();
101 
102 		final HashMap<String, String> replace = new HashMap<String, String>();
103 		if (ConfigEnum.JQMSG.getBoolean() && !SuperPermissions.isApiSet()) {
104 			replace.put("name", Utils.getPlayerName(p, null, true));
105 			event.setJoinMessage(Utils.I18n("joinMessage", replace));
106 		}
107 		if (player.hasPower(Type.INVISIBLE)) {
108 			event.setJoinMessage(null);
109 			Utils.sI18n(p, "stillInv");
110 			InvisibleWorker.getInstance().vanish(p, true);
111 
112 		}
113 		ACPluginManager.getScheduler().runTaskAsynchronously(
114 				ACPluginManager.getCorePlugin(), new Runnable() {
115 					@Override
116 					public void run() {
117 						DebugLog.INSTANCE
118 								.info("ASync Task for optimization for "
119 										+ p.getName());
120 						player.setInformation("last-ip", address.toString()
121 								.substring(1));
122 						DebugLog.INSTANCE.info("AFK start");
123 						if (ConfigEnum.AUTO_AFK.getBoolean()) {
124 							AFKWorker.getInstance().updateTimeStamp(p);
125 						}
126 						DebugLog.INSTANCE.info("AFK stop");
127 						DebugLog.INSTANCE.info("ImmunityLvl start");
128 						final int imLvl = ACHelper.getInstance().getLimit(p,
129 								Type.Limit.IMMUNITY, "defaultImmunityLvl");
130 						player.setInformation(
131 								"immunityLvl",
132 								imLvl == Integer.MAX_VALUE ? ConfigEnum.DIMMUNITY
133 										.getInt() : imLvl);
134 						DebugLog.INSTANCE.info("ImmunityLvl stop");
135 						DebugLog.INSTANCE.info("SPY start");
136 						if (player.hasPower(Type.SPYMSG)) {
137 							ACHelper.getInstance().addSpy(p);
138 						}
139 						DebugLog.INSTANCE.info("SPY stop");
140 						DebugLog.INSTANCE.info("LastConn start");
141 						final long lastConn = player.getInformation(
142 								"lastConnection").getLong(0);
143 						DebugLog.INSTANCE.info("LastConn stop");
144 						DebugLog.INSTANCE.info("TextLocale start");
145 						final long modifTime = TextLocale.NEWS.getModifTime();
146 						if (ConfigEnum.NEWS.getBoolean()
147 								&& (modifTime == 0 || lastConn <= modifTime)) {
148 							TextLocale.NEWS.sendText(p);
149 						}
150 						DebugLog.INSTANCE.info("TextLocale stop");
151 						DebugLog.INSTANCE.info("MOTD start");
152 						if (!player.getInformation("firstTime")
153 								.getBoolean(true)
154 								&& ConfigEnum.MOTD.getBoolean()) {
155 							TextLocale.MOTD.sendText(p);
156 						}
157 						DebugLog.INSTANCE.info("MOTD stop");
158 						DebugLog.INSTANCE.info("Rules start");
159 						if (ConfigEnum.RULES.getBoolean()
160 								&& !ConfigEnum.FJ_RULES.getBoolean()) {
161 							TextLocale.RULES.sendText(p);
162 						}
163 						DebugLog.INSTANCE.info("Rules stop");
164 						DebugLog.INSTANCE.info("TPREQUEST start");
165 						if (ConfigEnum.TPREQUEST.getBoolean()
166 								&& !player.hasPower(Type.TP_REQUEST)
167 								&& PermissionManager.hasPerm(p,
168 										"admincmd.tp.toggle.allow", false)) {
169 							player.setPower(Type.TP_REQUEST);
170 						}
171 						DebugLog.INSTANCE.info("TPREQUEST stop");
172 						player.setInformation("lastConnection",
173 								System.currentTimeMillis());
174 					}
175 				});
176 
177 		if (player.hasPower(Type.FAKEQUIT)) {
178 			event.setJoinMessage(null);
179 			ACHelper.getInstance().addFakeQuit(p);
180 		}
181 
182 		if (player.getInformation("firstTime").getBoolean(true)) {
183 			if (ConfigEnum.COMMANDS_ONJOIN.getBoolean()) {
184 				CommandManager.getInstance().executeFirstJoinCommands(p);
185 			}
186 			player.setInformation("firstTime", false);
187 			if (ConfigEnum.JQMSG.getBoolean() && !SuperPermissions.isApiSet()) {
188 				replace.clear();
189 				replace.put("name", Utils.getPlayerName(p, null, true));
190 				event.setJoinMessage(Utils
191 						.I18n("joinMessageFirstTime", replace));
192 			}
193 			if (ConfigEnum.FCSPAWN.getBoolean()) {
194 				ACHelper.getInstance().spawn(p);
195 			}
196 			if (!ConfigEnum.FCSPAWN.getBoolean()
197 					&& ConfigEnum.GSPAWN.getString().equalsIgnoreCase("group")) {
198 				ACHelper.getInstance().groupSpawn(p);
199 			}
200 			if (ConfigEnum.FJ_RULES.getBoolean()) {
201 				TextLocale.RULES.sendText(p);
202 			}
203 			if (ConfigEnum.MOTD.getBoolean()) {
204 				TextLocale.MOTD_NEW.sendText(p);
205 			}
206 		}
207 		DebugLog.INSTANCE.info("Player " + p.getName() + " joined sucessfully");
208 	}

252 	@EventHandler(priority = EventPriority.HIGH)
253 	public void onPlayerQuit(final PlayerQuitEvent event) {
254 		final Player p = event.getPlayer();
255 		final ACPlayer player = ACPlayer.getPlayer(p);
256 		ACPluginManager.runTaskLaterAsynchronously(new Runnable() {
257 
258 			@Override
259 			public void run() {
260 				player.setInformation("lastDisconnect",
261 						System.currentTimeMillis());
262 				player.setInformation("gameMode", p.getGameMode());
263 
264 			}
265 		});
266 
267 		if (player.hasPower(Type.FAKEQUIT)) {
268 			event.setQuitMessage(null);
269 		} else if (InvisibleWorker.getInstance().hasInvisiblePowers(p)) {
270 			event.setQuitMessage(null);
271 		}
272 		if (event.getQuitMessage() != null && ConfigEnum.JQMSG.getBoolean()
273 				&& !SuperPermissions.isApiSet()) {
274 			final HashMap<String, String> replace = new HashMap<String, String>();
275 			replace.put("name", Utils.getPlayerName(p, null, true));
276 			event.setQuitMessage(Utils.I18n("quitMessage", replace));
277 		}
278 
279 		PlayerManager.getInstance().setOffline(player);
280 		ACHelper.getInstance().removeDisconnectedPlayer(p);
281 	}

283 	@EventHandler(ignoreCancelled = true)
284 	public void onPlayerRespawn(final PlayerRespawnEvent event) {
285 		final Player player = event.getPlayer();
286 		final String spawn = ConfigEnum.GSPAWN.getString();
287 		Location loc = null;
288 		final World world = player.getWorld();
289 
290 		try {
291 			if (spawn.isEmpty() || spawn.equalsIgnoreCase("globalspawn")) {
292 
293 				loc = ACWorld.getWorld(world).getSpawn();
294 
295 				event.setRespawnLocation(loc);
296 			} else if (spawn.equalsIgnoreCase("home")) {
297 				loc = ACPlayer.getPlayer(player).getHome(world.getName());
298 				if (loc == null) {
299 					loc = ACWorld.getWorld(world).getSpawn();
300 				}
301 				event.setRespawnLocation(loc);
302 			} else if (spawn.equalsIgnoreCase("bed")) {
303 				try {
304 					loc = player.getBedSpawnLocation();
305 					if (loc == null) {
306 						loc = ACWorld.getWorld(world).getSpawn();
307 					}
308 				} catch (final NullPointerException e) {
309 					loc = ACWorld.getWorld(world).getSpawn();
310 				}
311 				event.setRespawnLocation(loc);
312 			} else if (spawn.equalsIgnoreCase("group")) {
313 				loc = ACHelper.getInstance().getGroupSpawnLocation(player);
314 				event.setRespawnLocation(loc);
315 			}
316 		} catch (final WorldNotLoaded e) {
317 
318 		}
319 	}
