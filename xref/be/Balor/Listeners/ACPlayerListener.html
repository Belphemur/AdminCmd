139 	@EventHandler(priority = EventPriority.NORMAL)
140 	public void onPlayerJoin(final PlayerJoinEvent event) {
141 		final Player p = event.getPlayer();
142 		final ACPlayer player = PlayerManager.getInstance().setOnline(p);
143 		player.setInformation("last-ip", p.getAddress().getAddress().toString());
144 		if (ConfigEnum.JQMSG.getBoolean() && !SuperPermissions.isApiSet()) {
145 			final HashMap<String, String> replace = new HashMap<String, String>();
146 			replace.put("name", Utils.getPlayerName(p, null, true));
147 			event.setJoinMessage(Utils.I18n("joinMessage", replace));
148 		}
149 		if (player.hasPower(Type.INVISIBLE)) {
150 			event.setJoinMessage(null);
151 			Utils.sI18n(event.getPlayer(), "stillInv");
152 			InvisibleWorker.getInstance().onJoinEvent(p);
153 		}
154 		ACPluginManager.getScheduler().scheduleAsyncDelayedTask(ACPluginManager.getCorePlugin(),
155 				new Runnable() {
156 
157 					@Override
158 					public void run() {
159 						if (ConfigEnum.AUTO_AFK.getBoolean())
160 							AFKWorker.getInstance().updateTimeStamp(p);
161 						final int imLvl = ACHelper.getInstance().getLimit(p, "immunityLvl",
162 								"defaultImmunityLvl");
163 						player.setInformation("immunityLvl",
164 								imLvl == Integer.MAX_VALUE ? ConfigEnum.DIMMUNITY.getInt() : imLvl);
165 						if (player.hasPower(Type.SPYMSG))
166 							ACHelper.getInstance().addSpy(p);
167 						player.setInformation("lastConnection", System.currentTimeMillis());
168 						if (ConfigEnum.NEWS.getBoolean())
169 							Utils.sParsedLocale(p, "NEWS");
170 						if (ConfigEnum.RULES.getBoolean() && !ConfigEnum.FJ_RULES.getBoolean())
171 							Utils.sParsedLocale(p, "Rules");
172 						if (ConfigEnum.TPREQUEST.getBoolean() && !player.hasPower(Type.TP_REQUEST)
173 								&& PermissionManager.hasPerm(p, "admincmd.tp.toggle.allow", false))
174 							player.setPower(Type.TP_REQUEST);
175 
176 					}
177 				});
178 
179 		if (player.hasPower(Type.FAKEQUIT)) {
180 			event.setJoinMessage(null);
181 			ACHelper.getInstance().addFakeQuit(p);
182 		}
183 
184 		if (player.getInformation("firstTime").getBoolean(true)) {
185 			player.setInformation("firstTime", false);
186 			if (ConfigEnum.JQMSG.getBoolean() && !SuperPermissions.isApiSet()) {
187 				final HashMap<String, String> replace = new HashMap<String, String>();
188 				replace.put("name", Utils.getPlayerName(p, null, true));
189 				event.setJoinMessage(Utils.I18n("joinMessageFirstTime", replace));
190 			}
191 			if (ConfigEnum.FCSPAWN.getBoolean())
192 				ACHelper.getInstance().spawn(p);
193 			if (!ConfigEnum.FCSPAWN.getBoolean()
194 					&& ConfigEnum.GSPAWN.getString().equalsIgnoreCase("group"))
195 				ACHelper.getInstance().groupSpawn(p);
196 			if (ConfigEnum.FJ_RULES.getBoolean())
197 				Utils.sParsedLocale(p, "Rules");
198 			if (ConfigEnum.MOTD.getBoolean())
199 				Utils.sParsedLocale(p, "MOTDNewUser");
200 		} else if (ConfigEnum.MOTD.getBoolean())
201 			Utils.sParsedLocale(p, "MOTD");
202 	}

281 	@EventHandler
282 	public void onPlayerRespawn(final PlayerRespawnEvent event) {
283 		final Player player = event.getPlayer();
284 		playerRespawnOrJoin(player);
285 		final String spawn = ConfigEnum.GSPAWN.getString();
286 		Location loc = null;
287 		final String worldName = player.getWorld().getName();
288 		if (spawn.isEmpty() || spawn.equalsIgnoreCase("globalspawn")) {
289 			loc = ACWorld.getWorld(worldName).getSpawn();
290 			event.setRespawnLocation(loc);
291 		} else if (spawn.equalsIgnoreCase("home")) {
292 			loc = ACPlayer.getPlayer(player).getHome(worldName);
293 			if (loc == null)
294 				loc = ACWorld.getWorld(worldName).getSpawn();
295 			event.setRespawnLocation(loc);
296 		} else if (spawn.equalsIgnoreCase("bed")) {
297 			try {
298 				loc = player.getBedSpawnLocation();
299 				if (loc == null)
300 					loc = ACWorld.getWorld(worldName).getSpawn();
301 			} catch (final NullPointerException e) {
302 				loc = ACWorld.getWorld(worldName).getSpawn();
303 			}
304 
305 			event.setRespawnLocation(loc);
306 		} else if (spawn.equalsIgnoreCase("group")) {
307 			final List<String> groups = ACHelper.getInstance().getGroupList();
308 			if (!groups.isEmpty()) {
309 				for (final String groupName : groups) {
310 					try {
311 						if (PermissionManager.isInGroup(groupName, player))
312 							loc = ACWorld.getWorld(worldName).getWarp(
313 									"spawn" + groupName.toLowerCase()).loc;
314 						break;
315 					} catch (final NoPermissionsPlugin e) {
316 						loc = ACWorld.getWorld(worldName).getSpawn();
317 						break;
318 					}
319 				}
320 			}
321 			if (loc == null)
322 				loc = ACWorld.getWorld(worldName).getSpawn();
323 			event.setRespawnLocation(loc);
324 		} /*
325 		 * else { loc = ACWorld.getWorld(worldName).getSpawn();
326 		 * event.setRespawnLocation(loc); }
327 		 */
328 
329 	}

341 	private boolean playerRespawnOrJoin(final Player newPlayer) {
342 		ACPluginManager
343 				.getServer()
344 				.getScheduler()
345 				.scheduleSyncDelayedTask(ACHelper.getInstance().getCoreInstance(),
346 						new UpdateInvisibleOnJoin(newPlayer), 15);
347 		if (InvisibleWorker.getInstance().hasInvisiblePowers(newPlayer.getName())) {
348 			ACPluginManager
349 					.getServer()
350 					.getScheduler()
351 					.scheduleSyncDelayedTask(ACHelper.getInstance().getCoreInstance(),
352 							new UpdateInvisible(newPlayer), 15);
353 			Utils.removePlayerFromOnlineList(newPlayer);
354 			return true;
355 		}
356 		return false;
357 	}
