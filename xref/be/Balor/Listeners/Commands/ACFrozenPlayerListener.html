36  public class ACFrozenPlayerListener implements Listener {
37  	@EventHandler(ignoreCancelled = true, priority = EventPriority.LOW)
38  	public void onMove(final PlayerMoveEvent event) {
39  		if (!ACPlayer.getPlayer(event.getPlayer()).hasPower(Type.FROZEN)) {
40  			return;
41  		}
42  		MinecraftReflection.teleportPlayer(event.getPlayer(), event.getFrom());
43  
44  	}
45  
46  	@EventHandler(ignoreCancelled = true, priority = EventPriority.LOW)
47  	public void onDrop(final PlayerDropItemEvent event) {
48  		if (ACPlayer.getPlayer(event.getPlayer()).hasPower(Type.FROZEN)) {
49  			event.setCancelled(true);
50  		}
51  	}
52  
53  	@EventHandler(ignoreCancelled = true, priority = EventPriority.LOW)
54  	public void onPickup(final PlayerPickupItemEvent event) {
55  		if (ACPlayer.getPlayer(event.getPlayer()).hasPower(Type.FROZEN)) {
56  			event.setCancelled(true);
57  		}
58  	}
59  
60  	@EventHandler(ignoreCancelled = true, priority = EventPriority.LOW)
61  	public void onInteract(final PlayerInteractEvent event) {
62  		if (ACPlayer.getPlayer(event.getPlayer()).hasPower(Type.FROZEN)) {
63  			event.setCancelled(true);
64  		}
65  	}
66  
67  	@EventHandler(ignoreCancelled = true)
68  	public void onTeleport(final PlayerTeleportEvent event) {
69  		if (ACPlayer.getPlayer(event.getPlayer()).hasPower(Type.FROZEN)) {
70  			event.setCancelled(true);
71  		}
72  	}
73  }
