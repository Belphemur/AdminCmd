44  public class ACNoDropListener implements Listener {
45  	private final Map<Player, PlayerInformation> itemsDrops = new HashMap<Player, PlayerInformation>();
46  	private final Map<String, PlayerInformation> itemsOfDeadDisconnected = new HashMap<String, PlayerInformation>();
47  
48  	@EventHandler(ignoreCancelled = true)
49  	public void onDrop(final PlayerDropItemEvent event) {
50  		if (ACPlayer.getPlayer(event.getPlayer()).hasPower(Type.NO_DROP)) {
51  			event.setCancelled(true);
52  		}
53  	}
54  
55  	@EventHandler
56  	public void onDeath(final EntityDeathEvent event) {
57  		if (!(event.getEntity() instanceof Player)) {
58  			return;
59  		}
60  		final Player p = (Player) event.getEntity();
61  		final ACPlayer player = ACPlayer.getPlayer(p);
62  		if (checkNoLoss(p, player)) {
63  			return;
64  		}
65  		for (final ItemStack item : event.getDrops()) {
66  			item.setAmount(0);
67  		}
68  		event.setDroppedExp(0);
69  		itemsDrops.put(p, new PlayerInformation(p));
70  	}
71  
72  	@EventHandler
73  	public void onRespawn(final PlayerRespawnEvent event) {
74  		final Player p = event.getPlayer();
75  		final ACPlayer player = ACPlayer.getPlayer(p);
76  		if (checkNoLoss(p, player)) {
77  			itemsDrops.remove(p);
78  			return;
79  		}
80  		final PlayerInformation inv = itemsDrops.get(p);
81  		if (inv == null) {
82  			return;
83  		}
84  		inv.setPlayerInfo(p);
85  		itemsDrops.remove(p);
86  	}
87  
88  	@EventHandler(priority = EventPriority.LOW)
89  	public void onQuit(final PlayerQuitEvent event) {
90  		final Player player = event.getPlayer();
91  		final PlayerInformation inv = itemsDrops.remove(player);
92  		if (inv != null && player.isDead()) {
93  			itemsOfDeadDisconnected.put(player.getName(), inv);
94  		}
95  	}
96  
97  	@EventHandler(priority = EventPriority.HIGH)
98  	public void onJoin(final PlayerJoinEvent event) {
99  		final Player p = event.getPlayer();
100 		final ACPlayer player = ACPlayer.getPlayer(p);
101 		final String name = p.getName();
102 		if (checkNoLoss(p, player)) {
103 			itemsOfDeadDisconnected.remove(name);
104 			return;
105 		}
106 		final PlayerInformation inv = itemsOfDeadDisconnected.remove(name);
107 		if (inv == null) {
108 			return;
109 		}
110 		itemsDrops.put(p, inv);
111 	}
112 
113 	/***
114 	 * Return if the player don't have the power to get back his informations
115 	 * (inventory, xp)
116 	 * 
117 	 * @param p
118 	 * @param player
119 	 * @return
120 	 */
121 	private boolean checkNoLoss(final Player p, final ACPlayer player) {
122 		return !player.hasPower(Type.NO_DROP)
123 				&& !PermissionManager.hasPerm(p, "admincmd.spec.noloss", false);
124 	}
125 
126 	private class PlayerInformation {
127 		final ItemStack items[];
128 		final ItemStack armor[];
129 		final float xp;
130 		final int level;
131 
132 		/***
133 		 * 
134 		 */
135 		public PlayerInformation(final Player p) {
136 			final PlayerInventory inventory = p.getInventory();
137 			items = Arrays.copyOf(inventory.getContents(),
138 					inventory.getContents().length);
139 			armor = Arrays.copyOf(inventory.getArmorContents(),
140 					inventory.getArmorContents().length);
141 			xp = p.getExp();
142 			level = p.getLevel();
143 		}
144 
145 		public void setPlayerInfo(final Player p) {
146 			final PlayerInventory inventory = p.getInventory();
147 			inventory.setArmorContents(armor);
148 			inventory.setContents(items);
149 			ACPluginManager.scheduleSyncTask(new Runnable() {
150 				@Override
151 				public void run() {
152 					p.setExp(xp);
153 					p.setLevel(level);
154 
155 				}
156 			});
157 
158 		}
159 
160 	}
