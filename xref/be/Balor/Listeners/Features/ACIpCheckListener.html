43  public class ACIpCheckListener implements Listener {
44  	private final ConcurrentMap<String, Player> ips = new MapMaker().makeMap();
45  
46  	@EventHandler
47  	public void onJoin(final PlayerJoinEvent event) {
48  		ACPluginManager.runTaskLaterAsynchronously(new Runnable() {
49  
50  			@Override
51  			public void run() {
52  				final Player p = event.getPlayer();
53  				final InetAddress address = p.getAddress().getAddress();
54  				final HashMap<String, String> replace = new HashMap<String, String>();
55  				final Player sameIP = addIP(p, address);
56  				if (sameIP != null) {
57  					replace.put("player", Utils.getPlayerName(p));
58  					replace.put("player2", Utils.getPlayerName(sameIP));
59  					replace.put("ip", address.toString().substring(1));
60  					broadcastIP(replace);
61  				}
62  
63  			}
64  		});
65  
66  	}
67  
68  	@EventHandler
69  	public void onQuit(final PlayerQuitEvent event) {
70  		removePlayer(event);
71  	}
72  
73  	@EventHandler
74  	public void onKick(final PlayerKickEvent event) {
75  		removePlayer(event);
76  	}

78  	/***
79  	 * Broadcasts a message if someone joined from the same IP to all players
80  	 * with a permission and to the console.
81  	 * 
82  	 * @param message
83  	 *            - The message
84  	 * @param players
85  	 *            - List of all players with the permission
86  	 * @author Lathanael (aka Philippe Leipold)
87  	 */
88  	private void broadcastIP(final HashMap<String, String> replace) {
89  		final String message = LocaleHelper.IP_BROADCAST.getLocale(replace);
90  		if (message == null) {
91  			return;
92  		}
93  		ACPluginManager.scheduleSyncTask(new Runnable() {
94  
95  			@Override
96  			public void run() {
97  				Bukkit.getServer().broadcast(message,
98  						"admincmd.spec.ipbroadcast");
99  			}
100 		});
101 
102 		ACLogger.info(message);
103 	}
104 
105 	private void removePlayer(final PlayerEvent event) {
106 		final Player player = event.getPlayer();
107 		updateIP(player, player.getAddress().getAddress());
108 	}
109 
110 	/***
111 	 * If a player quits loop through the online players and see if a second
112 	 * player uses the same IP and replace the quitting player with the online
113 	 * player. Otherwise remove the player from the list!
114 	 * 
115 	 * @param quits
116 	 * @param address
117 	 */
118 	private void updateIP(final Player quits, final InetAddress address) {
119 		ACPluginManager.runTaskLaterAsynchronously(new Runnable() {
120 			@Override
121 			public void run() {
122 				for (final Player p : Utils.getOnlinePlayers()) {
123 					if (p.equals(quits)) {
124 						continue;
125 					}
126 					if (p.getAddress().getAddress().equals(address)) {
127 						ips.replace(address.toString().substring(1), p);
128 						return;
129 					}
130 				}
131 				ips.remove(address.toString().substring(1));
132 
133 			}
134 		});
135 
136 	}
137 
138 	/***
139 	 * Adds a player to the ip list if his ip is not already in.
140 	 * 
141 	 * @param player
142 	 * @param address
143 	 * @return The player who also uses this IP or null if there was none
144 	 *         previously
145 	 */
146 	private Player addIP(final Player player, final InetAddress address) {
147 		final Player p = ips.putIfAbsent(address.toString().substring(1),
148 				player);
149 		return p;
150 	}
151 
152 	/***
153 	 * Checks if an IP is already in use
154 	 * 
155 	 * @param address
156 	 * @return {@code true} if an IP is found, {@code false} otherwise.
157 	 */
158 	public boolean ipInUse(final InetAddress address) {
159 		if (ips.containsKey(address.toString().substring(1))) {
160 			return true;
161 		}
162 		return false;
163 	}
