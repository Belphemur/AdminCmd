44  	protected PlayerInventoryProxy(final Player prop, final Object obj) {
45  		this.proprietary = prop;
46  		this.obj = obj;
47  		final Object playerHandle = MinecraftReflection.getHandle(proprietary);
48  		final Object inventory = FieldUtils.getField(playerHandle, "inventory");
49  		final Object[] armor = FieldUtils.getField(inventory, "armor");
50  		final Object[] items = FieldUtils.getField(inventory, "items");
51  		size = armor.length + items.length + extra.length;
52  		FieldUtils.setField(this.obj, "armor", armor);
53  		FieldUtils.setField(this.obj, "items", items);
54  	}

56  	public static Object newInstance(final Player prop, final Object obj) {
57  		if (!MinecraftReflection.getPlayerInventoryClass().isAssignableFrom(
58  				obj.getClass())) {
59  			throw new RuntimeException(
60  					"The object must be of the type of PlayerInventory");
61  		}
62  		return Proxy.newProxyInstance(obj.getClass().getClassLoader(), obj
63  				.getClass().getInterfaces(),
64  				new PlayerInventoryProxy(prop, obj));
65  	}

73  	@Override
74  	public Object invoke(final Object proxy, final Method method,
75  			final Object[] args) throws Throwable {
76  		final String methodName = method.getName();
77  		if (methodName.equals("onClose")) {
78  			this.onClose(args[0]);
79  			return null;
80  		} else if (methodName.equals("getContents")) {
81  			return getContents();
82  		} else if (methodName.equals("getSize")) {
83  			return getSize();
84  		} else if (methodName.equals("a_")) {
85  			return a_();
86  		} else if (methodName.equals("getName")) {
87  			return getName();
88  		} else if (methodName.equals("getItem")) {
89  			return getItem((Integer) args[0]);
90  		} else if (methodName.equals("splitStack")) {
91  			return splitStack((Integer) args[0], (Integer) args[1]);
92  		} else if (methodName.equals("splitWithoutUpdate")) {
93  			return splitWithoutUpdate((Integer) args[0]);
94  		} else if (methodName.equals("setItem")) {
95  			setItem((Integer) args[0], args[1]);
96  			return null;
97  		} else {
98  			return method.invoke(obj, args);
99  		}
100 	}

119 		final MethodHandler superOnClose = new MethodHandler(obj.getClass(),

125 	protected void checkCloseEvent() {
126 		final Object transactions = FieldUtils.getField(obj, "transaction");
127 		final MethodHandler isEmpty = new MethodHandler(
128 				transactions.getClass(), "isEmpty");
129 		final boolean empty = isEmpty.invoke(transactions);
130 		if (empty && !proprietary.isOnline()) {
131 			InventoryManager.INSTANCE.closeOfflineInv(proprietary);
132 		}
133 	}

172 	private int getReversedArmorSlotNum(final int i) {
173 		if (i == 0) {
174 			return 3;
175 		}
176 		if (i == 1) {

178 		}
179 		if (i == 2) {

190 		Object[] is = getItems();
191 
192 		if (i >= is.length) {
193 			i -= is.length;
194 			is = getArmor();
195 		} else {
196 			i = getReversedItemSlotNum(i);
197 		}
198 
199 		if (i >= is.length) {
200 			i -= is.length;
201 			is = this.extra;
202 		} else if (is == getArmor()) {
203 			i = getReversedArmorSlotNum(i);
204 		}
205 
206 		return is[i];

213 	protected Object splitStack(int i, final int j) {
214 		Object[] is = getItems();
215 
216 		if (i >= is.length) {
217 			i -= is.length;
218 			is = getArmor();
219 		} else {
220 			i = getReversedItemSlotNum(i);
221 		}
222 
223 		if (i >= is.length) {
224 			i -= is.length;
225 			is = this.extra;
226 		} else if (is == getArmor()) {
227 			i = getReversedArmorSlotNum(i);
228 		}
229 
230 		if (is[i] != null) {
231 			Object itemstack;
232 
233 			if (getCount(is[i]) <= j) {
234 				itemstack = is[i];
235 				is[i] = null;
236 				return itemstack;
237 			} else {
238 				final MethodHandler a = new MethodHandler(is[i].getClass(),
239 						"a", int.class);
240 				itemstack = a.invoke(is[i], j);
241 				if (getCount(is[i]) == 0) {
242 					is[i] = null;
243 				}
244 
245 				return itemstack;
246 			}
247 		} else {
248 			return null;
249 		}
250 	}
