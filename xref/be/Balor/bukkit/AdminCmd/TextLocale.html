40  public enum TextLocale {
41  	NEWS("NEWS"), RULES("Rules"), MOTD("MOTD"), MOTD_NEW(
42  			"MOTDNewUser",
43  			"motdNewUser");
44  	private final String locale;
45  	private final String file;
46  	private static ExtendedConfiguration version;
47  
48  	/***
49   * 
50   */
51  	private TextLocale(final String locale) {
52  		this.locale = locale;
53  		this.file = this.locale.toLowerCase();
54  	}
55  
56  	private TextLocale(final String locale, final String file) {
57  		this.locale = locale;
58  		this.file = file;
59  	}
60  
61  	/***
62  	 * Save the new content of the locale
63  	 * 
64  	 * @param content
65  	 *            new content
66  	 */
67  	public void saveContent(final String content) {
68  		final String parsedContent = parseContent(content);
69  		LocaleManager.getInstance().addLocale(locale, parsedContent, true);
70  		FileManager.getInstance().setTxtFile(file, parsedContent);
71  		version.set(file, System.currentTimeMillis());
72  		try {
73  			version.save();
74  		} catch (final IOException e) {
75  		}
76  	}
77  
78  	/***
79  	 * @param content
80  	 */
81  	private String parseContent(final String content) {
82  		return Utils.colorParser(content.replaceAll("//n", "\n").replaceAll(
83  				"//n", "\n"));
84  	}
85  
86  	/***
87  	 * Reload the text file
88  	 */
89  	public void reloadContent() {
90  		final String result = parseContent(FileManager.getInstance()
91  				.getTextFile(file + ".txt"));
92  		LocaleManager.getInstance().addLocale(locale, result, true);
93  		version.set(file, System.currentTimeMillis());
94  		try {
95  			version.save();
96  		} catch (final IOException e) {
97  		}
98  
99  	}

118 	/***
119 	 * Send the formated text of the locale to the player.
120 	 * 
121 	 * @return .
122 	 */
123 	public void sendText(final Player p) {
124 		final HashMap<String, String> replace = new HashMap<String, String>();
125 		final ACPlayer acPlayer = ACPlayer.getPlayer(p);
126 		final long total = acPlayer.getCurrentPlayedTime();
127 		replace.putAll(Utils.playedTime(Utils.getPlayerName(p), total));
128 		replace.put(
129 				"nb",
130 				String.valueOf(p.getServer().getOnlinePlayers().length
131 						- InvisibleWorker.getInstance().nbInvisibles()));
132 		replace.put("world", p.getWorld().getName());
133 		replace.put("x", String.valueOf(p.getLocation().getX()));
134 		replace.put("y", String.valueOf(p.getLocation().getY()));
135 		replace.put("z", String.valueOf(p.getLocation().getZ()));
136 		final Collection<String> list = Utils.getPlayerList(p);
137 		String connected = Joiner.on(", ").join(list);
138 		if (connected.length() >= ACMinecraftFontWidthCalculator.chatwidth) {
139 			final String tmp = connected.substring(0,
140 					ACMinecraftFontWidthCalculator.chatwidth);
141 			final String tmp2 = connected.substring(
142 					ACMinecraftFontWidthCalculator.chatwidth,
143 					connected.length());
144 			connected = tmp + "//n" + tmp2;
145 		}
146 		replace.put("connected", connected);
147 		final String serverTime = Utils.replaceDateAndTimeFormat(Utils
148 				.getServerRealTime("GMT" + ConfigEnum.DT_GMT.getString()));
149 		replace.put("time", serverTime);
150 		final String date = Utils.replaceDateAndTimeFormat(acPlayer,
151 				Whois.LOGIN);
152 		if (date == null) {
153 			replace.put("lastlogin", Utils.I18n("noLoginInformation"));
154 		} else {
155 			replace.put("lastlogin", date);
156 		}
157 		final String messageToSend = parseContent(Utils.I18n(locale, replace));
158 		if (messageToSend != null) {
159 			for (final String toSend : messageToSend.split("\n")) {
160 				if (toSend.isEmpty()) {
161 					continue;
162 				}
163 				p.sendMessage(toSend);
164 			}
165 		}
166 	}
