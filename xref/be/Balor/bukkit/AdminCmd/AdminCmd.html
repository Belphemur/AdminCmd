193 	public void onDisable() {
194 		final PluginDescriptionFile pdfFile = this.getDescription();
195 		getServer().getScheduler().cancelTasks(this);
196 		for (final ACPlayer p : PlayerManager.getInstance()
197 				.getOnlineACPlayers()) {
198 			PlayerManager.getInstance().setOffline(p);
199 		}
200 		ACPluginManager.getInstance().stopChildrenPlugins();
201 		CommandManager.getInstance().stopAllExecutorThreads();
202 		worker = null;
203 		ACHelper.killInstance();
204 		InvisibleWorker.killInstance();
205 		AFKWorker.killInstance();
206 		CommandManager.killInstance();
207 		HelpLister.killInstance();
208 		DebugLog.stopLogging();
209 		System.gc();
210 
211 		ACLogger.info("Plugin Disabled. (version " + pdfFile.getVersion() + ")");
212 	}

216 
217 		ExtendedConfiguration.setClassLoader(this.getClassLoader());
218 		DebugLog.setFile(getDataFolder().getPath());

220 		setMetrics();
221 
222 		final PluginDescriptionFile pdfFile = this.getDescription();
223 		DebugLog.INSTANCE.info("Plugin Version : " + pdfFile.getVersion());
224 		final PluginManager pm = getServer().getPluginManager();
225 		ACLogger.info("Plugin Enabled. (version " + pdfFile.getVersion() + ")");
226 		pm.registerEvents(new ACPluginListener(), this);
227 		worker = ACHelper.getInstance();
228 		worker.setCoreInstance(this);
229 
230 		worker.loadInfos();
231 		super.onEnable();
232 		TerminalCommandManager.getInstance().setPerm(this);
233 		pm.registerEvents(new ACBlockListener(), this);
234 		pm.registerEvents(new ACEntityListener(), this);
235 		pm.registerEvents(new ACPlayerListener(), this);
236 		pm.registerEvents(new ACWeatherListener(), this);
237 		checkModulableFeatures(pm);
238 		System.gc();
239 	}

244 	private void setMetrics() {
245 		try {
246 			metrics = new Metrics(this);
247 		} catch (final IOException e) {
248 			DebugLog.INSTANCE.log(Level.SEVERE, "Stats problem", e);
249 		}
250 		ACPluginManager.setMetrics(metrics);
251 
252 		metrics.addCustomData(new Metrics.Plotter() {
253 			@Override
254 			public String getColumnName() {
255 				return "Total Banned Players";
256 			}
257 
258 			@Override
259 			public int getValue() {
260 				return worker.countBannedPlayers();
261 			}
262 		});
263 		metrics.addCustomData(new Metrics.Plotter() {
264 			@Override
265 			public String getColumnName() {
266 				return "Total Kits";
267 			}
268 
269 			@Override
270 			public int getValue() {
271 				return worker.getNbKit();
272 			}
273 		});
274 		metrics.addCustomData(new Metrics.Plotter() {
275 			@Override
276 			public String getColumnName() {
277 				return "Total Blacklisted Items";
278 			}
279 
280 			@Override
281 			public int getValue() {
282 				return worker.countBlackListedItems();
283 			}
284 		});
285 		metrics.addCustomData(new Metrics.Plotter() {
286 			@Override
287 			public String getColumnName() {
288 				return "Total Invisible Players";
289 			}
290 
291 			@Override
292 			public int getValue() {
293 				return InvisibleWorker.getInstance().nbInvisibles();
294 			}
295 		});
296 		metrics.addCustomData(new Metrics.Plotter() {
297 			@Override
298 			public String getColumnName() {
299 				return "Total Afk Players";
300 			}
301 
302 			@Override
303 			public int getValue() {
304 				return AFKWorker.getInstance().nbAfk();
305 			}
306 		});
307 		getServer().getScheduler().runTaskLaterAsynchronously(this,
308 				new Runnable() {
309 
310 					@Override
311 					public void run() {
312 						metrics.start();
313 						DebugLog.INSTANCE.info("Stats started");
314 					}
315 				}, 30 * Utils.secInTick);
316 	}
