17  package be.Balor.bukkit.AdminCmd;
18  
19  import org.bukkit.plugin.java.JavaPlugin;
20  
21  import be.Balor.Manager.CommandManager;
22  import be.Balor.Manager.Permissions.PermissionLinker;
23  
24  /***
25   * @author Balor (aka Antoine Aflalo)
26   */
27  public abstract class AbstractAdminCmdPlugin extends JavaPlugin {
28  	protected final PermissionLinker permissionLinker;
29  	protected final String name;
30  	private final int hashCode;
31  
32  	/***
33  	 * Create the AdminCmd plugin.
34  	 * 
35  	 * @param name
36  	 *            the name used for the plugin.
37  	 * 
38  	 */
39  	public AbstractAdminCmdPlugin(final String name) {
40  		this.name = name;
41  		permissionLinker = PermissionLinker.getPermissionLinker(name);
42  		final int prime = 31;
43  		int result = 1;
44  		result = prime * result + ((name == null) ? 0 : name.hashCode());
45  		result = prime * result + ((permissionLinker == null) ? 0 : permissionLinker.hashCode());
46  		hashCode = result;
47  	}
48  
49  	public AbstractAdminCmdPlugin() {
50  		this.name = this.getClass().getSimpleName();
51  		permissionLinker = PermissionLinker.getPermissionLinker(name);
52  		final int prime = 31;
53  		int result = 1;
54  		result = prime * result + ((name == null) ? 0 : name.hashCode());
55  		result = prime * result + ((permissionLinker == null) ? 0 : permissionLinker.hashCode());
56  		hashCode = result;
57  	}
58  
59  	/***
60  	 * @return the name
61  	 */
62  	public String getAddonName() {
63  		return name;
64  	}
65  
66  	/***
67  	 * @return the permissionLinker
68  	 */
69  	public PermissionLinker getPermissionLinker() {
70  		return permissionLinker;
71  	}
72  
73  	/***
74  	 * Return the name of the plugin in the plugin.yml
75  	 * 
76  	 * @return the bukkit name of the plugin
77  	 */
78  	public String getPluginName() {
79  		return getDescription().getName();
80  	}
81  
82  	/*
83  	 * (non-Javadoc)
84  	 * 
85  	 * @see org.bukkit.plugin.Plugin#onEnable()
86  	 */
87  	@Override
88  	public void onEnable() {
89  		ACPluginManager.registerACPlugin(this);
90  		registerPermParents();
91  		CommandManager.getInstance().registerACPlugin(this);
92  		registerCmds();
93  		CommandManager.getInstance().checkAlias(this);
94  		setDefaultLocale();
95  	}
96  
97  	/***
98  	 * Definitions of the command used by the plugin
99  	 */
100 	public abstract void registerCmds();
101 
102 	/***
103 	 * Definition of the Permissions used by the plugin
104 	 */
105 	protected abstract void registerPermParents();
106 
107 	/***
108 	 * Definitions of the locale used by the plugin
109 	 */
110 	protected abstract void setDefaultLocale();
111 
112 	/*
113 	 * (non-Javadoc)
114 	 * 
115 	 * @see java.lang.Object#toString()
116 	 */
117 	@Override
118 	public String toString() {
119 		return "AbstractAdminCmdPlugin [permissionLinker=" + permissionLinker + ", name=" + name
120 				+ ", hashCode=" + hashCode + "]";
121 	}

123 }
