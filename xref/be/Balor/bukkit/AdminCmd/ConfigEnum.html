190 	/***
191 	 * @return the defaultvalues
192 	 */
193 	public static Map<String, Object> getDefaultvalues() {
194 		final Map<String, Object> values = new LinkedHashMap<String, Object>();
195 		for (final ConfigEnum ce : values())
196 			values.put(ce.confVal, ce.defaultVal);
197 		return values;
198 	}
199 
200 	public static String getHeader() {
201 		final StringBuffer buffer = new StringBuffer();
202 		buffer.append("Configuration file of ").append(pluginName).append('\n');
203 		buffer.append("Plugin Version: ").append(pluginVersion).append('\n').append('\n');
204 		for (final ConfigEnum ce : values())
205 			buffer.append(ce.confVal).append("\t:\t").append(ce.description).append(" (Default : ")
206 					.append(ce.defaultVal).append(')').append('\n');
207 		return buffer.toString();
208 	}
