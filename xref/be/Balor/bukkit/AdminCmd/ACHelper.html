397 	 * unBan the player or unban him if his time out.
398 	 * 
399 	 * @param player
400 	 * @return true if the ban is valid, false if invalid (expired)
401 	 */
402 	private boolean checkBan(final IBan player) {
403 		if (player instanceof ITempBan) {
404 			final ITempBan tempBan = (ITempBan) player;
405 			final Long timeLeft = tempBan.timeLeft();
406 			if (timeLeft <= 0) {
407 				unBanPlayer(player);
408 				return false;
409 			} else {
410 				ACPluginManager.getScheduler().runTaskLaterAsynchronously(
411 						coreInstance, new UnBanTask(tempBan, true),
412 						timeLeft / Utils.secondInMillis * Utils.secInTick);
413 				return true;
414 			}
415 		} else {
416 			return true;
417 		}
418 	}

539 	// translates a given color value/name into a real color value
540 	// also does some checking (error = -1)
541 	private short getColor(final String name, final Material mat) {
542 		short value = -1;
543 		// first try numbered colors
544 		try {
545 			value = Short.parseShort(name);
546 		} catch (final Exception e) {
547 			// try to find the name then
548 			for (short i = 0; i < materialsColors.get(mat).length; ++i) {
549 				if (materialsColors.get(mat)[i].equalsIgnoreCase(name)) {
550 					value = i;
551 					break;
552 				}
553 			}
554 		}
555 		// is the value OK?
556 		if (value < 0 || value >= materialsColors.get(mat).length) {
557 			return -1;
558 		}
559 		return value;
560 	}

1392 	public int undoLastModification(final String player)
1393 			throws EmptyStackException {
1394 		if (!undoQueue.containsKey(player)) {
1395 			throw new EmptyStackException();
1396 		}
1397 		final Stack<Stack<BlockRemanence>> blockQueue = undoQueue.get(player);
1398 		if (blockQueue.isEmpty()) {
1399 			throw new EmptyStackException();
1400 		}
1401 		final Stack<BlockRemanence> undo = blockQueue.pop();
1402 		final Stack<BlockRemanence> undoCache = new Stack<BlockRemanence>();
1403 		int i = 0;
1404 		try {
1405 			while (!undo.isEmpty()) {
1406 				undoCache.push(undo.pop());
1407 				if (undoCache.size() == Utils.MAX_BLOCKS) {
1408 					ACPluginManager.getScheduler().scheduleSyncDelayedTask(
1409 							coreInstance, new UndoBlockTask(undoCache), 1);
1410 				}
1411 				i++;
1412 			}
1413 
1414 		} catch (final Exception e) {
1415 			ACLogger.severe(e.getMessage(), e);
1416 			return i;
1417 		} finally {
1418 			ACPluginManager.getScheduler().scheduleSyncDelayedTask(
1419 					coreInstance, new UndoBlockTask(undoCache), 1);
1420 		}
1421 		return i;
