133 	/***
134 	 * Get registered plugin
135 	 * 
136 	 * @param name
137 	 *            name of the addon
138 	 * @return the addon or null if not registered
139 	 */
140 	protected AbstractAdminCmdPlugin getPlugin(final String name) {
141 		return pluginInstances.get(name);
142 	}
143 
144 	/***
145 	 * Register a AdminCmd addon
146 	 * 
147 	 * @param addon
148 	 */
149 	protected void registerPlugin(final AbstractAdminCmdPlugin addon)
150 			throws IllegalArgumentException {
151 		if (!pluginInstances.containsKey(addon.getAddonName())) {
152 			pluginInstances.put(addon.getAddonName(), addon);
153 			DebugLog.INSTANCE.info("Registering : " + addon);
154 			if (corePlugin == null || addon.equals(corePlugin))
155 				return;
156 			graph.addPlotter(new Plotter() {
157 
158 				@Override
159 				public int getValue() {
160 					return 1;
161 				}
162 
163 				@Override
164 				public String getColumnName() {
165 					return "Addon " + addon.getAddonName();
166 				}
167 			});
168 		} else
169 			throw new IllegalArgumentException("Plugin " + addon.getAddonName()
170 					+ " Already registered.");
171 	}
172 
173 	void stopChildrenPlugins() {
174 		ACLogger.info("Disabling all AdminCmd's plugins");
175 		for (final Entry<String, AbstractAdminCmdPlugin> plugin : pluginInstances.entrySet())
176 			if (plugin.getValue().isEnabled())
177 				server.getPluginManager().disablePlugin(plugin.getValue());
178 	}
179 
180 	/***
181 	 * Unregister an AdminCmd addon
182 	 * 
183 	 * @param addon
184 	 */
185 	protected void unRegisterPlugin(final AbstractAdminCmdPlugin addon) {
186 		pluginInstances.remove(addon.getAddonName());
187 		if (!addon.equals(corePlugin))
188 			graph.removePlotter(new Plotter() {
189 
190 				@Override
191 				public int getValue() {
192 					return 1;
193 				}
194 
195 				@Override
196 				public String getColumnName() {
197 					return "Addon " + addon.getAddonName();
198 				}
199 			});
200 	}
