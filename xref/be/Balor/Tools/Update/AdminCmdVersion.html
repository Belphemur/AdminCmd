32  public class AdminCmdVersion implements Comparable<AdminCmdVersion> {
33  	private final static Pattern admincmdVersion = Pattern
34  			.compile("([//d]*)//.([//d]*)//.([//d]*)(-SNAPSHOT | )//(BUILD (.*)//)");
35  	private int major, minor, build;
36  	private boolean dev = false;
37  	private Date buildDate;
38  	private final String version;
39  
40  	/***
41  	 * 
42  	 */
43  	public AdminCmdVersion(final String version) {
44  		this.version = version;
45  		try {
46  
47  			final Matcher regexMatcher = admincmdVersion.matcher(version);
48  			if (regexMatcher.find()) {
49  				major = Integer.parseInt(regexMatcher.group(1));
50  				minor = Integer.parseInt(regexMatcher.group(2));
51  				build = Integer.parseInt(regexMatcher.group(3));
52  				if (regexMatcher.group(4).equalsIgnoreCase("-SNAPSHOT ")) {
53  					dev = true;
54  				}
55  				buildDate = new SimpleDateFormat("dd.MM.yyyy '@' HH:mm:ss")
56  						.parse(regexMatcher.group(5));
57  			}
58  
59  		} catch (final PatternSyntaxException ex) {
60  			ex.printStackTrace();
61  		} catch (final ParseException e) {
62  			// TODO Auto-generated catch block
63  			e.printStackTrace();
64  		}
65  	}
66  
67  	/***
68  	 * Check if this version is newer than an other one
69  	 * 
70  	 * @param version
71  	 * @return true if newer.
72  	 */
73  	public boolean isNewerThan(final AdminCmdVersion version) {
74  		return compareTo(version) == 1;
75  	}

78  	 * (non-Javadoc)
79  	 * 
80  	 * @see java.lang.Object#hashCode()
81  	 */
82  	@Override
83  	public int hashCode() {
84  		final int prime = 31;
85  		int result = 1;
86  		result = prime * result + build;
87  		result = prime * result
88  				+ ((buildDate == null) ? 0 : buildDate.hashCode());
89  		result = prime * result + (dev ? 1231 : 1237);
90  		result = prime * result + major;
91  		result = prime * result + minor;
92  		return result;
93  	}

102 		if (this == obj) {
103 			return true;
104 		}
105 		if (obj == null) {
106 			return false;
107 		}
108 		if (getClass() != obj.getClass()) {
109 			return false;
110 		}
111 		final AdminCmdVersion other = (AdminCmdVersion) obj;
112 		if (build != other.build) {
113 			return false;
114 		}
115 		if (buildDate == null) {
116 			if (other.buildDate != null) {
117 				return false;
118 			}
119 		} else if (!buildDate.equals(other.buildDate)) {
120 			return false;
121 		}
122 		if (dev != other.dev) {
123 			return false;
124 		}
125 		if (major != other.major) {
126 			return false;
127 		}
128 		if (minor != other.minor) {
129 			return false;
130 		}
131 		return true;
132 	}

160 		if (o == null) {
