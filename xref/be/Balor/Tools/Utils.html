197 				return copy;

282 				return false;
283 			}
284 			if (pLvl >= tLvl)
285 				return true;

367 	/***
368 	 * Parse a string and replace the color in it
369 	 * 
370 	 * @author Speedy64
371 	 * @param toParse
372 	 * @return
373 	 */
374 	public static String colorParser(final String toParse) {
375 		String ResultString = null;
376 		try {
377 			Matcher regexMatcher = regexColorParser.matcher(toParse);
378 			String result = toParse;
379 			while (regexMatcher.find()) {
380 				ResultString = regexMatcher.group();
381 				int colorint = Integer.parseInt(ResultString.substring(1, 2), 16);
382 				if (ResultString.length() > 1) {
383 					if (colorint == 1 && ResultString.substring(2).matches("[012345]")) {
384 						colorint = colorint * 10 + Integer.parseInt(ResultString.substring(2));
385 					}
386 				}
387 				result = regexMatcher.replaceFirst(ChatColor.getByChar(
388 						Integer.toHexString(colorint)).toString());
389 				regexMatcher = regexColorParser.matcher(result);
390 			}
391 			return result;
392 		} catch (final Exception ex) {
393 			return toParse;
394 		}
395 	}

397 	/***
398 	 * Because water and lava are fluid, using another algo to "delete"
399 	 * 
400 	 * @param block
401 	 * @param radius
402 	 * @return
403 	 */
404 	private static Stack<BlockRemanence> drainFluid(final String playername, final Block block,
405 			final int radius) {
406 		final Stack<BlockRemanence> blocks = new Stack<BlockRemanence>();
407 		final Stack<SimplifiedLocation> processQueue = new Stack<SimplifiedLocation>();
408 		BlockRemanence current = null;
409 		final World w = block.getWorld();
410 		final Location start = block.getLocation();
411 		final HashSet<SimplifiedLocation> visited = new HashSet<SimplifiedLocation>();
412 		final Stack<BlockRemanence> blocksCache = new Stack<BlockRemanence>();
413 		for (int x = block.getX() - 2; x <= block.getX() + 2; x++) {
414 			for (int z = block.getZ() - 2; z <= block.getZ() + 2; z++) {
415 				for (int y = block.getY() - 2; y <= block.getY() + 2; y++) {
416 					final SimplifiedLocation newPos = new SimplifiedLocation(w, x, y, z);
417 					if (isFluid(newPos) && !visited.contains(newPos)) {
418 						visited.add(newPos);
419 						processQueue.push(newPos);
420 						current = IBlockRemanenceFactory.FACTORY.createBlockRemanence(newPos);
421 						blocks.push(current);
422 						blocksCache.push(current);
423 						if (blocksCache.size() == MAX_BLOCKS)
424 							ACPluginManager.getScheduler().scheduleSyncDelayedTask(
425 									ACHelper.getInstance().getCoreInstance(),
426 									new ReplaceBlockTask(blocksCache));
427 					}
428 
429 				}
430 			}
431 		}
432 		while (!processQueue.isEmpty()) {
433 			final SimplifiedLocation loc = processQueue.pop();
434 			for (int y = loc.getBlockY() - 1; y <= loc.getBlockY() + 1; y++) {
435 				for (int x = loc.getBlockX() - 1; x <= loc.getBlockX() + 1; x++) {
436 					for (int z = loc.getBlockZ() - 1; z <= loc.getBlockZ() + 1; z++) {
437 						final SimplifiedLocation newPos = new SimplifiedLocation(w, x, y, z);
438 						if (!visited.contains(newPos) && isFluid(newPos)
439 								&& start.distance(newPos) < radius) {
440 							processQueue.push(newPos);
441 							current = IBlockRemanenceFactory.FACTORY.createBlockRemanence(newPos);
442 							blocks.push(current);
443 							blocksCache.push(current);
444 							if (blocksCache.size() == MAX_BLOCKS)
445 								ACPluginManager.getScheduler().scheduleSyncDelayedTask(
446 										ACHelper.getInstance().getCoreInstance(),
447 										new ReplaceBlockTask(blocksCache));
448 							visited.add(newPos);
449 						}
450 					}
451 
452 				}
453 			}
454 		}
455 		ACPluginManager.getScheduler().scheduleSyncDelayedTask(
456 				ACHelper.getInstance().getCoreInstance(), new ReplaceBlockTask(blocksCache));
457 		return blocks;
458 	}

561 	public static Player getPlayer(final String name) {
562 		final Player[] players = ACPluginManager.getServer().getOnlinePlayers();
563 
564 		Player found = null;
565 		final String lowerName = name.toLowerCase();
566 		int delta = Integer.MAX_VALUE;
567 		for (final Player player : players) {
568 			if (player.getName().toLowerCase().startsWith(lowerName)) {
569 				int curDelta = player.getName().length() - lowerName.length();
570 				if (curDelta < delta) {
571 					found = player;
572 					delta = curDelta;
573 				} else {
574 					curDelta = player.getDisplayName().length() - lowerName.length();
575 					if (curDelta < delta) {
576 						found = player;
577 						delta = curDelta;
578 					}
579 				}
580 				if (curDelta == 0)
581 					break;
582 			}
583 		}
584 		return found;
585 
586 	}

924 	 * 
925 	 * @param mat
926 	 * @param block
927 	 * @param radius
928 	 * @return
929 	 */
930 	private static Stack<BlockRemanence> replaceInCuboid(final String playername,
931 			final List<Material> mat, final Block block, final int radius) {
932 		final Stack<BlockRemanence> blocks = new SynchronizedStack<BlockRemanence>();
933 		final Stack<BlockRemanence> blocksCache = new SynchronizedStack<BlockRemanence>();
934 		final int limitX = block.getX() + radius;
935 		final int limitY = block.getY() + radius;
936 		final int limitZ = block.getZ() + radius;
937 		BlockRemanence br = null;
938 		final Semaphore sema = new Semaphore(0, true);
939 		final List<SimplifiedLocation> okBlocks = new ArrayList<SimplifiedLocation>(50);
940 		ACPluginManager.scheduleSyncTask(new CheckingBlockTask(sema, okBlocks, block, radius,
941 				limitY, limitX, limitZ, mat));
942 		try {
943 			sema.acquire();
944 		} catch (final InterruptedException e) {
945 			DebugLog.INSTANCE.log(Level.SEVERE, "Problem with acquiring the semaphore", e);
946 		}
947 		for (final SimplifiedLocation loc : okBlocks) {
948 			br = IBlockRemanenceFactory.FACTORY.createBlockRemanence(loc);
949 			blocks.push(br);
950 			blocksCache.push(br);
951 			if (blocksCache.size() == MAX_BLOCKS)
952 				ACPluginManager.getScheduler().scheduleSyncDelayedTask(
953 						ACHelper.getInstance().getCoreInstance(),
954 						new ReplaceBlockTask(blocksCache), 1);
955 		}
956 		ACPluginManager.getScheduler().scheduleSyncDelayedTask(
957 				ACHelper.getInstance().getCoreInstance(), new ReplaceBlockTask(blocksCache), 1);
958 		return blocks;
959 	}

1048 					newtime += Integer.parseInt(arg);

1050 					sI18n(sender, "timeNotSet", replace);

1053 			}

1065 				new SetTime(w, newtime));
