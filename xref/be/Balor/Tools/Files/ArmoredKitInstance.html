39  public class ArmoredKitInstance extends KitInstance {
40  	protected final Map<Type.ArmorPart, MaterialContainer> armor = new EnumMap<Type.ArmorPart, MaterialContainer>(
41  			Type.ArmorPart.class);
42  
43  	/***
44  	 * @param name
45  	 * @param delay
46  	 * @param items
47  	 */
48  	public ArmoredKitInstance(final String name, final int delay,
49  			final List<MaterialContainer> items, final Map<Type.ArmorPart, MaterialContainer> armor) {
50  		super(name, delay, items);
51  		for (final Entry<Type.ArmorPart, MaterialContainer> e : armor.entrySet()) {
52  			if (!e.getKey().isValid(e.getValue().getMaterial().getId()))
53  				continue;
54  			this.armor.put(e.getKey(), e.getValue());
55  		}
56  
57  	}
58  
59  	public ArmoredKitInstance(final KitInstance kit) {
60  		super(kit.name, kit.delay, kit.items);
61  	}
62  
63  	/***
64  	 * Get the wanted part of the armor
65  	 * 
66  	 * @param part
67  	 * @return itemstack containing the part of the armor
68  	 */
69  	public ItemStack getArmorPart(final Type.ArmorPart part) {
70  		final MaterialContainer mat = armor.get(part);
71  		if (mat == null)
72  			return null;
73  		return mat.getItemStack();
74  	}
75  
76  	/***
77  	 * Set the player armor using the armor set in the kit
78  	 * 
79  	 * @param p
80  	 *            player to change the armor's inventory.
81  	 * @return contain the Part of the armor that couldn't be setted.
82  	 */
83  	public ArmorPart[] setPlayerArmorParts(final Player p) {
84  		final List<ArmorPart> armorParts = new ArrayList<Type.ArmorPart>(ArmorPart.values().length);
85  		final ItemStack[] armors = new ItemStack[ArmorPart.values().length];
86  		final PlayerInventory inventory = p.getInventory();
87  		if (ConfigEnum.ARMOR_KIT_OVERRIDE.getBoolean()) {
88  			for (final ArmorPart part : ArmorPart.values()) {
89  				ItemStack toadd = getArmorPart(part);
90  				if (toadd == null) {
91  					toadd = inventory.getItem(inventory.getSize() + part.getPlaceInInventory());
92  					armorParts.add(part);
93  				}
94  				armors[part.getPlaceInInventory()] = toadd;
95  			}
96  		} else {
97  			for (final ArmorPart part : ArmorPart.values()) {
98  				ItemStack toadd = getArmorPart(part);
99  				if (inventory.getItem(inventory.getSize() + part.getPlaceInInventory()) != null)
100 					continue;
101 				if (toadd == null) {
102 					toadd = inventory.getItem(inventory.getSize() + part.getPlaceInInventory());
103 					armorParts.add(part);
104 				}
105 				armors[part.getPlaceInInventory()] = toadd;
106 			}
107 		}
108 		ACPluginManager.scheduleSyncTask(new Runnable() {
109 
110 			@Override
111 			public void run() {
112 				inventory.setArmorContents(armors);
113 			}
114 		});
115 		return armorParts.toArray(new ArmorPart[0]);
116 	}

118 	/*
119 	 * (non-Javadoc)
120 	 * 
121 	 * @see
122 	 * be.Balor.Tools.Files.KitInstance#addParent(be.Balor.Tools.Files.KitInstance
123 	 * )
124 	 */
125 	@Override
126 	public void addParent(final KitInstance parent) {
127 		super.addParent(parent);
128 		if (!(parent instanceof ArmoredKitInstance))
129 			return;
130 		final ArmoredKitInstance armoredParent = (ArmoredKitInstance) parent;
131 		for (final ArmorPart part : ArmorPart.values()) {
132 			if (armor.get(part) == null)
133 				armor.put(part, armoredParent.armor.get(part));
134 		}
135 
136 	}
