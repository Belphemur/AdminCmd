8   http://www.unicode.org/unicode/faq/utf_bom.html
9   BOMs in byte length ordering:
10    00 00 FE FF    = UTF-32, big-endian
11    FF FE 00 00    = UTF-32, little-endian
12    EF BB BF       = UTF-8,
13    FE FF          = UTF-16, big-endian

19  package be.Balor.Tools.Files.Unicode;

21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.PushbackInputStream;
24  
25  /***
26   * This inputstream will recognize unicode BOM marks and will skip bytes if
27   * getEncoding() method is called before any of the read(...) methods.
28   * 
29   * Usage pattern: String enc = "ISO-8859-1"; // or NULL to use systemdefault
30   * FileInputStream fis = new FileInputStream(file); UnicodeInputStream uin = new
31   * UnicodeInputStream(fis, enc); enc = uin.getEncoding(); // check and skip
32   * possible BOM bytes InputStreamReader in; if (enc == null) in = new
33   * InputStreamReader(uin); else in = new InputStreamReader(uin, enc);
34   */
35  public class UnicodeInputStream extends InputStream {
36  	PushbackInputStream internalIn;
37  	boolean isInited = false;
38  	String defaultEnc;
39  	String encoding;
40  
41  	public UnicodeInputStream(final InputStream in, final String defaultEnc) {
42  		internalIn = new PushbackInputStream(in, UnicodeUtil.BOM_SIZE);
43  		this.defaultEnc = defaultEnc;
44  	}
45  
46  	public String getDefaultEncoding() {
47  		return defaultEnc;
48  	}
49  
50  	public String getEncoding() {
51  		if (!isInited) {
52  			try {
53  				init();
54  			} catch (final IOException ex) {
55  				final IllegalStateException ise = new IllegalStateException(
56  						"Init method failed.");
57  				ise.initCause(ise);
58  				throw ise;
59  			}
60  		}
61  		return encoding;
62  	}
63  
64  	/***
65  	 * Read-ahead four bytes and check for BOM marks. Extra bytes are unread
66  	 * back to the stream, only BOM bytes are skipped.
67  	 */
68  	protected void init() throws IOException {
69  		if (isInited) {
70  			return;
71  		}
72  
73  		final byte bom[] = new byte[UnicodeUtil.BOM_SIZE];
74  		int n, unread;
75  		n = internalIn.read(bom, 0, bom.length);
76  
77  		if ((bom[0] == (byte) 0x00) && (bom[1] == (byte) 0x00)
78  				&& (bom[2] == (byte) 0xFE) && (bom[3] == (byte) 0xFF)) {
79  			encoding = "UTF-32BE";
80  			unread = n - 4;
81  		} else if ((bom[0] == (byte) 0xFF) && (bom[1] == (byte) 0xFE)
82  				&& (bom[2] == (byte) 0x00) && (bom[3] == (byte) 0x00)) {
83  			encoding = "UTF-32LE";
84  			unread = n - 4;
85  		} else if ((bom[0] == (byte) 0xEF) && (bom[1] == (byte) 0xBB)
86  				&& (bom[2] == (byte) 0xBF)) {
87  			encoding = "UTF-8";
88  			unread = n - 3;
89  		} else if ((bom[0] == (byte) 0xFE) && (bom[1] == (byte) 0xFF)) {
90  			encoding = "UTF-16BE";
91  			unread = n - 2;
92  		} else if ((bom[0] == (byte) 0xFF) && (bom[1] == (byte) 0xFE)) {
93  			encoding = "UTF-16LE";
94  			unread = n - 2;
95  		} else {
96  			// Unicode BOM mark not found, unread all bytes
97  			encoding = defaultEnc;
98  			unread = n;
99  		}
100 		// System.out.println("read=" + n + ", unread=" + unread);
101 
102 		if (unread > 0) {
103 			internalIn.unread(bom, (n - unread), unread);
104 		}
105 
106 		isInited = true;
107 	}
108 
109 	@Override
110 	public void close() throws IOException {
111 		// init();
112 		isInited = true;
113 		internalIn.close();
114 	}
115 
116 	@Override
117 	public int read() throws IOException {
118 		// init();
119 		isInited = true;
120 		return internalIn.read();
121 	}
