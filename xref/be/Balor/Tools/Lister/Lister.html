30   * @author Balor (aka Antoine Aflalo)

33  public abstract class Lister {
34  	private final static Map<List, Lister> lastListed = new EnumMap<Lister.List, Lister>(
35  			Lister.List.class);
36  
37  	public enum List {
38  		BAN, MUTE;
39  	}
40  
41  	/***
42  	 * 
43  	 */
44  	Lister() {
45  	}
46  
47  	/***
48  	 * Get a Lister for the wanted type
49  	 * 
50  	 * @param type
51  	 *            type of the lister
52  	 * @return lister
53  	 */
54  	public static Lister getLister(final Lister.List type) {
55  		return getLister(type, true);
56  	}
57  
58  	/***
59  	 * Get a Lister for the wanted type
60  	 * 
61  	 * @param type
62  	 *            type of the lister
63  	 * @param instanciate
64  	 *            if we want to create it if doesn't exists.
65  	 * @return lister
66  	 */
67  	public static Lister getLister(final Lister.List type,
68  			final boolean instanciate) {
69  		Lister lister = lastListed.get(type);
70  		if (!instanciate) {
71  			return lister;
72  		} else if (lister == null) {
73  			switch (type) {
74  				case BAN :
75  					lister = new BanLister();
76  					break;
77  				case MUTE :
78  					lister = new MuteLister();
79  					break;
80  
81  				default :
82  					break;
83  			}
84  			lastListed.put(type, lister);
85  		}
86  
87  		return lister;
88  	}
89  
90  	/***
91  	 * Get the wanted page
92  	 * 
93  	 * @param page
94  	 *            nb of the page
95  	 * @return the items of the page to be displayed
96  	 * @throws EmptyListException
97  	 *             the list is empty
98  	 */
99  	public synchronized Collection<String> getPage(int page)
100 			throws EmptyListException {
101 		final int entryPerPage = ConfigEnum.LISTER_ITEMS.getInt();
102 		final Collection<String> list = getList();
103 
104 		if (list.isEmpty()) {
105 			throw new EmptyListException();
106 		}
107 
108 		final int maxPages = (int) Math.ceil(list.size()
109 				/ (double) entryPerPage);
110 		page = page > maxPages ? maxPages : page;
111 		final int start = (page - 1) * entryPerPage;
112 		final int end = start + entryPerPage > list.size()
113 				? list.size()
114 				: start + entryPerPage;
115 		final java.util.List<String> result = new ArrayList<String>();
116 		result.add(ChatColor.AQUA
117 				+ ACMinecraftFontWidthCalculator.strPadCenterChat(
118 						ChatColor.DARK_GREEN + " " + getType() + " (" + page
119 								+ "/" + maxPages + ") " + ChatColor.AQUA, '='));
120 		final String[] array = list.toArray(new String[]{});
121 		for (int i = start; i < end; i++) {
122 			result.add(array[i]);
123 		}
124 
125 		return result;
126 	}
127 
128 	/***
129 	 * Ask the lister to update its inner list
130 	 */
131 	public abstract void update();
132 
133 	abstract Collection<String> getList();
134 
135 	abstract List getType();
136 }
