24  import java.util.Queue;

31  /***
32   * @author Balor (aka Antoine Aflalo)

34   */
35  public class MuteLister extends Lister {
36  	private final Queue<String> mute = new PriorityQueue<String>();
37  
38  	/***
39  	 * 
40  	 */
41  	MuteLister() {
42  		update();
43  	}
44  
45  	/*
46  	 * (non-Javadoc)
47  	 * 
48  	 * @see be.Balor.Tools.Lister.Lister#update()
49  	 */
50  	@Override
51  	public synchronized void update() {
52  		mute.clear();
53  		final Set<ACPlayer> players = new HashSet<ACPlayer>();
54  		players.addAll(ACPlayer.getPlayers(Type.MUTED));
55  		players.addAll(ACPlayer.getPlayers(Type.MUTED_COMMAND));
56  		final HashMap<String, String> replace = new HashMap<String, String>();
57  		for (final ACPlayer p : players) {
58  			replace.clear();
59  			if (p.hasPower(Type.MUTED)) {
60  				replace.put("player", p.getName());
61  				replace.put("msg", p.getPower(Type.MUTED).getString());
62  				mute.add(LocaleHelper.MUTELIST.getLocale(replace));
63  			} else if (p.hasPower(Type.MUTED_COMMAND)) {
64  				replace.put("player", p.getName());
65  				replace.put("msg", p.getPower(Type.MUTED_COMMAND).getString());
66  				mute.add(LocaleHelper.MUTELIST.getLocale(replace));
67  			} else {
68  				continue;
69  			}
70  		}
71  
72  	}
73  
74  	/*
75  	 * (non-Javadoc)
76  	 * 
77  	 * @see be.Balor.Tools.Lister.Lister#getList()
78  	 */
79  	@Override
80  	Collection<String> getList() {
81  		return Collections.unmodifiableCollection(mute);
82  	}
83  
84  	/*
85  	 * (non-Javadoc)
86  	 * 
87  	 * @see be.Balor.Tools.Lister.Lister#getType()
88  	 */
89  	@Override
90  	List getType() {
91  		return List.MUTE;
92  	}
93  
94  }
