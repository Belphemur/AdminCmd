145 	public Metrics(final Plugin plugin) throws IOException {
146 		if (plugin == null) {
147 			throw new IllegalArgumentException("Plugin cannot be null");
148 		}
149 
150 		this.plugin = plugin;
151 
152 		// load the config
153 		configurationFile = getConfigFile();
154 		configuration = YamlConfiguration.loadConfiguration(configurationFile);
155 
156 		// add some defaults
157 		configuration.addDefault("opt-out", false);
158 		configuration.addDefault("guid", UUID.randomUUID().toString());
159 		configuration.addDefault("debug", false);
160 
161 		// Do we need to create the file?
162 		if (configuration.get("guid", null) == null) {
163 			configuration.options().header("http://mcstats.org")
164 					.copyDefaults(true);
165 			configuration.save(configurationFile);
166 		}
167 
168 		// Load the guid then
169 		guid = configuration.getString("guid");
170 		debug = configuration.getBoolean("debug", false);

231 	/***
232 	 * Start measuring statistics. This will immediately create an async
233 	 * repeating task as the plugin and send the initial data to the metrics
234 	 * backend, and then after that it will post in increments of PING_INTERVAL
235 	 * * 1200 ticks.
236 	 * 
237 	 * @return True if statistics measuring is running, otherwise false.
238 	 */
239 	public boolean start() {
240 		synchronized (optOutLock) {
241 			// Did we opt out?
242 			if (isOptOut()) {
243 				return false;
244 			}
245 
246 			// Is metrics already running?
247 			if (task != null) {
248 				return true;
249 			}
250 
251 			// Begin hitting the server with glorious data
252 			task = plugin.getServer().getScheduler()
253 					.runTaskTimerAsynchronously(plugin, new Runnable() {
254 
255 						private boolean firstPost = true;
256 
257 						@Override
258 						public void run() {
259 							try {
260 								// This has to be synchronized or it can collide
261 								// with the disable method.
262 								synchronized (optOutLock) {
263 									// Disable Task, if it is running and the
264 									// server owner decided to opt-out
265 									if (isOptOut() && task != null) {
266 										task.cancel();
267 										task = null;
268 										// Tell all plotters to stop gathering
269 										// information.
270 										for (final Graph graph : graphs) {
271 											graph.onOptOut();
272 										}
273 									}
274 								}
275 
276 								// We use the inverse of firstPost because if it
277 								// is the first time we are posting,
278 								// it is not a interval ping, so it evaluates to
279 								// FALSE
280 								// Each time thereafter it will evaluate to
281 								// TRUE, i.e PING!
282 								postPlugin(!firstPost);
283 
284 								// After the first post we set firstPost to
285 								// false
286 								// Each post thereafter will be a ping
287 								firstPost = false;
288 							} catch (final IOException e) {
289 								if (debug) {
290 									Bukkit.getLogger().log(Level.INFO,
291 											"[Metrics] " + e.getMessage());
292 								}
293 							}
294 						}
295 					}, 0, PING_INTERVAL * 1200);
296 
297 			return true;
298 		}
299 	}

302 	 * Has the server owner denied plugin metrics?
303 	 * 
304 	 * @return true if metrics should be opted out of it
305 	 */
306 	public boolean isOptOut() {
307 		synchronized (optOutLock) {
308 			try {
309 				// Reload the metrics file
310 				configuration.load(getConfigFile());
311 			} catch (final IOException ex) {
312 				if (debug) {
313 					Bukkit.getLogger().log(Level.INFO,
314 							"[Metrics] " + ex.getMessage());
315 				}
316 				return true;
317 			} catch (final InvalidConfigurationException ex) {
318 				if (debug) {
319 					Bukkit.getLogger().log(Level.INFO,
320 							"[Metrics] " + ex.getMessage());
321 				}
322 				return true;
323 			}
324 			return configuration.getBoolean("opt-out", false);
325 		}
326 	}

329 	 * Enables metrics for the server by setting "opt-out" to false in the
330 	 * config file and starting the metrics task.
331 	 * 
332 	 * @throws IOException
333 	 */
334 	public void enable() throws IOException {
335 		// This has to be synchronized or it can collide with the check in the
336 		// task.
337 		synchronized (optOutLock) {
338 			// Check if the server owner has already set opt-out, if not, set
339 			// it.
340 			if (isOptOut()) {
341 				configuration.set("opt-out", false);
342 				configuration.save(configurationFile);
343 			}
344 
345 			// Enable Task, if it is not running
346 			if (task == null) {
347 				start();
348 			}
349 		}
350 	}

353 	 * Disables metrics for the server by setting "opt-out" to true in the
354 	 * config file and canceling the metrics task.
355 	 * 
356 	 * @throws IOException
357 	 */
358 	public void disable() throws IOException {
359 		// This has to be synchronized or it can collide with the check in the
360 		// task.
361 		synchronized (optOutLock) {
362 			// Check if the server owner has already set opt-out, if not, set
363 			// it.
364 			if (!isOptOut()) {
365 				configuration.set("opt-out", true);
366 				configuration.save(configurationFile);
367 			}
368 
369 			// Disable Task, if it is running
370 			if (task != null) {
371 				task.cancel();
372 				task = null;
373 			}
374 		}
375 	}

393 		return new File(new File(pluginsFolder, "PluginMetrics"), "config.yml");

396 	/***
397 	 * Generic method that posts a plugin to the metrics website
398 	 */
399 	private void postPlugin(final boolean isPing) throws IOException {
400 		// Server software specific section
401 		final PluginDescriptionFile description = plugin.getDescription();
402 		final String pluginName = description.getName();
403 		final boolean onlineMode = Bukkit.getServer().getOnlineMode(); // TRUE
404 																		// if
405 																		// online
406 																		// mode
407 																		// is
408 																		// enabled
409 		final String pluginVersion = description.getVersion();
410 		final String serverVersion = Bukkit.getVersion();
411 		final int playersOnline = Bukkit.getServer().getOnlinePlayers().length;
412 
413 		// END server software specific section -- all code below does not use
414 		// any code outside of this class / Java
415 
416 		// Construct the post data
417 		final StringBuilder data = new StringBuilder();
418 
419 		// The plugin's description file containg all of the plugin data such as
420 		// name, version, author, etc
421 		data.append(encode("guid")).append('=').append(encode(guid));
422 		encodeDataPair(data, "version", pluginVersion);
423 		encodeDataPair(data, "server", serverVersion);
424 		encodeDataPair(data, "players", Integer.toString(playersOnline));
425 		encodeDataPair(data, "revision", String.valueOf(REVISION));
426 
427 		// New data as of R6
428 		final String osname = System.getProperty("os.name");
429 		String osarch = System.getProperty("os.arch");
430 		final String osversion = System.getProperty("os.version");
431 		final String java_version = System.getProperty("java.version");
432 		final int coreCount = Runtime.getRuntime().availableProcessors();
433 
434 		// normalize os arch .. amd64 -> x86_64
435 		if (osarch.equals("amd64")) {
436 			osarch = "x86_64";
437 		}
438 
439 		encodeDataPair(data, "osname", osname);
440 		encodeDataPair(data, "osarch", osarch);
441 		encodeDataPair(data, "osversion", osversion);
442 		encodeDataPair(data, "cores", Integer.toString(coreCount));
443 		encodeDataPair(data, "online-mode", Boolean.toString(onlineMode));
444 		encodeDataPair(data, "java_version", java_version);
445 
446 		// If we're pinging, append it
447 		if (isPing) {
448 			encodeDataPair(data, "ping", "true");
449 		}
450 
451 		// Acquire a lock on the graphs, which lets us make the assumption we
452 		// also lock everything
453 		// inside of the graph (e.g plotters)
454 		synchronized (graphs) {
455 			final Iterator<Graph> iter = graphs.iterator();
456 
457 			while (iter.hasNext()) {
458 				final Graph graph = iter.next();
459 
460 				for (final Plotter plotter : graph.getPlotters()) {
461 					// The key name to send to the metrics server
462 					// The format is C-GRAPHNAME-PLOTTERNAME where separator -
463 					// is defined at the top
464 					// Legacy (R4) submitters use the format Custom%s, or
465 					// CustomPLOTTERNAME
466 					final String key = String.format("C%s%s%s%s",
467 							CUSTOM_DATA_SEPARATOR, graph.getName(),
468 							CUSTOM_DATA_SEPARATOR, plotter.getColumnName());
469 
470 					// The value to send, which for the foreseeable future is
471 					// just the string
472 					// value of plotter.getValue()
473 					final String value = Integer.toString(plotter.getValue());
474 
475 					// Add it to the http post data :)
476 					encodeDataPair(data, key, value);
477 				}
478 			}
479 		}
480 
481 		// Create the url
482 		final URL url = new URL(BASE_URL
483 				+ String.format(REPORT_URL, encode(pluginName)));
484 
485 		// Connect to the website
486 		URLConnection connection;
487 
488 		// Mineshafter creates a socks proxy, so we can safely bypass it
489 		// It does not reroute POST requests so we need to go around it
490 		if (isMineshafterPresent()) {
491 			connection = url.openConnection(Proxy.NO_PROXY);
492 		} else {
493 			connection = url.openConnection();
494 		}
495 
496 		connection.setDoOutput(true);
497 
498 		// Write the data
499 		final OutputStreamWriter writer = new OutputStreamWriter(
500 				connection.getOutputStream());
501 		writer.write(data.toString());
502 		writer.flush();
503 
504 		// Now read the response
505 		final BufferedReader reader = new BufferedReader(new InputStreamReader(
506 				connection.getInputStream()));
507 		final String response = reader.readLine();
508 
509 		// close resources
510 		writer.close();
511 		reader.close();
512 
513 		if (response == null || response.startsWith("ERR")) {
514 			throw new IOException(response); // Throw the exception
515 		} else {
516 			// Is this the first update this hour?
517 			if (response.contains("OK This is your first update this hour")) {
518 				synchronized (graphs) {
519 					final Iterator<Graph> iter = graphs.iterator();
520 
521 					while (iter.hasNext()) {
522 						final Graph graph = iter.next();
523 
524 						for (final Plotter plotter : graph.getPlotters()) {
525 							plotter.reset();
526 						}
527 					}
528 				}
529 			}
530 		}
531 	}

548 	/***
549 	 * <p>
550 	 * Encode a key/value data pair to be used in a HTTP post request. This
551 	 * INCLUDES a & so the first key/value pair MUST be included manually, e.g:
552 	 * </p>
553 	 * <code>
554 	 * StringBuffer data = new StringBuffer();
555 	 * data.append(encode("guid")).append('=').append(encode(guid));
556 	 * encodeDataPair(data, "version", description.getVersion());
557 	 * </code>
558 	 * 
559 	 * @param buffer
560 	 *            the stringbuilder to append the data pair onto
561 	 * @param key
562 	 *            the key value
563 	 * @param value
564 	 *            the value
565 	 */
566 	private static void encodeDataPair(final StringBuilder buffer,
567 			final String key, final String value)
568 			throws UnsupportedEncodingException {
569 		buffer.append('&').append(encode(key)).append('=')
570 				.append(encode(value));
571 	}

667 	/***
668 	 * Interface used to collect custom data for a plugin
669 	 */
670 	public static abstract class Plotter {
671 
672 		/***
673 		 * The plot's name
674 		 */
675 		private final String name;
676 
677 		/***
678 		 * Construct a plotter with the default plot name
679 		 */
680 		public Plotter() {
681 			this("Default");
682 		}
683 
684 		/***
685 		 * Construct a plotter with a specific plot name
686 		 * 
687 		 * @param name
688 		 *            the name of the plotter to use, which will show up on the
689 		 *            website
690 		 */
691 		public Plotter(final String name) {
692 			this.name = name;
693 		}
694 
695 		/***
696 		 * Get the current value for the plotted point. Since this function
697 		 * defers to an external function it may or may not return immediately
698 		 * thus cannot be guaranteed to be thread friendly or safe. This
699 		 * function can be called from any thread so care should be taken when
700 		 * accessing resources that need to be synchronized.
701 		 * 
702 		 * @return the current value for the point to be plotted.
703 		 */
704 		public abstract int getValue();
705 
706 		/***
707 		 * Get the column name for the plotted point
708 		 * 
709 		 * @return the plotted point's column name
710 		 */
711 		public String getColumnName() {
712 			return name;
713 		}
714 
715 		/***
716 		 * Called after the website graphs have been updated
717 		 */
718 		public void reset() {
719 		}
720 
721 		@Override
722 		public int hashCode() {
723 			return getColumnName().hashCode();
724 		}
725 
726 		@Override
727 		public boolean equals(final Object object) {
728 			if (!(object instanceof Plotter)) {
729 				return false;
730 			}
731 
732 			final Plotter plotter = (Plotter) object;
733 			return plotter.name.equals(name)
734 					&& plotter.getValue() == getValue();
735 		}

737 	}
