30  public class MaterialContainer implements Comparable<MaterialContainer> {
31  	private Material material = null;
32  	private short dmg = 0;
33  	private int amount = 1;
34  	private final Map<Enchantment, Integer> enchantments;
35  
36  	public MaterialContainer(final ItemStack is) {
37  		material = is.getType();
38  		dmg = is.getDurability();
39  		this.enchantments = is.getEnchantments();
40  	}
41  
42  	public MaterialContainer(final String mat) {
43  		String[] info = new String[2];
44  		if (mat.contains(":"))
45  			info = mat.split(":");
46  		else {
47  			info[0] = mat;
48  			info[1] = "0";
49  		}
50  		parseMat(info[0]);
51  		parseDmg(info[1]);
52  		this.enchantments = new HashMap<Enchantment, Integer>();
53  	}
54  
55  	/***
56  	 * @param amount
57  	 *            the amount to set
58  	 */
59  	public void setAmount(final int amount) {
60  		this.amount = amount;
61  	}
62  
63  	public MaterialContainer(final String mat, final String damage) {
64  		parseMat(mat);
65  		parseDmg(damage);
66  		this.enchantments = new HashMap<Enchantment, Integer>();
67  	}
68  
69  	public MaterialContainer() {
70  		this.enchantments = new HashMap<Enchantment, Integer>();
71  	}
72  
73  	private void parseMat(final String material) {
74  		Material m = null;
75  		try {
76  			final int id = Integer.parseInt(material);
77  			m = Material.getMaterial(id);
78  		} catch (final NumberFormatException e) {
79  			m = Material.matchMaterial(material);
80  		}
81  		this.material = m;
82  	}
83  
84  	private void parseDmg(final String damage) {
85  		short d = 0;
86  		try {
87  			d = Short.parseShort(damage);
88  		} catch (final NumberFormatException e) {
89  		}
90  		this.dmg = d;
91  	}
92  
93  	public boolean isNull() {
94  		return material == null;
95  	}
96  
97  	/***
98  	 * Transform the MaterialContainer to an ItemStack
99  	 * 
100 	 * @param amount
101 	 * @return
102 	 */
103 
104 	public ItemStack getItemStack() {
105 		final ItemStack toReturn = new ItemStack(material, amount, dmg);
106 		toReturn.addUnsafeEnchantments(enchantments);
107 		return toReturn;
108 	}

112 		if (material != null)

172 	 * @see java.lang.Comparable#compareTo(java.lang.Object)

174 	@Override
175 	public int compareTo(final MaterialContainer o) {
176 		if (material.equals(o.getMaterial()))
177 			return dmg - o.getDmg();
178 		return material.compareTo(o.getMaterial());
179 	}

182 	 * (non-Javadoc)

186 	@Override

188 		final int prime = 31;
