164 	private void parseMat(final String material) {
165 		Material m = null;
166 		try {
167 			final int id = Integer.parseInt(material);
168 			m = Material.getMaterial(id);
169 		} catch (final NumberFormatException e) {
170 			m = Material.matchMaterial(material);
171 		}
172 		this.material = m;
173 	}

188 	/***
189 	 * Add the wanted enchantment to the item with the wanted lvl
190 	 * 
191 	 * @param enchant
192 	 *            enchantment
193 	 * @param lvl
194 	 *            level
195 	 * @return false if the enchantment is already set
196 	 * @throws EnchantmentConflictException
197 	 *             if there is a conflict between an already existed enchantment
198 	 *             on the item.
199 	 * @throws CantEnchantItemException
200 	 *             if the item can't be enchanted by this enchantment.
201 	 */
202 	public boolean addEnchantment(final Enchantment enchant, final int lvl)
203 			throws EnchantmentConflictException, CantEnchantItemException {
204 		try {
205 			if (getMeta().hasEnchant(enchant)) {
206 				return false;
207 			}
208 		} catch (final Exception ex) {
209 			if (enchantments.containsKey(enchant)) {
210 				return false;
211 			}
212 		}
213 		if (!enchant.canEnchantItem(getItemStack())) {
214 			throw new CantEnchantItemException(enchant, this.material);
215 		}
216 		Set<Enchantment> enchants = new HashSet<Enchantment>();
217 		try {
218 			enchants = getMeta().getEnchants().keySet();
219 		} catch (final Exception e) {
220 			enchants = enchantments.keySet();
221 		}
222 		for (final Enchantment e : enchants) {
223 			if (e.conflictsWith(enchant)) {
224 				throw new EnchantmentConflictException(enchant, e);
225 			}
226 		}
227 		try {
228 			getMeta().addEnchant(enchant, lvl, true);
229 		} catch (final Exception ex) {
230 			enchantments.put(enchant, lvl);
231 		}
232 
233 		return true;
234 	}

236 	/***
237 	 * Try to add an Enchantment to the item by parsing it
238 	 * 
239 	 * @param enchant
240 	 *            string containing the enchantment like
241 	 *            <code>enchant:lvl</code>
242 	 * @return true if we can add the enchantment on the item, false if the
243 	 *         enchantment don't exist or already present.
244 	 * @throws EnchantmentConflictException
245 	 *             if there is a conflict between an already existed enchantment
246 	 *             on the item.
247 	 * @throws CantEnchantItemException
248 	 *             if the item can't be enchanted by this enchantment.
249 	 */
250 	public boolean addEnchantment(final String enchant)
251 			throws EnchantmentConflictException, CantEnchantItemException {
252 		final String split[] = enchant.split(":");
253 		if (split.length < 2) {
254 			return false;
255 		}
256 		int lvl;
257 		try {
258 			lvl = Integer.parseInt(split[1]);
259 		} catch (final NumberFormatException e) {
260 			lvl = 1;
261 		}
262 		final Enchantment e = ENCHANT_LIST.get(split[0]);
263 		if (e != null) {
264 			return addEnchantment(e, lvl);
265 		}
266 		final String found = Str.matchString(ENCHANT_LIST.keySet(), split[0]);
267 		if (found == null) {
268 			return false;
269 		}
270 		return addEnchantment(ENCHANT_LIST.get(found), lvl);
271 
272 	}

403 	 * @return false if we can't find the wanted color
404 	 * @throws IllegalArgumentException
405 	 *             if the material can't be colored

407 	public boolean setColor(final String color) throws IllegalArgumentException {
408 		if (color == null) {
409 			return false;
410 		}
411 		Color foundColor = COLORS.get(color);
412 		if (foundColor != null) {
413 			setColorMeta(foundColor);
414 			return true;
415 		}
416 		final String found = Str.matchString(COLORS.keySet(), color);
417 		if (found == null) {
418 			return false;
419 		}
420 		foundColor = COLORS.get(found);
421 		setColorMeta(foundColor);
422 		return true;
423 	}

425 	private void setColorMeta(final Color color)
426 			throws IllegalArgumentException {
427 		switch (material) {
428 		case LEATHER_HELMET:
429 		case LEATHER_CHESTPLATE:
430 		case LEATHER_LEGGINGS:
431 		case LEATHER_BOOTS:
432 			((LeatherArmorMeta) getMeta()).setColor(color);
433 			break;
434 		default:
435 			throw new IllegalArgumentException("This material :" + material
436 					+ " can't be modified");
437 		}
438 	}

444 	 * org.bukkit.configuration.serialization.ConfigurationSerializable#serialize

446 	 */
447 	@Override
448 	public Map<String, Object> serialize() {
449 		final Map<String, Object> result = new LinkedHashMap<String, Object>();
450 
451 		result.put("type", material.name());
452 
453 		if (this.dmg != 0) {
454 			result.put("damage", this.dmg);
455 		}
456 
457 		if (getAmount() != 1) {
458 			result.put("amount", getAmount());
459 		}
460 
461 		final ItemMeta meta = this.meta;
462 		if (!Bukkit.getItemFactory().equals(meta, null)) {
463 			result.put("meta", meta);
464 		} else if (!enchantments.isEmpty()) {
465 			final Map<String, Integer> convert = new LinkedHashMap<String, Integer>();
466 			for (final Entry<Enchantment, Integer> entry : enchantments
467 					.entrySet()) {
468 				convert.put(entry.getKey().getName(), entry.getValue());
469 			}
470 			result.put("enchantements", convert);
471 		}
472 		return result;
473 	}

475 	/***
476 	 * Required method for configuration serialization
477 	 * 
478 	 * @param args
479 	 *            map to deserialize
480 	 * @return deserialized item stack
481 	 * @see ConfigurationSerializable
482 	 */
483 	public static MaterialContainer deserialize(final Map<String, Object> args) {
484 		final Material type = Material.getMaterial((String) args.get("type"));
485 		short damage = 0;
486 		int amount = 1;
487 
488 		if (args.containsKey("damage")) {
489 			damage = ((Number) args.get("damage")).shortValue();
490 		}
491 
492 		if (args.containsKey("amount")) {
493 			amount = (Integer) args.get("amount");
494 		}
495 
496 		final ItemStack result = new ItemStack(type, amount, damage);
497 
498 		if (args.containsKey("enchantments")) { // Backward compatiblity,
499 												// @deprecated
500 			final Object raw = args.get("enchantments");
501 
502 			if (raw instanceof Map) {
503 				final Map<?, ?> map = (Map<?, ?>) raw;
504 
505 				for (final Map.Entry<?, ?> entry : map.entrySet()) {
506 					final Enchantment enchantment = Enchantment.getByName(entry
507 							.getKey().toString());
508 
509 					if ((enchantment != null)
510 							&& (entry.getValue() instanceof Integer)) {
511 						result.addUnsafeEnchantment(enchantment,
512 								(Integer) entry.getValue());
513 					}
514 				}
515 			}
516 		} else if (args.containsKey("meta")) { // We cannot and will not have
517 												// meta when enchantments
518 												// (pre-ItemMeta) exist
519 			final Object raw = args.get("meta");
520 			if (raw instanceof ItemMeta) {
521 				result.setItemMeta((ItemMeta) raw);
522 			}
523 		}
524 
525 		return new MaterialContainer(result);
526 	}
