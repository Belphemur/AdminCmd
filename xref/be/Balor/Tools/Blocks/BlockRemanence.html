26  public class BlockRemanence {
27  	protected Location loc;
28  	private final int oldType;
29  	private byte data = 0;
30  	private boolean useData;
31  
32  	/***
33  	 *
34  	 */
35  	public BlockRemanence(final Location loc) {
36  		this.loc = loc;
37  		final Block b = loc.getWorld().getBlockAt(loc);
38  		this.oldType = b.getTypeId();
39  		if ((useData = usesData(oldType))) {
40  			data = b.getState().getRawData();
41  		}
42  
43  	}
44  
45  	public Block returnToThePast() {
46  		final Block b = loc.getWorld().getBlockAt(loc);
47  		if (useData) {
48  			b.setTypeIdAndData(oldType, data, true);
49  		} else {
50  			b.setTypeId(oldType, true);
51  		}
52  		return b;
53  	}
54  
55  	public void setBlockType(final int type) {
56  		loc.getWorld().getBlockAt(loc).setTypeId(type, true);
57  	}
58  
59  	/***
60  	 * @return the oldType
61  	 */
62  	public int getOldType() {
63  		return oldType;
64  	}
65  
66  	/***
67  	 * @return the data
68  	 */
69  	public byte getData() {
70  		return data;
71  	}
