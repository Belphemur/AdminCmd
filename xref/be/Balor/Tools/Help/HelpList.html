61  	public boolean removeEntry(final String commandName) {
62  		DebugLog.INSTANCE.info("Remove " + commandName + " help from plugin : " + pluginName);
63  		HelpEntry toRemove = null;
64  		for (final HelpEntry he : pluginHelp)
65  			if (he.getCommandName().equals(commandName)) {
66  				toRemove = he;
67  				break;
68  			}
69  		if (toRemove != null)
70  			return pluginHelp.remove(toRemove);
71  		return false;
72  
73  	}

75  	public HelpList(final Plugin plugin) throws IllegalArgumentException {
76  		final TreeSet<HelpEntry> list = new TreeSet<HelpEntry>(new EntryNameComparator());
77  		final Map<String, Map<String, Object>> cmds = plugin.getDescription().getCommands();
78  		this.pluginName = plugin.getDescription().getName();
79  		if (cmds == null)
80  			throw new IllegalArgumentException(pluginName + " don't have any commands to list");
81  		final List<String> perms = new ArrayList<String>();
82  		try {
83  			for (final Entry<String, Map<String, Object>> k : cmds.entrySet()) {
84  				final Map<String, Object> value = k.getValue();
85  				if (value.containsKey("permission") && value.get("permission") != null
86  						&& !(value.get("permission").equals("")))
87  					perms.add(value.get("permission").toString());
88  				else if (value.containsKey("permissions") && value.get("permissions") != null
89  						&& !(value.get("permissions").equals("")))
90  					perms.add(value.get("permissions").toString());
91  				final String desc = value.get("description").toString();
92  				list.add(new HelpEntry(k.getKey(), desc == null ? "" : desc, "",
93  						new ArrayList<String>(perms), k.getKey()));
94  				perms.clear();
95  			}
96  			this.pluginHelp = list;
97  		} catch (final Exception e) {
98  			ACLogger.warning("[HELP] Problem with commands of " + pluginName);
99  			DebugLog.INSTANCE.warning("[HELP] " + e.toString());
100 			final StackTraceElement[] trace = e.getStackTrace();
101 			for (final StackTraceElement element : trace)
102 				DebugLog.INSTANCE.warning("[HELP] " + element.toString());
103 			this.pluginHelp = new TreeSet<HelpEntry>(new EntryNameComparator());
104 		}
105 
106 	}

115 	/***
116 	 * Process all help to check get only the command that the player have
117 	 * access
118 	 * 
119 	 * @param sender
120 	 */
121 	private void checkPermissions(final CommandSender sender) {
122 		if (lastCommandSender != null && sender.equals(lastCommandSender))
123 			return;
124 		lastHelpEntries = new ArrayList<HelpEntry>();
125 		lastCommandSender = sender;
126 		for (final HelpEntry he : pluginHelp)
127 			if (he.hasPerm(sender))
128 				lastHelpEntries.add(he);
129 		Collections.sort(lastHelpEntries, new EntryCommandComparator());
130 	}

132 	/***
133 	 * Get a list of the string to display for the wanted page, and the given
134 	 * user
135 	 * 
136 	 * @param page
137 	 *            int the wanted page
138 	 * @param sender
139 	 *            CommandSender the sender of the command
140 	 * @param detailed
141 	 *            If true the detailed description will be displayed if one
142 	 *            exists
143 	 * @return
144 	 */
145 	public List<String> getPage(int page, final CommandSender sender, final boolean detailed) {
146 		final int entryPerPage = ConfigEnum.H_ENTRY.getInt();
147 		final List<String> helpList = new ArrayList<String>();
148 		checkPermissions(sender);
149 		final int maxPages = (int) Math.ceil(lastHelpEntries.size() / (double) entryPerPage);
150 		page = page > maxPages ? maxPages : page;
151 		final int start = (page - 1) * entryPerPage;
152 		final int end = start + entryPerPage > lastHelpEntries.size() ? lastHelpEntries.size()
153 				: start + entryPerPage;
154 		helpList.add(ChatColor.AQUA
155 				+ ACMinecraftFontWidthCalculator.strPadCenterChat(ChatColor.DARK_GREEN + " "
156 						+ pluginName + " (" + page + "/" + maxPages + ") " + ChatColor.AQUA, '='));
157 		final HelpEntry[] array = lastHelpEntries.toArray(new HelpEntry[] {});
158 		if (Utils.isPlayer(sender, false)) {
159 			for (int i = start; i < end; i++) {
160 				final HelpEntry he = array[i];
161 				helpList.add(he.chatString(detailed));
162 			}
163 		} else {
164 			for (int i = start; i < end; i++) {
165 				final HelpEntry he = array[i];
166 				helpList.add(he.consoleString(detailed));
167 			}
168 		}
169 		return helpList;
170 	}

176 	/***
177 	 * Get the command help of the wanted command by matching it in the list of
178 	 * avaible commands.
179 	 * 
180 	 * @param cmd
181 	 *            command to search
182 	 * @param sender
183 	 *            sender of the command (used for checking the permission)
184 	 * @param detailed
185 	 *            If true the detailed description will be displayed if one
186 	 *            exists
187 	 * @return the chat String to display to the user, <b>null</b> if not found
188 	 */
189 	public List<HelpEntry> getCommandMatch(final String cmd, final CommandSender sender,
190 			final boolean detailed) {
191 		final List<HelpEntry> result = new ArrayList<HelpEntry>();
192 		if (cmd == null)
193 			return null;
194 		if (lastCommandSearched != null && lastCommandSearched.getCmd().equals(cmd))
195 			return lastCommandSearched.getResult();
196 		final String lowerSearch = cmd.toLowerCase().trim();
197 		for (final HelpEntry entry : pluginHelp) {
198 			final String str = entry.getCommand().trim();
199 			if (str.toLowerCase().startsWith(lowerSearch) && entry.hasPerm(sender))
200 				result.add(entry);
201 
202 		}
203 		if (result.isEmpty()) {
204 			for (final HelpEntry entry : pluginHelp) {
205 				if (entry.hasPerm(sender)
206 						&& entry.getDescription().toLowerCase().contains(lowerSearch))
207 					result.add(entry);
208 
209 			}
210 		}
211 
212 		lastCommandSearched = new CmdMatch(cmd, result);
213 		return result;
214 	}

216 	public List<HelpEntry> getCommandMatch(final String cmd, final CommandSender sender) {
