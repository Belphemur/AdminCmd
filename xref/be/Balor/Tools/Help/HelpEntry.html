58  	public boolean hasPerm(final CommandSender p) {
59  		for (final String perm : permissions) {
60  			if (perm.equals("OP")) {
61  				if (Utils.isPlayer(p, false)) {
62  					return p.isOp();
63  				} else {
64  					return true;
65  				}
66  			}
67  			return PermissionManager.hasPerm(p, perm, false);
68  		}
69  		return true;
70  	}

81  	private String addColors(final String msg, final boolean description) {
82  		return msg
83  				.replace(
84  						"[",
85  						ChatColor.getByChar(ConfigEnum.H_C_REQ_PARAM
86  								.getString()) + "[")
87  				.replace(
88  						"]",
89  						"]"
90  								+ ChatColor
91  										.getByChar(description ? ConfigEnum.H_C_NORMAL
92  												.getString()
93  												: ConfigEnum.H_C_CMD
94  														.getString()))
95  				.replace(
96  						"<",
97  						ChatColor.getByChar(ConfigEnum.H_C_OPT_PARAM
98  								.getString()) + "<")
99  				.replace(
100 						">",
101 						">"
102 								+ ChatColor
103 										.getByChar(description ? ConfigEnum.H_C_NORMAL
104 												.getString()
105 												: ConfigEnum.H_C_CMD
106 														.getString()));
107 	}

109 	public String chatString(final boolean detailed) {
110 		String line = getFormatedCmd();
111 
112 		final int sizeRemaining = ACMinecraftFontWidthCalculator.chatwidth
113 				- ACMinecraftFontWidthCalculator.getStringWidth(line);
114 		int descriptionSize;
115 		if (detailed && !detailedDesc.equals("")) {
116 			descriptionSize = ACMinecraftFontWidthCalculator
117 					.strLen(detailedDesc);
118 			line += ACMinecraftFontWidthCalculator.strPadLeftChat(
119 					addColors(detailedDesc, true), sizeRemaining, ' ');
120 		} else {
121 			descriptionSize = ACMinecraftFontWidthCalculator
122 					.strLen(description);
123 			line += ACMinecraftFontWidthCalculator.strPadLeftChat(
124 					addColors(description, true), sizeRemaining, ' ');
125 		}
126 
127 		if (ConfigEnum.H_SHORTE.getBoolean()) {
128 			return ACMinecraftFontWidthCalculator.strChatTrim(line);
129 		} else if (sizeRemaining > descriptionSize
130 				|| !ConfigEnum.H_WRAP.getBoolean()) {
131 			return line;
132 		} else if (ConfigEnum.H_RWRAP.getBoolean()) {
133 			return ACMinecraftFontWidthCalculator.strChatWordWrapRight(line,
134 					10, ' ', ':');
135 		} else {
136 			return ACMinecraftFontWidthCalculator.strChatWordWrap(line, 10);
137 		}
138 	}

140 	public String consoleString(final boolean detailed) {
141 		final int width = System.getProperty("os.name").startsWith("Windows") ? 80 - 17
142 				: 90;
143 		String line = getFormatedCmd();
144 
145 		final int sizeRemaining = width
146 				- ACMinecraftFontWidthCalculator.strLen(line);
147 		int descriptionSize;
148 		if (detailed && !detailedDesc.equals("")) {
149 			descriptionSize = ACMinecraftFontWidthCalculator
150 					.strLen(detailedDesc);
151 			line += ACMinecraftFontWidthCalculator.unformattedPadLeft(
152 					addColors(detailedDesc, true), sizeRemaining, ' ');
153 		} else {
154 			descriptionSize = ACMinecraftFontWidthCalculator
155 					.strLen(description);
156 			line += ACMinecraftFontWidthCalculator.unformattedPadLeft(
157 					addColors(description, true), sizeRemaining, ' ');
158 		}
159 
160 		if (ConfigEnum.H_SHORTE.getBoolean()) {
161 			return ACMinecraftFontWidthCalculator.strTrim(line, width);
162 		} else if (sizeRemaining > descriptionSize
163 				|| !ConfigEnum.H_WRAP.getBoolean()) {
164 			return line;
165 		} else if (ConfigEnum.H_RWRAP.getBoolean()) {
166 			return ACMinecraftFontWidthCalculator.strWordWrapRight(line, width,
167 					10, ' ', ':');
168 		} else {
169 			return ACMinecraftFontWidthCalculator.strWordWrap(line, width, 10);
170 		}
171 
172 	}
