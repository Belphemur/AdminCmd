313 	 * @param search

316 	public static String matchString(final Collection<String> container,
317 			final String search) {
318 		String found = null;
319 		if (search == null) {
320 			return found;
321 		}
322 		final String lowerSearch = search.toLowerCase();
323 		int delta = Integer.MAX_VALUE;
324 		for (final String str : container) {
325 			if (str.toLowerCase().startsWith(lowerSearch)) {
326 				final int curDelta = str.length() - lowerSearch.length();
327 				if (curDelta < delta) {
328 					found = str;
329 					delta = curDelta;
330 				}
331 				if (curDelta == 0) {
332 					break;
333 				}
334 			}
335 		}
336 		return found;
