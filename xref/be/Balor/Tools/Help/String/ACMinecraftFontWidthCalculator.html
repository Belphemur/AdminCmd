44  	public static int getStringWidth(final String s) {
45  		int i = 0;
46  		if (s != null) {
47  			for (final char c : s.replaceAll("//u00A7.", "").toCharArray()) {
48  				i += getCharWidth(c);
49  			}
50  		}
51  		return i;
52  	}

54  	public static int getCharWidth(final char c) {
55  		// return getCharWidth(c, 0);
56  		final int k = charWidthIndexIndex.indexOf(c);
57  		if (c != '\247' && k >= 0) {
58  			return charWidths[k];
59  		}
60  		return 0;
61  	}

221 	public static String strTrim(final String str, final int length) {
222 		if (uncoloredStr(str).length() > length) {
223 			int width = length;
224 			String ret = "";
225 			boolean lastCol = false;
226 			for (final char c : str.toCharArray()) {
227 				if (c == '\u00A7') {
228 					ret += c;
229 					lastCol = true;
230 				} else {
231 					if (!lastCol) {
232 						if (width - 1 >= 0) {
233 							width -= 1;
234 							ret += c;
235 						} else {
236 							return ret;
237 						}
238 					} else {
239 						ret += c;
240 						lastCol = false;
241 					}
242 				}
243 			}
244 		}
245 		return str;
246 	}

252 	public static String strChatTrim(final String str, final int absLen) {
253 		int width = getStringWidth(str);
254 		if (width > absLen) {
255 			width = absLen;
256 			String ret = "";
257 			boolean lastCol = false;
258 			for (final char c : str.toCharArray()) {
259 				if (c == '\u00A7') {
260 					ret += c;
261 					lastCol = true;
262 				} else {
263 					if (!lastCol) {
264 						final int w = getCharWidth(c);
265 						if (width - w >= 0) {
266 							width -= w;
267 							ret += c;
268 						} else {
269 							return ret;
270 						}
271 					} else {
272 						ret += c;
273 						lastCol = false;
274 					}
275 				}
276 			}
277 		}
278 		return str;
279 	}

289 	public static String strChatWordWrap(String str, final int tab, final char tabChar) {
290 		String ret = "";
291 		while (str.length() > 0) {
292 			// find last char of first line
293 			if (getStringWidth(str) <= chatwidth) {
294 				return (ret.length() > 0 ? ret + "\n" + unformattedStrRepeat(tabChar, tab) : "")
295 						.concat(str);
296 			}
297 			final String line1 = strChatTrim(str);
298 			int lastPos = line1.length() - (ret.length() > 0 ? tab + 1 : 1);
299 			while (lastPos > 0 && line1.charAt(lastPos) != ' ') {
300 				--lastPos;
301 			}
302 			if (lastPos == 0) {
303 				lastPos = line1.length() - (ret.length() > 0 ? tab + 1 : 1);
304 			}
305 			// ret += strPadRightChat((ret.length() > 0 ?
306 			// unformattedStrRepeat(tabChar, tab) : "") + str.substring(0,
307 			// lastPos));
308 			ret += (ret.length() > 0 ? "\n" + unformattedStrRepeat(tabChar, tab) : "")
309 					+ str.substring(0, lastPos);
310 			str = str.substring(lastPos + 1);
311 		}
312 		return ret;
313 	}

543 	private static boolean containsAlignTag(final String str, final String tag) {
544 		final int pos = str.indexOf("<" + tag);
545 		if (pos >= 0) {
546 			return str.length() > pos + ("<" + tag).length()
547 					&& (str.charAt(pos + ("<" + tag).length()) == '>' || str.charAt(pos
548 							+ ("<" + tag).length() + 1) == '>');
549 		}
550 		return false;
551 	}
