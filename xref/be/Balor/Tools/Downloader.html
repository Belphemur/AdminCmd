45  		/***
46  		 * @param version

55  		/*
56  		 * (non-Javadoc)
57  		 * 
58  		 * @see java.lang.Object#hashCode()
59  		 */
60  		@Override
61  		public int hashCode() {
62  			final int prime = 31;
63  			int result = 1;
64  			result = prime * result + (int) (size ^ (size >>> 32));
65  			result = prime * result
66  					+ ((version == null) ? 0 : version.hashCode());
67  			return result;
68  		}

70  		/*
71  		 * (non-Javadoc)
72  		 * 
73  		 * @see java.lang.Object#equals(java.lang.Object)
74  		 */
75  		@Override
76  		public boolean equals(final Object obj) {
77  			if (this == obj) {
78  				return true;
79  			}
80  			if (obj == null) {
81  				return false;
82  			}
83  			if (!(obj instanceof Version)) {
84  				return false;
85  			}
86  			final Version other = (Version) obj;
87  			if (size != other.size) {
88  				return false;
89  			}
90  			if (version == null) {
91  				if (other.version != null) {
92  					return false;
93  				}
94  			} else if (!version.equals(other.version)) {
95  				return false;
96  			}
97  			return true;
98  		}

123 	/***
124 	 * Download the remote file
125 	 * 
126 	 * @param urlString
127 	 *            url to the remote file
128 	 * @param downloaded
129 	 *            where to download the file
130 	 * @throws IOException
131 	 *             if something goes wrong in the process of downloading the
132 	 *             file
133 	 */
134 	public static void download(final String urlString, final File downloaded)
135 			throws IOException {
136 		BufferedOutputStream bout = null;
137 		BufferedInputStream in = null;
138 		HttpURLConnection connection = null;
139 		if (!exists(urlString)) {
140 			throw new FileNotFoundException("The remote file " + urlString
141 					+ " can't be found.");
142 		}
143 
144 		if (downloaded.getParentFile() != null
145 				&& !downloaded.getParentFile().exists()) {
146 			downloaded.getParentFile().mkdirs();
147 		}
148 		if (!checkVersionToDownload(urlString, downloaded)) {
149 			return;
150 		}
151 		if (downloaded.exists()) {
152 			downloaded.delete();
153 		}
154 
155 		DebugLog.INSTANCE.info("Downloading file : " + urlString);
156 		try {
157 
158 			final URL url = new URL(urlString);
159 			connection = (HttpURLConnection) url.openConnection();
160 			final int filesize = connection.getContentLength();
161 			float totalDataRead = 0;
162 			in = new BufferedInputStream(connection.getInputStream());
163 			final FileOutputStream fos = new FileOutputStream(downloaded);
164 			final byte[] data = new byte[1024];
165 			bout = new BufferedOutputStream(fos, data.length);
166 			int i = 0;
167 			int oldPercent = 0;
168 			while ((i = in.read(data, 0, data.length)) >= 0) {
169 				totalDataRead = totalDataRead + i;
170 				bout.write(data, 0, i);
171 				final int percent = (int) ((totalDataRead * 100) / filesize);
172 				if (percent - oldPercent >= 10) {
173 					DebugLog.INSTANCE.info("Downloaded at " + percent + "%");
174 					oldPercent = percent;
175 				}
176 			}
177 		} finally {
178 			if (bout != null) {
179 				bout.close();
180 			}
181 			if (in != null) {
182 				in.close();
183 			}
184 			if (connection != null) {
185 				connection.disconnect();
186 			}
187 		}
188 		DebugLog.INSTANCE.info("File " + urlString + " downloaded");
189 	}

191 	/***
192 	 * Check for the presence of a .version file to indicate the version of the
193 	 * file to download <br />
194 	 * Example of version file : <br />
195 	 * <blockquote>1.1.1 <br/>
196 	 * 9009104 </blockquote>
197 	 * 
198 	 * @param fileUrl
199 	 *            url to the remote file
200 	 * @param download
201 	 *            where the file will be downloaded
202 	 * @return true if the file can be downloaded else false.
203 	 * @throws IOException
204 	 * 
205 	 */
206 	private static final boolean checkVersionToDownload(final String fileUrl,
207 			final File download) throws IOException {
208 		final String urlString = fileUrl + ".version";
209 		final File versionFile = new File(download.getParent(),
210 				download.getName() + ".version");
211 		if (!exists(urlString)) {
212 			return true;
213 		}
214 		HttpURLConnection connection = null;
215 		try {
216 			connection = (HttpURLConnection) new URL(urlString)
217 					.openConnection();
218 			final Version version = readVersion(connection.getInputStream());
219 			if (versionFile.exists()) {
220 				final Version curVersion = readVersion(new FileInputStream(
221 						versionFile));
222 				if (curVersion.equals(version) && download.exists()
223 						&& version.checkSize(download)) {
224 					return false;
225 				}
226 			}
227 			version.writeOnFile(versionFile);
228 			return true;
229 
230 		} finally {
231 
232 			if (connection != null) {
233 				connection.disconnect();
234 			}
235 		}
236 
237 	}

239 	private static boolean exists(final String URLName) throws IOException {
240 		HttpURLConnection con = null;
241 		try {
242 			HttpURLConnection.setFollowRedirects(false);
243 			// note : you may also need
244 			// HttpURLConnection.setInstanceFollowRedirects(false)
245 			con = (HttpURLConnection) new URL(URLName).openConnection();
246 			con.setRequestMethod("HEAD");
247 			return (con.getResponseCode() == HttpURLConnection.HTTP_OK);
248 		} finally {
249 			if (con != null) {
250 				con.disconnect();
251 			}
252 		}
253 	}

255 	private static Version readVersion(final InputStream stream) {
256 		final BufferedReader reader = new BufferedReader(new InputStreamReader(
257 				stream));
258 		try {
259 			final String version = reader.readLine();
260 			Long size;
261 			try {
262 				size = Long.parseLong(reader.readLine());
263 			} catch (final Exception e) {
264 				size = 0L;
265 			}
266 			return new Version(version, size);
267 		} catch (final IOException e) {
268 			return null;
269 		} finally {
270 			try {
271 				reader.close();
272 			} catch (final IOException e) {}
273 		}
