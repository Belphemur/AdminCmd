59  	/***
60  	 * Creates a new {@link ExtendedConfiguration}, loading from the given file.
61  	 * <p>
62  	 * Any errors loading the Configuration will be logged and then ignored. If
63  	 * the specified input is not a valid config, a blank config will be
64  	 * returned.
65  	 * 
66  	 * @param file
67  	 *            Input file
68  	 * @return Resulting configuration
69  	 * @throws IllegalArgumentException
70  	 *             Thrown is file is null
71  	 */
72  	public static ExtendedConfiguration loadConfiguration(final File file) {
73  		if (file == null) {
74  			throw new IllegalArgumentException("File cannot be null");
75  		}
76  
77  		final ExtendedConfiguration config = new ExtendedConfiguration();
78  
79  		try {
80  			config.load(file);
81  		} catch (final FileNotFoundException ex) {
82  		} catch (final IOException ex) {
83  			Bukkit.getLogger().log(Level.SEVERE, "Cannot load " + file, ex);
84  		} catch (final InvalidConfigurationException ex) {
85  			if (ex.getCause() instanceof YAMLException) {
86  				ACLogger.severe("Config file " + file + " isn't valid! " + ex.getCause());
87  			} else if ((ex.getCause() == null) || (ex.getCause() instanceof ClassCastException)) {
88  				ACLogger.severe("Config file " + file + " isn't valid!");
89  			} else {
90  				ACLogger.severe("Cannot load " + file + ": " + ex.getCause().getClass(), ex);
91  			}
92  		}
93  
94  		return config;
95  	}

101 	@Override
102 	protected String buildHeader() {
103 		final String header = options().header();
104 
105 		if (options().copyHeader()) {
106 			final Configuration def = getDefaults();
107 
108 			if ((def != null) && (def instanceof ExFileConfiguration)) {
109 				final ExFileConfiguration filedefaults = (ExFileConfiguration) def;
110 				final String defaultsHeader = filedefaults.buildHeader();
111 
112 				if ((defaultsHeader != null) && (defaultsHeader.length() > 0)) {
113 					return defaultsHeader;
114 				}
115 			}
116 		}
117 
118 		if (header == null) {
119 			return "";
120 		}
121 
122 		final StringBuilder builder = new StringBuilder();
123 		final String[] lines = header.split("\r?\n", -1);
124 		boolean startedHeader = false;
125 
126 		for (int i = lines.length - 1; i >= 0; i--) {
127 			builder.insert(0, "\n");
128 
129 			if ((startedHeader) || (lines[i].length() != 0)) {
130 				builder.insert(0, lines[i]);
131 				builder.insert(0, COMMENT_PREFIX);
132 				startedHeader = true;
133 			}
134 		}
135 
136 		return builder.toString();
137 	}

139 	protected void convertMapsToSections(final Map<Object, Object> input,
140 			final ConfigurationSection section) {
141 		for (final Map.Entry<Object, Object> entry : input.entrySet()) {
142 			final String key = entry.getKey().toString();
143 			final Object value = entry.getValue();
144 
145 			if (value instanceof Map<?, ?>) {
146 				convertMapsToSections((Map<Object, Object>) value, section.createSection(key));
147 			} else {
148 				section.set(key, value);
149 			}
150 		}
151 	}

160 	@Override
161 	public void loadFromString(String contents) throws InvalidConfigurationException {
162 		lock.lock();
163 		try {
164 
165 			if (contents == null) {
166 				throw new IllegalArgumentException("Contents cannot be null");
167 			}
168 
169 			Map<Object, Object> input = null;
170 			contents = contents.replaceAll("\uFFFD", "?").replaceAll("\t", "    ");
171 			try {
172 				input = (Map<Object, Object>) yaml.load(contents);
173 			} catch (final ScannerException e) {
174 				if (e.getContextMark() == null) {
175 					ACLogger.severe("File : " + file
176 							+ "\n You have to correct the error manualy in the file.", e);
177 					corrupted = true;
178 					return;
179 				}
180 				removeLineFromFile(e.getContextMark().getLine());
181 				ACLogger.info("File : " + file + "\n" + e.toString() + "\nLINE "
182 						+ (e.getContextMark().getLine() + 1) + " DELETED");
183 				try {
184 					load(file);
185 				} catch (final FileNotFoundException e1) {
186 				} catch (final IOException e1) {
187 				}
188 			} catch (final ParserException e) {
189 				ACLogger.severe("File : " + file
190 						+ "\n You have to correct the error manualy in the file.", e);
191 				corrupted = true;
192 				return;
193 
194 			} catch (final Throwable ex) {
195 				corrupted = true;
196 				throw new InvalidConfigurationException(
197 						"Specified contents is not a valid Configuration", ex);
198 			}
199 
200 			final int size = (input == null) ? 0 : input.size();
201 			final Map<String, Object> result = new LinkedHashMap<String, Object>(size);
202 
203 			if (size > 0) {
204 				for (final Map.Entry<Object, Object> entry : input.entrySet()) {
205 					result.put(entry.getKey().toString(), entry.getValue());
206 				}
207 			}
208 
209 			final String header = parseHeader(contents);
210 			if (header.length() > 0) {
211 				options().header(header);
212 			}
213 
214 			if (input != null) {
215 				convertMapsToSections(input, this);
216 			}
217 			corrupted = false;
218 		} finally {
219 			lock.unlock();
220 		}
221

233 	protected String parseHeader(final String input) {
234 		final String[] lines = input.split("\r?\n", -1);
235 		final StringBuilder result = new StringBuilder();
236 		boolean readingHeader = true;
237 
238 		for (int i = 0; (i < lines.length) && (readingHeader); i++) {
239 			final String line = lines[i];
240 
241 			if (line.startsWith(COMMENT_PREFIX)) {
242 				if (i > 0) {
243 					result.append("\n");
244 				}
245 
246 				if (line.length() > COMMENT_PREFIX.length()) {
247 					result.append(line.substring(COMMENT_PREFIX.length()));
248 				}
249 			} else if (line.length() == 0) {
250 				result.append("\n");
251 			} else {
252 				readingHeader = false;
253 			}
254 		}
255 
256 		return result.toString();
257 	}

270 	private void removeLineFromFile(final int lineToRemove) {
271 		BufferedReader br = null;
272 		PrintWriter pw = null;
273 		try {
274 
275 			final File inFile = file;
276 			// Construct the new file that will later be renamed to the original
277 			// filename.
278 			final File tempFile = File.createTempFile(file.getName(), null);
279 			br = new BufferedReader(new UnicodeReader(new FileInputStream(file)));
280 			pw = new PrintWriter(new OutputStreamWriter(new FileOutputStream(tempFile), "UTF-8"));
281 
282 			String line = null;
283 
284 			// Read from the original file and write to the new
285 			// unless content matches data to be removed.
286 			int i = 0;
287 			while ((line = br.readLine()) != null) {
288 				try {
289 					if (i == lineToRemove)
290 						continue;
291 					pw.println(line);
292 				} finally {
293 					i++;
294 				}
295 			}
296 			pw.flush();
297 			pw.close();
298 			br.close();
299 			// Delete the original file
300 			if (!inFile.delete()) {
301 				System.out.println("Could not delete file");
302 				return;
303 			}
304 
305 			// Rename the new file to the filename the original file had.
306 			if (!tempFile.renameTo(inFile))
307 				System.out.println("Could not rename file");
308 
309 		} catch (final FileNotFoundException ex) {
310 			ex.printStackTrace();
311 		} catch (final IOException ex) {
312 			ex.printStackTrace();
313 		} finally {
314 			if (pw != null)
315 				pw.close();
316 			try {
317 				if (br != null)
318 					br.close();
319 			} catch (final IOException e) {
320 			}
321 		}
322 	}

340 	 * 
341 	 * @see org.bukkit.configuration.file.FileConfiguration#saveToString()
342 	 */
343 	@Override
344 	public String saveToString() {
345 		lock.lock();
346 		String header = "";
347 		String dump = "";
348 		try {
349 			yamlOptions.setIndent(options().indent());
350 			yamlOptions.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
351 			yamlRepresenter.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
352 
353 			header = buildHeader();
354 			dump = yaml.dump(getValues(false));
355 
356 			if (dump.equals(BLANK_CONFIG)) {
357 				dump = "";
358 			}
359 		} finally {
360 			lock.unlock();
361 		}
362 		return header + dump;
363 	}
