41  public class ExtendedRepresenter extends Representer {
42  
43  	protected class EmptyRepresentNull implements Represent {
44  		@Override
45  		public Node representData(final Object data) {
46  			return representScalar(Tag.NULL, ""); // Changed "null" to "" so as
47  													// to avoid writing nulls
48  		}
49  	}
50  
51  	private class RepresentConfigurationSection extends RepresentMap {
52  		@Override
53  		public Node representData(final Object data) {
54  			return super.representData(((ConfigurationSection) data)
55  					.getValues(false));
56  		}
57  	}
58  
59  	private class RepresentConfigurationSerializable extends RepresentMap {
60  		@Override
61  		public Node representData(final Object data) {
62  			ConfigurationSerializable serializable = (ConfigurationSerializable) data;
63  			Map<String, Object> values = new LinkedHashMap<String, Object>();
64  			values.put(
65  					ConfigurationSerialization.SERIALIZED_TYPE_KEY,
66  					ConfigurationSerialization.getAlias(serializable.getClass()));
67  			values.putAll(serializable.serialize());
68  
69  			return super.representData(values);
70  		}
71  	}
72  
73  	private class RepresentTpRequest implements Represent {
74  		@Override
75  		public Node representData(final Object data) {
76  			final TpRequest tpRequest = (TpRequest) data;
77  			final String value = tpRequest.getFrom() + ";" + tpRequest.getTo();
78  			return representScalar(new Tag("!tpRequest"), value);
79  		}
80  	}
81  
82  	public ExtendedRepresenter() {
83  		super();
84  		this.nullRepresenter = new EmptyRepresentNull();
85  		this.representers.put(TpRequest.class, new RepresentTpRequest());
86  		this.multiRepresenters.put(ConfigurationSection.class,
87  				new RepresentConfigurationSection());
88  		this.multiRepresenters.put(ConfigurationSerializable.class,
89  				new RepresentConfigurationSerializable());
90  	}

92  	// Code borrowed from snakeyaml
93  	// (http://code.google.com/p/snakeyaml/source/browse/src/test/java/org/yaml/snakeyaml/issues/issue60/SkipBeanTest.java)
94  	@Override
95  	protected NodeTuple representJavaBeanProperty(final Object javaBean,
96  			final Property property, final Object propertyValue,
97  			final Tag customTag) {
98  		final NodeTuple tuple = super.representJavaBeanProperty(javaBean,
99  				property, propertyValue, customTag);
100 		final Node valueNode = tuple.getValueNode();
101 		if (valueNode instanceof CollectionNode) {
102 			// Removed null check
103 			if (Tag.SEQ.equals(valueNode.getTag())) {
104 				final SequenceNode seq = (SequenceNode) valueNode;
105 				if (seq.getValue().isEmpty()) {
106 					return null; // skip empty lists
107 				}
108 			}
109 			if (Tag.MAP.equals(valueNode.getTag())) {
110 				final MappingNode seq = (MappingNode) valueNode;
111 				if (seq.getValue().isEmpty()) {
112 					return null; // skip empty maps
113 				}
114 			}
115 		}
116 		return tuple;
117 	}
