24   * @author Balor (aka Antoine Aflalo)

26   */
27  public interface ExConfigurationSection extends ConfigurationSection {
28  	/***
29  	 * Create an entry in the {@link ExConfigurationSection} if it not existing
30  	 * else return the existing one.
31  	 * 
32  	 * @param path
33  	 *            Path to create/get the entry at.
34  	 * @return false if the path already exists, true if it's not set.
35  	 */
36  	public boolean add(String path, Object value);
37  
38  	/***
39  	 * Create a {@link ExConfigurationSection} if it not existing else return
40  	 * the existing one.
41  	 * 
42  	 * @param path
43  	 *            Path to create/get the section at.
44  	 * @return the ConfigurationSection
45  	 */
46  	public ExConfigurationSection addSection(String path);
47  
48  	@Override
49  	public ExConfigurationSection createSection(String path);
50  
51  	/***
52  	 * Gets a list of booleans. Non-valid entries will not be in the list. There
53  	 * will be no null slots. If the list is not defined, the default will be
54  	 * returned. 'null' can be passed for the default and an empty list will be
55  	 * returned instead. The node must be an actual list and cannot be just a
56  	 * boolean,
57  	 * 
58  	 * @param path
59  	 *            path to node (dot notation)
60  	 * @param def
61  	 *            default value or null for an empty list as default
62  	 * @return list of integers
63  	 */
64  	public List<Boolean> getBooleanList(String path, List<Boolean> def);
65  
66  	@Override
67  	public ExConfigurationSection getConfigurationSection(String path);
68  
69  	/***
70  	 * Gets a list of doubles. Non-valid entries will not be in the list. There
71  	 * will be no null slots. If the list is not defined, the default will be
72  	 * returned. 'null' can be passed for the default and an empty list will be
73  	 * returned instead. The node must be an actual list and cannot be just a
74  	 * double.
75  	 * 
76  	 * @param path
77  	 *            path to node (dot notation)
78  	 * @param def
79  	 *            default value or null for an empty list as default
80  	 * @return list of integers
81  	 */
82  	public List<Double> getDoubleList(String path, List<Double> def);
83  
84  	/***
85  	 * Gets a list of integers. Non-valid entries will not be in the list. There
86  	 * will be no null slots. If the list is not defined, the default will be
87  	 * returned. 'null' can be passed for the default and an empty list will be
88  	 * returned instead. The node must be an actual list and not just an
89  	 * integer.
90  	 * 
91  	 * @param path
92  	 *            path to node (dot notation)
93  	 * @param def
94  	 *            default value or null for an empty list as default
95  	 * @return list of integers
96  	 */
97  	public List<Integer> getIntList(String path, List<Integer> def);
98  
99  	/***
100 	 * Gets a list of strings. Non-valid entries will not be in the list. There
101 	 * will be no null slots. If the list is not defined, the default will be
102 	 * returned. 'null' can be passed for the default and an empty list will be
103 	 * returned instead. If an item in the list is not a string, it will be
104 	 * converted to a string. The node must be an actual list and not just a
105 	 * string.
106 	 * 
107 	 * @param path
108 	 *            path to node (dot notation)
109 	 * @param def
110 	 *            default value or null for an empty list as default
111 	 * @return list of strings
112 	 */
113 	public List<String> getStringList(String path, List<String> def);
114 
115 	/***
116 	 * Shortcut to remove an item by setting it null
117 	 * 
118 	 * @param path
119 	 *            Path to remove the entry at.
120 	 */
121 	public void remove(String path);
122 
123 }
