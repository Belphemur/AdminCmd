56  	/***
57  	 * Retrieve the name of the Minecraft server package.
58  	 * 
59  	 * @return Full canonical name of the Minecraft server package.
60  	 */
61  	public static String getMinecraftPackage() {
62  		// Speed things up
63  		if (MINECRAFT_FULL_PACKAGE != null) {
64  			return MINECRAFT_FULL_PACKAGE;
65  		}
66  
67  		final Server craftServer = Bukkit.getServer();
68  
69  		// This server should have a "getHandle" method that we can use
70  		if (craftServer != null) {
71  			try {
72  				final Class<?> craftClass = craftServer.getClass();
73  				final Method getHandle = craftClass.getMethod("getHandle");
74  
75  				final Class<?> returnType = getHandle.getReturnType();
76  				final String returnName = returnType.getCanonicalName();
77  
78  				// The return type will tell us the full package, regardless of
79  				// formating
80  				CRAFTBUKKIT_PACKAGE = getPackage(craftClass.getCanonicalName());
81  				MINECRAFT_FULL_PACKAGE = getPackage(returnName);
82  				return MINECRAFT_FULL_PACKAGE;
83  
84  			} catch (final SecurityException e) {
85  				throw new RuntimeException(
86  						"Security violation. Cannot get handle method.", e);
87  			} catch (final NoSuchMethodException e) {
88  				throw new IllegalStateException(
89  						"Cannot find getHandle() method on server. Is this a modified CraftBukkit version?",
90  						e);
91  			}
92  
93  		} else {
94  			throw new IllegalStateException(
95  					"Could not find Bukkit. Is it running?");
96  		}
97  	}

136 	 * Determine if a given object can be found within the package
137 	 * net.minecraft.server.
138 	 * 
139 	 * @param obj
140 	 *            - the object to test.
141 	 * @return TRUE if it can, FALSE otherwise.
142 	 */
143 	public static boolean isMinecraftObject(@Nonnull final Object obj) {
144 		if (obj == null) {
145 			throw new IllegalArgumentException(
146 					"Cannot determine the type of a null object.");
147 		}
148 
149 		// Doesn't matter if we don't check for the version here
150 		return obj.getClass().getName().startsWith(MINECRAFT_PREFIX_PACKAGE);
151 	}

153 	/***
154 	 * Determine if a given object is found in net.minecraft.server, and has the
155 	 * given name.
156 	 * 
157 	 * @param obj
158 	 *            - the object to test.
159 	 * @param className
160 	 *            - the class name to test.
161 	 * @return TRUE if it can, FALSE otherwise.
162 	 */
163 	public static boolean isMinecraftObject(@Nonnull final Object obj,
164 			final String className) {
165 		if (obj == null) {
166 			throw new IllegalArgumentException(
167 					"Cannot determine the type of a null object.");
168 		}
169 
170 		final String javaName = obj.getClass().getName();
171 		return javaName.startsWith(MINECRAFT_PREFIX_PACKAGE)
172 				&& javaName.endsWith(className);
173 	}

602 	 * Retrieve a CraftItemStack from a given ItemStack.
603 	 * 
604 	 * @param bukkitItemStack
605 	 *            - the Bukkit ItemStack to convert.
606 	 * @return A CraftItemStack as an ItemStack.
607 	 */
608 	@SuppressWarnings("unchecked")
609 	public static ItemStack getBukkitItemStack(final ItemStack bukkitItemStack) {
610 		// Delegate this task to the method that can execute it
611 		if (craftBukkitMethod != null) {
612 			return getBukkitItemByMethod(bukkitItemStack);
613 		}
614 
615 		if (craftBukkitConstructor == null) {
616 			try {
617 				craftBukkitConstructor = getCraftItemStackClass()
618 						.getConstructor(ItemStack.class);
619 			} catch (final Exception e) {
620 				// See if this method works
621 				if (!craftItemStackFailed) {
622 					return getBukkitItemByMethod(bukkitItemStack);
623 				}
624 
625 				throw new RuntimeException(
626 						"Cannot find CraftItemStack(org.bukkit.inventory.ItemStack).",
627 						e);
628 			}
629 		}
630 
631 		// Try to create the CraftItemStack
632 		try {
633 			return (ItemStack) craftBukkitConstructor
634 					.newInstance(bukkitItemStack);
635 		} catch (final Exception e) {
636 			throw new RuntimeException("Cannot construct CraftItemStack.", e);
637 		}
638 	}

640 	@SuppressWarnings("unchecked")
641 	private static ItemStack getBukkitItemByMethod(
642 			final ItemStack bukkitItemStack) {
643 		if (craftBukkitMethod == null) {
644 			try {
645 				craftBukkitMethod = getCraftItemStackClass().getMethod(
646 						"asCraftCopy", ItemStack.class);
647 			} catch (final Exception e) {
648 				craftItemStackFailed = true;
649 				throw new RuntimeException(
650 						"Cannot find CraftItemStack.asCraftCopy(org.bukkit.inventory.ItemStack).",
651 						e);
652 			}
653 		}
654 
655 		// Next, construct it
656 		try {
657 			return (ItemStack) craftBukkitMethod.invoke(null, bukkitItemStack);
658 		} catch (final Exception e) {
659 			throw new RuntimeException("Cannot construct CraftItemStack.", e);
660 		}
661 	}

727 	 * Retrieve the net.minecraft.server ItemStack from a Bukkit ItemStack.
728 	 * 
729 	 * @param stack
730 	 *            - the Bukkit ItemStack to convert.
731 	 * @return The NMS ItemStack.
732 	 */
733 	public static Object getMinecraftItemStack(ItemStack stack) {
734 		// Make sure this is a CraftItemStack
735 		if (!isCraftItemStack(stack)) {
736 			stack = getBukkitItemStack(stack);
737 		}
