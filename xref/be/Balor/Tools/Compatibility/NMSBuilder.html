19  import java.lang.reflect.Constructor;
20  
21  import org.bukkit.entity.Player;
22  
23  import be.Balor.Tools.Compatibility.Reflect.FieldUtils;
24  import be.Balor.Tools.Compatibility.Reflect.MethodHandler;

26  public class NMSBuilder {
27  
28  	/***
29  	 * Create a Packet201PlayerInfo
30  	 * 
31  	 * @param player
32  	 *            - Bukkit Player
33  	 * @param online
34  	 *            - online or offline
35  	 * @param ping
36  	 *            - ping
37  	 * @return instance of Packet201PlayerInfo
38  	 */
39  	public static Object buildPacket201PlayerInfo(final Player player,
40  			final boolean online, final int ping) {
41  		final Object playerHandle = MinecraftReflection.getHandle(player);
42  		try {
43  			final String listName = FieldUtils.getField(playerHandle,
44  					"listName");
45  			final Class<?> packetClass = MinecraftReflection
46  					.getPacket201PlayerInfoClass();
47  			final Constructor<?> packetConstructor = packetClass
48  					.getConstructor(String.class, boolean.class, int.class);
49  			return packetConstructor.newInstance(listName, online, ping);
50  		} catch (final Exception e) {
51  			throw new RuntimeException("Can't create the wanted packet", e);
52  		}
53  	}
54  
55  	/***
56  	 * Build a NMS PlayerInventory for the given Bukkit Player.
57  	 * 
58  	 * @param player
59  	 *            - bukkit player
60  	 * @return instance of PlayerInventory
61  	 */
62  	public static Object buildPlayerInventory(final Player player) {
63  		final Object playerHandle = MinecraftReflection.getHandle(player);
64  		try {
65  			final Class<?> playerInventoryClass = MinecraftReflection
66  					.getPlayerInventoryClass();
67  			final Constructor<?> invConstructor = playerInventoryClass
68  					.getConstructor(MinecraftReflection.getEntityHumanClass());
69  			return invConstructor.newInstance(playerHandle);
70  		} catch (final Exception e) {
71  			throw new RuntimeException("Can't build PlayerInventory", e);
72  		}
73  	}
74  
75  	/***
76  	 * Build a FAKE entity player
77  	 * 
78  	 * @param name
79  	 *            - name of the player
80  	 * @return instance NMS of EntityPlayer
81  	 */
82  	public static Object buildEntityPlayer(final String name) {
83  		final Object minecraftServer = MinecraftReflection.getMinecraftServer();
84  		try {
85  			final MethodHandler getWorldServer = new MethodHandler(
86  					minecraftServer.getClass(), "getWorldServer", int.class);
87  			final Object worldServer = getWorldServer
88  					.invoke(minecraftServer, 0);
89  			final Object itemInWorldManager = buildItemInWorldManager(worldServer);
90  			final Class<?> entityPlayer = MinecraftReflection
91  					.getEntityPlayerClass();
92  			final Constructor<?> ePConstructor = entityPlayer.getConstructor(
93  					MinecraftReflection.getMinecraftServerClass(),
94  					MinecraftReflection.getNMSWorldClass(), String.class,
95  					MinecraftReflection.getItemInWorldManagerClass());
96  			return ePConstructor.newInstance(minecraftServer, worldServer,
97  					name, itemInWorldManager);
98  		} catch (final Exception e) {
99  			throw new RuntimeException("Can't build PlayerInventory", e);
100 		}
101 	}
102 
103 	/***
104 	 * Build a ItemInWorldManager
105 	 * 
106 	 * @param worldServer
107 	 * @return return instance of ItemInWorldManager
108 	 */
109 	public static Object buildItemInWorldManager(final Object worldServer) {
110 		if (!MinecraftReflection.getWorldServerClass().isAssignableFrom(
111 				worldServer.getClass())) {
112 			throw new RuntimeException("The constructor need a "
113 					+ MinecraftReflection.getWorldServerClass().getSimpleName());
114 		}
115 		final Class<?> clazz = MinecraftReflection.getItemInWorldManagerClass();
116 		try {
117 			final Constructor<?> constructor = clazz
118 					.getConstructor(MinecraftReflection.getWorldServerClass());
119 			return constructor.newInstance(worldServer);
120 		} catch (final Exception e) {
121 			throw new RuntimeException("Can't build ItemInWorldManager", e);
122 		}
123 
124 	}
