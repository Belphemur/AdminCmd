12  class CachedPackage {

14  	private final String packageName;

17  		this.packageName = packageName;
18  		this.cache = Maps.newConcurrentMap();
19  	}
20  
21  	/***
22  	 * Associate a given class with a class name.
23  	 * 
24  	 * @param className
25  	 *            - class name.
26  	 * @param clazz
27  	 *            - type of class.
28  	 */
29  	@SuppressWarnings("rawtypes")
30  	public void setPackageClass(final String className, final Class clazz) {
31  		cache.put(className, clazz);
32  	}
33  
34  	/***
35  	 * Retrieve the class object of a specific class in the current package.
36  	 * 
37  	 * @param className
38  	 *            - the specific class.
39  	 * @return Class object.
40  	 * @throws RuntimeException
41  	 *             If we are unable to find the given class.
42  	 */
43  	@SuppressWarnings("rawtypes")
44  	public Class getPackageClass(final String className) {
45  		try {
46  			Class result = cache.get(className);
47  
48  			// Concurrency is not a problem - we don't care if we look up a
49  			// class twice
50  			if (result == null) {
51  				// Look up the class dynamically
52  				result = CachedPackage.class.getClassLoader().loadClass(
53  						packageName + "." + className);
54  				cache.put(className, result);
55  			}
56  
57  			return result;
58  
59  		} catch (ClassNotFoundException e) {
60  			throw new RuntimeException("Cannot find class " + className, e);
61  		}
62  	}
