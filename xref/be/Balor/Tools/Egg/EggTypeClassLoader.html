42  public class EggTypeClassLoader extends ClassLoader {
43  	private static final Map<String, Class<? extends EggType<?>>> classes = new HashMap<String, Class<? extends EggType<?>>>();
44  	private static final Map<String, Class<? extends EggType<?>>> classesSimpleName = new HashMap<String, Class<? extends EggType<?>>>();
45  	private static final Pattern regex = Pattern.compile("//w.+Egg");

47  	/***
48  	 * Add a Package that containing the EggType class. The Class must have a
49  	 * name that finish by Egg.
50  	 * 
51  	 * @param packageName
52  	 */
53  	@SuppressWarnings("unchecked")
54  	public synchronized static void addPackage(final Plugin plugin, final String packageName) {
55  
56  		for (final Class<?> clazz : getClassesInPackage(packageName, plugin.getClass()
57  				.getClassLoader())) {
58  			Permission perm = null;
59  			if (EggType.class.isAssignableFrom(clazz)) {
60  				if (clazz.isAnnotationPresent(EggPermission.class)) {
61  					final EggPermission annotation = clazz.getAnnotation(EggPermission.class);
62  					if (!annotation.permission().isEmpty())
63  						perm = PermissionLinker.addOnTheFly(annotation.permission(),
64  								annotation.permissionParent());
65  				} else {
66  					final String simpleName = clazz.getSimpleName();
67  					perm = PermissionLinker.addOnTheFly(
68  							"admincmd.egg."
69  									+ simpleName.substring(0, simpleName.length() - 3)
70  											.toLowerCase(), "admincmd.egg.*");
71  				}
72  				classes.put(clazz.getName(), (Class<? extends EggType<?>>) clazz);
73  				classesSimpleName.put(clazz.getSimpleName(), (Class<? extends EggType<?>>) clazz);
74  				EggPermissionManager.INSTANCE.addPermission((Class<? extends EggType<?>>) clazz,
75  						perm);
76  			}
77  		}
78  	}

80  	private Class<? extends EggType<?>> matchClassName(final String search)
81  			throws ClassNotFoundException {
82  		Class<? extends EggType<?>> found = null;
83  		final String lowerSearch = search.toLowerCase();
84  		int delta = Integer.MAX_VALUE;
85  		for (final Entry<String, Class<? extends EggType<?>>> entry : classesSimpleName.entrySet()) {
86  			final String str = entry.getKey();
87  			if (str.toLowerCase().startsWith(lowerSearch)) {
88  				final int curDelta = str.length() - lowerSearch.length();
89  				if (curDelta < delta) {
90  					found = entry.getValue();
91  					delta = curDelta;
92  				}
93  				if (curDelta == 0)
94  					break;
95  			}
96  		}
97  		if (found == null)
98  			throw new ClassNotFoundException("Can't find the class " + search);
99  		return found;
100 	}

107 	@Override
108 	protected Class<? extends EggType<?>> findClass(final String name)
109 			throws ClassNotFoundException {
110 		Class<? extends EggType<?>> clazz = classes.get(name);
111 		if (clazz == null)
112 			clazz = classesSimpleName.get(name);
113 		if (clazz == null)
114 			clazz = matchClassName(name);
115 		if (clazz == null)
116 			throw new ClassNotFoundException();
117 		return clazz;
118 	}

120 	/***
121 	 * /** Scans all classes accessible from the context class loader which
122 	 * belong to the given package and subpackages. Adapted from
123 	 * http://snippets.dzone.com/posts/show/4831 and extended to support use of
124 	 * JAR files
125 	 * 
126 	 * @param packageName
127 	 *            The base package
128 	 * 
129 	 * @param classLoader
130 	 *            the parent class loader.
131 	 * @return
132 	 */
133 	private static List<Class<?>> getClassesInPackage(final String packageName,
134 			final ClassLoader classLoader) {
135 		try {
136 			assert classLoader != null;
137 			final String path = packageName.replace('.', '/');
138 			final Enumeration<URL> resources = classLoader.getResources(path);
139 			final List<String> dirs = new ArrayList<String>();
140 			while (resources.hasMoreElements()) {
141 				final URL resource = resources.nextElement();
142 				dirs.add(resource.getFile());
143 			}
144 			final SortedSet<String> classes = new TreeSet<String>();
145 			for (final String directory : dirs) {
146 				classes.addAll(findClasses(directory, packageName));
147 			}
148 			final List<Class<?>> classList = new ArrayList<Class<?>>();
149 			for (final String clazz : classes) {
150 				classList.add(Class.forName(clazz, true, classLoader));
151 			}
152 			return classList;
153 		} catch (final Exception e) {
154 			e.printStackTrace();
155 			return new ArrayList<Class<?>>();
156 		}
157 	}

159 	/***
160 	 * Recursive method used to find all classes in a given path (directory or
161 	 * zip file url). Directories are searched recursively. (zip files are
162 	 * Adapted from http://snippets.dzone.com/posts/show/4831 and extended to
163 	 * support use of JAR files
164 	 * 
165 	 * @param path
166 	 *            The base directory or url from which to search.
167 	 * @param packageName
168 	 *            The package name for classes found inside the base directory
169 	 * @param regex
170 	 *            an optional class name pattern. e.g. .*Test
171 	 * @return The classes
172 	 */
173 	private static SortedSet<String> findClasses(final String path, final String packageName)
174 			throws Exception {
175 		final TreeSet<String> classes = new TreeSet<String>();
176 		if (path.startsWith("file:") && path.contains("!")) {
177 			final String[] split = path.split("!");
178 			final URL jar = new URL(split[0]);
179 			final ZipInputStream zip = new ZipInputStream(jar.openStream());
180 			ZipEntry entry;
181 			while ((entry = zip.getNextEntry()) != null) {
182 				if (entry.getName().endsWith(".class")) {
183 					final String className = entry.getName().replaceAll("[$].*", "")
184 							.replaceAll("[.]class", "").replace('/', '.');
185 					if (className.startsWith(packageName)
186 							&& (regex == null || regex.matcher(className).matches()))
187 						classes.add(className);
188 				}
189 			}
190 		}
191 		final File dir = new File(path);
192 		if (!dir.exists()) {
193 			return classes;
194 		}
195 		final File[] files = dir.listFiles();
196 		for (final File file : files) {
197 			if (file.isDirectory()) {
198 				assert !file.getName().contains(".");
199 				classes.addAll(findClasses(file.getAbsolutePath(),
200 						packageName + "." + file.getName()));
201 			} else if (file.getName().endsWith(".class")) {
202 				final String className = packageName + '.'
203 						+ file.getName().substring(0, file.getName().length() - 6);
204 				if (regex == null || regex.matcher(className).matches())
205 					classes.add(className);
206 			}
207 		}
208 		return classes;
209 	}
