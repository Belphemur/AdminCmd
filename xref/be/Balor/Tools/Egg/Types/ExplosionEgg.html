17  package be.Balor.Tools.Egg.Types;
18  
19  import org.bukkit.entity.Egg;
20  import org.bukkit.entity.Player;
21  import org.bukkit.event.player.PlayerEggThrowEvent;
22  
23  import be.Balor.Manager.Commands.CommandArgs;
24  import be.Balor.Tools.Utils;
25  import be.Balor.Tools.Egg.EggType;
26  import be.Balor.Tools.Egg.Exceptions.ProcessingArgsException;
27  import be.Balor.bukkit.AdminCmd.ConfigEnum;
28  
29  /***
30   * @author Balor (aka Antoine Aflalo)
31   * 
32   */
33  public class ExplosionEgg extends EggType<Float> {
34  
35  	/***
36  	 * 
37  	 */
38  	private static final long serialVersionUID = 729116147611485304L;
39  
40  	/*
41  	 * (non-Javadoc)
42  	 * 
43  	 * @see be.Balor.Tools.Egg.EggType#execute(org.bukkit.event.player.
44  	 * PlayerEggThrowEvent)
45  	 */
46  	@Override
47  	public void onEvent(final PlayerEggThrowEvent event) {
48  		final Egg egg = event.getEgg();
49  		egg.remove();
50  		event.setHatching(false);
51  		egg.getWorld().createExplosion(egg.getLocation(), value);
52  	}
53  
54  	/*
55  	 * (non-Javadoc)
56  	 * 
57  	 * @see
58  	 * be.Balor.Tools.Egg.EggType#processArguments(be.Balor.Manager.Commands
59  	 * .CommandArgs)
60  	 */
61  	@Override
62  	protected void processArguments(final Player sender, final CommandArgs args)
63  			throws ProcessingArgsException {
64  		float power = ConfigEnum.DEGG_EX_RADIUS.getFloat();
65  		if (args.hasFlag('p')) {
66  			final String flag = args.getValueFlag('p');
67  			try {
68  				power = Float.parseFloat(flag);
69  			} catch (final NumberFormatException e) {
70  				Utils.sI18n(sender, "NaN", "number", flag);
71  				return;
72  			}
73  		}
74  		value = power > ConfigEnum.MAXEGG_EX_RADIUS.getFloat() ? ConfigEnum.MAXEGG_EX_RADIUS
75  				.getFloat() : power;
76  
77  	}
78  
79  }
