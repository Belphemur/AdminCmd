41  public class KillerEgg extends SimpleRadiusEgg {
42  
43  	/***
44  	 * 
45  	 */
46  	private static final long serialVersionUID = -7763897329319981939L;
47  
48  	/***
49  	 * 
50  	 */
51  	public KillerEgg() {
52  		super(ConfigEnum.DEGG_KILL_RADIUS.getInt(), ConfigEnum.MAXEGG_KILL_RADIUS.getInt());
53  	}
54  
55  	/*
56  	 * (non-Javadoc)
57  	 * 
58  	 * @see be.Balor.Tools.Egg.EggType#onEvent(org.bukkit.event.player.
59  	 * PlayerEggThrowEvent)
60  	 */
61  	@Override
62  	public void onEvent(final PlayerEggThrowEvent event) {
63  		event.setHatching(false);
64  		final Location loc = event.getEgg().getLocation();
65  		event.getEgg().remove();
66  		final List<EntityLiving> entities = new ArrayList<EntityLiving>();
67  		final CraftPlayer p = (CraftPlayer) event.getPlayer();
68  		final World w = p.getWorld();
69  		final int radius = value * value;
70  		for (final Object entity : ((CraftWorld) w).getHandle().entityList) {
71  			if (entity instanceof EntityLiving)
72  				entities.add((EntityLiving) entity);
73  		}
74  		ACPluginManager.getScheduler().scheduleAsyncDelayedTask(ACPluginManager.getCorePlugin(),
75  				new Runnable() {
76  
77  					@Override
78  					public void run() {
79  						int count = 0;
80  						for (final EntityLiving entity : entities) {
81  							if (entity.equals(p.getHandle())) {
82  								continue;
83  							}
84  							final Location entityLoc = new Location(w, entity.locX, entity.locY,
85  									entity.locZ, entity.yaw, entity.pitch);
86  							if (entityLoc.distanceSquared(loc) > radius)
87  								continue;
88  							if (entity instanceof EntityPlayer) {
89  								final Player player = (Player) entity.getBukkitEntity();
90  								player.setHealth(0);
91  								count++;
92  								continue;
93  							}
94  							entity.die(DamageSource.playerAttack(p.getHandle()));
95  							entity.die();
96  							count++;
97  						}
98  						p.sendMessage(String.valueOf(count) + " killed.");
99  					}
100 				});
101 	}
