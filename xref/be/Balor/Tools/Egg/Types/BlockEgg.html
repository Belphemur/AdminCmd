51  public class BlockEgg extends RadiusEgg<BlockChangeInfo> {
52  
53  	/***
54  	 * 
55  	 */
56  	private static final long serialVersionUID = 1046757929821301999L;
57  	private int eggNb = 0;
58  	private final Map<Integer, SynchronizedStack<BlockRemanence>> blocksPerEvent = Collections
59  			.synchronizedMap(new HashMap<Integer, SynchronizedStack<BlockRemanence>>());
60  
61  	/***
62  	 * 
63  	 */
64  	public BlockEgg() {
65  		super(ConfigEnum.DEGG_BLOCK_RADIUS.getInt(),
66  				ConfigEnum.MAXEGG_BLOCK_RADIUS.getInt());
67  	}
68  
69  	/***
70  	 * @param defaultRadius
71  	 * @param maxRadius
72  	 */
73  	public BlockEgg(final int defaultRadius, final int maxRadius) {
74  		super(defaultRadius, maxRadius);
75  	}
76  
77  	/*
78  	 * (non-Javadoc)
79  	 * 
80  	 * @see be.Balor.Tools.Egg.EggType#onEvent(org.bukkit.event.player.
81  	 * PlayerEggThrowEvent)
82  	 */
83  	@Override
84  	public void onEvent(final PlayerEggThrowEvent event) {
85  		final int radius = value.getRadius() / 2;
86  		event.getEgg().remove();
87  		event.setHatching(false);
88  		final Location loc = event.getEgg().getLocation();
89  		final SynchronizedStack<BlockRemanence> blocks = new SynchronizedStack<BlockRemanence>();
90  		final World w = loc.getWorld();
91  		if (!Material.getMaterial(value.getBlockTypeId()).isBlock()) {
92  			LocaleHelper.NOT_A_BLOCK.sendLocale(event.getPlayer());
93  			return;
94  		}
95  		if (blockTimeOut() == 0) {
96  			for (int x = loc.getBlockX() - radius; x < loc.getBlockX() + radius; x++) {
97  				for (int z = loc.getBlockZ() - radius; z < loc.getBlockZ()
98  						+ radius; z++) {
99  					for (int y = loc.getBlockY() - radius; y < loc.getBlockY()
100 							+ radius; y++) {
101 						final Block block = w.getBlockAt(x, y, z);
102 						if (block.getTypeId() != Material.AIR.getId()
103 								&& block.getTypeId() != Material.SNOW.getId()) {
104 							continue;
105 						}
106 						IBlockRemanenceFactory.FACTORY.createBlockRemanence(
107 								new SimplifiedLocation(w, x, y, z))
108 								.setBlockType(value.getBlockTypeId());
109 					}
110 				}
111 			}
112 			return;
113 		}
114 		final Integer eventId = eggNb++;
115 		for (int x = loc.getBlockX() - radius; x < loc.getBlockX() + radius; x++) {
116 			for (int z = loc.getBlockZ() - radius; z < loc.getBlockZ() + radius; z++) {
117 				for (int y = loc.getBlockY() - radius; y < loc.getBlockY()
118 						+ radius; y++) {
119 					final int blckId = w.getBlockTypeIdAt(x, y, z);
120 					if (blckId != Material.AIR.getId()
121 							&& blckId != Material.SNOW.getId()) {
122 						continue;
123 					}
124 					final BlockRemanence blk = IBlockRemanenceFactory.FACTORY
125 							.createBlockRemanence(new SimplifiedLocation(w, x,
126 									y, z));
127 					blk.setBlockType(value.getBlockTypeId());
128 					blocks.add(blk);
129 
130 				}
131 			}
132 		}
133 		blocksPerEvent.put(eventId, blocks);
134 		ACPluginManager.getScheduler().scheduleSyncDelayedTask(
135 				ACPluginManager.getCorePlugin(), new Runnable() {
136 
137 					@Override
138 					public void run() {
139 						final SynchronizedStack<BlockRemanence> blocks = blocksPerEvent
140 								.get(eventId);
141 						while (!blocks.empty()) {
142 							blocks.pop().returnToThePast();
143 						}
144 						blocksPerEvent.remove(eventId);
145 
146 					}
147 				}, blockTimeOut() * Utils.secInTick);
148 
149 	}
150 
151 	protected int blockTimeOut() {
152 		return ConfigEnum.EGG_BLOCK_TIMEOUT.getInt();
153 	}
154 
155 	/*
156 	 * (non-Javadoc)
157 	 * 
158 	 * @see
159 	 * be.Balor.Tools.Egg.EggType#processArguments(org.bukkit.entity.Player,
160 	 * be.Balor.Manager.Commands.CommandArgs)
161 	 */
162 	@Override
163 	protected void processArguments(final Player sender, final CommandArgs args)
164 			throws ProcessingArgsException {
165 		final String block = args.getValueFlag('b');
166 		if (block == null) {
167 			throw new ParameterMissingException('b',
168 					LocaleHelper.EGG_PARAM_BLOCK.getLocale());
169 		}
170 		final int radius = getRadius(sender, args);
171 		if (radius == -1) {
172 			return;
173 		}
174 		final MaterialContainer mat = ACHelper.getInstance().checkMaterial(
175 				sender, block);
176 		if (mat.isNull()) {
177 			final HashMap<String, String> replace = new HashMap<String, String>();
178 			replace.put("type", LocaleHelper.TYPE_MAT.getLocale());
179 			replace.put("value", block);
180 			throw new ProcessingArgsException(ExceptionType.CUSTOM,
181 					LocaleHelper.DONT_EXISTS.getLocale(replace));
182 		}
183 		value = new BlockChangeInfo(mat.getMaterial().getId(), radius);
184 
185 	}
