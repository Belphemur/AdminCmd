17  package be.Balor.Tools.Debug;
18  
19  import java.util.logging.Level;
20  import java.util.logging.Logger;
21  
22  import org.bukkit.ChatColor;
23  
24  /***
25   * @author Balor (aka Antoine Aflalo)
26   * 
27   */
28  public class ACLogger {
29  
30  	protected static final Logger logger = Logger.getLogger("Minecraft");
31  	protected static final String prefix = "[AdminCmd] ";
32  
33  	public static void severe(final String string, final Throwable ex) {
34  		logger.log(Level.SEVERE, prefix.concat(ChatColor.stripColor(string)),
35  				ex);
36  	}
37  
38  	public static void severe(final String string) {
39  		logger.log(Level.SEVERE, prefix.concat(ChatColor.stripColor(string)));
40  	}
41  
42  	public static void info(final String string) {
43  		logger.log(Level.INFO, prefix.concat(ChatColor.stripColor(string)));
44  	}
45  
46  	public static void warning(final String string) {
47  		logger.log(Level.WARNING, prefix.concat(ChatColor.stripColor(string)));
48  	}
49  
50  	public static void Log(final String txt) {
51  		logger.log(Level.INFO,
52  				String.format(prefix + "%s", ChatColor.stripColor(txt)));
53  	}
54  
55  	public static void Log(final Level loglevel, final String txt) {
56  		Log(loglevel, txt, true);
57  	}
58  
59  	public static void Log(final Level loglevel, final String txt,
60  			final boolean sendReport) {
61  		logger.log(
62  				loglevel,
63  				String.format(prefix + "%s",
64  						txt == null ? "" : ChatColor.stripColor(txt)));
65  	}
66  
67  	public static void Log(final Level loglevel, final String txt,
68  			final Throwable params) {
69  		if (txt == null) {
70  			Log(loglevel, params);
71  		} else {
72  			logger.log(
73  					loglevel,
74  					String.format(prefix + "%s",
75  							txt == null ? "" : ChatColor.stripColor(txt)),
76  					params);
77  		}
78  	}
79  
80  	public static void Log(final Level loglevel, final Throwable err) {
81  		logger.log(loglevel, String.format(prefix + "%s", err == null
82  				? "? unknown exception ?"
83  				: err.getMessage()), err);
84  	}
85  }
