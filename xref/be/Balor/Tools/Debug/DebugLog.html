17  package be.Balor.Tools.Debug;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.util.Stack;
22  import java.util.logging.FileHandler;
23  import java.util.logging.Handler;
24  import java.util.logging.Level;
25  import java.util.logging.Logger;
26  
27  import com.google.common.io.Files;
28  
29  /***
30   * @author Balor (aka Antoine Aflalo)
31   * 
32   */
33  public class DebugLog {
34  	private static class DebugInfo {
35  		private final String blockMsg;
36  		private final Long timeStamp;
37  
38  		/***
39  		 * @param blockMsg
40  		 * @param timeStamp
41  		 */
42  		public DebugInfo(final String blockMsg) {
43  			this.blockMsg = blockMsg;
44  			this.timeStamp = System.currentTimeMillis();
45  		}
46  
47  		/***
48  		 * @return the blockMsg
49  		 */
50  		public String getBlockMsg() {
51  			return blockMsg;
52  		}
53  
54  		/***
55  		 * @return the timeStamp
56  		 */
57  		public Long getElapsedTime() {
58  			return System.currentTimeMillis() - timeStamp;
59  		}
60  
61  	}
62  
63  	public static final Logger INSTANCE = Logger.getLogger("AdminCmd");
64  	public static final String BEGIN_PREFIX = "[BEGIN] ";
65  	public static final String END_PREFIX = "[END] ";
66  	private static final Stack<DebugInfo> debugInfos = new Stack<DebugInfo>();
67  	static {
68  		INSTANCE.setUseParentHandlers(false);
69  		INSTANCE.setLevel(Level.ALL);
70  	}
71  
72  	public static void setFile(final String path) {
73  		FileHandler fh;
74  		try {
75  			// This block configure the logger with handler and formatter
76  			final File file = new File(path + File.separator + "debug.log");
77  			if (file.exists()) {
78  				file.delete();
79  			} else {
80  				Files.createParentDirs(file);
81  			}
82  
83  			fh = new FileHandler(file.getPath(), true);
84  			INSTANCE.addHandler(fh);
85  			fh.setFormatter(new LogFormatter());
86  
87  			// the following statement is used to log any messages
88  			INSTANCE.info("Logger created");
89  
90  		} catch (final SecurityException e) {
91  			e.printStackTrace();
92  		} catch (final IOException e) {
93  			e.printStackTrace();
94  		}
95  	}
96  
97  	public static void stopLogging() {
98  		for (final Handler h : INSTANCE.getHandlers()) {
99  			h.close();
100 			INSTANCE.removeHandler(h);
101 		}
102 	}
103 
104 	/***
105 	 * Begin logging of a block of code
106 	 * 
107 	 * @param msg
108 	 */
109 	public static void beginInfo(final String msg) {
110 		debugInfos.add(new DebugInfo(msg));
111 		INSTANCE.info(getSpaces() + BEGIN_PREFIX + msg);
112 
113 	}
114 
115 	/***
116 	 * @return
117 	 */
118 	private static String getSpaces() {
119 		final StringBuffer space = new StringBuffer();
120 		for (int i = 0; i < debugInfos.size(); i++) {
121 			space.append("\t");
122 		}
123 		return space.toString();
124 	}
125 
126 	/***
127 	 * End logging a block of code
128 	 */
129 	public static void endInfo() {
130 		try {
131 			final String spaces = getSpaces();
132 			final DebugInfo info = debugInfos.pop();
133 			INSTANCE.info(spaces + END_PREFIX + info.getBlockMsg() + " => "
134 					+ info.getElapsedTime() + " milisec");
135 		} catch (final Exception e) {
136 		}
137 
138 	}
139 }
