55  	@Override
56  	public void initImport() {
57  		int number = 0;
58  		final File file = new File(importPath);
59  		if (!file.exists()) {
60  			ConfigEnum.IMPORT_ESSENTIALS.setValue(false);
61  			try {
62  				ConfigEnum.save();
63  			} catch (final IOException e) {
64  			}
65  			ACLogger.info("Import deactivated. Did not find folder 'Essentials' in plugins.");
66  			return;
67  		}
68  		ACLogger.info("Starting import of Essentials data.\n Trying to import User-data.....");
69  		number = importUserData();
70  		ACLogger.info("Data of " + number
71  				+ " user(s) imported. Trying to import spawn point(s)....");
72  		number = importSpawnPoints();
73  		ACLogger.info(number
74  				+ " spawn point(s) imported. Trying to import warp point(s)....");
75  		number = importWarpPoints();
76  		ACLogger.info(number
77  				+ "Warp point(s) imported. Trying to import text-files....");
78  		try {
79  			importTextFiles();
80  		} catch (final IOException e) {
81  			ACLogger.info("[ERROR] Failed to import rules.txt and motd.txt. For more information refer to the debug.log!");
82  			DebugLog.INSTANCE.log(Level.INFO,
83  					"[ERROR] Failed to import rules.txt and motd.txt.", e);
84  			return;
85  		}
86  		ACLogger.info("Text files imported.\n Import finished successfully, deactivating"
87  				+ "import-option in the configuration file...");
88  		ConfigEnum.IMPORT_ESSENTIALS.setValue(false);
89  		try {
90  			ConfigEnum.save();
91  		} catch (final IOException e) {
92  		}
93  		ACLogger.info("Import deactivated, have fun with your old data in AdminCmd.");
94  	}

101 	@Override
102 	public int importUserData() {
103 		int counter = 0;
104 		final List<File> files = filter.getFiles(new File(importPath
105 				+ File.separator + "userdata"), filter.new PatternFilter(
106 				Type.FILE, ".yml"), false);
107 		ExtendedConfiguration esUserFile = null;
108 		String playerName;
109 		boolean homes = false, lastLoc = false, connect = false, newPlayer = false, ip = false, mute = false, god = false;
110 		for (final File f : files) {
111 			playerName = f.getName().substring(0, f.getName().length() - 4);
112 			ACPlayer p = ACPlayer.getPlayer(playerName);
113 			if (p == null) {
114 				playerName = playerName.substring(0, 1).toUpperCase()
115 						+ playerName.substring(1);
116 			}
117 			p = ACPlayer.getPlayer(playerName);
118 			esUserFile = ExtendedConfiguration.loadConfiguration(f);
119 			homes = ImportTools.importESHomes(esUserFile, playerName);
120 			lastLoc = ImportTools.importESLastLocation(esUserFile, playerName);
121 			if (p != null && esUserFile != null) {
122 				p.setInformation("lastConnection",
123 						esUserFile.getLong("timestamps.login"));
124 				p.setInformation("lastDisconnect",
125 						esUserFile.getLong("timestamps.logout"));
126 				connect = true;
127 				p.setInformation("firstTime",
128 						esUserFile.getBoolean("newplayer"));
129 				newPlayer = true;
130 				p.setInformation("last-ip", esUserFile.getString("ipAddress"));
131 				ip = true;
132 				if (esUserFile.getBoolean("muted")) {
133 					p.setPower(be.Balor.Tools.Type.MUTED,
134 							"Permanently muted by Server Admin");
135 					mute = true;
136 				}
137 				if (esUserFile.getBoolean("godmode")) {
138 					p.setPower(be.Balor.Tools.Type.GOD);
139 					god = true;
140 				}
141 			}
142 			if (homes || lastLoc || connect || ip || newPlayer || mute || god) {
143 				counter++;
144 			}
145 		}
146 		return counter;
147 	}

154 	@Override
155 	public int importWarpPoints() {
156 		List<File> files = filter.getFiles(new File(importPath + File.separator
157 				+ "warps"), filter.new PatternFilter(Type.FILE, ".yml"), false);
158 		ExtendedConfiguration esWarp = null;
159 		Location acWarp = null;
160 		ACWorld world = null;
161 		String name = "";
162 		int counter = 0;
163 
164 		for (final File f : files) {
165 			esWarp = ExtendedConfiguration.loadConfiguration(f);
166 			if (esWarp == null) {
167 				ACLogger.info("[ERROR] Could not import WarpPoint: "
168 						+ f.getName().substring(0, f.getName().length() - 4));
169 				continue;
170 			}
171 			world = ACWorld.getWorld(esWarp.getString("world"));
172 			if (world == null) {
173 				continue;
174 			}
175 			try {
176 				name = esWarp.getString("name");
177 				acWarp = ImportTools.buildLocation(esWarp, world);
178 			} catch (final Exception e) {
179 				ACLogger.info("[ERROR] Could not import WarpPoint: "
180 						+ f.getName().substring(0, f.getName().length() - 4));
181 				continue;
182 			}
183 			world.addWarp(name, acWarp);
184 			counter++;
185 		}
186 		files = filter.getFiles(new File(importPath), filter.new PatternFilter(
187 				Type.FILE, ".yml"), false);
188 		ExtendedConfiguration esJails = null;
189 		for (final File f : files) {
190 			if (!f.getName().contains("jails")) {
191 				continue;
192 			}
193 			esJails = ExtendedConfiguration.loadConfiguration(f);
194 			if (esJails == null) {
195 				ACLogger.info("[ERROR] Could not import Jails.");
196 				continue;
197 			}
198 			ConfigurationSection jails = null;
199 			jails = esJails.getConfigurationSection("jails");
200 			if (jails == null) {
201 				ACLogger.info("[ERROR] Could not import Jails.");
202 				continue;
203 			}
204 			final Set<String> keys = jails.getKeys(false);
205 			if (keys == null) {
206 				ACLogger.info("[ERROR] Could not import Jails.");
207 				continue;
208 			}
209 			ConfigurationSection jail = null;
210 			for (final String jName : keys) {
211 				if (jName == null) {
212 					ACLogger.info("[ERROR] Could not import jail "
213 							+ String.valueOf(jName));
214 					continue;
215 				}
216 				jail = jails.getConfigurationSection(jName);
217 				world = ACWorld.getWorld(esWarp.getString("world"));
218 				if (world == null) {
219 					continue;
220 				}
221 				try {
222 					acWarp = ImportTools.buildLocation(jail, world);
223 				} catch (final Exception e) {
224 					ACLogger.info("[ERROR] Could not import jail: " + jName);
225 					continue;
226 				}
227 				world.addWarp("jail:" + jName, acWarp);
228 				counter++;
229 			}
230 		}
231 		return counter;
232 	}

239 	@Override
240 	public int importSpawnPoints() {
241 		final List<File> files = filter.getFiles(new File(importPath),
242 				filter.new PatternFilter(Type.FILE, ".yml"), false);
243 		ExtendedConfiguration esSpawns = null;
244 		ConfigurationSection esSpawn = null;
245 		Location acSpawn = null;
246 		ACWorld world = null;
247 		int counter = 0;
248 		for (final File f : files) {
249 			if (!f.getName().contains("spawn")) {
250 				continue;
251 			}
252 			esSpawns = ExtendedConfiguration.loadConfiguration(f);
253 			esSpawn = esSpawns.getConfigurationSection("spawns");
254 			if (esSpawn == null) {
255 				ACLogger.info("[ERROR] Could not import Spawns.");
256 				continue;
257 			}
258 			final Set<String> keys = esSpawn.getKeys(false);
259 			if (keys == null) {
260 				ACLogger.info("[ERROR] Could not import Spawns.");
261 				continue;
262 			}
263 			ConfigurationSection spawn = null;
264 			for (final String sName : keys) {
265 				if (sName == null) {
266 					ACLogger.info("[ERROR] Could not import spawn "
267 							+ String.valueOf(sName));
268 					continue;
269 				}
270 				spawn = esSpawn.getConfigurationSection(sName);
271 				world = ACWorld.getWorld(spawn.getString("world"));
272 				if (world == null) {
273 					continue;
274 				}
275 				try {
276 					acSpawn = ImportTools.buildLocation(spawn, world);
277 				} catch (final Exception e) {
278 					ACLogger.info("[ERROR] Could not import spawn: " + sName);
279 					continue;
280 				}
281 				if (sName.equalsIgnoreCase("default")) {
282 					world.setSpawn(acSpawn);
283 				} else {
284 					world.setGroupSpawn(sName, acSpawn);
285 				}
286 				counter++;
287 			}
288 		}
289 		return counter;
290 	}

298 	public void importTextFiles() throws IOException {
299 		final List<File> files = filter.getFiles(new File(importPath),
300 				filter.new PatternFilter(Type.FILE, ".txt"), false);
301 		for (final File f : files) {
302 			if (f.getName().contains("rules")) {
303 				ImportTools.copyTextFile(
304 						f, FileManager.getInstance().getFile(ACHelper.getInstance()
305 								.getCoreInstance().getDataFolder().getAbsolutePath(),
306 								"rules.txt"));
307 			} else if (f.getName().contains("motd")) {
308 				ImportTools.copyTextFile(
309 						f, FileManager.getInstance().getFile(ACHelper.getInstance()
310 								.getCoreInstance().getDataFolder().getAbsolutePath(),
311 								"motd.txt"));
312 			} else {
313 				continue;
314 			}
315 		}
316 	}
