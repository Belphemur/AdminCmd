51  	public static void copyTextFile(final File sourcefile, final File targeFile)
52  			throws IOException {
53  		assert sourcefile != null && targeFile != null;
54  		final BufferedReader in = new BufferedReader(new UnicodeReader(
55  				new FileInputStream(sourcefile), "UTF-8"));
56  		final StringBuffer result = new StringBuffer();
57  		String temp;
58  		while ((temp = in.readLine()) != null) {
59  			result.append(temp + System.getProperty("line.separator"));
60  		}
61  		in.close();
62  		UnicodeUtil.saveUTF8File(targeFile, result.toString(), false);
63  	}

65  	public static boolean importESHomes(final ExtendedConfiguration userFile,
66  			final String playerName) {
67  		Set<String> nodeList;
68  		ConfigurationSection home = null, homes = null;
69  		Location homeLoc = null;
70  		if (userFile == null) {
71  			ACLogger.info("[ERROR] Could not import data of user: "
72  					+ playerName);
73  			return false;
74  		}
75  		homes = userFile.getConfigurationSection("homes");
76  		if (homes != null) {
77  			nodeList = homes.getKeys(false);
78  			for (final String name : nodeList) {
79  				home = homes.getConfigurationSection(name);
80  				final ACWorld w = ACWorld.getWorld(home.getString("world"));
81  				try {
82  					homeLoc = buildLocation(home, w);
83  				} catch (final Exception e) {
84  					ACLogger.info("[ERROR] Could not import homes of user: "
85  							+ playerName);
86  					DebugLog.INSTANCE.log(Level.WARNING,
87  							"[ERROR] Could not import homes of user: "
88  									+ playerName, e);
89  					return false;
90  				}
91  				ACPlayer.getPlayer(playerName).setHome(name, homeLoc);
92  			}
93  		} else {
94  			if (ConfigEnum.VERBOSE.getBoolean()) {
95  				ACLogger.info("User " + playerName + " did not have any homes.");
96  			}
97  			return false;
98  		}
99  		return false;
100 	}

102 	public static Location buildLocation(final ConfigurationSection section,
103 			final ACWorld world) {
104 		Location loc = null;
105 		assert section != null && world != null;
106 		double x = 0D, y = 0D, z = 0D;
107 		float pitch = 0F, yaw = 0F;
108 		x = section.getDouble("x");
109 		y = section.getDouble("y");
110 		z = section.getDouble("z");
111 		yaw = Float.parseFloat(section.getString("yaw"));
112 		pitch = Float.parseFloat(section.getString("pitch"));
113 		loc = new Location(world.getHandle(), x, y, z, yaw, pitch);
114 		return loc;
115 	}

117 	public static boolean importESLastLocation(
118 			final ExtendedConfiguration uerFile, final String playerName) {
119 		Location lastLoc = null;
120 		ConfigurationSection iLoc = null;
121 		iLoc = uerFile.getConfigurationSection("lastlocation");
122 		if (iLoc == null) {
123 			DebugLog.INSTANCE
124 					.info("Can find the lastLocation in the configuration file for "
125 							+ playerName + " in file " + uerFile);
126 			return false;
127 		}
128 		final ACWorld w = ACWorld.getWorld(iLoc.getString("world"));
129 		try {
130 			lastLoc = buildLocation(iLoc, w);
131 		} catch (final Exception e) {
132 			ACLogger.info("[ERROR] Could not import the last location of user: "
133 					+ playerName);
134 			DebugLog.INSTANCE.log(Level.WARNING,
135 					"[ERROR] Could not import last location of user: "
136 							+ playerName, e);
137 			return false;
138 		}
139 		ACPlayer.getPlayer(playerName).setLastLocation(lastLoc);
140 		return true;
141 	}
