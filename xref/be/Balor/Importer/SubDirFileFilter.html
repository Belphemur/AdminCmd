19  package be.Balor.Importer;
20  
21  import java.io.File;
22  import java.io.FileFilter;
23  import java.util.ArrayList;
24  import java.util.Arrays;
25  import java.util.List;
26  import java.util.regex.Pattern;
27  
28  /***
29   * Implements a {@link java.io.FileFilter FileFilter} to return files matching a
30   * given pattern.</br> It also can search all sub-directories by a recursive
31   * implementation.</br> You can also use the Type to only search for DIRS. </p>
32   * Example:
33   * 
34   * <pre>
35   * import be.Balor.Importer.SubDirFileFilter.Type;
36   * import be.Balor.Importer.SubDirFileFilter.PatternFilter;
37   * 
38   * SubDirFileFilter filter = new SubDirFileFilter();
39   * List< File > files = filter.getFiles(new File(""),
40   *                       filter.new PatternFilter(Type.ALL, ".java"),
41   *                       true);
42   * </pre>
43   * 
44   * @author Lathanael (aka Philippe Leipold)
45   */
46  public class SubDirFileFilter {
47  
48  	/***
49  	 * Returns a List<File> of all Dirs/Files depending on the filter. If
50  	 * recursive is set to true it will get dirs/files in sub-diretories too.
51  	 * 
52  	 * @param basedir
53  	 * @param filter
54  	 * @param recursive
55  	 * @return A list of files and/or directories matching the input pattern
56  	 */
57  	public final List<File> getFiles(final File basedir,
58  			final FileFilter filter, boolean recursive) {
59  		List<File> files = new ArrayList<File>();
60  		if (basedir != null && basedir.isDirectory()) {
61  			if (recursive)
62  				for (File subdir : basedir.listFiles())
63  					files.addAll(this.getFiles(subdir, filter, recursive));
64  			files.addAll(Arrays.asList(basedir.listFiles(filter)));
65  		}
66  		return files;
67  	}
68  
69  	/***
70  	 * Defines for what type of file should be looked for!
71  	 */
72  	public enum Type implements FileFilter {
73  		FILE, DIR, ALL;
74  
75  		public boolean accept(final File file) {
76  			return file != null
77  					&& (this == ALL || (this == FILE && file.isFile()) || (this == DIR && file
78  							.isDirectory()));
79  		}
80  	}
81  
82  	/***
83  	 * Gets a file matching a given suffix pattern like ".java"
84  	 */
85  	public class PatternFilter implements FileFilter {
86  		private final Type type;
87  		private final String pattern;
88  
89  		public PatternFilter(final Type type, final String pattern) {
90  			this.type = type;
91  			this.pattern = "^.*" + Pattern.quote(pattern) + "$";
92  		}
93  
94  		public boolean accept(final File file) {
95  			return type.accept(file) && file.getName().matches(pattern);
96  		}
97  	}
98  }
