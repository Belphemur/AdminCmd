47  	@Override
48  	public void open() throws SQLException {
49  		initialize();
50  		String url = "";
51  		url = "jdbc:mysql://" + this.hostname + ":" + this.portnmbr + "/"
52  				+ this.database;
53  		this.connection = DriverManager.getConnection(url, this.username,

57  	@Override
58  	public ResultSet query(final String query) {
59  		Statement statement = null;
60  		ResultSet result = null;
61  		try {
62  			autoReconnect();
63  			synchronized (connection) {
64  				statement = this.connection.createStatement();
65  
66  				switch (this.getStatement(query)) {
67  				case SELECT:
68  					result = statement.executeQuery(query);
69  					break;
70  
71  				case INSERT:
72  				case UPDATE:
73  				case DELETE:
74  				case CREATE:
75  				case ALTER:
76  				case DROP:
77  				case TRUNCATE:
78  				case RENAME:
79  				case DO:
80  				case REPLACE:
81  				case LOAD:
82  				case HANDLER:
83  				case CALL:
84  					this.lastUpdate = statement.executeUpdate(query);
85  					break;
86  
87  				default:
88  					result = statement.executeQuery(query);
89  				}
90  			}
91  			return result;
92  		} catch (final SQLException e) {
93  			this.writeError("SQL exception in query(): " + e.getMessage(),
94  					false);
95  		}
96  		return result;
97  	}

99  	@Override
100 	public boolean wipeTable(final String table) {
101 		Statement statement = null;
102 		String query = null;
103 		try {
104 			if (!this.checkTable(table)) {
105 				this.writeError("Table \"" + table
106 						+ "\" in wipeTable() does not exist.", true);
107 				return false;
108 			}
109 			query = "DELETE FROM " + table + ";";
110 			synchronized (connection) {
111 				statement = connection.createStatement();
112 				statement.executeQuery(query);
113 			}
114 			return true;
115 		} catch (final SQLException e) {
116 			if (!e.toString().contains("not return ResultSet")) {
117 				return false;
118 			}
119 		}
120 		return false;
121 	}

140 		try {

142 			autoReconnect();
143 			synchronized (connection) {
144 				if (getStatement(query) == Statements.INSERT) {
145 					ps = connection.prepareStatement(query,
146 							PreparedStatement.RETURN_GENERATED_KEYS);
147 				} else {
148 					ps = connection.prepareStatement(query);
149 				}
150 			}
151 			return ps;
152 		} catch (final SQLException e) {
153 			if (!e.toString().contains("not return ResultSet")) {
154 				this.writeError(
155 						"SQL exception in prepare(): " + e.getMessage(), false);
156 			}
157 		}
