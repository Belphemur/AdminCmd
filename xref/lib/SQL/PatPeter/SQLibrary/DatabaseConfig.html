17  package lib.SQL.PatPeter.SQLibrary;
18  
19  import java.util.EnumMap;
20  import java.util.HashSet;
21  import java.util.Map;
22  import java.util.Set;
23  import java.util.logging.Logger;
24  
25  import org.bukkit.configuration.InvalidConfigurationException;
26  
27  /***
28   * @author Balor (aka Antoine Aflalo)
29   * 
30   */
31  public class DatabaseConfig {
32  	private final Map<Parameter, String> config = new EnumMap<Parameter, String>(
33  			Parameter.class);
34  	private DatabaseType type;
35  	private Logger log;
36  
37  	public enum DatabaseType {
38  		MYSQL,
39  		SQLITE,
40  		ALL;
41  	}
42  
43  	public enum Parameter {
44  		HOSTNAME(
45  				DatabaseType.MYSQL),
46  		USER(
47  				DatabaseType.MYSQL),
48  		PASSWORD(
49  				DatabaseType.MYSQL),
50  		PORT_NUMBER(
51  				DatabaseType.MYSQL),
52  		DATABASE(
53  				DatabaseType.MYSQL),
54  		DB_PREFIX(
55  				DatabaseType.ALL),
56  		DB_LOCATION(
57  				DatabaseType.SQLITE),
58  		DB_NAME(
59  				DatabaseType.SQLITE);
60  		private final Set<DatabaseType> dbTypes = new HashSet<DatabaseType>();
61  		private static Map<DatabaseType, Integer> count;
62  
63  		/***
64  		 * 
65  		 */
66  		private Parameter(final DatabaseType... type) {
67  			for (final DatabaseType element : type) {
68  				dbTypes.add(element);
69  				updateCount(element);
70  			}
71  
72  		}
73  
74  		public boolean validParam(final DatabaseType toCheck) {
75  			if (dbTypes.contains(DatabaseType.ALL)) {
76  				return true;
77  			}
78  			if (dbTypes.contains(toCheck)) {
79  				return true;
80  			}
81  			return false;
82  
83  		}
84  
85  		private static void updateCount(final DatabaseType type) {
86  			if (count == null) {
87  				count = new EnumMap<DatabaseType, Integer>(DatabaseType.class);
88  			}
89  			Integer nb = count.get(type);
90  			if (nb == null) {
91  				nb = 1;
92  			} else {
93  				nb++;
94  			}
95  			count.put(type, nb);
96  		}
97  
98  		public static int getCount(final DatabaseType type) {
99  			final int nb = count.get(DatabaseType.ALL) + count.get(type);
100 			return nb;
101 		}
102 	}
103 
104 	/***
105 	 * @param type
106 	 *            the type to set
107 	 */
108 	public void setType(final DatabaseType type)
109 			throws IllegalArgumentException {
110 		if (type == DatabaseType.ALL) {
111 			throw new IllegalArgumentException(
112 					"You can't set your database type to ALL");
113 		}
114 		this.type = type;
115 	}
116 
117 	/***
118 	 * @param log
119 	 *            the log to set
120 	 */
121 	public void setLog(final Logger log) {
122 		this.log = log;
123 	}
124 
125 	/***
126 	 * @return the type
127 	 */
128 	public DatabaseType getType() {
129 		return type;
130 	}
131 
132 	/***
133 	 * @return the log
134 	 */
135 	public Logger getLog() {
136 		return log;
137 	}
138 
139 	public DatabaseConfig setParameter(final Parameter param, final String value)
140 			throws NullPointerException, InvalidConfigurationException {
141 		if (this.type == null) {
142 			throw new NullPointerException(
143 					"You must set the type of the database first");
144 		}
145 		if (!param.validParam(type)) {
146 			throw new InvalidConfigurationException(param.toString()
147 					+ " is invalid for a database type of : " + type.toString());
148 		}
149 		config.put(param, value);
150 		return this;
151 
152 	}
153 
154 	public String getParameter(final Parameter param) {
155 		return config.get(param);
156 	}
157 
158 	public boolean isValid() throws InvalidConfigurationException {
159 		if (log == null) {
160 			throw new InvalidConfigurationException(
161 					"You need to set the logger.");
162 		}
163 		return config.size() == Parameter.getCount(type);
164 	}
165 }
