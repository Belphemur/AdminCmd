70  	public ResultSet query(final String query) {
71  		Statement statement = null;
72  		ResultSet result = null;
73  
74  		try {
75  			synchronized (connection) {
76  				statement = connection.createStatement();
77  
78  				switch (this.getStatement(query)) {
79  				case SELECT:
80  					result = statement.executeQuery(query);
81  					break;
82  
83  				case INSERT:
84  				case UPDATE:
85  				case DELETE:
86  				case CREATE:
87  				case ALTER:
88  				case DROP:
89  				case TRUNCATE:
90  				case RENAME:
91  				case DO:
92  				case REPLACE:
93  				case LOAD:
94  				case HANDLER:
95  				case CALL:
96  					this.lastUpdate = statement.executeUpdate(query);
97  					break;
98  
99  				default:
100 					result = statement.executeQuery(query);
101 
102 				}
103 			}
104 			return result;
105 		} catch (final SQLException e) {
106 			if (e.getMessage().toLowerCase().contains("locking")
107 					|| e.getMessage().toLowerCase().contains("locked")) {
108 				return retry(query);
109 			} else {
110 				this.writeError("SQL exception in query(): " + e.getMessage(),
111 						false);
112 			}
113 
114 		}

118 	@Override
119 	public boolean wipeTable(final String table) {
120 		Statement statement = null;
121 		String query = null;
122 		try {
123 			if (!this.checkTable(table)) {
124 				this.writeError("Table \"" + table
125 						+ "\" in wipeTable() does not exist.", true);
126 				return false;
127 			}
128 			query = "DELETE FROM " + table + ";";
129 			synchronized (connection) {
130 				statement = connection.createStatement();
131 				statement.executeQuery(query);
132 			}
133 			return true;
134 		} catch (final SQLException ex) {
135 			if (!(ex.getMessage().toLowerCase().contains("locking") || ex
136 					.getMessage().toLowerCase().contains("locked"))
137 					&& !ex.toString().contains("not return ResultSet")) {
138 				this.writeError("Error at SQL Wipe Table Query: " + ex, false);
139 			}
140 			return false;
141 		}
142 	}

155 	private ResultSet retry(final String query) {
156 		Statement statement = null;
157 		ResultSet result = null;
158 
159 		try {
160 			synchronized (connection) {
161 				statement = connection.createStatement();
162 				result = statement.executeQuery(query);
163 			}
164 			return result;
165 		} catch (final SQLException ex) {
166 			if (ex.getMessage().toLowerCase().contains("locking")
167 					|| ex.getMessage().toLowerCase().contains("locked")) {
168 				this.writeError(
169 						"Please close your previous ResultSet to run the query: \n\t"
170 								+ query, false);
171 			} else {
172 				this.writeError("SQL exception in retry(): " + ex.getMessage(),
173 						false);
174 			}
175 		}
176 
177 		return null;
