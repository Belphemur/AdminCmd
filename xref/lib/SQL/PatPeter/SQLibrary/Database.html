54  	public static void initDb() {
55  		if (DATABASE != null) {
56  			DATABASE.close();
57  		}
58  		final DatabaseConfig config = new DatabaseConfig();
59  		Database db;
60  		final String dbWrapper = ConfigEnum.DATA_WRAPPER.getString();
61  		if (dbWrapper.equalsIgnoreCase("mysql")) {
62  			config.setType(DatabaseType.MYSQL);
63  			try {
64  				config.setParameter(Parameter.HOSTNAME,
65  						ConfigEnum.MYSQL_HOST.getString());
66  				config.setParameter(Parameter.PASSWORD,
67  						ConfigEnum.MYSQL_PASS.getString());
68  				config.setParameter(Parameter.USER,
69  						ConfigEnum.MYSQL_USER.getString());
70  				config.setParameter(Parameter.PORT_NUMBER, "3306");
71  				config.setParameter(Parameter.DATABASE,
72  						ConfigEnum.MYSQL_DB.getString());
73  			} catch (final NullPointerException e) {
74  			} catch (final InvalidConfigurationException e) {
75  			}
76  		} else if (dbWrapper.equalsIgnoreCase("sqlite")) {
77  			config.setType(DatabaseType.SQLITE);
78  			try {
79  				config.setParameter(Parameter.DB_LOCATION, ACHelper
80  						.getInstance().getCoreInstance().getDataFolder()
81  						.getAbsolutePath());
82  				config.setParameter(Parameter.DB_NAME, "admincmd");
83  
84  			} catch (final NullPointerException e) {
85  			} catch (final InvalidConfigurationException e) {
86  			}
87  		}
88  		config.setLog(ACHelper.getInstance().getCoreInstance().getLogger());
89  		try {
90  			config.setParameter(Parameter.DB_PREFIX, "[AdminCmd Database]");
91  		} catch (final NullPointerException e1) {
92  
93  		} catch (final InvalidConfigurationException e1) {
94  
95  		}
96  
97  		try {
98  			if (config.getType() == null) {
99  				db = null;
100 			} else {
101 				db = DatabaseFactory.createDatabase(config);
102 			}
103 		} catch (final InvalidConfigurationException e) {
104 			ACLogger.severe("Problem while trying to load the Database", e);
105 			db = null;
106 		}
107 		DATABASE = db;
108 		DebugLog.INSTANCE.info("Database initialization done");
109 	}

333 			return Statements.DROP;

363 	public boolean createTable(final String query) {
364 		Statement statement = null;
365 		try {
366 			if (query.equals("") || query == null) {
367 				this.writeError(
368 						"Parameter 'query' empty or null in createTable().",
369 						true);
370 				return false;
371 			}
372 			synchronized (connection) {
373 				statement = connection.createStatement();
374 				statement.execute(query);
375 			}
376 			return true;
377 		} catch (final SQLException ex) {
378 			this.writeError(ex.getMessage(), true);
379 			return false;
380 		}
381 	}

384 	 * <b>checkTable</b><br>
385 	 * <br>
386 	 * &nbsp;&nbsp;Checks a table in a database based on the table's name. <br>
387 	 * <br>
388 	 * 
389 	 * @param table
390 	 *            - name of the table to check.
391 	 * @return success of the method.
392 	 */
393 	public boolean checkTable(final String table) {
394 		DatabaseMetaData dbm = null;
395 		try {
396 			synchronized (connection) {
397 				dbm = this.connection.getMetaData();
398 			}
399 
400 			final ResultSet tables = dbm.getTables(null, null, table, null);
401 			if (tables.next()) {
402 				return true;
403 			} else {
404 				return false;
405 			}
406 		} catch (final SQLException e) {
407 			this.writeError("Failed to check if table \"" + table
408 					+ "\" exists: " + e.getMessage(), true);
409 			return false;
410 		}
411 	}
